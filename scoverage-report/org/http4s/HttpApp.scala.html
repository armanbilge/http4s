19 import cats.Applicative

23 /** Functions for creating [[HttpApp]] kleislis. */
24 object HttpApp {
25 
26   /** Lifts a function into an [[HttpApp]].  The application of `run` is
27     * suspended in `F` to permit more efficient combination of routes
28     * via `SemigroupK`.
29     *
30     * @tparam F the effect of the [[HttpApp]].
31     * @param run the function to lift
32     * @return an [[HttpApp]] that wraps `run`
33     */
34   def apply[F[_]: Monad](run: Request[F] => F[Response[F]]): HttpApp[F] =
35     Http(run)
36 
37   /** Lifts an effectful [[Response]] into an [[HttpApp]].
38     *
39     * @tparam F the effect of the [[HttpApp]]
40     * @param fr the effectful [[Response]] to lift
41     * @return an [[HttpApp]] that always returns `fr`
42     */
43   def liftF[F[_]](fr: F[Response[F]]): HttpApp[F] =
44     Kleisli.liftF(fr)
45 
46   /** Lifts a [[Response]] into an [[HttpApp]].
47     *
48     * @tparam F the effect of the [[HttpApp]]
49     * @param r the [[Response]] to lift
50     * @return an [[Http]] that always returns `r` in effect `F`
51     */
52   def pure[F[_]: Applicative](r: Response[F]): HttpApp[F] =
53     Kleisli.pure(r)
54 
55   /** Transforms an [[HttpApp]] on its input.  The application of the
56     * transformed function is suspended in `F` to permit more
57     * efficient combination of routes via `SemigroupK`.
58     *
59     * @tparam F the effect of the [[HttpApp]]
60     * @param f a function to apply to the [[Request]]
61     * @param fa the [[HttpApp]] to transform
62     * @return An [[HttpApp]] whose input is transformed by `f` before
63     * being applied to `fa`
64     */
65   def local[F[_]](f: Request[F] => Request[F])(fa: HttpApp[F])(implicit F: Monad[F]): HttpApp[F] =
66     Http.local(f)(fa)
67 
68   /** An app that always returns `404 Not Found`. */
69   def notFound[F[_]: Applicative]: HttpApp[F] = pure(Response.notFound)
70 }
