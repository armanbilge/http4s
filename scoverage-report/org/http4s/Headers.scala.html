32 /** A collection of HTTP Headers */
33 final class Headers(val headers: List[Header.Raw]) extends AnyVal {
34 
35   def transform(f: List[Header.Raw] => List[Header.Raw]): Headers =
36     Headers(f(headers))
37 
38   /** Attempt to get a (potentially repeating) header from this collection of headers.
39     *
40     * @return a scala.Option possibly containing the resulting (potentially repeating) header.
41     */
42   def get[A](implicit ev: Header.Select[A]): Option[ev.F[A]] =
43     ev.from(headers).flatMap(_.toOption)
44 
45   /** Attempt to get a (potentially repeating) header and/or any parse errors from this collection of headers.
46     *
47     * @return a scala.Option possibly containing the resulting (potentially repeating) header
48     *         and/or any parse errors.
49     */
50   def getWithWarnings[A](implicit
51       ev: Header.Select[A]
52   ): Option[Ior[NonEmptyList[ParseFailure], ev.F[A]]] =
53     ev.from(headers)
54 
55   /** Returns true if there is at least one header by the specified name. */
56   def contains[A](implicit ev: Header[A, _]): Boolean =
57     headers.exists(_.name == ev.name)
58 
59   /** Attempt to get headers by key from this collection of headers.
60     *
61     * @param key name of the headers to find.
62     * @return a scala.Option possibly containing the resulting collection
63     *         [[cats.data.NonEmptyList]] of [[org.http4s.Header.Raw]].
64     */
65   def get(key: CIString): Option[NonEmptyList[Header.Raw]] = headers.filter(_.name == key).toNel
66 
67   /** Make a new collection adding the specified headers, replacing existing `Single` headers.
68     *
69     * @param in multiple heteregenous headers [[Header]] to append to the new collection, see [[Header.ToRaw]]
70     * @return a new [[Headers]] containing the sum of the initial and input headers
71     */
72   def put(in: Header.ToRaw*): Headers =
73     this ++ Headers(in.values)
74 
75   def ++(those: Headers): Headers =
76     if (those.headers.isEmpty) this
77     else if (this.headers.isEmpty) those
78     else {
79       val thoseNames = mutable.Set.empty[CIString]
80       those.headers.foreach(h => thoseNames.add(h.name))
81       Headers(headers.filterNot(h => thoseNames.contains(h.name)) ++ those.headers)
82     }
83 
84   def add[H: Header[*, Header.Recurring]](h: H): Headers =
85     Headers(this.headers ++ Header.ToRaw.modelledHeadersToRaw(h).values)
86 
87   /** Removes the `Content-Length`, `Content-Range`, `Trailer`, and
88     * `Transfer-Encoding` headers.
89     *
90     *  https://datatracker.ietf.org/doc/html/rfc7231#section-3.3
91     */
92   def removePayloadHeaders: Headers =
93     transform(_.filterNot(h => Headers.PayloadHeaderKeys(h.name)))
94 
95   /** Puts a `Content-Length` header, replacing any existing.  Removes
96     * any existing `chunked` value from the `Transfer-Encoding`
97     * header.  It is critical that the supplied content length
98     * accurately describe the length of the body stream.
99     *
100     * {{{
101     * scala> import org.http4s.headers._
102     * scala> val chunked = Headers(
103     *      |   `Transfer-Encoding`(TransferCoding.chunked),
104     *      |   `Content-Type`(MediaType.text.plain))
105     * scala> chunked.withContentLength(`Content-Length`.unsafeFromLong(1024))
106     * res0: Headers = Headers(Content-Length: 1024, Content-Type: text/plain)
107     *
108     * scala> val chunkedGzipped = Headers(
109     *      |   `Transfer-Encoding`(TransferCoding.chunked, TransferCoding.gzip),
110     *      |   `Content-Type`(MediaType.text.plain))
111     * scala> chunkedGzipped.withContentLength(`Content-Length`.unsafeFromLong(1024))
112     * res1: Headers = Headers(Content-Length: 1024, Transfer-Encoding: gzip, Content-Type: text/plain)
113     *
114     * scala> val const = Headers(
115     *      |   `Content-Length`(2048),
116     *      |   `Content-Type`(MediaType.text.plain))
117     * scala> const.withContentLength(`Content-Length`.unsafeFromLong(1024))
118     * res1: Headers = Headers(Content-Length: 1024, Content-Type: text/plain)
119     * }}}
120     */
121   def withContentLength(contentLength: `Content-Length`): Headers =
122     transform { hs =>
123       val b = List.newBuilder[Header.Raw]
124       b += contentLength.toRaw1
125       hs.foreach { h =>
126         h.name match {
127           case `Transfer-Encoding`.name =>
128             `Transfer-Encoding`
129               .parse(h.value)
130               .redeem(
131                 _ => b += h,
132                 _.filter(_ != TransferCoding.chunked)
133                   .foreach(b += _.toRaw1),
134               )
135           case `Content-Length`.name =>
136             ()
137           case _ =>
138             b += h
139         }
140       }
141       b.result()
142     }
143 
144   def redactSensitive(
145       redactWhen: CIString => Boolean = Headers.SensitiveHeaders.contains
146   ): Headers =
147     transform {
148       _.map {
149         case h if redactWhen(h.name) => Header.Raw(h.name, "<REDACTED>")
150         case h => h
151       }
152     }
153 
154   def foreach(f: Header.Raw => Unit): Unit = headers.foreach(f)
155 
156   override def toString: String =
157     this.show
158 }
159 
160 object Headers {
161   val empty: Headers = Headers(List.empty[Header.Raw])
162 
163   /** Creates a new Headers collection.
164     * The [[Header.ToRaw]] machinery allows the creation of Headers with
165     * variadic and heteregenous arguments, provided they are either:
166     * - A value of type `A`  which has a `Header[A]` in scope
167     * - A (name, value) pair of `String`
168     * - A `Header.Raw`
169     * - A `Foldable` (`List`, `Option`, etc) of the above.
170     */
171   def apply(headers: Header.ToRaw*): Headers =
172     new Headers(headers.values)
173 
174   @deprecated("use Headers.apply", "0.22.0")
175   def of(headers: Header.ToRaw*): Headers = apply(headers: _*)
176 
177   implicit val headersShow: Show[Headers] =
178     _.headers.iterator.map(_.show).mkString("Headers(", ", ", ")")
179 
180   implicit lazy val HeadersOrder: Order[Headers] =
181     Order.by(_.headers)
182 
183   implicit val headersMonoid: Monoid[Headers] = new Monoid[Headers] {
184     def empty: Headers = Headers.empty
185     def combine(xa: Headers, xb: Headers): Headers =
186       xa ++ xb
187   }
188 
189   private val PayloadHeaderKeys = Set(
190     `Content-Length`.name,
191     `Content-Range`.name,
192     Trailer.name,
193     `Transfer-Encoding`.name,
194   )
195 
196   val SensitiveHeaders: Set[CIString] = Set(
197     Authorization.name,
198     Cookie.name,
199     `Set-Cookie`.name,
200   )
201 }
