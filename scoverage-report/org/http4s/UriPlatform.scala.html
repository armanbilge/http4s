17 package org.http4s

19 import scala.reflect.macros.whitebox
20 
21 trait UriPlatform {
22 
23   /** Literal syntax for URIs.  Invalid or non-literal arguments are rejected
24     * at compile time.
25     */
26   @deprecated("""use uri"" string interpolation instead""", "0.20")
27   def uri(s: String): Uri = macro UriPlatform.Macros.uriLiteral
28 }
29 
30 object UriPlatform {
31   @deprecated(
32     """URI literal is deprecated.  Import `org.http4s.implicits._` and use the uri"" string context""",
33     "0.22.2",
34   )
35   def deprecatedLiteralImpl(s: String): Uri =
36     Uri.unsafeFromString(s)
37 
38   private[UriPlatform] class Macros(val c: whitebox.Context) {
39     import c.universe._
40 
41     def uriLiteral(s: c.Expr[String]): Tree =
42       s.tree match {
43         case Literal(Constant(s: String)) =>
44           Uri
45             .fromString(s)
46             .fold(
47               e => c.abort(c.enclosingPosition, e.details),
48               _ => q"_root_.org.http4s.UriPlatform.deprecatedLiteralImpl($s)",
49             )
50         case _ =>
51           c.abort(
52             c.enclosingPosition,
53             s"This method uses a macro to verify that a String literal is a valid URI. Use Uri.fromString if you have a dynamic String that you want to parse as a Uri.",
54           )
55       }
56   }
57 }
