30 /** An HTTP method.
31   *
32   * @param name The name of the method
33   *
34   * @param isSafe Request methods are considered "safe" if their defined
35   * semantics are essentially read-only; i.e., the client does not request, and
36   * does not expect, any state change on the origin server as a result of
37   * applying a safe method to a target resource.
38   *
39   * @param isIdempotent A request method is considered "idempotent" if the
40   * intended effect on the server of multiple identical requests with that
41   * method is the same as the effect for a single such request.
42   *
43   * @see [[https://datatracker.ietf.org/doc/html/rfc7231#section-4 RFC 7321, Section 4, Request Methods]]
44   * @see [[http://www.iana.org/assignments/http-methods/http-methods.xhtml IANA HTTP Method Registry]]
45   */
46 final class Method private (val name: String, val isSafe: Boolean, val isIdempotent: Boolean)
47     extends Renderable
48     with Serializable {
49   override def equals(that: Any): Boolean =
50     that match {
51       case that: Method => this.name == that.name
52       case _ => false
53     }
54 
55   override def hashCode(): Int = MurmurHash3.stringHash(name, Method.HashSeed)
56 
57   override def toString(): String = name
58 
59   override final def render(writer: Writer): writer.type = writer << name
60 }
