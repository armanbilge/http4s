43   /** A kleisli with a [[Request]] input and a [[Response]] output.  This type
44     * is useful for writing middleware that are polymorphic over the return
45     * type F.
46     *
47     * @tparam F the effect type in which the [[Response]] is returned
48     * @tparam G the effect type of the [[Request]] and [[Response]] bodies
49     */
50   type Http[F[_], G[_]] = Kleisli[F, Request[G], Response[G]]
51 
52   /** A kleisli with a [[Request]] input and a [[Response]] output, such
53     * that the response effect is the same as the request and response bodies'.
54     * An HTTP app is total on its inputs.  An HTTP app may be run by a server,
55     * and a client can be converted to or from an HTTP app.
56     *
57     * @tparam F the effect type in which the [[Response]] is returned, and also
58     * of the [[Request]] and [[Response]] bodies.
59     */
60   type HttpApp[F[_]] = Http[F, F]
61 
62   /** A kleisli with a [[Request]] input and a [[Response]] output, such
63     * that the response effect is an optional inside the effect of the
64     * request and response bodies.  HTTP routes can conveniently be
65     * constructed from a partial function and combined as a
66     * `SemigroupK`.
67     *
68     * @tparam F the effect type of the [[Request]] and [[Response]] bodies,
69     * and the base monad of the `OptionT` in which the response is returned.
70     */
71   type HttpRoutes[F[_]] = Http[OptionT[F, *], F]
72 
73   type AuthedRequest[F[_], T] = ContextRequest[F, T]
