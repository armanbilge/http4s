31 class UrlForm private (val values: Map[String, Chain[String]]) extends AnyVal {
32   override def toString: String = values.toString()
33 
34   def get(key: String): Chain[String] =
35     this.getOrElse(key, Chain.empty[String])
36 
37   def getOrElse(key: String, default: => Chain[String]): Chain[String] =
38     values.getOrElse(key, default)
39 
40   def getFirst(key: String): Option[String] =
41     values.get(key).flatMap(_.uncons).map { case (s, _) => s }
42 
43   def getFirstOrElse(key: String, default: => String): String =
44     this.getFirst(key).getOrElse(default)
45 
46   def +(kv: (String, String)): UrlForm = {
47     val newValues = values.get(kv._1).fold(Chain(kv._2))(_ :+ kv._2)
48     UrlForm(values.updated(kv._1, newValues))
49   }
50 
51   /** @param key name of the field
52     * @param value value of the field
53     * @param ev evidence of the existence of `QueryParamEncoder[T]`
54     * @return `UrlForm` updated with `key` and `value` pair if key does not exist in `values`. Otherwise `value` will be added to the existing entry.
55     */
56   def updateFormField[T](key: String, value: T)(implicit ev: QueryParamEncoder[T]): UrlForm =
57     this + (key -> ev.encode(value).value)
58 
59   /** @param key name of the field
60     * @param value optional value of the field
61     * @param ev evidence of the existence of `QueryParamEncoder[T]`
62     * @return `UrlForm` updated as it is updated with `updateFormField(key, v)` if `value` is `Some(v)`, otherwise it is unaltered
63     */
64   def updateFormField[T](key: String, value: Option[T])(implicit
65       ev: QueryParamEncoder[T]
66   ): UrlForm =
67     value.fold(this)(updateFormField(key, _))
68 
69   /** @param key name of the field
70     * @param vals a Chain of values for the field
71     * @param ev evidence of the existence of `QueryParamEncoder[T]`
72     * @return `UrlForm` updated with `key` and `vals` if key does not exist in `values`, otherwise `vals` will be appended to the existing entry. If `vals` is empty, `UrlForm` will remain as is
73     */
74   def updateFormFields[T](key: String, vals: Chain[T])(implicit ev: QueryParamEncoder[T]): UrlForm =
75     vals.foldLeft(this)(_.updateFormField(key, _)(ev))
76 
77   /* same as `updateFormField(key, value)` */
78   def +?[T: QueryParamEncoder](key: String, value: T): UrlForm =
79     updateFormField(key, value)
80 
81   /* same as `updateParamEncoder`(key, value) */
82   def +?[T: QueryParamEncoder](key: String, value: Option[T]): UrlForm =
83     updateFormField(key, value)
84 
85   /* same as `updatedParamEncoders`(key, vals) */
86   def ++?[T: QueryParamEncoder](key: String, vals: Chain[T]): UrlForm =
87     updateFormFields(key, vals)
88 }

90 object UrlForm {
91   val empty: UrlForm = new UrlForm(Map.empty)
92 
93   def apply(values: Map[String, Chain[String]]): UrlForm =
94     // value "" -> Chain() is just noise and it is not maintain during encoding round trip
95     if (values.get("").fold(false)(_.isEmpty)) new UrlForm(values - "")
96     else new UrlForm(values)
97 
98   def apply(values: (String, String)*): UrlForm =
99     values.foldLeft(empty)(_ + _)
100 
101   def fromChain(values: Chain[(String, String)]): UrlForm =
102     apply(values.toList: _*)
103 
104   implicit def entityEncoder[F[_]](implicit charset: Charset = `UTF-8`): EntityEncoder[F, UrlForm] =
105     EntityEncoder
106       .stringEncoder[F]
107       .contramap[UrlForm](encodeString(charset))
108       .withContentType(`Content-Type`(MediaType.application.`x-www-form-urlencoded`, charset))
109 
110   implicit def entityDecoder[F[_]](implicit
111       F: Concurrent[F],
112       defaultCharset: Charset = `UTF-8`,
113   ): EntityDecoder[F, UrlForm] =
114     EntityDecoder.decodeBy(MediaType.application.`x-www-form-urlencoded`) { m =>
115       DecodeResult(
116         EntityDecoder
117           .decodeText(m)
118           .map(decodeString(m.charset.getOrElse(defaultCharset)))
119       )
120     }
121 
122   implicit val eqInstance: Eq[UrlForm] = Eq.instance { (x: UrlForm, y: UrlForm) =>
123     x.values === y.values
124   }
125 
126   implicit val monoidInstance: Monoid[UrlForm] = new Monoid[UrlForm] {
127     override def empty: UrlForm = UrlForm.empty
128 
129     override def combine(x: UrlForm, y: UrlForm): UrlForm =
130       UrlForm(x.values |+| y.values)
131   }
132 
133   /** Attempt to decode the `String` to a [[UrlForm]] */
134   def decodeString(
135       charset: Charset
136   )(urlForm: String): Either[MalformedMessageBodyFailure, UrlForm] =
137     QueryParser
138       .parseQueryString(urlForm.replace("+", "%20"), new Codec(charset.nioCharset))
139       .map(q => UrlForm(CollectionCompat.mapValues(q.multiParams)(Chain.fromSeq)))
140       .leftMap { parseFailure =>
141         MalformedMessageBodyFailure(parseFailure.message, None)
142       }
143 
144   /** Encode the [[UrlForm]] into a `String` using the provided `Charset` */
145   def encodeString(charset: Charset)(urlForm: UrlForm): String = {
146     def encode(s: String): String =
147       Uri.encode(s, charset.nioCharset, spaceIsPlus = true, toSkip = Uri.Unreserved)
148 
149     val sb = new StringBuilder(urlForm.values.size * 20)
150     urlForm.values.foreach { case (k, vs) =>
151       if (sb.nonEmpty) sb.append('&')
152       val encodedKey = encode(k)
153       if (vs.isEmpty) sb.append(encodedKey)
154       else {
155         var first = true
156         vs.map { v =>
157           if (!first) sb.append('&')
158           else first = false
159           sb.append(encodedKey)
160             .append('=')
161             .append(encode(v))
162         }
163       }
164     }
165     sb.result()
166   }
