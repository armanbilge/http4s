29 /** Simple representation of a URI Template that can be rendered as RFC6570
30   * conform string.
31   *
32   * This model reflects only a subset of RFC6570.
33   *
34   * Level 1 and Level 2 are completely modeled and
35   * Level 3 features are limited to:
36   *  - Path segments, slash-prefixed
37   *  - Form-style query, ampersand-separated
38   *  - Fragment expansion
39   */
40 final case class UriTemplate(
41     scheme: Option[Scheme] = None,
42     authority: Option[Authority] = None,
43     path: Path = Nil,
44     query: UriTemplate.Query = Nil,
45     fragment: Fragment = Nil,
46 ) {
47 
48   /** Replaces any expansion type that matches the given `name`. If no matching
49     * `expansion` could be found the same instance will be returned.
50     */
51   def expandAny[T: QueryParamEncoder](name: String, value: T): UriTemplate =
52     expandPath(name, value).expandQuery(name, value).expandFragment(name, value)
53 
54   /** Replaces any expansion type in `fragment` that matches the given `name`.
55     * If no matching `expansion` could be found the same instance will be
56     * returned.
57     */
58   def expandFragment[T: QueryParamEncoder](name: String, value: T): UriTemplate =
59     if (fragment.isEmpty) this
60     else copy(fragment = expandFragmentN(fragment, name, QueryParamEncoder[T].encode(value).value))
61 
62   /** Replaces any expansion type in `path` that matches the given `name`. If no
63     * matching `expansion` could be found the same instance will be returned.
64     */
65   def expandPath[T: QueryParamEncoder](name: String, values: List[T]): UriTemplate =
66     copy(path = expandPathN(path, name, values.map(QueryParamEncoder[T].encode)))
67 
68   /** Replaces any expansion type in `path` that matches the given `name`. If no
69     * matching `expansion` could be found the same instance will be returned.
70     */
71   def expandPath[T: QueryParamEncoder](name: String, value: T): UriTemplate =
72     copy(path = expandPathN(path, name, QueryParamEncoder[T].encode(value) :: Nil))
73 
74   /** Replaces any expansion type in `query` that matches the specified `name`.
75     * If no matching `expansion` could be found the same instance will be
76     * returned.
77     */
78   def expandQuery[T: QueryParamEncoder](name: String, values: List[T]): UriTemplate =
79     if (query.isEmpty) this
80     else copy(query = expandQueryN(query, name, values.map(QueryParamEncoder[T].encode(_).value)))
81 
82   /** Replaces any expansion type in `query` that matches the specified `name`.
83     * If no matching `expansion` could be found the same instance will be
84     * returned.
85     */
86   def expandQuery(name: String): UriTemplate = expandQuery(name, List[String]())
87 
88   /** Replaces any expansion type in `query` that matches the specified `name`.
89     * If no matching `expansion` could be found the same instance will be
90     * returned.
91     */
92   def expandQuery[T: QueryParamEncoder](name: String, values: T*): UriTemplate =
93     expandQuery(name, values.toList)
94 
95   override lazy val toString: String =
96     renderUriTemplate(this)
97 
98   /** If no expansion is available an `Uri` will be created otherwise the
99     * current instance of `UriTemplate` will be returned.
100     */
101   def toUriIfPossible: Try[Uri] =
102     if (containsExpansions(this))
103       Failure(
104         new IllegalStateException(s"all expansions must be resolved to be convertable: $this")
105       )
106     else Success(toUri(this))
107 }

114   protected val unreserved: Set[Char] =
115     (('a' to 'z') ++ ('A' to 'Z') ++ ('0' to '9') :+ '-' :+ '.' :+ '_' :+ '~').toSet

117   //  protected val genDelims = ':' :: '/' :: '?' :: '#' :: '[' :: ']' :: '@' :: Nil

119   //  protected val reserved = genDelims ::: subDelims

121   def isUnreserved(s: String): Boolean = s.forall(unreserved.contains)
122 
123   def isUnreservedOrEncoded(s: String): Boolean =
124     Uri
125       .encode(s, spaceIsPlus = true, toSkip = unreserved)
126       .forall(c => unreserved.contains(c) || c == '%')

128   protected def expandPathN(path: Path, name: String, values: List[QueryParameterValue]): Path = {
129     val acc = new ArrayBuffer[PathDef]()
130     def appendValues(): Unit =
131       values.foreach { v =>
132         acc.append(PathElm(v.value))
133       }
134     path.foreach {
135       case p @ PathElm(_) => acc.append(p)
136       case p @ VarExp(collection.Seq(n)) =>
137         if (n == name) appendValues()
138         else acc.append(p)
139       case p @ VarExp(ns) =>
140         if (ns.contains(name)) {
141           appendValues()
142           acc.append(VarExp(ns.filterNot(_ == name)))
143         } else acc.append(p)
144       case p @ ReservedExp(collection.Seq(n)) =>
145         if (n == name) appendValues()
146         else acc.append(p)
147       case p @ ReservedExp(ns) =>
148         if (ns.contains(name)) {
149           appendValues()
150           acc.append(VarExp(ns.filterNot(_ == name)))
151         } else acc.append(p)
152       case p @ PathExp(collection.Seq(n)) =>
153         if (n == name) appendValues()
154         else acc.append(p)
155       case p @ PathExp(ns) =>
156         if (ns.contains(name)) {
157           appendValues()
158           acc.append(PathExp(ns.filterNot(_ == name)))
159         } else acc.append(p)
160     }
161     acc.toList
162   }

164   protected def expandQueryN(query: Query, name: String, values: List[String]): Query = {
165     val acc = new ArrayBuffer[QueryDef]()
166     query.foreach {
167       case p @ ParamElm(_, _) => acc.append(p)
168       case p @ ParamVarExp(r, List(n)) =>
169         if (n == name) acc.append(ParamElm(r, values))
170         else acc.append(p)
171       case p @ ParamVarExp(r, ns) =>
172         if (ns.contains(name)) {
173           acc.append(ParamElm(r, values))
174           acc.append(ParamVarExp(r, ns.filterNot(_ == name)))
175         } else acc.append(p)
176       case p @ ParamReservedExp(r, List(n)) =>
177         if (n == name) acc.append(ParamElm(r, values))
178         else acc.append(p)
179       case p @ ParamReservedExp(r, ns) =>
180         if (ns.contains(name)) {
181           acc.append(ParamElm(r, values))
182           acc.append(ParamReservedExp(r, ns.filterNot(_ == name)))
183         } else acc.append(p)
184       case p @ ParamExp(collection.Seq(n)) =>
185         if (n == name) acc.append(ParamElm(name, values))
186         else acc.append(p)
187       case p @ ParamExp(ns) =>
188         if (ns.contains(name)) {
189           acc.append(ParamElm(name, values))
190           acc.append(ParamExp(ns.filterNot(_ == name)))
191         } else acc.append(p)
192       case p @ ParamContExp(collection.Seq(n)) =>
193         if (n == name) acc.append(ParamElm(name, values))
194         else acc.append(p)
195       case p @ ParamContExp(ns) =>
196         if (ns.contains(name)) {
197           acc.append(ParamElm(name, values))
198           acc.append(ParamContExp(ns.filterNot(_ == name)))
199         } else acc.append(p)
200     }
201     acc.toList
202   }

204   protected def expandFragmentN(fragment: Fragment, name: String, value: String): Fragment = {
205     val acc = new ArrayBuffer[FragmentDef]()
206     fragment.foreach {
207       case p @ FragmentElm(_) => acc.append(p)
208       case p @ SimpleFragmentExp(n) =>
209         if (n == name) acc.append(FragmentElm(value)) else acc.append(p)
210       case p @ MultiFragmentExp(collection.Seq(n)) =>
211         if (n == name) acc.append(FragmentElm(value)) else acc.append(p)
212       case p @ MultiFragmentExp(ns) =>
213         if (ns.contains(name)) {
214           acc.append(FragmentElm(value))
215           acc.append(MultiFragmentExp(ns.filterNot(_ == name)))
216         } else acc.append(p)
217     }
218     acc.toList
219   }

221   protected def renderAuthority(a: Authority): String =
222     a match {
223       case Authority(Some(u), h, None) => s"${renderUserInfo(u)}@${renderHost(h)}"
224       case Authority(Some(u), h, Some(p)) => s"${renderUserInfo(u)}@${renderHost(h)}:${p}"
225       case Authority(None, h, Some(p)) => renderHost(h) + ":" + p
226       case Authority(_, h, _) => renderHost(h)
227     }
228 
229   protected def renderUserInfo(u: UserInfo): String =
230     u match {
231       case UserInfo(username, Some(password)) =>
232         (new StringWriter << username << ":" << password).result
233       case UserInfo(username, None) => username
234     }

236   protected def renderHost(h: Host): String =
237     h match {
238       case RegName(n) => n.toString
239       case a: Ipv4Address => a.value
240       case a: Ipv6Address => "[" + a.value + "]"
241     }

243   protected def renderScheme(s: Scheme): String =
244     (new StringWriter << s << ":").result
245 
246   protected def renderSchemeAndAuthority(t: UriTemplate): String =
247     t match {
248       case UriTemplate(None, None, _, _, _) => ""
249       case UriTemplate(Some(s), Some(a), _, _, _) => renderScheme(s) + "//" + renderAuthority(a)
250       case UriTemplate(Some(s), None, _, _, _) => renderScheme(s)
251       case UriTemplate(None, Some(a), _, _, _) => renderAuthority(a)
252     }

254   protected def renderQuery(ps: Query): String = {
255     val parted = ps.partition {
256       case ParamElm(_, _) => false
257       case ParamVarExp(_, _) => false
258       case ParamReservedExp(_, _) => false
259       case ParamExp(_) => true
260       case ParamContExp(_) => true
261     }
262     val elements = new ArrayBuffer[String]()
263     parted._2.foreach {
264       case ParamElm(n, Nil) => elements.append(n)
265       case ParamElm(n, List(v)) => elements.append(n + "=" + v)
266       case ParamElm(n, vs) => vs.foreach(v => elements.append(n + "=" + v))
267       case ParamVarExp(n, vs) => elements.append(n + "=" + "{" + vs.mkString(",") + "}")
268       case ParamReservedExp(n, vs) => elements.append(n + "=" + "{+" + vs.mkString(",") + "}")
269       case u => throw new IllegalStateException(s"type ${u.getClass.getName} not supported")
270     }
271     val exps = new ArrayBuffer[String]()
272     def separator = if (elements.isEmpty && exps.isEmpty) "?" else "&"
273     parted._1.foreach {
274       case ParamExp(ns) => exps.append("{" + separator + ns.mkString(",") + "}")
275       case ParamContExp(ns) => exps.append("{" + separator + ns.mkString(",") + "}")
276       case u => throw new IllegalStateException(s"type ${u.getClass.getName} not supported")
277     }
278     if (elements.isEmpty) exps.mkString
279     else "?" + elements.mkString("&") + exps.mkString
280   }

282   protected def renderFragment(f: Fragment): String = {
283     val elements = new mutable.ArrayBuffer[String]()
284     val expansions = new mutable.ArrayBuffer[String]()
285     f.foreach {
286       case FragmentElm(v) => elements.append(v)
287       case SimpleFragmentExp(n) => expansions.append(n)
288       case MultiFragmentExp(ns) => expansions.append(ns.mkString(","))
289     }
290     if (elements.nonEmpty && expansions.nonEmpty)
291       "#" + elements.mkString(",") + "{#" + expansions.mkString(",") + "}"
292     else if (elements.nonEmpty)
293       "#" + elements.mkString(",")
294     else if (expansions.nonEmpty)
295       "{#" + expansions.mkString(",") + "}"
296     else
297       "#"
298   }

300   protected def renderFragmentIdentifier(f: Fragment): String = {
301     val elements = new mutable.ArrayBuffer[String]()
302     f.foreach {
303       case FragmentElm(v) => elements.append(v)
304       case SimpleFragmentExp(_) =>
305         throw new IllegalStateException("SimpleFragmentExp cannot be converted to a Uri")
306       case MultiFragmentExp(_) =>
307         throw new IllegalStateException("MultiFragmentExp cannot be converted to a Uri")
308     }
309     if (elements.isEmpty) ""
310     else elements.mkString(",")
311   }
312 
313   protected def buildQuery(q: Query): org.http4s.Query = {
314     val vec = q.foldLeft(Vector.empty[(String, Option[String])]) {
315       case (elements, ParamElm(n, Nil)) => elements :+ (n -> None)
316       case (elements, ParamElm(n, List(v))) => elements :+ (n -> Some(v))
317       case (elements, ParamElm(n, vs)) =>
318         vs.foldLeft(elements) { case (elements, v) => elements :+ (n -> Some(v)) }
319       case u =>
320         throw new IllegalStateException(s"${u.getClass.getName} cannot be converted to a Uri")
321     }
322 
323     org.http4s.Query.fromVector(vec)
324   }

326   protected def renderPath(p: Path): String =
327     p match {
328       case Nil => "/"
329       case ps =>
330         val elements = new ArrayBuffer[String]()
331         ps.foreach {
332           case PathElm(n) => elements.append("/" + n)
333           case VarExp(ns) => elements.append("{" + ns.mkString(",") + "}")
334           case ReservedExp(ns) => elements.append("{+" + ns.mkString(",") + "}")
335           case PathExp(ns) => elements.append("{/" + ns.mkString(",") + "}")
336         }
337         elements.mkString
338     }

340   protected def renderPathAndQueryAndFragment(t: UriTemplate): String =
341     t match {
342       case UriTemplate(_, _, Nil, Nil, Nil) => "/"
343       case UriTemplate(_, _, Nil, Nil, f) => "/" + renderFragment(f)
344       case UriTemplate(_, _, Nil, query, Nil) => "/" + renderQuery(query)
345       case UriTemplate(_, _, Nil, query, f) => "/" + renderQuery(query) + renderFragment(f)
346       case UriTemplate(_, _, path, Nil, Nil) => renderPath(path)
347       case UriTemplate(_, _, path, query, Nil) => renderPath(path) + renderQuery(query)
348       case UriTemplate(_, _, path, Nil, f) => renderPath(path) + renderFragment(f)
349       case UriTemplate(_, _, path, query, f) =>
350         renderPath(path) + renderQuery(query) + renderFragment(f)
351     }
352 
353   protected def renderUriTemplate(t: UriTemplate): String =
354     t match {
355       case UriTemplate(None, None, Nil, Nil, Nil) => "/"
356       case UriTemplate(Some(_), Some(_), Nil, Nil, Nil) => renderSchemeAndAuthority(t)
357       case UriTemplate(scheme @ _, authority @ _, path @ _, params @ _, fragment @ _) =>
358         renderSchemeAndAuthority(t) + renderPathAndQueryAndFragment(t)
359     }

361   protected def fragmentExp(f: FragmentDef): Boolean =

363       case FragmentElm(_) => false
364       case SimpleFragmentExp(_) => true
365       case MultiFragmentExp(_) => true

368   protected def pathExp(p: PathDef): Boolean =
369     p match {
370       case PathElm(_) => false
371       case VarExp(_) => true
372       case ReservedExp(_) => true
373       case PathExp(_) => true
374     }

376   protected def queryExp(q: QueryDef): Boolean =
377     q match {
378       case ParamElm(_, _) => false
379       case ParamVarExp(_, _) => true
380       case ParamReservedExp(_, _) => true
381       case ParamExp(_) => true
382       case ParamContExp(_) => true
383     }

385   protected def containsExpansions(t: UriTemplate): Boolean =
386     t match {
387       case UriTemplate(_, _, Nil, Nil, Nil) => false
388       case UriTemplate(_, _, Nil, Nil, f) => f.exists(fragmentExp)
389       case UriTemplate(_, _, Nil, q, Nil) => q.exists(queryExp)
390       case UriTemplate(_, _, Nil, q, f) => (q.exists(queryExp)) || (f.exists(fragmentExp))
391       case UriTemplate(_, _, p, Nil, Nil) => p.exists(pathExp)
392       case UriTemplate(_, _, p, Nil, f) => (p.exists(pathExp)) || (f.exists(fragmentExp))
393       case UriTemplate(_, _, p, q, Nil) => (p.exists(pathExp)) || (q.exists(queryExp))
394       case UriTemplate(_, _, p, q, f) =>
395         (p.exists(pathExp)) || (q.exists(queryExp)) || (f.exists(fragmentExp))
396     }

398   protected def toUri(t: UriTemplate): Uri =
399     t match {
400       case UriTemplate(s, a, Nil, Nil, Nil) => Uri(s, a)
401       case UriTemplate(s, a, Nil, Nil, f) => Uri(s, a, fragment = Some(renderFragmentIdentifier(f)))
402       case UriTemplate(s, a, Nil, q, Nil) => Uri(s, a, query = buildQuery(q))
403       case UriTemplate(s, a, Nil, q, f) =>
404         Uri(s, a, query = buildQuery(q), fragment = Some(renderFragmentIdentifier(f)))
405       case UriTemplate(s, a, p, Nil, Nil) => Uri(s, a, Uri.Path.unsafeFromString(renderPath(p)))
406       case UriTemplate(s, a, p, q, Nil) =>
407         Uri(s, a, Uri.Path.unsafeFromString(renderPath(p)), buildQuery(q))
408       case UriTemplate(s, a, p, Nil, f) =>
409         Uri(
410           s,
411           a,
412           Uri.Path.unsafeFromString(renderPath(p)),
413           fragment = Some(renderFragmentIdentifier(f)),
414         )
415       case UriTemplate(s, a, p, q, f) =>
416         Uri(
417           s,
418           a,
419           Uri.Path.unsafeFromString(renderPath(p)),
420           buildQuery(q),
421           Some(renderFragmentIdentifier(f)),
422         )
423     }

463   /** URI Templates are similar to a macro language with a fixed set of macro
464     * definitions: the expression type determines the expansion process.
465     *
466     * The default expression type is simple string expansion (Level 1), wherein a
467     * single named variable is replaced by its value as a string after
468     * pct-encoding any characters not in the set of unreserved URI characters
469     * (<a href="https://datatracker.ietf.org/doc/html/rfc6570#section-1.5">Section 1.5</a>).
470     *
471     * Level 2 templates add the plus ("+") operator, for expansion of values that
472     * are allowed to include reserved URI characters
473     * (<a href="https://datatracker.ietf.org/doc/html/rfc6570#section-1.5">Section 1.5</a>),
474     * and the crosshatch ("#") operator for expansion of fragment identifiers.
475     *
476     * Level 3 templates allow multiple variables per expression, each
477     * separated by a comma, and add more complex operators for dot-prefixed
478     * labels, slash-prefixed path segments, semicolon-prefixed path
479     * parameters, and the form-style construction of a query syntax
480     * consisting of name=value pairs that are separated by an ampersand
481     * character.
482     */
483   sealed trait ExpansionType
