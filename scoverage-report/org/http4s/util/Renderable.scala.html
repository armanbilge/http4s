32 /** A type class that describes how to efficiently render a type
33   * @tparam T the type which will be rendered
34   */
35 trait Renderer[T] {
36 
37   /** Renders the object to the writer
38     * @param writer [[org.http4s.util.Writer]] to render to
39     * @param t object to render
40     * @return the same [[org.http4s.util.Writer]] provided
41     */
42   def render(writer: Writer, t: T): writer.type
43 }
44 
45 object Renderer {
46   @inline def apply[A](implicit ev: Renderer[A]): Renderer[A] = ev
47 
48   def renderString[T: Renderer](t: T): String = new StringWriter().append(t).result
49 
50   implicit val RFC7231InstantRenderer: Renderer[Instant] = new Renderer[Instant] {
51     private val dateFormat =
52       DateTimeFormatter
53         .ofPattern("EEE, dd MMM yyyy HH:mm:ss zzz")
54         .withLocale(Locale.US)
55         .withZone(ZoneId.of("GMT"))
56 
57     override def render(writer: Writer, t: Instant): writer.type =
58       writer << dateFormat.format(t)
59   }
60 
61   implicit val stringRenderer: Renderer[String] = new Renderer[String] {
62     override def render(writer: Writer, string: String): writer.type =
63       writer << string
64   }
65 
66   // Render a finite duration in seconds
67   implicit val finiteDurationRenderer: Renderer[FiniteDuration] = new Renderer[FiniteDuration] {
68     override def render(writer: Writer, d: FiniteDuration): writer.type =
69       writer << d.toSeconds.toString
70   }
71 
72   // Render a long value, e.g. on the Age header
73   implicit val longRenderer: Renderer[Long] = new Renderer[Long] {
74     override def render(writer: Writer, d: Long): writer.type =
75       writer << d.toString
76   }
77 
78   implicit def eitherRenderer[A, B](implicit
79       ra: Renderer[A],
80       rb: Renderer[B],
81   ): Renderer[Either[A, B]] =
82     new Renderer[Either[A, B]] {
83       override def render(writer: Writer, e: Either[A, B]): writer.type =
84         e match {
85           case Left(a) => ra.render(writer, a)
86           case Right(b) => rb.render(writer, b)
87         }
88     }
89 
90   implicit val ciStringRenderer: Renderer[CIString] = new Renderer[CIString] {
91     override def render(writer: Writer, ciString: CIString): writer.type =
92       writer << ciString
93   }
94 
95   implicit def nelRenderer[T: Renderer]: Renderer[NonEmptyList[T]] =
96     new Renderer[NonEmptyList[T]] {
97       override def render(writer: Writer, values: NonEmptyList[T]): writer.type =
98         writer.addNel(values)
99     }
100 
101   implicit def listRenderer[T: Renderer]: Renderer[List[T]] =
102     new Renderer[List[T]] {
103       override def render(writer: Writer, values: List[T]): writer.type =
104         writer.addList(values)
105     }
106 
107   implicit def setRenderer[T: Renderer]: Renderer[Set[T]] =

119 /** Mixin that makes a type writable by a [[Writer]] without needing a [[Renderer]] instance */
120 trait Renderable extends Any {
121 
122   /** Base method for rendering this object efficiently */
123   def render(writer: Writer): writer.type
124 
125   /** Generates a String rendering of this object */
126   def renderString: String = Renderer.renderString(this)
127 
128   override def toString: String = renderString
129 }

143   val HeaderValueDQuote: BitSet = BitSet("\\\"".map(_.toInt): _*)

146 /** Efficiently accumulate [[Renderable]] representations */
147 trait Writer { self =>
148   def append(s: String): this.type
149   def append(ci: CIString): this.type = append(ci.toString)
150   def append(char: Char): this.type = append(char.toString)
151   def append(float: Float): this.type = append(float.toString)
152   def append(double: Double): this.type = append(double.toString)
153   def append(int: Int): this.type = append(int.toString)
154   def append(long: Long): this.type = append(long.toString)
155 
156   def append[T](r: T)(implicit R: Renderer[T]): this.type = R.render(this, r)
157 
158   def quote(
159       s: String,
160       escapedChars: BitSet = Writer.HeaderValueDQuote,
161       escapeChar: Char = '\\',
162   ): this.type = {
163     this << '"'
164 
165     @tailrec
166     def go(i: Int): Unit =
167       if (i < s.length) {
168         val c = s.charAt(i)
169         if (escapedChars.contains(c.toInt)) this << escapeChar
170         this << c
171         go(i + 1)
172       }
173 
174     go(0)
175     this << '"'
176   }
177   // Adapted from https://github.com/akka/akka-http/blob/b071bd67547714bd8bed2ccd8170fbbc6c2dbd77/akka-http-core/src/main/scala/akka/http/impl/util/Rendering.scala#L219-L229
178   def eligibleOnly(s: String, keep: CharPredicate, placeholder: Char): this.type = {
179     @tailrec def rec(ix: Int = 0): this.type =
180       if (ix < s.length) {
181         val c = s.charAt(ix)
182         if (keep(c)) this << c
183         else this << placeholder
184         rec(ix + 1)
185       } else this
186     rec()
187   }
188 
189   def addStrings(
190       s: collection.Seq[String],
191       sep: String = "",
192       start: String = "",
193       end: String = "",
194   ): this.type = {
195     append(start)
196     if (s.nonEmpty) {
197       append(s.head)
198       s.tail.foreach(s => append(sep).append(s))
199     }
200     append(end)
201   }
202 
203   def addList[T: Renderer](
204       s: List[T],
205       sep: String = ", ",
206       start: String = "",
207       end: String = "",
208   ): this.type =
209     NonEmptyList.fromList(s) match {
210       case Some(s) => addNel(s, sep, start, end)
211       case None =>
212         append(start)
213         append(end)
214     }
215 
216   def addNel[T: Renderer](
217       s: NonEmptyList[T],
218       sep: String = ", ",
219       start: String = "",
220       end: String = "",
221   ): this.type = {
222     append(start)
223     append(s.head)
224     s.tail.foreach(s => append(sep).append(s))
225     append(end)
226   }
227 
228   def addSet[T: Renderer](
229       s: collection.Set[T],
230       sep: String = ", ",
231       start: String = "",
232       end: String = "",
233   ): this.type = {
234     append(start)
235     if (s.nonEmpty) {
236       append(s.head)
237       s.tail.foreach(s => append(sep).append(s))
238     }
239     append(end)
240   }
241 
242   final def <<(s: String): this.type = append(s)
243   final def <<#(s: String): this.type = quote(s)
244   final def <<(s: CIString): this.type = append(s)
245   final def <<(char: Char): this.type = append(char)
246   final def <<(float: Float): this.type = append(float)
247   final def <<(double: Double): this.type = append(double)
248   final def <<(int: Int): this.type = append(int)
249   final def <<(long: Long): this.type = append(long)
250   final def <<[T: Renderer](r: T): this.type = append(r)
251 
252   def sanitize(f: Writer => Writer): this.type = {
253     val w = new Writer {
254       def append(s: String): this.type = {
255         s.foreach(append(_))
256         this
257       }
258       override def append(c: Char): this.type = {
259         if (c == 0x0.toChar || c == '\r' || c == '\n')
260           self.append(' ')
261         else
262           self.append(c)
263         this
264       }
265     }
266     f(w)
267     this
268   }
269 }

271 /** [[Writer]] that will result in a `String`
272   * @param size initial buffer size of the underlying `StringBuilder`
273   */
274 class StringWriter(size: Int = StringWriter.InitialCapacity) extends Writer { self =>
275   private val sb = new java.lang.StringBuilder(size)
276 
277   def append(s: String): this.type = { sb.append(s); this }
278   override def append(char: Char): this.type = { sb.append(char); this }
279   override def append(float: Float): this.type = { sb.append(float); this }
280   override def append(double: Double): this.type = { sb.append(double); this }
281   override def append(int: Int): this.type = { sb.append(int); this }
282   override def append(long: Long): this.type = { sb.append(long); this }
283 
284   override def sanitize(f: Writer => Writer): this.type = {
285     val w = new Writer {
286       def append(s: String): this.type = {
287         val start = sb.length
288         self.append(s)
289         for (i <- start until sb.length) {
290           val c = sb.charAt(i)
291           if (c == 0x0.toChar || c == '\r' || c == '\n') {
292             sb.setCharAt(i, ' ')
293           }
294         }
295         this
296       }
297       override def append(c: Char): this.type = {
298         if (c == 0x0.toChar || c == '\r' || c == '\n')
299           self.append(' ')
300         else
301           self.append(c)
302         this
303       }
304     }
305     f(w)
306     this
307   }
308 
309   def result: String = sb.toString
310 }
