21 final case class ContextResponse[F[_], A](context: A, response: Response[F]) {
22   def mapContext[B](f: A => B): ContextResponse[F, B] =
23     ContextResponse(f(context), response)
24 
25   def mapK[G[_]](fk: F ~> G): ContextResponse[G, A] =
26     ContextResponse(context, response.mapK(fk))
27 }
