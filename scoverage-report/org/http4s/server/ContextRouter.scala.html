33   object Segment {
34     implicit def functor[F[_]: Functor, X]: Functor[Segment[F, X, *]] =
35       new Functor[Segment[F, X, *]] {
36         override def map[A, B](fa: Segment[F, X, A])(f: A => B): Segment[F, X, B] =
37           new Segment(fa.run(_, _).map(f))
38       }
39 
40     implicit def contravariant[F[_], X]: Contravariant[Segment[F, *, X]] =
41       new Contravariant[Segment[F, *, X]] {
42         override def contramap[A, B](fa: Segment[F, A, X])(f: B => A): Segment[F, B, X] =
43           new Segment((b, segment) => fa.run(f(b), segment))
44       }
45 
46     class Partial[A] {
47       def apply[F[_], B](run: (A, Uri.Path.Segment) => OptionT[F, B]): Segment[F, A, B] =
48         new Segment(run)
49     }
50     // partially apply for inference
51     def apply[A] = new Partial[A]
52   }
53 
54   final class Segment[F[_], A, B](val run: (A, Uri.Path.Segment) => OptionT[F, B]) {
55     def apply(routes: ContextRoutes[B, F])(implicit F: Monad[F]): ContextRoutes[A, F] =
56       Kleisli { case ContextRequest(a, req) =>
57         for {
58           head <- OptionT.fromOption[F](req.pathInfo.segments.headOption)
59           b <- run(a, head)
60           caret = req.attributes.lookup(Request.Keys.PathInfoCaret).getOrElse(0)
61           response <- routes(
62             ContextRequest(b, req.withAttribute(Request.Keys.PathInfoCaret, caret + 1))
63           )
64         } yield response
65       }
66 
67     def ->(routes: ContextRoutes[B, F]): Routable[F, A] =
68       Routable.Dynamic(this, routes)
69   }
70 
71   object Routable {
72     final case class Static[F[_], A](tupled: (String, ContextRoutes[A, F])) extends Routable[F, A]
73     final case class Dynamic[F[_], A, B](segment: Segment[F, A, B], routes: ContextRoutes[B, F])
74         extends Routable[F, A]
75 
76     implicit def tuple[F[_], A](tupled: (String, ContextRoutes[A, F])): Routable[F, A] =
77       Static(tupled)
78   }
79   sealed trait Routable[F[_], A]

94   /** Defines an [[ContextRoutes]] based on list of mappings and
95     * a default Service to be used when none in the list match incoming requests.
96     *
97     * The mappings are processed in descending order (longest first) of prefix length.
98     */
99   def define[F[_]: Monad, A](
100       mappings: (String, ContextRoutes[A, F])*
101   )(default: ContextRoutes[A, F]): ContextRoutes[A, F] =
102     defineHelper(mappings, default)
103 
104   @deprecated("Kept for binary compatiblity.  Use the define with the Monad constraint.", "0.23.12")
105   def define[F[_], A](
106       mappings: Seq[(String, ContextRoutes[A, F])]
107   )(default: ContextRoutes[A, F], sync: Sync[F]): ContextRoutes[A, F] =
108     defineHelper(mappings, default)(sync)
109 
110   // Supports the deprecated methods.  We can't call the Sync
111   // version of define after defining the new Monad version.
112   private def defineHelper[F[_]: Monad, A](
113       mappings: Seq[(String, ContextRoutes[A, F])],
114       default: ContextRoutes[A, F],
115   ): ContextRoutes[A, F] =
116     mappings.sortBy(_._1.length).foldLeft(default) { case (acc, (prefix, routes)) =>
117       val prefixSegments = Uri.Path.unsafeFromString(prefix)
118       if (prefixSegments.isEmpty) routes <+> acc
119       else
120         Kleisli { case creq @ ContextRequest(_, req) =>
121           if (req.pathInfo.startsWith(prefixSegments)) {
122             val treq = Router.translate(prefixSegments)(creq.req)
123             routes(creq.copy(req = treq)).orElse(acc(creq))
124           } else acc(creq)
125         }
126     }
127 
128   def of[F[_]: Sync, A](mappings: Routable[F, A]*): ContextRoutes[A, F] =
129     dynamic(mappings: _*)(ContextRoutes.empty[A, F])
130 
131   def dynamic[F[_]: Sync, A](mappings: Routable[F, A]*)(
132       default: ContextRoutes[A, F]
133   ): ContextRoutes[A, F] = {
134     val (statics, dynamic) = mappings.toList.partitionEither {
135       case Routable.Static(tuple) => Left(tuple)
136       case Routable.Dynamic(segment, route) => Right(segment(route))
137     }
138     dynamic.foldLeft(define(statics: _*)(default))(_ <+> _)
139   }
