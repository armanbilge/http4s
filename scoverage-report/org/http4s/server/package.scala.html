102   /** A middleware is a function of one [[cats.data.Kleisli]] to another, possibly of a
103     * different [[Request]] and [[Response]] type.  http4s comes with several
104     * middlewares for composing common functionality into services.
105     *
106     * @tparam F the effect type of the services
107     * @tparam A the request type of the original service
108     * @tparam B the response type of the original service
109     * @tparam C the request type of the resulting service
110     * @tparam D the response type of the resulting service
111     */
112   type Middleware[F[_], A, B, C, D] = Kleisli[F, A, B] => Kleisli[F, C, D]
113 
114   /** An HTTP middleware converts an [[HttpRoutes]] to another.
115     */
116   type HttpMiddleware[F[_]] =
117     Middleware[OptionT[F, *], Request[F], Response[F], Request[F], Response[F]]

119   /** An HTTP middleware that authenticates users.
120     */
121   type AuthMiddleware[F[_], T] =
122     Middleware[OptionT[F, *], AuthedRequest[F, T], Response[F], Request[F], Response[F]]
123 
124   /** An HTTP middleware that adds a context.

129   object AuthMiddleware {
130     def apply[F[_]: Monad, T](
131         authUser: Kleisli[OptionT[F, *], Request[F], T]
132     ): AuthMiddleware[F, T] =
133       noSpider[F, T](authUser, defaultAuthFailure[F])
134 
135     def withFallThrough[F[_]: Monad, T](
136         authUser: Kleisli[OptionT[F, *], Request[F], T]
137     ): AuthMiddleware[F, T] =
138       _.compose(Kleisli((r: Request[F]) => authUser(r).map(AuthedRequest(_, r))))
139 
140     def noSpider[F[_]: Monad, T](
141         authUser: Kleisli[OptionT[F, *], Request[F], T],
142         onAuthFailure: Request[F] => F[Response[F]],
143     ): AuthMiddleware[F, T] = { service =>
144       Kleisli { (r: Request[F]) =>
145         val resp = authUser(r).value.flatMap {
146           case Some(authReq) =>
147             service(AuthedRequest(authReq, r)).getOrElse(Response[F](Status.NotFound))
148           case None => onAuthFailure(r)
149         }
150         OptionT.liftF(resp)
151       }
152     }
153 
154     def defaultAuthFailure[F[_]](implicit F: Applicative[F]): Request[F] => F[Response[F]] =
155       _ => F.pure(Response[F](Status.Unauthorized))
156 
157     def apply[F[_], Err, T](
158         authUser: Kleisli[F, Request[F], Either[Err, T]],
159         onFailure: AuthedRoutes[Err, F],
160     )(implicit F: Monad[F]): AuthMiddleware[F, T] =
161       (routes: AuthedRoutes[T, F]) =>
162         Kleisli { (req: Request[F]) =>
163           OptionT {
164             authUser(req).flatMap {
165               case Left(err) => onFailure(AuthedRequest(err, req)).value
166               case Right(suc) => routes(AuthedRequest(suc, req)).value
167             }
168           }
169         }
170   }

182   def inDefaultServiceErrorHandler[F[_], G[_]](implicit
183       F: Monad[F]
184   ): Request[G] => PartialFunction[Throwable, F[Response[G]]] =
185     req => {
186       case mf: MessageFailure =>
187         messageFailureLogger.debug(mf)(
188           s"""Message failure handling request: ${req.method} ${req.pathInfo} from ${req.remoteAddr
189               .getOrElse("<unknown>")}"""
190         )
191         mf.toHttpResponse[G](req.httpVersion).pure[F]
192       case NonFatal(t) =>
193         serviceErrorLogger.error(t)(
194           s"""Error servicing request: ${req.method} ${req.pathInfo} from ${req.remoteAddr
195               .getOrElse("<unknown>")}"""
196         )
197         F.pure(
198           Response(
199             Status.InternalServerError,
200             req.httpVersion,
201             Headers(
202               Connection.close,
203               `Content-Length`.zero,
204             ),
205           )
206         )
207     }
