27   object Segment {
28     implicit def instances[F[_]: Functor]: Functor[Segment[F, *]] =
29       new Functor[Segment[F, *]] {
30         override def map[A, B](fa: Segment[F, A])(f: A => B): Segment[F, B] =
31           Segment(fa.run(_).map(f))
32       }
33   }
34 
35   final case class Segment[F[_], A](run: Uri.Path.Segment => OptionT[F, A]) {
36     def apply(routes: ContextRoutes[A, F])(implicit F: Monad[F]): HttpRoutes[F] =
37       Kleisli { req =>
38         for {
39           head <- OptionT.fromOption[F](req.pathInfo.segments.headOption)
40           a <- run(head)
41           caret = req.attributes.lookup(Request.Keys.PathInfoCaret).getOrElse(0)
42           response <- routes(
43             ContextRequest(a, req.withAttribute(Request.Keys.PathInfoCaret, caret + 1))
44           )
45         } yield response
46       }
47 
48     def ->(routes: ContextRoutes[A, F]): Routable[F] =
49       Routable.Dynamic(this, routes)
50   }
51 
52   object Routable {
53     final case class Static[F[_]](tupled: (String, HttpRoutes[F])) extends Routable[F]
54     final case class Dynamic[F[_], A](segment: Segment[F, A], routes: ContextRoutes[A, F])
55         extends Routable[F]
56 
57     implicit def tuple[F[_]](tupled: (String, HttpRoutes[F])): Routable[F] =
58       Static(tupled)
59   }
60   sealed trait Routable[F[_]]
61 
62   /** Defines an [[HttpRoutes]] based on list of mappings.
63     * @see define
64     */
65   def apply[F[_]: Monad](mappings: (String, HttpRoutes[F])*): HttpRoutes[F] =
66     define(mappings: _*)(HttpRoutes.empty[F])
67 
68   /** Defines an [[HttpRoutes]] based on list of mappings and
69     * a default Service to be used when none in the list match incomming requests.
70     *
71     * The mappings are processed in descending order (longest first) of prefix length.
72     */
73   def define[F[_]: Monad](
74       mappings: (String, HttpRoutes[F])*
75   )(default: HttpRoutes[F]): HttpRoutes[F] =
76     mappings.sortBy(_._1.length).foldLeft(default) { case (acc, (prefix, routes)) =>
77       val prefixPath = Uri.Path.unsafeFromString(prefix)
78       if (prefixPath.isEmpty) routes <+> acc
79       else
80         Kleisli { req =>
81           if (req.pathInfo.startsWith(prefixPath))
82             routes(translate(prefixPath)(req)).orElse(acc(req))
83           else
84             acc(req)
85         }
86     }
87 
88   def of[F[_]: Monad](mappings: Routable[F]*): HttpRoutes[F] =
89     dynamic(mappings: _*)(HttpRoutes.empty[F])
90 
91   def dynamic[F[_]: Monad](
92       mappings: Routable[F]*
93   )(default: HttpRoutes[F]): HttpRoutes[F] = {
94     val (statics, dynamic) = mappings.toList.partitionEither {
95       case Routable.Static(tupled) => Left(tupled)
96       case Routable.Dynamic(segment, route) => Right(segment(route))
97     }
98     dynamic.foldLeft(define(statics: _*)(default))(_ <+> _)
99   }
100 
101   private[server] def translate[F[_]](prefix: Uri.Path)(req: Request[F]): Request[F] = {
102     val newCaret = req.pathInfo.findSplit(prefix)
103     val oldCaret = req.attributes.lookup(Request.Keys.PathInfoCaret)
104     val resultCaret = oldCaret |+| newCaret
105     resultCaret match {
106       case Some(index) => req.withAttribute(Request.Keys.PathInfoCaret, index)
107       case None => req
108     }
109   }
