26 object ErrorAction {
27   def apply[F[_]: ApplicativeThrow, G[_], B](
28       k: Kleisli[F, Request[G], B],
29       f: (Request[G], Throwable) => F[Unit],
30   ): Kleisli[F, Request[G], B] =
31     Kleisli { req =>
32       k.run(req).onError { case e => f(req, e) }
33     }
34 
35   def log[F[_]: ApplicativeThrow, G[_], B](
36       http: Kleisli[F, Request[G], B],
37       messageFailureLogAction: (Throwable, => String) => F[Unit],
38       serviceErrorLogAction: (Throwable, => String) => F[Unit],
39   ): Kleisli[F, Request[G], B] =
40     apply(
41       http,
42       {
43         case (req, mf: MessageFailure) =>
44           messageFailureLogAction(
45             mf,
46             s"""Message failure handling request: ${req.method} ${req.pathInfo} from ${req.remoteAddr
47                 .getOrElse("<unknown>")}""",
48           )
49         case (req, e) =>
50           serviceErrorLogAction(
51             e,
52             s"""Error servicing request: ${req.method} ${req.pathInfo} from ${req.remoteAddr
53                 .getOrElse("<unknown>")}""",
54           )
55       },
56     )
57 
58   object httpApp {
59     def apply[F[_]: ApplicativeThrow](
60         httpApp: HttpApp[F],
61         f: (Request[F], Throwable) => F[Unit],
62     ): HttpApp[F] =
63       ErrorAction(httpApp, f)
64 
65     def log[F[_]: ApplicativeThrow, G[_], B](
66         httpApp: HttpApp[F],
67         messageFailureLogAction: (Throwable, => String) => F[Unit],
68         serviceErrorLogAction: (Throwable, => String) => F[Unit],
69     ): HttpApp[F] =
70       ErrorAction.log(httpApp, messageFailureLogAction, serviceErrorLogAction)
71   }
72 
73   object httpRoutes {
74     def apply[F[_]: MonadThrow](
75         httpRoutes: HttpRoutes[F],

79 
80     def log[F[_]: MonadThrow](
81         httpRoutes: HttpRoutes[F],
82         messageFailureLogAction: (Throwable, => String) => F[Unit],

89       )
90   }
