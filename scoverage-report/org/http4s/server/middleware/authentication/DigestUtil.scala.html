51   /** Computes the response value used in Digest Authentication.
52     * @param method
53     * @param ha1
54     * @param uri
55     * @param nonce
56     * @param nc
57     * @param cnonce
58     * @param qop
59     * @return
60     */
61   def computeHashedResponse[F[_]: Monad: Hash](
62       method: String,
63       ha1: String,
64       uri: String,
65       nonce: String,
66       nc: String,
67       cnonce: String,
68       qop: String,
69   ): F[String] = for {
70     ha2str <- (method + ":" + uri).pure[F]
71     ha2 <- md5(ha2str)
72     respstr = ha1 + ":" + nonce + ":" + nc + ":" + cnonce + ":" + qop + ":" + ha2
73     result <- md5(respstr)
74   } yield result

76   /** Computes the ha1 component of the Digest Authentication scheme
77     * @param username
78     * @param realm
79     * @param password
80     * @return The hash of the supplied fields when concatenated together
81     */
82   def computeHa1[F[_]: Monad: Hash](username: String, realm: String, password: String): F[String] =
83     for {
84       ha1str <- (username + ":" + realm + ":" + password).pure[F]
85       ha1 <- md5(ha1str)
