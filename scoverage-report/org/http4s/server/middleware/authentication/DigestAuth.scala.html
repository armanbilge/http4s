38   @deprecated(
39     "AuthenticationStore is going away, in favor of explicit subclasses of AuthStore. PlainTextAuthStore maintains the previous, insecure behaviour, whereas Md5HashedAuthStore is the new advised implementation going forward.",
40     "0.23.12",
41   )
42   type AuthenticationStore[F[_], A] = String => F[Option[(A, String)]]
43 
44   sealed trait AuthStore[F[_], A]
45 
46   /** A function mapping username to a user object and password, or
47     * None if no user exists.
48     *
49     * Requires that the server can recover the password in clear text,
50     * which is _strongly_ discouraged. Please use Md5HashedAuthStore
51     * if you can.
52     */
53   object PlainTextAuthStore {
54     def apply[F[_], A](func: String => F[Option[(A, String)]]): AuthStore[F, A] =
55       new PlainTextAuthStore(func)
56   }
57   final class PlainTextAuthStore[F[_], A](val func: String => F[Option[(A, String)]])
58       extends AuthStore[F, A]
59 
60   /** A function mapping username to a user object and precomputed md5
61     * hash of the username, realm, and password, or None if no user exists.
62     *
63     * More secure than PlainTextAuthStore due to only needing to
64     * store the digested hash instead of the password in plain text.
65     */
66   object Md5HashedAuthStore {
67     def apply[F[_], A](func: String => F[Option[(A, String)]]): AuthStore[F, A] =
68       new Md5HashedAuthStore(func)
69 
70     def precomputeHash[F[_]: Monad: Hash](
71         username: String,
72         realm: String,
73         password: String,
74     ): F[String] =
75       DigestUtil.computeHa1(username, realm, password)
76   }
77   final class Md5HashedAuthStore[F[_], A](val func: String => F[Option[(A, String)]])
78       extends AuthStore[F, A]

80   private trait AuthReply[+A]
81   private final case class OK[A](authInfo: A) extends AuthReply[A]
82   private case object StaleNonce extends AuthReply[Nothing]
83   private case object BadNC extends AuthReply[Nothing]
84   private case object WrongResponse extends AuthReply[Nothing]
85   private case object BadParameters extends AuthReply[Nothing]
86   private case object UserUnknown extends AuthReply[Nothing]
87   private case object NoCredentials extends AuthReply[Nothing]
88   private case object NoAuthorizationHeader extends AuthReply[Nothing]

90   @deprecated("Calling apply is side-effecting, please use applyF", "0.23.12")
91   def apply[F[_]: Sync, A](
92       realm: String,
93       store: String => F[Option[(A, String)]],
94       nonceCleanupInterval: Duration = 1.hour,
95       nonceStaleTime: Duration = 1.hour,
96       nonceBits: Int = 160,
97   ): AuthMiddleware[F, A] = {
98     val nonceKeeper =
99       new NonceKeeper(nonceStaleTime.toMillis, nonceCleanupInterval.toMillis, nonceBits)
100     challenged(challenge(realm, store, nonceKeeper))
101   }

103   /** @param realm The realm used for authentication purposes.
104     * @param store A partial function mapping (realm, user) to the
105     *              appropriate password.
106     * @param nonceCleanupInterval Interval (in milliseconds) at which stale
107     *                             nonces should be cleaned up.
108     * @param nonceStaleTime Amount of time (in milliseconds) after which a nonce
109     *                       is considered stale (i.e. not used for authentication
110     *                       purposes anymore).
111     * @param nonceBits The number of random bits a nonce should consist of.
112     */
113   def applyF[F[_], A](
114       realm: String,
115       store: AuthStore[F, A],
116       nonceCleanupInterval: Duration = 1.hour,
117       nonceStaleTime: Duration = 1.hour,
118       nonceBits: Int = 160,
119   )(implicit F: Async[F]): F[AuthMiddleware[F, A]] =
120     challenge[F, A](
121       realm = realm,
122       store = store,
123       nonceCleanupInterval = nonceCleanupInterval,
124       nonceStaleTime = nonceStaleTime,
125       nonceBits = nonceBits,
126     ).map { runChallenge =>
127       challenged(runChallenge)
128     }

130   @deprecated(
131     "Uses a side-effecting NonceKeeper. Use challenge(String, AuthStore, Blocker, Duration, Int, Int).",

134   def challenge[F[_], A](
135       realm: String,
136       store: String => F[Option[(A, String)]],

148   /** Similar to [[apply]], but exposing the underlying [[challenge[F[_],A](realm:String,store:org*]]
149     * [[cats.data.Kleisli]] instead of an entire [[AuthMiddleware]]
150     *
151     * Side-effect of running the returned task: If req contains a valid
152     * AuthorizationHeader, the corresponding nonce counter (nc) is increased.
153     *
154     * @param realm The realm used for authentication purposes.
155     * @param store A partial function mapping (realm, user) to the
156     *              appropriate password.
157     * @param nonceCleanupInterval Interval (in milliseconds) at which stale
158     *                             nonces should be cleaned up.
159     * @param nonceStaleTime Amount of time (in milliseconds) after which a nonce
160     *                       is considered stale (i.e. not used for authentication
161     *                       purposes anymore).
162     * @param nonceBits The number of random bits a nonce should consist of.
163     */
164   def challenge[F[_], A](
165       realm: String,
166       store: AuthStore[F, A],
167       nonceCleanupInterval: Duration = 1.hour,
168       nonceStaleTime: Duration = 1.hour,
169       nonceBits: Int = 160,
170   )(implicit
171       F: Async[F]
172   ): F[Kleisli[F, Request[F], Either[Challenge, AuthedRequest[F, A]]]] =
173     NonceKeeperF[F](nonceStaleTime, nonceCleanupInterval, nonceBits)
174       .map { nonceKeeper =>
175         challengeInterop[F, A](realm, store, nonceKeeper.newNonce(), nonceKeeper.receiveNonce _)
176       }
177 
178   private[this] def challengeInterop[F[_], A](
179       realm: String,
180       store: AuthStore[F, A],
181       newNonce: F[String],
182       receiveNonce: (String, Int) => F[NonceKeeper.Reply],
183   )(implicit
184       F: Sync[F]
185   ): Kleisli[F, Request[F], Either[Challenge, AuthedRequest[F, A]]] =
186     Kleisli { req =>
187       def paramsToChallenge(params: Map[String, String]) =
188         Either.left(Challenge("Digest", realm, params))
189 
190       checkAuth(realm, store, receiveNonce, req).flatMap {
191         case OK(authInfo) => F.pure(Either.right(AuthedRequest(authInfo, req)))
192         case StaleNonce =>
193           getChallengeParams(newNonce, staleNonce = true).map(paramsToChallenge)
194         case _ => getChallengeParams(newNonce, staleNonce = false).map(paramsToChallenge)
195       }
196     }
197 
198   private def checkAuth[F[_]: Hash, A](
199       realm: String,
200       store: AuthStore[F, A],
201       receiveNonce: (String, Int) => F[NonceKeeper.Reply],
202       req: Request[F],
203   )(implicit F: Monad[F]): F[AuthReply[A]] =
204     req.headers.get[Authorization] match {
205       case Some(Authorization(Credentials.AuthParams(AuthScheme.Digest, params))) =>
206         checkAuthParams(realm, store, receiveNonce, req, params)
207       case Some(_) =>
208         F.pure(NoCredentials)
209       case None =>
210         F.pure(NoAuthorizationHeader)
211     }
212 
213   private def getChallengeParams[F[_]](newNonce: F[String], staleNonce: Boolean)(implicit
214       F: Functor[F]
215   ): F[Map[String, String]] =
216     newNonce
217       .map { nonce =>
218         val m = Map("qop" -> "auth", "nonce" -> nonce)
219         if (staleNonce)
220           m + ("stale" -> "TRUE")
221         else
222           m
223       }
224 
225   private def checkAuthParams[F[_]: Hash, A](
226       realm: String,
227       store: AuthStore[F, A],
228       receiveNonce: (String, Int) => F[NonceKeeper.Reply],
229       req: Request[F],
230       paramsNel: NonEmptyList[(String, String)],
231   )(implicit F: Monad[F]): F[AuthReply[A]] = {
232     val params = paramsNel.toList.toMap
233     if (!Set("realm", "nonce", "nc", "username", "cnonce", "qop").subsetOf(params.keySet)) {
234       F.pure(BadParameters)
235     } else {
236       val method = req.method.toString
237 
238       if (!params.get("realm").contains(realm)) {
239         F.pure(BadParameters)
240       } else {
241         val nonce = params("nonce")
242         val nc = params("nc")
243         receiveNonce(nonce, Integer.parseInt(nc, 16)).flatMap {
244           case NonceKeeper.StaleReply => F.pure(StaleNonce)
245           case NonceKeeper.BadNCReply => F.pure(BadNC)
246           case NonceKeeper.OKReply =>
247             store match {
248               case authStore: PlainTextAuthStore[F, A] =>
249                 authStore.func(params("username")).flatMap {
250                   case None => F.pure(UserUnknown)
251                   case Some((authInfo, password)) =>
252                     DigestUtil
253                       .computeResponse(
254                         method,
255                         params("username"),
256                         realm,
257                         password,
258                         req.uri,
259                         nonce,
260                         nc,
261                         params("cnonce"),
262                         params("qop"),
263                       )
264                       .map { resp =>
265                         if (resp == params("response")) OK(authInfo)
266                         else WrongResponse
267                       }
268                 }
269               case authStore: Md5HashedAuthStore[F, A] =>
270                 authStore.func(params("username")).flatMap {
271                   case None => F.pure(UserUnknown)
272                   case Some((authInfo, ha1Hash)) =>
273                     DigestUtil
274                       .computeHashedResponse(
275                         method,
276                         ha1Hash,
277                         req.uri,
278                         nonce,
279                         nc,
280                         params("cnonce"),
281                         params("qop"),
282                       )
283                       .map { resp =>
284                         if (resp == params("response")) OK(authInfo)
285                         else WrongResponse
286                       }
287                 }
288             }
289         }
290       }
291     }
292   }
