29       challenge: Kleisli[F, Request[F], Either[Challenge, AuthedRequest[F, A]]]
30   )(routes: AuthedRoutes[A, F])(implicit F: Sync[F]): HttpRoutes[F] =
31     Kleisli { req =>
32       OptionT[F, Response[F]] {
33         F.flatMap(challenge(req)) {
34           case Left(challenge) => F.pure(Some(unauthorized(challenge)))
35           case Right(authedRequest) => routes(authedRequest).value
36         }
37       }
38     }
39 
40   private[this] def unauthorized[F[_]](challenge: Challenge): Response[F] =
41     Response(Status.Unauthorized).putHeaders(`WWW-Authenticate`(challenge))
