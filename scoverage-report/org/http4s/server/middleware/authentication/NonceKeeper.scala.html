32 /** A thread-safe class used to manage a database of nonces.
33   *
34   * @param staleTimeout Amount of time (in milliseconds) after which a nonce
35   *                     is considered stale (i.e. not used for authentication
36   *                     purposes anymore).
37   * @param bits The number of random bits a nonce should consist of.
38   */
39 @deprecated("Side-effecting. Use NonceKeeperF.", "0.23.12")
40 private[authentication] class NonceKeeper(
41     staleTimeout: Long,
42     nonceCleanupInterval: Long,
43     bits: Int,
44 ) {
45   require(bits > 0, "Please supply a positive integer for bits.")
46   private val nonces = new LinkedHashMap[String, Nonce]
47   private var lastCleanup = System.currentTimeMillis()
48 
49   /** Removes nonces that are older than staleTimeout
50     * Note: this _MUST_ be executed inside a block synchronized on `nonces`
51     */
52   private def checkStale() = {
53     val d = System.currentTimeMillis()
54     if (d - lastCleanup > nonceCleanupInterval) {
55       lastCleanup = d
56 
57       // Because we are using an LinkedHashMap, the keys will be returned in the order they were
58       // inserted. Therefor, once we reach a non-stale value, the remaining values are also not stale.
59       val it = nonces.values().iterator()
60       @tailrec
61       def dropStale(): Unit =
62         if (it.hasNext && staleTimeout > d - it.next().created.getTime) {
63           it.remove()
64           dropStale()
65         }
66       dropStale()
67     }
68   }
69 
70   /** Get a fresh nonce in form of a {@link String}.
71     * @return A fresh nonce.
72     */
73   def newNonce(): String = {
74     var n: Nonce = null
75     nonces.synchronized {
76       checkStale()
77       n = Nonce.gen(bits)
78       while (nonces.get(n.data) != null) n = Nonce.gen(bits)
79       nonces.put(n.data, n)
80     }
81     n.data
82   }
83 
84   /** Checks if the nonce {@link data} is known and the {@link nc} value is
85     * correct. If this is so, the nc value associated with the nonce is increased
86     * and the appropriate status is returned.
87     * @param data The nonce.
88     * @param nc The nonce counter.
89     * @return A reply indicating the status of (data, nc).
90     */
91   def receiveNonce(data: String, nc: Int): NonceKeeper.Reply =
92     nonces.synchronized {
93       checkStale()
94       nonces.get(data) match {
95         case null => NonceKeeper.StaleReply
96         case n: Nonce =>
97           if (nc > n.nc) {
98             n.nc = n.nc + 1
99             NonceKeeper.OKReply
100           } else
101             NonceKeeper.BadNCReply
102       }
103     }
104 }
