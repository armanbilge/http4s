30 private[authentication] object NonceKeeperF {
31   def apply[F[_]](
32       staleTimeout: Duration,
33       nonceCleanupInterval: Duration,
34       bits: Int,
35   )(implicit F: Async[F]): F[NonceKeeperF[F]] = for {
36     // This semaphore controls who has access to `nonces` during stale nonce eviction. This must never be set above one.
37     semaphore <- Semaphore[F](1)
38     current <- F.monotonic
39     lastCleanupMillis <- Ref[F].of(current)
40     nonces = new LinkedHashMap[String, NonceF[F]]
41     random <- Random.javaSecuritySecureRandom[F]
42   } yield new NonceKeeperF(
43     staleTimeout,
44     nonceCleanupInterval,
45     bits,
46     semaphore,
47     lastCleanupMillis,
48     nonces,
49     random,
50   )
51 }

53 /** A thread-safe class used to manage a database of nonces.
54   *
55   * @param staleTimeout Amount of time (in milliseconds) after which a nonce
56   *                     is considered stale (i.e. not used for authentication
57   *                     purposes anymore).
58   * @param bits The number of random bits a nonce should consist of.
59   */
60 private[authentication] class NonceKeeperF[F[_]](
61     staleTimeout: Duration,
62     nonceCleanupInterval: Duration,
63     bits: Int,
64     semaphore: Semaphore[F],
65     lastCleanupMillis: Ref[F, FiniteDuration],
66     nonces: LinkedHashMap[String, NonceF[F]],
67     random: Random[F],
68 )(implicit F: Async[F]) {
69   require(bits > 0, "Please supply a positive integer for bits.")
70 
71   /** Removes nonces that are older than staleTimeout
72     * Note: this _MUST_ be executed with the singleton permit from the semaphore
73     */
74   private def unsafeCheckStale(): F[Unit] =
75     for {
76       now <- F.monotonic
77       lastCleanupTime <- lastCleanupMillis.get
78       result <-
79         if (now - lastCleanupTime > nonceCleanupInterval) {
80           lastCleanupMillis
81             .set(now)
82             .flatMap { _ =>
83               // Because we are using an LinkedHashMap, the keys will be returned in the order they were
84               // inserted. Therefore, once we reach a non-stale value, the remaining values are also not stale.
85               F.tailRecM[ju.Iterator[NonceF[F]], Unit](nonces.values().iterator()) {
86                 case it if it.hasNext && staleTimeout > now - it.next().created =>
87                   F.delay(Left {
88                     it.remove()
89                     it
90                   })
91                 case _ => F.delay(Right(()))
92               }
93             }
94         } else F.pure(())
95     } yield result
96 
97   /** Get a fresh nonce in form of a {@link String}.
98     * @return A fresh nonce.
99     */
100   def newNonce(): F[String] =
101     semaphore.permit.surround {
102       for {
103         _ <- unsafeCheckStale()
104         n <- NonceF.gen[F](random, bits).iterateUntil(n => nonces.get(n.data) == null)
105       } yield {
106         nonces.put(n.data, n)
107         n.data
108       }
109     }
110 
111   /** Checks if the nonce {@link data} is known and the {@link nc} value is
112     * correct. If this is so, the nc value associated with the nonce is increased
113     * and the appropriate status is returned.
114     * @param data The nonce.
115     * @param nc The nonce counter.
116     * @return A reply indicating the status of (data, nc).
117     */
118   def receiveNonce(data: String, nc: Int): F[NonceKeeper.Reply] =
119     semaphore.permit.surround {
120       for {
121         _ <- unsafeCheckStale()
122         res <- nonces.get(data) match {
123           case null => F.pure(NonceKeeper.StaleReply)
124           case n: NonceF[F] =>
125             n.nc.modify { lastNc =>
126               if (nc > lastNc) {
127                 (lastNc + 1, NonceKeeper.OKReply)
128               } else
129                 (lastNc, NonceKeeper.BadNCReply)
130             }
131         }
132       } yield res
133     }
134 }
