35 
36   private def bytesToHex(arr: Array[Byte]): String = {
37     val sb = new StringBuilder(arr.size * 2)
38     for (i <- 0 until arr.size) {
39       val b = arr(i) & 0xff
40       sb.append(hexAlphabet(b >>> 4))
41       sb.append(hexAlphabet(b & 0x0f))
42     }
43     sb.toString
44   }
45 
46   private def getRandomBits[F[_]: Functor](random: Random[F], bits: Int): F[Array[Byte]] = {
47     val bytes = (bits + 7) / 8
48     random.nextBytes(bytes).map { arr =>
49       if (arr.nonEmpty) {
50         val extraBits = 8 * bytes - bits
51         arr(0) = (arr(0) & ((1 << (8 - extraBits)) - 1)).toByte
52       }
53       arr
54     }
55   }
56 
57   def getRandomData[F[_]: Functor](random: Random[F], bits: Int): F[String] =
58     getRandomBits[F](random, bits).map(bytesToHex)
59 
60   def gen[F[_]](random: Random[F], bits: Int)(implicit
61       F: Async[F]
62   ): F[NonceF[F]] = for {
63     nc <- Ref[F].of(0)
64     data <- getRandomData[F](random, bits)
65     created <- F.monotonic
66   } yield new NonceF(created, nc, data)
