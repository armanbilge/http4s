28 /** Provides Basic Authentication from RFC 2617.
29   */
30 object BasicAuth {
31 
32   private val authParams = Map("charset" -> "UTF-8")
33 
34   /** Validates a plaintext password (presumably by comparing it to a
35     * hashed value).  A Some value indicates success; None indicates
36     * the password failed to validate.
37     */
38   type BasicAuthenticator[F[_], A] = BasicCredentials => F[Option[A]]
39 
40   /** Construct authentication middleware that can validate the client-provided
41     * plaintext password against something else (like a stored, hashed password).
42     * @param realm The realm used for authentication purposes.
43     * @param validate Function that validates a plaintext password
44     * @return
45     */
46   def apply[F[_]: Sync, A](
47       realm: String,
48       validate: BasicAuthenticator[F, A],
49   ): AuthMiddleware[F, A] =
50     challenged(challenge(realm, validate))
51 
52   def challenge[F[_]: Applicative, A](
53       realm: String,
54       validate: BasicAuthenticator[F, A],
55   ): Kleisli[F, Request[F], Either[Challenge, AuthedRequest[F, A]]] =
56     Kleisli { req =>
57       validatePassword(validate, req).map {
58         case Some(authInfo) =>
59           Right(AuthedRequest(authInfo, req))
60         case None =>
61           Left(Challenge("Basic", realm, authParams))
62       }
63     }
64 
65   private def validatePassword[F[_], A](validate: BasicAuthenticator[F, A], req: Request[F])(
66       implicit F: Applicative[F]
67   ): F[Option[A]] =
68     req.headers.get[Authorization] match {
69       case Some(Authorization(BasicCredentials(username, password))) =>
70         validate(BasicCredentials(username, password))
71       case _ =>
72         F.pure(None)
73     }
74 }
