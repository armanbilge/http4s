26 /** [[Middleware]] for lifting application/x-www-form-urlencoded bodies into the
27   * request query params.
28   *
29   * The params are merged into the existing paras _after_ the existing query params. This
30   * means that if the query already contains the pair "foo" -> Some("bar"), parameters on
31   * the body must be acessed through `multiParams`.
32   */
33 object UrlFormLifter {
34   def apply[F[_]: Sync, G[_]: Concurrent](f: G ~> F)(
35       http: Kleisli[F, Request[G], Response[G]],
36       strictDecode: Boolean = false,
37   ): Kleisli[F, Request[G], Response[G]] =
38     Kleisli { req =>
39       def addUrlForm(form: UrlForm): F[Response[G]] = {
40         val flatForm = form.values.toVector.flatMap { case (k, vs) =>
41           vs.toVector.map(v => (k, Some(v)))
42         }
43         val params = req.uri.query.toVector ++ flatForm: Vector[(String, Option[String])]
44         val newQuery = Query(params: _*)
45 
46         val newRequest = req
47           .withUri(req.uri.copy(query = newQuery))
48           .withAttributes(req.attributes)
49           .withEmptyBody
50         http(newRequest)
51       }
52 
53       req.headers.get[headers.`Content-Type`] match {
54         case Some(headers.`Content-Type`(MediaType.application.`x-www-form-urlencoded`, _))
55             if checkRequest(req) =>
56           for {
57             decoded <- f(UrlForm.entityDecoder[G].decode(req, strictDecode).value)
58             resp <- decoded.fold(
59               mf => f(mf.toHttpResponse[G](req.httpVersion).pure[G]),
60               addUrlForm,
61             )
62           } yield resp
63 
64         case _ => http(req)
65       }
66     }
67 
68   private def checkRequest[F[_]](req: Request[F]): Boolean =
69     req.method == Method.POST || req.method == Method.PUT
70 }
