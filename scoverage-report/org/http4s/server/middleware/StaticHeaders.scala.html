26 /** Simple middleware for adding a static set of headers to responses
27   * returned by a kleisli.
28   */
29 object StaticHeaders {
30   def apply[F[_]: Functor, G[_], A](
31       headers: Headers
32   )(http: Kleisli[F, A, Response[G]]): Kleisli[F, A, Response[G]] =
33     Kleisli { req =>
34       http(req).map(resp => resp.copy(headers = headers ++ resp.headers))
35     }
36 
37   private val noCacheHeader = `Cache-Control`(NonEmptyList.of(CacheDirective.`no-cache`()))
38 
39   def `no-cache`[F[_]: Functor, G[_], A](
40       http: Kleisli[F, A, Response[G]]
41   ): Kleisli[F, A, Response[G]] =
42     StaticHeaders(Headers(noCacheHeader))(http)
43 }
