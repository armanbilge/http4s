50 /** Middleware to avoid Cross-site request forgery attacks.
51   * More info on CSRF at: https://www.owasp.org/index.php/Cross-Site_Request_Forgery_(CSRF)
52   *
53   * This middleware is modeled after the double submit cookie pattern:
54   * https://www.owasp.org/index.php/Cross-Site_Request_Forgery_(CSRF)_Prevention_Cheat_Sheet#Double_Submit_Cookie
55   *
56   * When a user authenticates, `embedNew` is used to send a random CSRF value as a cookie.  (Alternatively,
57   * an authenticating service can be wrapped in `withNewToken`).
58   *
59   * By default, for requests that are unsafe (PUT, POST, DELETE, PATCH), services protected by the `validated` method in the
60   * middleware will check that the csrf token is present in both the header `headerName` and the cookie `cookieName`.
61   * Due to the Same-Origin policy, an attacker will be unable to reproduce this value in a
62   * custom header, resulting in a `403 Forbidden` response.
63   *
64   * By default, requests with safe methods (such as GET, OPTIONS, HEAD) will have a new token embedded in them if there isn't one,
65   * or will receive a refreshed token based off of the previous token to mitigate the BREACH vulnerability. If a request
66   * contains an invalid token, regardless of whether it is a safe method, this middleware will fail it with
67   * `403 Forbidden`. In this situation, your user(s) should clear their cookies for your page, to receive a new
68   * token.
69   *
70   * The default can be overridden by modifying the `predicate` in `validate`. It will, by default, check if the method is safe.
71   * Thus, you can provide some whitelisting capability for certain kinds of requests.
72   *
73   * We'd like to emphasize that you please follow proper design principles in creating endpoints, as to
74   * not mutate in what should otherwise be idempotent methods (i.e no dropping your DB in a GET method, or altering
75   * user data). Please do not use the CSRF protection from this middleware as a safety net for bad design.
76   *
77   * @param headerName your CSRF header name
78   * @param cookieSettings the CSRF cookie settings
79   * @param key the CSRF signing key
80   * @param clock clock used as a nonce
81   */

96   /** Sign our token using the current time in milliseconds as a nonce
97     * Signing and generating a token is potentially a unsafe operation
98     * if constructed with a bad key.
99     */
100   def signToken[M[_]](rawToken: String)(implicit F: Sync[M]): M[CSRFToken] =
101     for {
102       joined <- F.delay(rawToken + "-" + clock.millis())
103       data <- F.fromEither(ByteVector.encodeUtf8(joined))
104       out <- Hmac[M].digest(key, data)
105     } yield lift(joined + "-" + encodeHexString(out.toArray))

164   /** Decode our CSRF token, check the signature
165     * and extract the original token string to sign
166     */
167   def extractRaw(rawToken: String): Either[CSRFCheckFailed, String] =
168     rawToken.split("-") match {
169       case Array(raw, nonce, signed) =>
170         val out = Hmac[SyncIO]
171           .digest(key, ByteVector.view((raw + "-" + nonce).getBytes(StandardCharsets.UTF_8)))
172           .unsafeRunSync()
173         decodeHexString(signed) match {
174           case Some(decoded) =>
175             if (SecureEq[ByteVector].eqv(out, ByteVector.view(decoded)))
176               Right(raw)
177             else
178               Left(CSRFCheckFailed)
179           case None =>
180             Left(CSRFCheckFailed)
181         }
182       case _ =>
183         Left(CSRFCheckFailed)
184     }

186   /** To be only used on safe methods: if the method is safe (i.e doesn't modify data)
187     * and a token is present, validate and regenerate it for BREACH to be impractical
188     */
189   private[middleware] def validate(r: Request[G], response: F[Response[G]])(implicit
190       F: Sync[F]
191   ): F[Response[G]] =
192     CSRF.cookieFromHeaders(r, cookieSettings.cookieName) match {
193       case Some(c) =>
194         (for {
195           raw <- F.fromEither(extractRaw(c.content))
196           res <- response
197           newToken <- signToken[F](raw)
198         } yield res.addCookie(createResponseCookie(newToken)))
199           .recover { case CSRFCheckFailed =>
200             onFailure
201           }
202       case None =>
203         if (createIfNotFound)
204           response.flatMap(r => embedNewInResponseCookie(r))
205         else response
206     }

208   /** Check for CSRF validity for an unsafe action.
209     *
210     * Exposed to users in case of manual plumbing of csrf token
211     * (i.e websocket or query param)
212     */
213   def checkCSRFToken(r: Request[G], respAction: F[Response[G]], rawToken: String)(implicit
214       F: Sync[F]
215   ): F[Response[G]] =
216     if (!headerCheck(r))
217       F.pure(onFailure)
218     else
219       (for {
220         c1 <- CSRF.cookieFromHeadersF[F, G](r, cookieSettings.cookieName)
221         raw1 <- F.fromEither(extractRaw(c1.content))
222         raw2 <- F.fromEither(extractRaw(rawToken))
223         response <-
224           if (CSRF.isEqual(raw1, raw2)) respAction
225           else F.raiseError[Response[G]](CSRFCheckFailed)
226         newToken <- signToken[F](raw1) // Generate a new token to guard against BREACH.
227       } yield response.addCookie(createResponseCookie(newToken)))
228         .recover { case CSRFCheckFailed =>
229           onFailure
230         }

235   /** Constructs a middleware that will check for the csrf token
236     * presence on both the proper cookie, and header values,
237     * if the predicate is not satisfied
238     *
239     * If it is a valid token, it will then embed a new one,
240     * to effectively randomize the complete token while
241     * avoiding the generation of a new secure random Id, to guard
242     * against [BREACH](http://breachattack.com/)
243     */
244   def validate(
245       predicate: Request[G] => Boolean = _.method.isSafe
246   ): Middleware[F, Request[G], Response[G], Request[G], Response[G]] = { http =>
247     Kleisli { (r: Request[G]) =>
248       if (predicate(r)) validate(r, http(r)) else checkCSRF(r, http(r))

411   def checkCSRFinHeaderAndForm[F[_], G[_]: Concurrent](fieldName: String, nt: G ~> F)(implicit
412       F: Sync[F]
413   ): CSRF[F, G] => CSRFCheck[F, G] = { csrf => (r, http) =>
414     def getFormToken: F[Option[String]] = {
415       def extractToken: G[Option[String]] =
416         r.attemptAs[UrlForm]
417           .value
418           .map(_.fold(_ => none[String], _.values.get(fieldName).flatMap(_.uncons.map(_._1))))
419 
420       r.headers.get[`Content-Type`] match {
421         case Some(headers.`Content-Type`(MediaType.application.`x-www-form-urlencoded`, _)) =>
422           nt(extractToken)
423         case _ => F.pure(none[String])
424       }
425     }
426 
427     for {
428       fst <- F.pure(csrf.getHeaderToken(r))
429       snd <- if (fst.isDefined) F.pure(fst) else getFormToken
430       tok <- snd.fold(csrf.onfailureF)(csrf.checkCSRFToken(r, http, _))
431     } yield tok
432   }

453   def defaultOriginCheck[F[_]](
454       r: Request[F],
455       host: String,
456       sc: Scheme,
457       port: Option[Int],
458   ): Boolean =
459     r.headers
460       .get(ci"Origin")
461       .flatMap(o =>
462         // Hack to get around 2.11 compat
463         Uri.fromString(o.head.value) match {
464           case Right(uri) => Some(uri)
465           case Left(_) => None
466         }
467       )
468       .exists(u =>
469         u.host.exists(_.value == host) && u.scheme.contains(sc) && u.port == port
470       ) || r.headers
471       .get[Referer]
472       .exists(u =>
473         u.uri.host.exists(_.value == host) && u.uri.scheme.contains(sc) && u.uri.port == port
474       )
