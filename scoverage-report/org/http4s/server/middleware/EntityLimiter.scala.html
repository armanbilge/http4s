27 object EntityLimiter {
28   final case class EntityTooLarge(limit: Long) extends Exception with NoStackTrace
29 
30   val DefaultMaxEntitySize: Long = 2L * 1024L * 1024L // 2 MB default
31 
32   def apply[F[_], G[_], B](http: Kleisli[F, Request[G], B], limit: Long = DefaultMaxEntitySize)(
33       implicit G: ApplicativeThrow[G]
34   ): Kleisli[F, Request[G], B] =
35     http.local(_.pipeBodyThrough(takeLimited(limit)))
36 
37   def httpRoutes[F[_]: ApplicativeThrow](
38       httpRoutes: HttpRoutes[F],
39       limit: Long = DefaultMaxEntitySize,
40   ): HttpRoutes[F] =
41     apply(httpRoutes, limit)
42 
43   def httpApp[F[_]: ApplicativeThrow](
44       httpApp: HttpApp[F],
45       limit: Long = DefaultMaxEntitySize,
46   ): HttpApp[F] =
47     apply(httpApp, limit)
48 
49   private def takeLimited[F[_]](n: Long)(implicit F: ApplicativeThrow[F]): Pipe[F, Byte, Byte] =
50     _.pull
51       .take(n)
52       .flatMap {
53         case Some(_) => Pull.raiseError[F](EntityTooLarge(n))
54         case None => Pull.done
55       }
56       .stream
57 }
