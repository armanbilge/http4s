37 /** Simple middleware for logging responses as they are processed
38   */
39 object ResponseLogger {
40   private[this] val logger = getLogger

42   def apply[G[_], F[_], A](
43       logHeaders: Boolean,
44       logBody: Boolean,
45       fk: F ~> G,

53   private[server] def impl[G[_], F[_], A](
54       logHeaders: Boolean,
55       logBodyText: Either[Boolean, Stream[F, Byte] => Option[F[String]]],
56       fk: F ~> G,
57       redactHeadersWhen: CIString => Boolean,
58       logAction: Option[String => F[Unit]],
59   )(
60       http: Kleisli[G, A, Response[F]]
61   )(implicit G: MonadCancelThrow[G], F: Async[F]): Kleisli[G, A, Response[F]] = {
62     val fallback: String => F[Unit] = s => Sync[F].delay(logger.info(s))
63     val log = logAction.fold(fallback)(identity)
64 
65     def logMessage(resp: Response[F]): F[Unit] =
66       logBodyText match {
67         case Left(bool) =>
68           Logger.logMessage[F, Response[F]](resp)(logHeaders, bool, redactHeadersWhen)(log(_))
69         case Right(f) =>
70           org.http4s.internal.Logger
71             .logMessageWithBodyText[F, Response[F]](resp)(logHeaders, f, redactHeadersWhen)(log(_))
72       }
73 
74     val logBody: Boolean = logBodyText match {
75       case Left(bool) => bool
76       case Right(_) => true
77     }
78 
79     def logResponse(response: Response[F]): F[Response[F]] =
80       if (!logBody)
81         logMessage(response)
82           .as(response)
83       else
84         F.ref(Vector.empty[Chunk[Byte]]).map { vec =>
85           val newBody = Stream.eval(vec.get).flatMap(v => Stream.emits(v)).unchunks
86           // Cannot Be Done Asynchronously - Otherwise All Chunks May Not Be Appended Previous to Finalization
87           val logPipe: Pipe[F, Byte, Byte] =
88             _.observe(_.chunks.flatMap(c => Stream.exec(vec.update(_ :+ c))))
89               .onFinalizeWeak(logMessage(response.withBodyStream(newBody)))
90 
91           response.pipeBodyThrough(logPipe)
92         }
93 
94     Kleisli[G, A, Response[F]] { req =>
95       http(req)
96         .flatMap((response: Response[F]) => fk(logResponse(response)))
97         .guaranteeCase {
98           case Outcome.Errored(t) => fk(log(s"service raised an error: ${t.getClass}"))
99           case Outcome.Canceled() => fk(log(s"service canceled response for request"))
100           case Outcome.Succeeded(_) => G.unit
101         }
102     }
103   }
104
