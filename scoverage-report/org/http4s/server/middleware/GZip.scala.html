29 object GZip {
30   private[this] val logger = getLogger
31 
32   // TODO: It could be possible to look for F.pure type bodies, and change the Content-Length header after
33   // TODO      zipping and buffering all the input. Just a thought.
34   def apply[F[_]: Functor, G[_]: Sync](
35       http: Http[F, G],
36       bufferSize: Int = 32 * 1024,
37       level: DeflateParams.Level = DeflateParams.Level.DEFAULT,
38       isZippable: Response[G] => Boolean = defaultIsZippable[G](_: Response[G]),
39   ): Http[F, G] =
40     Kleisli { (req: Request[G]) =>
41       req.headers.get[`Accept-Encoding`] match {
42         case Some(acceptEncoding) if satisfiedByGzip(acceptEncoding) =>
43           http(req).map(zipOrPass(_, bufferSize, level, isZippable))
44         case _ => http(req)
45       }
46     }
47 
48   def defaultIsZippable[F[_]](resp: Response[F]): Boolean = {
49     val contentType = resp.headers.get[`Content-Type`]
50     !resp.headers.contains[`Content-Encoding`] &&
51     resp.status.isEntityAllowed &&
52     (contentType.isEmpty || contentType.get.mediaType.compressible ||
53       (contentType.get.mediaType eq MediaType.application.`octet-stream`))
54   }
55 
56   private def satisfiedByGzip(acceptEncoding: `Accept-Encoding`) =
57     acceptEncoding.satisfiedBy(ContentCoding.gzip) || acceptEncoding.satisfiedBy(
58       ContentCoding.`x-gzip`
59     )
60 
61   private def zipOrPass[F[_]: Sync](
62       response: Response[F],
63       bufferSize: Int,
64       level: DeflateParams.Level,
65       isZippable: Response[F] => Boolean,
66   ): Response[F] =
67     response match {
68       case resp if isZippable(resp) => zipResponse(bufferSize, level, resp)
69       case resp => resp // Don't touch it, Content-Encoding already set
70     }
71 
72   private def zipResponse[F[_]: Sync](
73       bufferSize: Int,
74       level: DeflateParams.Level,
75       resp: Response[F],
76   ): Response[F] = {
77     val compressPipe =
78       Compression[F].gzip(
79         fileName = None,
80         modificationTime = None,
81         comment = None,
82         DeflateParams(
83           bufferSize = bufferSize,
84           level = level,
85           header = ZLibParams.Header.GZIP,
86         ),
87       )
88     logger.trace("GZip middleware encoding content")
89     resp
90       .removeHeader[`Content-Length`]
91       .putHeaders(`Content-Encoding`(ContentCoding.gzip))
92       .pipeBodyThrough(compressPipe)
93   }
94 }
