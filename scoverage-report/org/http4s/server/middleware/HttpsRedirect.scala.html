36 /** [[Middleware]] to redirect http traffic to https.
37   * Inspects `X-Forwarded-Proto` header and if it is set to `http`,
38   * redirects to `Host` with same URL with https schema; otherwise does nothing.
39   * This middleware is useful when a service is deployed behind a load balancer
40   * which does not support such redirect feature, e.g. Heroku.
41   */
42 object HttpsRedirect {
43   private[HttpsRedirect] val logger = getLogger
44 
45   def apply[F[_], G[_]](http: Http[F, G])(implicit F: Applicative[F]): Http[F, G] =
46     Kleisli { req =>
47       (req.headers.get(ci"X-Forwarded-Proto"), req.headers.get[Host]) match {
48         case (Some(NonEmptyList(proto, _)), Some(host))
49             if Scheme.fromString(proto.value).contains(Scheme.http) =>
50           logger.debug(s"Redirecting ${req.method} ${req.uri} to https on $host")
51           val authority = Authority(host = RegName(host.value))
52           val location = req.uri.copy(scheme = Some(Scheme.https), authority = Some(authority))
53           val headers = Headers(Location(location), `Content-Type`(MediaType.text.xml))
54           val response = Response[G](status = MovedPermanently, headers = headers)
55           F.pure(response)
56 
57         case _ =>
58           http(req)
59       }
60     }
61 
62   def httpRoutes[F[_]: Monad](httpRoutes: HttpRoutes[F]): HttpRoutes[F] =
63     apply(httpRoutes)
64 
65   def httpApp[F[_]: Applicative](httpApp: HttpApp[F]): HttpApp[F] =
66     apply(httpApp)
67 }
