26 /** Removes a trailing slash from [[Request]] path
27   *
28   * If a route exists with a file style [[Uri]], eg "/foo",
29   * this middleware will cause [[Request]]s with uri = "/foo" and
30   * uri = "/foo/" to match the route.
31   */
32 object AutoSlash {
33   def apply[F[_], G[_], B](
34       http: Kleisli[F, Request[G], B]
35   )(implicit F: MonoidK[F]): Kleisli[F, Request[G], B] =
36     Kleisli { req =>
37       http(req) <+> {
38         val pathInfo = req.pathInfo
39 
40         if (pathInfo.isEmpty)
41           F.empty
42         else
43           http.apply(req.withPathInfo(pathInfo.dropEndsWithSlash))
44       }
45     }
46 
47   def httpRoutes[F[_]: Monad](httpRoutes: HttpRoutes[F]): HttpRoutes[F] =
48     apply(httpRoutes)
49 }
