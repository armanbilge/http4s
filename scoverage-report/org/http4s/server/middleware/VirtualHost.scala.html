27 /** Middleware for virtual host mapping
28   *
29   * The `VirtualHost` middleware allows multiple services to be mapped
30   * based on the [[org.http4s.headers.Host]] header of the [[org.http4s.Request]].
31   */
32 object VirtualHost {
33 
34   /** Specification of the virtual host service and predicate.
35     *
36     * The predicate receives the the Host header information with the port
37     * filled in, if possible, using the request Uri or knowledge of the
38     * security of the underlying transport protocol.
39     */
40   final case class HostService[F[_], G[_]](http: Http[F, G], p: Host => Boolean)
41 
42   /** Create a [[HostService]] that will match based on the exact host string
43     * (discounting case) and port, if the port is given. If the port is not
44     * given, it is ignored.
45     */
46   def exact[F[_], G[_]](
47       http: Http[F, G],
48       requestHost: String,
49       port: Option[Int] = None,
50   ): HostService[F, G] =
51     HostService(http, h => h.host.equalsIgnoreCase(requestHost) && (port.isEmpty || port == h.port))
52 
53   /** Create a [[HostService]] that will match based on the host string allowing
54     * for wildcard matching of the lowercase host string and port, if the port is
55     * given. If the port is not given, it is ignored.
56     */
57   def wildcard[F[_], G[_]](
58       http: Http[F, G],
59       wildcardHost: String,
60       port: Option[Int] = None,
61   ): HostService[F, G] =
62     regex(http, wildcardHost.replace("*", "\\w+").replace(".", "\\.").replace("-", "\\-"), port)
63 
64   /** Create a [[HostService]] that uses a regular expression to match the host
65     * string (which will be provided in lower case form) and port, if the port
66     * is given. If the port is not given, it is ignored.
67     */
68   def regex[F[_], G[_]](
69       http: Http[F, G],
70       hostRegex: String,
71       port: Option[Int] = None,
72   ): HostService[F, G] = {
73     val r = hostRegex.r
74     HostService(
75       http,
76       h => r.findFirstIn(h.host.toLowerCase).nonEmpty && (port.isEmpty || port == h.port),
77     )
78   }
79 
80   def apply[F[_], G[_]](first: HostService[F, G], rest: HostService[F, G]*)(implicit
81       F: Applicative[F],
82       W: EntityEncoder[G, String],
83   ): Http[F, G] =
84     Kleisli { req =>
85       req.headers
86         .get[Host]
87         .fold(F.pure(Response[G](BadRequest).withEntity("Host header required."))) { h =>
88           // Fill in the host port if possible
89           val host: Host = h.port match {
90             case Some(_) => h
91             case None =>
92               h.copy(port = req.uri.port.orElse(req.isSecure.map(if (_) 443 else 80)))
93           }
94           (first +: rest).toVector
95             .collectFirst { case HostService(s, p) if p(host) => s(req) }
96             .getOrElse(F.pure(Response[G](NotFound).withEntity(s"Host '$host' not found.")))
97         }
98     }
99 }
