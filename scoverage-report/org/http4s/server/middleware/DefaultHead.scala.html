30 /** Handles HEAD requests as a GET without a body.
31   *
32   * If the service returns the fallthrough response, the request is resubmitted
33   * as a GET.  The resulting response's body is killed, but all headers are
34   * preserved.  This is a naive, but correct, implementation of HEAD.  Routes
35   * requiring more optimization should implement their own HEAD handler.
36   */
37 object DefaultHead {
38   def apply[F[_]: Functor, G[_]: Concurrent](http: Http[F, G])(implicit F: MonoidK[F]): Http[F, G] =
39     Kleisli { req =>
40       req.method match {
41         case HEAD => http(req) <+> http(req.withMethod(GET)).map(drainBody[G])
42         case _ => http(req)
43       }
44     }
45 
46   def httpRoutes[F[_]: Concurrent](httpRoutes: HttpRoutes[F]): HttpRoutes[F] =
47     apply(httpRoutes)
48 
49   private[this] def drainBody[G[_]: Concurrent](response: Response[G]): Response[G] =
50     response.pipeBodyThrough(_.interruptWhen[G](Stream(true)).drain)
51 }
