25 /** Date Middleware, adds the Date Header to All Responses generated
26   * by the service.
27   */
28 object Date {
29   def apply[G[_]: Temporal, F[_], A](k: Kleisli[G, A, Response[F]]): Kleisli[G, A, Response[F]] =
30     Kleisli { a =>
31       for {
32         resp <- k(a)
33         header <-
34           resp.headers
35             .get[HDate]
36             .fold(
37               HttpDate.current[G].map(HDate(_))
38             )(_.pure[G])
39       } yield resp.putHeaders(header)
40     }
41 
42   def httpRoutes[F[_]: Temporal](routes: HttpRoutes[F]): HttpRoutes[F] =
43     apply(routes)
44 
45   def httpApp[F[_]: Temporal](app: HttpApp[F]): HttpApp[F] =
46     apply(app)
47 }
