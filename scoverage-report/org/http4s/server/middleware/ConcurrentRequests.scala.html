25 /** Middlewares for tracking the quantity of concurrent requests.
26   *
27   * These are generalized middlewares and can be used to implement metrics,
28   * logging, max concurrent requests, etc.
29   *
30   * @note The concurrent request count is decremented on the completion of the
31   *       Response body, or in the event of any error, and is guaranteed to
32   *       only occur once.
33   */
34 object ConcurrentRequests {
35 
36   /** Run a side effect each time the concurrent request count increments and
37     * decrements.
38     *
39     * @note Each side effect is given the current number of concurrent
40     *       requests as an argument.
41     *
42     * @note `onIncrement` should never be < 1 and `onDecrement` should never
43     *       be value < 0.
44     *
45     * @note This is the same as [[route]], but allows for the inner and outer
46     *       effect types to differ.
47     */
48   def route2[F[_]: Sync, G[_]: Async]( // TODO (ce3-ra): Sync + MonadCancel
49       onIncrement: Long => G[Unit],
50       onDecrement: Long => G[Unit],
51   ): F[ContextMiddleware[G, Long]] =
52     Ref
53       .in[F, G, Long](0L)
54       .map(ref =>
55         BracketRequestResponse.bracketRequestResponseRoutes[G, Long](
56           ref.updateAndGet(_ + 1L).flatTap(onIncrement)
57         )(Function.const(ref.updateAndGet(_ - 1L).flatMap(onDecrement)))
58       )
59 
60   /** Run a side effect each time the concurrent request count increments and
61     * decrements.
62     *
63     * @note Each side effect is given the current number of concurrent
64     *       requests as an argument.
65     *
66     * @note `onIncrement` should never be < 1 and `onDecrement` should never
67     *       be value < 0.
68     */
69   def route[F[_]: Async](
70       onIncrement: Long => F[Unit],
71       onDecrement: Long => F[Unit],
72   ): F[ContextMiddleware[F, Long]] =
73     route2[F, F](onIncrement, onDecrement)
74 
75   /** As [[route]], but runs the same effect on increment and decrement of the concurrent request count. */
76   def onChangeRoute[F[_]: Async](onChange: Long => F[Unit]): F[ContextMiddleware[F, Long]] =
77     route[F](onChange, onChange)
78 
79   /** Run a side effect each time the concurrent request count increments and
80     * decrements.
81     *
82     * @note Each side effect is given the current number of concurrent
83     *       requests as an argument.
84     *
85     * @note `onIncrement` should never be < 1 and `onDecrement` should never
86     *       be value < 0.
87     *
88     * @note This is the same as [[app]], but allows for the inner and outer
89     *       effect types to differ.
90     */
91   def app2[F[_]: Sync, G[_]: Async](
92       onIncrement: Long => G[Unit],
93       onDecrement: Long => G[Unit],
94   ): F[Kleisli[G, ContextRequest[G, Long], Response[G]] => Kleisli[G, Request[G], Response[G]]] =
95     Ref
96       .in[F, G, Long](0L)
97       .map(ref =>
98         BracketRequestResponse.bracketRequestResponseApp[G, Long](
99           ref.updateAndGet(_ + 1L).flatTap(onIncrement)
100         )(Function.const(ref.updateAndGet(_ - 1L).flatMap(onDecrement)))
101       )
102 
103   /** Run a side effect each time the concurrent request count increments and
104     * decrements.
105     *
106     * @note Each side effect is given the current number of concurrent
107     *       requests as an argument.
108     *
109     * @note `onIncrement` should never be < 1 and `onDecrement` should never
110     *       be value < 0.
111     */
112   def app[F[_]: Async](
113       onIncrement: Long => F[Unit],
114       onDecrement: Long => F[Unit],
115   ): F[Kleisli[F, ContextRequest[F, Long], Response[F]] => Kleisli[F, Request[F], Response[F]]] =
116     app2[F, F](onIncrement, onDecrement)
117 
118   /** As [[app]], but runs the same effect on increment and decrement of the concurrent request count. */
119   def onChangeApp[F[_]: Async](
120       onChange: Long => F[Unit]
121   ): F[Kleisli[F, ContextRequest[F, Long], Response[F]] => Kleisli[F, Request[F], Response[F]]] =
122     app[F](onChange, onChange)
123 }
