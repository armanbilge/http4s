194   @deprecated(
195     "Depends on a deficient `CORSConfig`. See https://github.com/http4s/http4s/security/advisories/GHSA-52cf-226f-rhr6. If config.anyOrigin is true and config.allowCredentials is true, then the `Access-Control-Allow-Credentials` header will be suppressed starting with 0.22.3.",

198   @nowarn("cat=deprecation")
199   def apply[F[_], G[_]](http: Http[F, G], config: CORSConfig = CORSConfig.default)(implicit
200       F: Applicative[F]
201   ): Http[F, G] = {
202     if (config.anyOrigin && config.allowCredentials)
203       logger.warn(
204         "Insecure CORS config detected: `anyOrigin=true` and `allowCredentials=true` are mutually exclusive. `Access-Control-Allow-Credentials` header will not be sent. Change either flag to false to remove this warning."
205       )
206     Kleisli { req =>
207       // In the case of an options request we want to return a simple response with the correct Headers set.
208       def createOptionsResponse(
209           origin: Origin,
210           acrm: `Access-Control-Request-Method`,
211       ): Response[G] =
212         corsHeaders(origin, acrm.method, isPreflight = true)(Response())
213 
214       def methodBasedHeader(isPreflight: Boolean) =
215         if (isPreflight)
216           config.allowedHeaders.map(headerFromStrings("Access-Control-Allow-Headers", _))
217         else
218           config.exposedHeaders.map(headerFromStrings("Access-Control-Expose-Headers", _))
219 
220       def varyHeader(response: Response[G]): Response[G] =
221         response.headers.get(ci"Vary") match {
222           case None => response.putHeaders(defaultVaryHeader)
223           case _ => response
224         }
225 
226       def allowCredentialsHeader(resp: Response[G]): Response[G] =
227         if (!config.anyOrigin && config.allowCredentials)
228           resp.putHeaders("Access-Control-Allow-Credentials" -> "true")
229         else
230           resp
231 
232       def corsHeaders(origin: Origin, method: Method, isPreflight: Boolean)(
233           resp: Response[G]
234       ): Response[G] = {
235         val withMethodBasedHeader = methodBasedHeader(isPreflight)
236           .fold(resp)(h => resp.putHeaders(h))
237 
238         varyHeader(allowCredentialsHeader(withMethodBasedHeader))
239           .putHeaders(
240             // TODO model me
241             "Access-Control-Allow-Methods" -> config.allowedMethods.fold(method.renderString)(
242               _.mkString("", ", ", "")
243             ),
244             // TODO model me
245             "Access-Control-Allow-Origin" -> origin.value,
246             // TODO model me
247             "Access-Control-Max-Age" -> config.maxAge.toSeconds.toString,
248           )
249       }
250 
251       def allowCORS(origin: Origin, method: Method): Boolean = {
252         def allowOrigin = config.anyOrigin || config.allowedOrigins(origin.value)
253         def allowMethod = config.anyMethod || config.allowedMethods.exists(_.exists(_ === method))
254         allowOrigin && allowMethod
255       }
256 
257       def headerFromStrings(headerName: String, values: Set[String]): Header.Raw =
258         Header.Raw(CIString(headerName), values.mkString("", ", ", ""))
259 
260       (
261         req.method,
262         req.headers.get[Origin],
263         req.headers.get[`Access-Control-Request-Method`],
264       ) match {
265         case (OPTIONS, Some(origin), Some(acrm)) if allowCORS(origin, acrm.method) =>
266           logger.debug(s"Serving OPTIONS with CORS headers for $acrm ${req.uri}")
267           createOptionsResponse(origin, acrm).pure[F]
268         case (_, Some(origin), _) =>
269           if (allowCORS(origin, req.method))
270             http(req).map { resp =>
271               logger.debug(s"Adding CORS headers to ${req.method} ${req.uri}")
272               corsHeaders(origin, req.method, isPreflight = false)(resp)
273             }
274           else {
275             logger.debug(s"CORS headers were denied for ${req.method} ${req.uri}")
276             Response(status = Status.Forbidden).pure[F]
277           }
278         case _ =>
279           // This request is out of scope for CORS
280           http(req)
281       }
282     }
283   }

300 /** A middleware that applies the CORS protocol to any `Http` value.
301   * Obtain a reference to a `CORSPolicy` via the [[CORS]] object,
302   * which represents a default policy.
303   *
304   * Requests with an Origin header will receive the appropriate CORS
305   * headers.  More headers are available for "pre-flight" requests,
306   * those whose method is `OPTIONS` and has an
307   * `Access-Control-Request-Method` header.
308   *
309   * Requests without the required headers, or requests that fail a
310   * CORS origin, method, or headers check are passed through to the
311   * underlying Http function, but do not receive any CORS headers in
312   * the response.  The user agent will then block sharing the resource
313   * across origins according to the CORS protocol.
314   */
315 sealed class CORSPolicy(
316     allowOrigin: CORSPolicy.AllowOrigin,
317     allowCredentials: CORSPolicy.AllowCredentials,
318     exposeHeaders: CORSPolicy.ExposeHeaders,
319     allowMethods: CORSPolicy.AllowMethods,
320     allowHeaders: CORSPolicy.AllowHeaders,
321     maxAge: CORSPolicy.MaxAge,
322 ) {
323   import CORSPolicy._
324 
325   def apply[F[_]: Applicative, G[_]](http: Http[F, G]): Http[F, G] =
326     applicatively(http)
327 
328   // Hack because httpRoutes and httpApp convenience constructors want this
329   private def applicatively[F[_]: Applicative, G[_]](http: Http[F, G]): Http[F, G] =
330     impl(http, Http.pure(Response(Status.Ok)))
331 
332   @deprecated("Does not return 200 on preflight requests. Use the Applicative version", "0.21.28")
333   protected[CORSPolicy] def apply[F[_]: Functor, G[_]](http: Http[F, G]): Http[F, G] = {
334     logger.warn(
335       "This CORSPolicy does not return 200 on preflight requests. It's kept for binary compatibility, but it's buggy. If you see this, upgrade to v0.23.3 or greater."
336     )
337     impl(http, http)
338   }
339 
340   def impl[F[_]: Functor, G[_]](http: Http[F, G], preflightResponder: Http[F, G]): Http[F, G] = {
341 
342     val allowCredentialsHeader =
343       allowCredentials match {
344         case AllowCredentials.Allow =>
345           CommonHeaders.someAllowCredentials
346         case AllowCredentials.Deny =>
347           None
348       }
349 
350     val exposeHeadersHeader =
351       exposeHeaders match {
352         case ExposeHeaders.All =>
353           CommonHeaders.someExposeHeadersWildcard
354         case ExposeHeaders.In(names) =>
355           Header.Raw(Header[`Access-Control-Expose-Headers`].name, names.mkString(", ")).some
356         case ExposeHeaders.None =>
357           None
358       }
359 
360     val someAllowMethodsSpecificHeader =
361       allowMethods match {
362         case AllowMethods.All => None
363         case AllowMethods.In(methods) =>
364           Header
365             .Raw(ci"Access-Control-Allow-Methods", methods.map(_.renderString).mkString(", "))
366             .some
367       }
368 
369     val maxAgeHeader =
370       maxAge match {
371         case MaxAge.Some(deltaSeconds) =>
372           Header.Raw(ci"Access-Control-Max-Age", deltaSeconds.toString).some
373         case MaxAge.Default =>
374           None
375         case MaxAge.DisableCaching =>
376           Header.Raw(ci"Access-Control-Max-Age", "-1").some
377       }
378 
379     val varyHeaderNonOptions =
380       allowOrigin match {
381         case AllowOrigin.Match(_) =>
382           Header.Raw(ci"Vary", Header[Origin].name.toString).some
383         case _ =>
384           None
385       }
386 
387     val varyHeaderOptions = {
388       def origin = allowOrigin match {
389         case AllowOrigin.All => Nil
390         case AllowOrigin.Match(_) => List(Header[Origin].name)
391       }
392       def methods = allowMethods match {
393         case AllowMethods.All => Nil
394         case AllowMethods.In(_) => List(Header[`Access-Control-Request-Method`].name)
395       }
396       def headers = allowHeaders match {
397         case AllowHeaders.All | AllowHeaders.Static(_) => Nil
398         case AllowHeaders.In(_) | AllowHeaders.Reflect =>
399           List(ci"Access-Control-Request-Headers")
400       }
401       (origin ++ methods ++ headers) match {
402         case Nil =>
403           None
404         case nonEmpty =>
405           Header.Raw(ci"Vary", nonEmpty.map(_.toString).mkString(", ")).some
406       }
407     }
408 
409     def dispatch(req: Request[G]) =
410       req.headers.get[Origin] match {
411         case Some(origin) =>
412           req.method match {
413             case Method.OPTIONS =>
414               req.headers.get[`Access-Control-Request-Method`] match {
415                 case Some(acrm) =>
416                   val headers = req.headers.get(ci"Access-Control-Request-Headers") match {
417                     case Some(acrHeaders) =>
418                       acrHeaders.map(_.value.split("\\s*,\\s*").map(CIString(_)).toSet).fold
419                     case None =>
420                       Set.empty[CIString]
421                   }
422                   preflight(req, origin, acrm.method, headers)
423                 case None =>
424                   nonPreflight(req, origin)
425               }
426             case _ =>
427               nonPreflight(req, origin)
428           }
429         case None =>
430           nonCors(req)
431       }
432 
433     def nonPreflight(req: Request[G], origin: Origin) = {
434       val buff = List.newBuilder[Header.Raw]
435       allowOriginHeader(origin).map { allowOrigin =>
436         buff += allowOrigin
437         allowCredentialsHeader.foreach(buff.+=)
438         exposeHeadersHeader.foreach(buff.+=)
439         buff
440       }
441       http(req)
442         .map(_.putHeaders(buff.result().map(Header.ToRaw.rawToRaw): _*))
443         .map(varyHeader(req.method))
444     }
445 
446     def preflight(req: Request[G], origin: Origin, method: Method, headers: Set[CIString]) = {
447       val buff = List.newBuilder[Header.Raw]
448       (allowOriginHeader(origin), allowMethodsHeader(method), allowHeadersHeader(headers)).mapN {
449         case (allowOrigin, allowMethods, allowHeaders) =>
450           buff += allowOrigin
451           allowCredentialsHeader.foreach(buff.+=)
452           buff += allowMethods
453           buff += allowHeaders
454           maxAgeHeader.foreach(buff.+=)
455       }
456       preflightResponder(req)
457         .map(_.putHeaders(buff.result().map(Header.ToRaw.rawToRaw): _*))
458         .map(varyHeader(Method.OPTIONS))
459     }
460 
461     def nonCors(req: Request[G]) =
462       http(req).map(varyHeader(req.method))
463 
464     def allowOriginHeader(origin: Origin) =
465       allowOrigin match {
466         case AllowOrigin.All =>
467           CommonHeaders.someAllowOriginWildcard
468         case AllowOrigin.Match(p) =>
469           if (p(origin))
470             Header.Raw(ci"Access-Control-Allow-Origin", origin.value).some
471           else
472             None
473       }
474 
475     def allowMethodsHeader(method: Method) =
476       allowMethods match {
477         case AllowMethods.All =>
478           if (allowCredentials == AllowCredentials.Deny || method === wildcardMethod)
479             CommonHeaders.someAllowMethodsWildcard
480           else
481             None
482         case AllowMethods.In(methods) =>
483           if (methods.contains(method))
484             someAllowMethodsSpecificHeader
485           else
486             None
487       }
488 
489     def someAllowHeadersHeader(headers: Set[CIString]) =
490       Header
491         .Raw(Header[`Access-Control-Allow-Headers`].name, headers.map(_.toString).mkString(", "))
492         .some
493 
494     def allowHeadersHeader(requestHeaders: Set[CIString]) =
495       allowHeaders match {
496         case AllowHeaders.All =>
497           if (allowCredentials == AllowCredentials.Deny || requestHeaders === wildcardHeadersSet)
498             CommonHeaders.someAllowHeadersWildcard
499           else
500             None
501         case AllowHeaders.Static(allowedHeaders) =>
502           someAllowHeadersHeader(allowedHeaders)
503         case AllowHeaders.In(allowedHeaders) =>
504           if ((requestHeaders -- allowedHeaders).isEmpty)
505             someAllowHeadersHeader(allowedHeaders)
506           else
507             None
508         case AllowHeaders.Reflect =>
509           someAllowHeadersHeader(requestHeaders)
510       }
511 
512     // Working around the poor model in 0.21.  If we just add a Vary
513     // header, it clobbers the existing one, because it's not properly
514     // flagged as recurring.  This doesn't need to be special when the
515     // Vary model is fixed.
516     def varyHeader(method: Method)(resp: Response[G]) =
517       (method match {
518         case Method.OPTIONS => varyHeaderOptions
519         case _ => varyHeaderNonOptions
520       }) match {
521         case Some(vary) =>
522           resp.putHeaders(
523             resp.headers.get(ci"Vary") match {
524               case None =>
525                 vary
526               case Some(oldVary) =>
527                 Header.Raw(ci"Vary", oldVary.map(_.value).toList.mkString(", ") + ", " + vary.value)
528             }
529           )
530         case None =>
531           resp
532       }
533 
534     if (allowOrigin == AllowOrigin.All && allowCredentials == AllowCredentials.Allow) {
535       logger.warn(
536         "CORS disabled due to insecure config prohibited by spec. Call withCredentials(false) to avoid sharing credential-tainted responses with arbitrary origins, or call withAllowOrigin* method to be explicit who you trust with credential-tainted responses."
537       )
538       http
539     } else
540       Kleisli(dispatch)
541   }

572   /** Allow requests from any origin matching the predicate `p`.  On
573     * matching requests, the request origin is reflected as the
574     * `Access-Control-Allow-Origin` header.
575     *
576     * The Origin header contains some arcane settings, like multiple
577     * origins, or a `null` origin. `withAllowOriginHost` is generally
578     * more convenient.
579     */
580   def withAllowOriginHeader(p: Origin => Boolean): CORSPolicy =
581     copy(AllowOrigin.Match(p))
582 
583   /** Allow requests from any origin host matching the predicate `p`.
584     * The origin host is the first value in the request's `Origin`
585     * header, if not `null` header, unless it is `null`.  Examples:
586     *
587     * - `Origin: http://www.example.com` => `http://www.example.com`
588     * - `Origin: http://www.example.com, http://example.net` =>
589     *   `http://www.example.com`
590     * - `Origin: null` => always false
591     *
592     * A `Set[Origin.Host]` is often a good choice here, but a predicate is
593     * offered to support more advanced matching.
594     */
595   def withAllowOriginHost(p: Origin.Host => Boolean): CORSPolicy =
596     withAllowOriginHeader(_ match {
597       case Origin.HostList(NonEmptyList(h, _)) => p(h)
598       case Origin.Null => false
599     })

601   /** Allow requests from any origin host whose case-insensitive
602     * rendering matches predicate `p`.  A concession to the fact
603     * that constructing [[org.http4s.headers.Origin.Host]] values is verbose.
604     *
605     * @see [[withAllowOriginHost]]
606     */
607   def withAllowOriginHostCi(p: CIString => Boolean): CORSPolicy =

636   /** Exposes no response headers to the origin beyond the
