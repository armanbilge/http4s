25 object HeaderEcho {
26 
27   /** Simple server middleware that adds selected headers present on the request to the response.
28     *
29     * @param echoHeadersWhen the function that selects which headers to echo on the response
30     * @param http [[Http]] to transform
31     */
32   def apply[F[_]: Functor, G[_]](
33       echoHeadersWhen: CIString => Boolean
34   )(http: Http[F, G]): Http[F, G] =
35     Kleisli { (req: Request[G]) =>
36       val headersToEcho = Headers(req.headers.headers.filter(h => echoHeadersWhen(h.name)))
37 
38       http(req).map(_.putHeaders(headersToEcho.headers))
39     }
40 
41   def httpRoutes[F[_]: Functor](echoHeadersWhen: CIString => Boolean)(
42       httpRoutes: HttpRoutes[F]
43   ): HttpRoutes[F] =
44     apply(echoHeadersWhen)(httpRoutes)
45 
46   def httpApp[F[_]: Functor](echoHeadersWhen: CIString => Boolean)(
47       httpApp: HttpApp[F]
48   ): HttpApp[F] =
49     apply(echoHeadersWhen)(httpApp)
50 }
