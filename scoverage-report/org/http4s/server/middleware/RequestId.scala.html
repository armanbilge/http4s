37 /** Propagate a `X-Request-Id` header to the response, generate a UUID
38   * when the `X-Request-Id` header is unset.
39   * https://devcenter.heroku.com/articles/http-request-id
40   */
41 object RequestId {
42 
43   private[this] val requestIdHeader = ci"X-Request-ID"
44 
45   val requestIdAttrKey: Key[String] = Key.newKey[SyncIO, String].unsafeRunSync()
46 
47   def apply[G[_], F[_]](http: Http[G, F])(implicit G: Sync[G]): Http[G, F] =
48     apply(requestIdHeader)(http)
49 
50   @deprecated("Preserved for bincompat", "0.23.12")
51   def apply[G[_], F[_]](headerName: CIString, http: Http[G, F], G: Sync[G]): Http[G, F] =
52     apply(headerName)(http)(G, UUIDGen.fromSync(G))
53 
54   def apply[G[_], F[_]](
55       headerName: CIString
56   )(http: Http[G, F])(implicit G: Sync[G], gen: UUIDGen[G]): Http[G, F] =
57     Kleisli[G, Request[F], Response[F]] { req =>
58       for {
59         header <- req.headers.get(headerName).map(_.head) match {
60           case None => UUIDGen.randomString[G].map(Header.Raw(headerName, _))
61           case Some(header) => G.pure(header)
62         }
63         reqId = header.value
64         response <- http(req.withAttribute(requestIdAttrKey, reqId).putHeaders(header))
65       } yield response.withAttribute(requestIdAttrKey, reqId).putHeaders(header)
66     }
67 
68   def apply[G[_], F[_]](
69       fk: F ~> G,
70       headerName: CIString = requestIdHeader,
71       genReqId: F[UUID],
72   )(http: Http[G, F])(implicit G: FlatMap[G], F: Sync[F]): Http[G, F] =
73     Kleisli[G, Request[F], Response[F]] { req =>
74       for {
75         header <- fk(req.headers.get(headerName) match {
76           case None => genReqId.map(reqId => Header.Raw(headerName, reqId.show))
77           case Some(NonEmptyList(header, _)) => F.pure(header)
78         })
79         reqId = header.value
80         response <- http(req.withAttribute(requestIdAttrKey, reqId).putHeaders(header))
81       } yield response.withAttribute(requestIdAttrKey, reqId).putHeaders(header)
82     }
