46   def apply[F[_]: Applicative, G[_]](callbackParam: String)(http: Http[F, G]): Http[F, G] =
47     Kleisli { req =>
48       req.params.get(callbackParam) match {
49         case Some(ValidCallback(callback)) =>
50           http(req).map { response =>
51             if (hasJsonContent(response)) jsonp(response, callback) else response
52           }
53         case Some(invalidCallback) =>
54           logger.warn(s"Jsonp requested with invalid callback function name $invalidCallback")
55           Response[G](Status.BadRequest).withEntity(s"Not a valid callback name.").pure[F]
56         case None => http(req)
57       }
58     }

60   private def hasJsonContent[F[_]](resp: Response[F]): Boolean =
61     resp.contentType.map(_.mediaType).contains(MediaType.application.json)
62 
63   private def jsonp[F[_]](resp: Response[F], callback: String) = {
64     val begin = beginJsonp(callback)
65     val end = EndJsonp
66     val jsonpBody = chunk(begin) ++ resp.body ++ chunk(end)
67     val newLengthHeaderOption = resp.headers.get[`Content-Length`].flatMap { old =>
68       old.modify(_ + begin.size + end.size)
69     }
70     resp
71       .copy(body = jsonpBody)
72       .transformHeaders(_ ++ Headers(newLengthHeaderOption))
73       .withContentType(`Content-Type`(MediaType.application.javascript))
74   }
75 
76   private def beginJsonp(callback: String) =
77     Chunk.array((callback + "(").getBytes(StandardCharsets.UTF_8))
78 
79   private val EndJsonp =
80     Chunk.array(");".getBytes(StandardCharsets.UTF_8))
