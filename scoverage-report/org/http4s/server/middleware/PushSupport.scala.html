29 object PushSupport {
30   private[this] val logger = getLogger
31 
32   implicit def http4sPushOps[F[_]](response: Response[F]): PushOps[F] =
33     new PushOps[F](response)
34 
35   final class PushOps[F[_]](response: Response[F]) extends AnyRef {
36     def push(url: String, cascade: Boolean = true)(implicit req: Request[F]): Response[F] = {
37       val newUrl = {
38         val script = req.scriptName
39         if (script.nonEmpty) {
40           val sb = new StringBuilder()
41           sb.append(script.toAbsolute)
42           sb.append(url).result()
43         } else url
44       }
45 
46       logger.trace(s"Adding push resource: $newUrl")
47 
48       val newPushResouces = response.attributes
49         .lookup(pushLocationKey)
50         .map(_ :+ PushLocation(newUrl, cascade))
51         .getOrElse(Vector(PushLocation(newUrl, cascade)))
52 
53       response.copy(
54         body = response.body,
55         attributes = response.attributes.insert(PushSupport.pushLocationKey, newPushResouces),
56       )
57     }
58   }
59 
60   private def collectResponse[F[_]](
61       r: Vector[PushLocation],
62       req: Request[F],
63       verify: String => Boolean,
64       routes: HttpRoutes[F],
65   )(implicit F: Monad[F]): F[Vector[PushResponse[F]]] = {
66     val emptyCollect: F[Vector[PushResponse[F]]] = F.pure(Vector.empty[PushResponse[F]])
67 
68     def fetchAndAdd(facc: F[Vector[PushResponse[F]]], v: PushLocation): F[Vector[PushResponse[F]]] =
69       routes(req.withPathInfo(Uri.Path.unsafeFromString(v.location))).value.flatMap {
70         case None => emptyCollect
71         case Some(response) =>
72           if (v.cascade) {
73             val pr = PushResponse(v.location, response)
74             response.attributes.lookup(pushLocationKey) match {
75               case Some(pushed) => // Need to gather the sub resources
76                 val fsubs = collectResponse(pushed, req, verify, routes)
77                 F.map2(facc, fsubs)(_ ++ _ :+ pr)
78               case None => facc.map(_ :+ pr)
79             }
80           } else {
81             facc.map(_ :+ PushResponse(v.location, response))
82           }
83       }
84 
85     r.filter(x => verify(x.location)).foldLeft(emptyCollect)(fetchAndAdd)
86   }
87 
88   /** Transform the route such that requests will gather pushed resources
89     *
90     * @param routes HttpRoutes to transform
91     * @param verify method that determines if the location should be pushed
92     * @return      Transformed route
93     */
94   def apply[F[_]: Monad](
95       routes: HttpRoutes[F],
96       verify: String => Boolean = _ => true,
97   ): HttpRoutes[F] = {
98     def gather(req: Request[F])(resp: Response[F]): Response[F] =
99       resp.attributes
100         .lookup(pushLocationKey)
101         .map { fresource =>
102           val collected = collectResponse(fresource, req, verify, routes)
103           resp.copy(
104             attributes = resp.attributes.insert(pushResponsesKey[F], collected)
105           )
106         }
107         .getOrElse(resp)
108 
109     Kleisli(req => routes(req).map(gather(req)))
110   }
111 
112   private[PushSupport] final case class PushLocation(location: String, cascade: Boolean)
113   private[http4s] final case class PushResponse[F[_]](location: String, resp: Response[F])
114 
115   private[PushSupport] val pushLocationKey =
116     Key.newKey[SyncIO, Vector[PushLocation]].unsafeRunSync()
117   private[http4s] def pushResponsesKey[F[_]]: Key[F[Vector[PushResponse[F]]]] =
118     Keys.PushResponses.asInstanceOf[Key[F[Vector[PushResponse[F]]]]]
119 
120   private[this] object Keys {
121     val PushResponses: Key[Any] = Key.newKey[SyncIO, Any].unsafeRunSync()
122   }
