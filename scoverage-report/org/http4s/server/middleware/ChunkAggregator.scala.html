35 /** Generic middleware to aggregate chunked response into memory.
36   *
37   * This middleware wraps any function that may produce an Http response.
38   * Upon receiving a chunk-encoded response, the composed function will
39   * pull and assemble the full response into memory before emitting the
40   * whole response. It also removes the "Chunked encoding" headers, and
41   * may add a Content-Length header for the whole payload.
42   *
43   * Reference: "Chunked Transfer Encoding", Section 4.1 of RFC 7230
44   *  https://datatracker.ietf.org/doc/html/rfc7230#section-4.1
45   */
46 object ChunkAggregator {
47   def apply[F[_]: FlatMap, G[_]: Sync, A](
48       f: G ~> F
49   )(http: Kleisli[F, A, Response[G]]): Kleisli[F, A, Response[G]] =
50     http.flatMapF(response => f(aggregate(response)))
51 
52   private[this] def aggregate[G[_]: Sync](r: Response[G]): G[Response[G]] =
53     r.body.chunks.compile.toVector // scalafix:ok Http4sFs2Linters.noFs2SyncCompiler; bincompat until 1.0
54       .map { vec =>
55         val body = Chunk.concat(vec)
56         r
57           .withBodyStream(Stream.chunk(body))
58           .transformHeaders(removeChunkedTransferEncoding(body.size.toLong))
59       }
60 
61   def httpRoutes[F[_]: Sync](httpRoutes: HttpRoutes[F]): HttpRoutes[F] =
62     apply(OptionT.liftK[F])(httpRoutes)
63 
64   def httpApp[F[_]: Sync](httpApp: HttpApp[F]): HttpApp[F] =
65     apply(FunctionK.id[F])(httpApp)
66 
67   /* removes the `TransferCoding.chunked` value from the `Transfer-Encoding` header,
68    removes the `Content-Length` header, and leaves the other headers unchanged */
69   private[this] def removeChunkedTransferEncoding(len: Long)(headers: Headers): Headers = {
70     val hh: ListBuffer[Header.ToRaw] = ListBuffer.empty[Header.ToRaw]
71     headers.headers.foreach {
72       case h @ Header.Raw(ci"Transfer-Encoding", value) =>
73         `Transfer-Encoding`.parse(value) match {
74           case Right(te) =>
75             te.values.filterNot(_ === TransferCoding.chunked) match {
76               case v :: vs =>
77                 hh += `Transfer-Encoding`(NonEmptyList(v, vs))
78               case Nil =>
79               // do nothing
80             }
81           case Left(_) =>
82             hh += h
83         }
84       case Header.Raw(ci"Content-Length", _) => // do nothing
85       case header => hh += header
86     }
87     if (len > 0L)
88       hh += `Content-Length`.unsafeFromLong(len)
89     Headers(hh.toList)
90   }
91 }
