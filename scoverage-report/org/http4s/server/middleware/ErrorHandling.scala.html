27 object ErrorHandling {
28   def apply[F[_], G[_]](
29       k: Kleisli[F, Request[G], Response[G]]
30   )(implicit F: MonadThrow[F]): Kleisli[F, Request[G], Response[G]] =
31     Kleisli { req =>
32       val pf: PartialFunction[Throwable, F[Response[G]]] =
33         inDefaultServiceErrorHandler[F, G](F)(req)
34       k.run(req).handleErrorWith { e =>
35         pf.lift(e) match {
36           case Some(resp) => resp
37           case None => F.raiseError(e)
38         }
39       }
40     }
41 
42   def httpRoutes[F[_]: MonadThrow](httpRoutes: HttpRoutes[F]): HttpRoutes[F] =
43     apply(httpRoutes)
44 
45   def httpApp[F[_]: MonadThrow](httpApp: HttpApp[F]): HttpApp[F] =
46     apply(httpApp)
47 
48   object Custom {
49     def recoverWith[F[_]: MonadThrow, G[_], A](
50         http: Kleisli[F, A, Response[G]]
51     )(pf: PartialFunction[Throwable, F[Response[G]]]): Kleisli[F, A, Response[G]] =
52       Kleisli { (a: A) =>
53         http.run(a).recoverWith(pf)
54       }
55   }
56 
57   object Recover {
58 
59     def total[F[_]: MonadThrow, G[_], A](
60         http: Kleisli[F, Request[G], Response[G]]
61     ): Kleisli[F, Request[G], Response[G]] =
62       Kleisli { (a: Request[G]) =>
63         http.run(a).handleError(totalRecover(a.httpVersion))
64       }
65 
66     def messageFailure[F[_]: MonadThrow, G[_], A](
67         http: Kleisli[F, Request[G], Response[G]]
68     ): Kleisli[F, Request[G], Response[G]] =
69       Kleisli { (a: Request[G]) =>
70         http.run(a).recover(messageFailureRecover(a.httpVersion))
71       }
72 
73     def messageFailureRecover[G[_]](
74         httpVersion: HttpVersion
75     ): PartialFunction[Throwable, Response[G]] = { case m: MessageFailure =>
76       m.toHttpResponse[G](httpVersion)
77     }
78 
79     def totalRecover[G[_]](
80         httpVersion: HttpVersion
81     ): Throwable => Response[G] = {
82       case m: MessageFailure => m.toHttpResponse[G](httpVersion)
83       case _ =>
84         Response(
85           Status.InternalServerError,
86           httpVersion,
87           Headers(
88             Connection(ci"close"),
89             `Content-Length`.zero,
90           ),
91         )
92     }
93 
94   }
