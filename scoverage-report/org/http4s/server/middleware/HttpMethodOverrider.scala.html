42         overrideStrategy: OverrideStrategy[F, G] = overrideStrategy,
43         overridableMethods: Set[Method] = overridableMethods,
44     ): Self =

46 
47     def withOverrideStrategy(overrideStrategy: OverrideStrategy[F, G]): Self =

54   object HttpMethodOverriderConfig {
55     def apply[F[_], G[_]](
56         overrideStrategy: OverrideStrategy[F, G],
57         overridableMethods: Set[Method],
58     ): HttpMethodOverriderConfig[F, G] =
59       new HttpMethodOverriderConfig[F, G](overrideStrategy, overridableMethods)
60   }

62   sealed trait OverrideStrategy[F[_], G[_]]
63   final case class HeaderOverrideStrategy[F[_], G[_]](headerName: CIString)
64       extends OverrideStrategy[F, G]
65   final case class QueryOverrideStrategy[F[_], G[_]](paramName: String)
66       extends OverrideStrategy[F, G]
67   final case class FormOverrideStrategy[F[_], G[_]](
68       fieldName: String,
69       naturalTransformation: G ~> F,
70   ) extends OverrideStrategy[F, G]
71 
72   def defaultConfig[F[_], G[_]]: HttpMethodOverriderConfig[F, G] =
73     HttpMethodOverriderConfig[F, G](
74       HeaderOverrideStrategy(ci"X-HTTP-Method-Override"),
75       Set(Method.POST),
76     )

78   val overriddenMethodAttrKey: Key[Method] = Key.newKey[SyncIO, Method].unsafeRunSync()

80   /** Simple middleware for HTTP Method Override.
81     *
82     * This middleware lets you use  HTTP verbs such as PUT or DELETE in places where the client
83     * doesn't support it. Camouflage your request with another HTTP verb(usually POST) and sneak
84     * the desired one using a custom header or request parameter. The middleware will '''override'''
85     * the original verb with the new one for you, allowing the request the be dispatched properly.
86     *
87     * @param http [[Http]] to transform
88     * @param config http method overrider config
89     */
90   def apply[F[_], G[_]](http: Http[F, G], config: HttpMethodOverriderConfig[F, G])(implicit
91       F: Monad[F],
92       S: Concurrent[G],
93   ): Http[F, G] = {
94     val parseMethod = (m: String) => Method.fromString(m.toUpperCase)
95 
96     def processRequestWithMethod(
97         req: Request[G],
98         parseResult: ParseResult[Method],
99     ): F[Response[G]] =
100       parseResult match {
101         case Left(_) => F.pure(Response[G](Status.BadRequest))
102         case Right(om) => http(updateRequestWithMethod(req, om)).map(updateVaryHeader)
103       }
104 
105     def updateVaryHeader(resp: Response[G]): Response[G] = {
106       val varyHeaderName = ci"Vary"
107       config.overrideStrategy match {
108         case HeaderOverrideStrategy(headerName) =>
109           val updatedVaryHeader =
110             resp.headers
111               .get(varyHeaderName)
112               .map(_.head)
113               .map((h: Header.Raw) => Header.Raw(h.name, s"${h.value}, ${headerName.toString}"))
114               .getOrElse(Header.Raw(varyHeaderName, headerName.toString))
115 
116           resp.withHeaders(resp.headers.put(updatedVaryHeader))
117         case _ => resp
118       }
119     }
120 
121     def updateRequestWithMethod(req: Request[G], om: Method): Request[G] = {
122       val attrs = req.attributes.insert(overriddenMethodAttrKey, req.method)
123       req.withAttributes(attrs).withMethod(om)
124     }
125 
126     def getUnsafeOverrideMethod(req: Request[G]): F[Option[String]] =
127       config.overrideStrategy match {
128         case HeaderOverrideStrategy(headerName) =>
129           F.pure(req.headers.get(headerName).map(_.head.value))
130         case QueryOverrideStrategy(parameter) => F.pure(req.params.get(parameter))
131         case FormOverrideStrategy(field, f) =>
132           for {
133             formFields <- f(
134               UrlForm
135                 .entityDecoder[G]
136                 .decode(req, strict = true)
137                 .value
138                 .map(_.toOption.map(_.values))
139             )
140           } yield formFields.flatMap(_.get(field).flatMap(_.uncons.map(_._1)))
141       }
142 
143     def processRequest(req: Request[G]): F[Response[G]] =
144       getUnsafeOverrideMethod(req).flatMap {
145         case Some(m: String) => parseMethod.andThen(processRequestWithMethod(req, _)).apply(m)
146         case None => http(req)
147       }
148 
149     Kleisli { (req: Request[G]) =>
150       val isOverridden = config.overridableMethods.contains(req.method)
151       if (isOverridden) processRequest(req) else http(req)
152     }
153   }
