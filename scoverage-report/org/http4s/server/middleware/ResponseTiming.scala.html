28 object ResponseTiming {
29 
30   /** Simple middleware for adding a custom header with timing information to a response.
31     *
32     * This middleware captures the time starting from when the request headers are parsed and supplied
33     * to the wrapped service and ending when the response is started. Metrics middleware, like this one,
34     * work best as the outer layer to ensure work done by other middleware is also included.
35     *
36     * @param http [[HttpApp]] to transform
37     * @param timeUnit the units of measure for this timing
38     * @param headerName the name to use for the header containing the timing info
39     */
40   def apply[F[_]](
41       http: HttpApp[F],
42       timeUnit: TimeUnit = MILLISECONDS,
43       headerName: CIString = ci"X-Response-Time",
44   )(implicit F: Sync[F], clock: Clock[F]): HttpApp[F] =
45     Kleisli { req =>
46       val getTime = clock.monotonic.map(_.toUnit(timeUnit).toLong)
47       for {
48         before <- getTime
49         resp <- http(req)
50         after <- getTime
51         header = Header.Raw(headerName, s"${after - before}")
52       } yield resp.putHeaders(header)
53     }
54 }
