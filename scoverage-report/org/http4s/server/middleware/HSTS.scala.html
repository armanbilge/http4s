27 /** [[Middleware]] to add HTTP Strict Transport Security (HSTS) support adding
28   * the Strict Transport Security headers
29   */
30 object HSTS {
31   // Default HSTS policy of waiting for 1 year and include sub domains
32   private val defaultHSTSPolicy = `Strict-Transport-Security`.unsafeFromDuration(
33     365.days,
34     includeSubDomains = true,
35     preload = false,
36   )
37 
38   def apply[F[_]: Functor, A, G[_]](
39       routes: Kleisli[F, A, Response[G]]
40   ): Kleisli[F, A, Response[G]] =
41     apply(routes, defaultHSTSPolicy)
42 
43   def apply[F[_]: Functor, A, G[_]](
44       http: Kleisli[F, A, Response[G]],
45       header: `Strict-Transport-Security`,
46   ): Kleisli[F, A, Response[G]] =
47     Kleisli { req =>
48       http.map(_.putHeaders(header)).apply(req)
49     }
50 
51   def unsafeFromDuration[F[_]: Functor, A, G[_]](
52       http: Kleisli[F, A, Response[G]],
53       maxAge: FiniteDuration = 365.days,
54       includeSubDomains: Boolean = true,
55       preload: Boolean = false,
56   ): Kleisli[F, A, Response[G]] = {
57     val header = `Strict-Transport-Security`.unsafeFromDuration(maxAge, includeSubDomains, preload)
58     apply(http, header)
59   }
60 
61   object httpRoutes {
62     def apply[F[_]: Functor](httpRoutes: HttpRoutes[F]): HttpRoutes[F] =
63       HSTS.apply(httpRoutes)
64 
65     def apply[F[_]: Functor](
66         httpRoutes: HttpRoutes[F],
67         header: `Strict-Transport-Security`,
68     ): HttpRoutes[F] =
69       HSTS.apply(httpRoutes, header)
70 
71     def unsafeFromDuration[F[_]: Functor](
72         httpRoutes: HttpRoutes[F],
73         maxAge: FiniteDuration = 365.days,
74         includeSubDomains: Boolean = true,
75         preload: Boolean = false,
76     ): HttpRoutes[F] =
77       HSTS.unsafeFromDuration(httpRoutes, maxAge, includeSubDomains, preload)
78 
79   }
80 
81   object httpApp {
82     def apply[F[_]: Functor](httpApp: HttpApp[F]): HttpApp[F] =
83       HSTS.apply(httpApp)
84 
85     def apply[F[_]: Functor](httpApp: HttpApp[F], header: `Strict-Transport-Security`): HttpApp[F] =
86       HSTS.apply(httpApp, header)
87 
88     def unsafeFromDuration[F[_]: Functor](
89         httpApp: HttpApp[F],
90         maxAge: FiniteDuration = 365.days,
91         includeSubDomains: Boolean = true,
92         preload: Boolean = false,
93     ): HttpApp[F] =
94       HSTS.unsafeFromDuration(httpApp, maxAge, includeSubDomains, preload)
95   }
