42   def apply[G[_], F[_]](
43       logHeaders: Boolean,
44       logBody: Boolean,
45       fk: F ~> G,
46       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
47       logAction: Option[String => F[Unit]] = None,
48   )(http: Http[G, F])(implicit
49       F: Async[F],
50       G: MonadCancelThrow[G],

52     impl[G, F](logHeaders, Left(logBody), fk, redactHeadersWhen, logAction)(http)

54   private[server] def impl[G[_], F[_]](
55       logHeaders: Boolean,
56       logBodyText: Either[Boolean, Stream[F, Byte] => Option[F[String]]],
57       fk: F ~> G,
58       redactHeadersWhen: CIString => Boolean,
59       logAction: Option[String => F[Unit]],
60   )(http: Http[G, F])(implicit
61       F: Async[F],
62       G: MonadCancelThrow[G],
63   ): Http[G, F] = {
64     val log = logAction.fold { (s: String) =>
65       Sync[F].delay(logger.info(s))
66     }(identity)
67 
68     def logMessage(r: Request[F]): F[Unit] =
69       logBodyText match {
70         case Left(bool) =>
71           Logger.logMessage[F, Request[F]](r)(logHeaders, bool, redactHeadersWhen)(log(_))
72         case Right(f) =>
73           org.http4s.internal.Logger
74             .logMessageWithBodyText[F, Request[F]](r)(logHeaders, f, redactHeadersWhen)(log(_))
75       }
76 
77     val logBody: Boolean = logBodyText match {
78       case Left(bool) => bool
79       case Right(_) => true
80     }
81 
82     Kleisli { req =>
83       if (!logBody) {
84         def logAct = logMessage(req)
85         // This construction will log on Any Error/Cancellation
86         // The Completed Case is Unit, as we rely on the semantics of G
87         // As None Is Successful, but we oly want to log on Some
88         http(req)
89           .guaranteeCase {
90             case Outcome.Succeeded(_) => G.unit
91             case _ => fk(logAct)
92           } <* fk(logAct)
93       } else
94         fk(F.ref(Vector.empty[Chunk[Byte]]))
95           .flatMap { vec =>
96             val collectChunks: Pipe[F, Byte, Nothing] =
97               _.chunks.flatMap(c => Stream.exec(vec.update(_ :+ c)))
98 
99             val changedRequest = req.pipeBodyThrough(_.observe(collectChunks))
100 
101             val newBody = Stream.eval(vec.get).flatMap(v => Stream.emits(v)).unchunks
102             val logRequest: F[Unit] = logMessage(req.withBodyStream(newBody))
103 
104             http(changedRequest)
105               .guaranteeCase {
106                 case Outcome.Succeeded(_) => G.unit
107                 case _ => fk(logRequest)
108               }
109               .map(_.pipeBodyThrough(_.onFinalizeWeak(logRequest)))
110           }
111     }
112   }
