27 /** Middleware for caching the request body for multiple compilations
28   *
29   * As the body of the request is the [[fs2.Stream]] of bytes,
30   * compiling it several times (e.g. with middlewares) is unsafe.
31   * This middleware forbids such behaviour, compiling the body only once.
32   * It does so only for the "inner" middlewares:
33   *
34   * {{{
35   * val route = AMiddleware(BodyCache(SomeOtherMiddleware(myRoute)))
36   * }}}
37   *
38   * In this example only `myRoute` & `SomeOtherMiddleware` will receive cached request body,
39   * while the `AMiddleware` will get the raw one.
40   *
41   * As the entire request body will be allocated in memory,
42   * there is a possibility of OOM error with a large body.
43   * Because of that, using the `EntityLimiter` middleware is strongly advised.
44   *
45   * @note This middleware has nothing to do with the HTTP caching mechanism
46   *       and it does not cache bodies between multiple requests.
47   */
48 object BodyCache {
49 
50   def apply[G[_]: Concurrent, F[_]: Concurrent, R](
51       old: Kleisli[G, R, Response[F]]
52   )(reqGet: R => Request[F], reqSet: R => Request[F] => R)(
53       lift: F ~> G
54   ): Kleisli[G, R, Response[F]] = Kleisli {
55     case req if hasNoBody(reqGet(req)) => old(req)
56     case req => lift(compileBody(reqGet(req))).flatMap(reqSet(req).andThen(old.run))
57   }
58 
59   def httpRoutes[F[_]: Concurrent](routes: HttpRoutes[F]): HttpRoutes[F] =
60     apply(routes)(identity, _ => identity)(OptionT.liftK)
61 
62   def contextRoutes[T, F[_]: Concurrent](routes: ContextRoutes[T, F]): ContextRoutes[T, F] =
63     apply(routes)(_.req, in => cached => in.copy(req = cached))(OptionT.liftK)
64 
65   def httpApp[F[_]: Concurrent](app: HttpApp[F]): HttpApp[F] =
66     apply(app)(identity, _ => identity)(FunctionK.id)
67 
68   private def compileBody[F[_]: Concurrent](req: Request[F]): F[Request[F]] =
69     req.toStrict(None)
70 
71   def hasNoBody[F[_]](req: Request[F]): Boolean =
72     req.contentLength.contains(0L)
73 }
