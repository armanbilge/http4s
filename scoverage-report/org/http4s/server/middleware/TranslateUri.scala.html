24 /** Removes the given prefix from the beginning of the path of the [[Request]].
25   */
26 object TranslateUri {
27   def apply[F[_], G[_], B](
28       prefix: String
29   )(http: Kleisli[F, Request[G], B])(implicit F: MonoidK[F]): Kleisli[F, Request[G], B] =
30     if (prefix.isEmpty || prefix == "/") http
31     else {
32       val prefixAsPath = Uri.Path.unsafeFromString(prefix)
33 
34       Kleisli { (req: Request[G]) =>
35         val newCaret = req.pathInfo.findSplit(prefixAsPath)
36         val shouldTranslate = req.pathInfo.startsWith(prefixAsPath)
37         if (shouldTranslate) http(setCaret(req, newCaret))
38         else F.empty
39       }
40     }
41 
42   private def setCaret[F[_]](req: Request[F], newCaret: Option[Int]): Request[F] = {
43     val oldCaret = req.attributes.lookup(Request.Keys.PathInfoCaret)
44     val combined = oldCaret |+| newCaret
45     combined match {
46       case Some(value) => req.withAttribute(Request.Keys.PathInfoCaret, value)
47       case None => req
48     }
49   }
50 }
