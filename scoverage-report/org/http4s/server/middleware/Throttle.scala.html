34 /** Transform a service to reject any calls the go over a given rate.
35   */
36 object Throttle {
37   sealed abstract class TokenAvailability extends Product with Serializable
38   case object TokenAvailable extends TokenAvailability
39   final case class TokenUnavailable(retryAfter: Option[FiniteDuration]) extends TokenAvailability
40 
41   /** A token bucket for use with the [[Throttle]] middleware.  Consumers can take tokens which will be refilled over time.
42     * Implementations are required to provide their own refill mechanism.
43     *
44     * Possible implementations include a remote TokenBucket service to coordinate between different application instances.
45     */
46   trait TokenBucket[F[_]] {
47     def takeToken: F[TokenAvailability]
48     def mapK[G[_]](fk: F ~> G): TokenBucket[G] = new TokenBucket.Translated[F, G](this, fk)
49   }
50 
51   object TokenBucket {
52     private class Translated[F[_], G[_]](t: TokenBucket[F], fk: F ~> G) extends TokenBucket[G] {
53       def takeToken: G[TokenAvailability] = fk(t.takeToken)
54     }
55 
56     /** Creates an in-memory [[TokenBucket]].
57       *
58       * @param capacity the number of tokens the bucket can hold and starts with.
59       * @param refillEvery the frequency with which to add another token if there is capacity spare.
60       * @return A task to create the [[TokenBucket]].
61       */
62     def local[F[_]](capacity: Int, refillEvery: FiniteDuration)(implicit
63         F: Temporal[F]
64     ): F[TokenBucket[F]] = {
65       def getTime = F.monotonic.map(_.toNanos)
66       val bucket = getTime.flatMap(time => F.ref((capacity.toDouble, time)))
67 
68       bucket.map { counter =>
69         new TokenBucket[F] {
70           override def takeToken: F[TokenAvailability] =
71             counter.access.flatMap { case ((previousTokens, previousTime), setter) =>
72               getTime.flatMap { currentTime =>
73                 val timeDifference = currentTime - previousTime
74                 val tokensToAdd = timeDifference.toDouble / refillEvery.toNanos.toDouble
75                 val newTokenTotal = Math.min(previousTokens + tokensToAdd, capacity.toDouble)
76                 def unavailable = {
77                   val timeToNextToken = refillEvery.toNanos - timeDifference
78                   TokenUnavailable(timeToNextToken.nanos.some)
79                 }
80 
81                 // If setter fails (yields the value false), then retry with recursive call
82                 if (newTokenTotal >= 1)
83                   setter((newTokenTotal - 1, currentTime)).ifM(F.pure(TokenAvailable), takeToken)
84                 else
85                   setter((newTokenTotal, currentTime)).ifM(F.pure(unavailable), takeToken)
86               }
87             }
88         }
89       }
90     }
91   }
92 
93   private def createBucket[F[_]](amount: Int, per: FiniteDuration)(implicit F: Temporal[F]) = {
94     val refillFrequency = per / amount.toLong
95     TokenBucket.local(amount, refillFrequency)
96   }
97 
98   /** Limits the supplied service to a given rate of calls using an in-memory [[TokenBucket]]
99     *
100     * @param amount the number of calls to the service to permit within the given time period.
101     * @param per the time period over which a given number of calls is permitted.
102     * @param http the service to transform.
103     * @return a task containing the transformed service.
104     */
105   def apply[F[_], G[_]](amount: Int, per: FiniteDuration)(
106       http: Http[F, G]
107   )(implicit F: Temporal[F]): F[Http[F, G]] =
108     createBucket(amount, per).map(bucket => apply(bucket, defaultResponse[G] _)(http))
109 
110   /** As [[[apply[F[_],G[_]](amount:Int,per:scala\.concurrent\.duration\.FiniteDuration* apply(amount,per)]]], but for HttpRoutes[F]
111     */
112   def httpRoutes[F[_]](amount: Int, per: FiniteDuration)(
113       httpRoutes: HttpRoutes[F]
114   )(implicit F: Temporal[F]): F[HttpRoutes[F]] =
115     createBucket(amount, per).map(bucket =>
116       Throttle.httpRoutes(bucket, defaultResponse[F] _)(httpRoutes)
117     )
118 
119   /** As [[[apply[F[_],G[_]](amount:Int,per:scala\.concurrent\.duration\.FiniteDuration* apply(amount,per)]]], but for HttpApp[F]
120     */
121   def httpAapp[F[_]](amount: Int, per: FiniteDuration)(httpApp: HttpApp[F])(implicit
122       F: Temporal[F]
123   ): F[HttpApp[F]] =
124     apply(amount, per)(httpApp)

126   def defaultResponse[F[_]](@nowarn retryAfter: Option[FiniteDuration]): Response[F] =
127     Response[F](Status.TooManyRequests)
128 
129   /** Limits the supplied service using a provided [[TokenBucket]]
130     *
131     * @param bucket a [[TokenBucket]] to use to track the rate of incoming requests.
132     * @param throttleResponse a function that defines the response when throttled, may be supplied a suggested retry time depending on bucket implementation.
133     * @param http the service to transform.
134     * @return a task containing the transformed service.
135     */
136   def apply[F[_], G[_]](
137       bucket: TokenBucket[F],
138       throttleResponse: Option[FiniteDuration] => Response[G] = defaultResponse[G] _,
139   )(http: Http[F, G])(implicit F: Monad[F]): Http[F, G] =
140     Kleisli { req =>
141       bucket.takeToken.flatMap {
142         case TokenAvailable => http(req)
143         case TokenUnavailable(retryAfter) => throttleResponse(retryAfter).pure[F]
144       }
145     }
146 
147   /** As [[[apply[F[_],G[_]](bucket:org\.http4s\.server\.middleware\.Throttle\.TokenBucket[F],throttleResponse:Option[scala\.concurrent\.duration\.FiniteDuration]=>org\.http4s\.Response[G]* apply(bucket,throttleResponse)]]], but for HttpRoutes[F]
148     */
149   def httpRoutes[F[_]: Monad](
150       bucket: TokenBucket[F],
151       throttleResponse: Option[FiniteDuration] => Response[F] = defaultResponse[F] _,
152   )(httpRoutes: HttpRoutes[F]): HttpRoutes[F] =
153     apply(bucket.mapK(OptionT.liftK), throttleResponse)(httpRoutes)
154 
155   /** As [[[apply[F[_],G[_]](bucket:org\.http4s\.server\.middleware\.Throttle\.TokenBucket[F],throttleResponse:Option[scala\.concurrent\.duration\.FiniteDuration]=>org\.http4s\.Response[G]* apply(bucket,throttleResponse)]]], but for HttpApp[F]
156     */
157   def httpApp[F[_]: Monad](
158       bucket: TokenBucket[F],
159       throttleResponse: Option[FiniteDuration] => Response[F] = defaultResponse[F] _,
160   )(httpApp: HttpApp[F]): HttpApp[F] =
161     apply(bucket, throttleResponse)(httpApp)
