28 /** Caching contains middlewares to support caching functionality.
29   *
30   * Helper functions to support [[Caching.cache]] can be found in
31   * [[Caching.Helpers]]
32   */
33 object Caching {
34 
35   /** Middleware that implies responses should NOT be cached.
36     * This is a best attempt, many implementors of caching have done so differently.
37     */
38   def `no-store`[G[_]: Temporal, F[_], A](
39       http: Kleisli[G, A, Response[F]]
40   ): Kleisli[G, A, Response[F]] =
41     Kleisli { (a: A) =>
42       for {
43         resp <- http(a)
44         out <- `no-store-response`[G](resp)
45       } yield out
46     }
47 
48   /** Transform a Response so that it will not be cached.
49     */
50   def `no-store-response`[G[_]]: PartiallyAppliedNoStoreCache[G] =
51     new PartiallyAppliedNoStoreCache[G] {
52       def apply[F[_]](resp: Response[F])(implicit G: Temporal[G]): G[Response[F]] =
53         HttpDate.current[G].map(now => resp.putHeaders(HDate(now)).putHeaders(noStoreStaticHeaders))
54     }
55 
56   // These never change, so don't recreate them each time.
57   private val noStoreStaticHeaders: List[Header.ToRaw] = List(
58     `Cache-Control`(
59       NonEmptyList.of[CacheDirective](
60         CacheDirective.`no-store`,
61         CacheDirective.`no-cache`(),
62         CacheDirective.`max-age`(0.seconds),
63       )
64     ),
65     "Pragma" -> "no-cache",
66     Expires(HttpDate.Epoch), // Expire at the epoch for no time confusion
67   )
68 
69   /** Helpers Contains the default arguments used to help construct middleware
70     * with caching. They serve to support the default arguments for
71     * [[publicCache]] and [[privateCache]].
72     */
73   object Helpers {
74     def defaultStatusToSetOn(s: Status): Boolean =
75       s match {
76         case Status.NotModified => true
77         case otherwise => otherwise.isSuccess
78       }
79 
80     def defaultMethodsToSetOn(m: Method): Boolean =
81       m == Method.GET || m == Method.HEAD
82   }
83 
84   /** Sets headers for response to be publicly cached for the specified duration.
85     *
86     * Note: If set to Duration.Inf, lifetime falls back to
87     * 10 years for support of Http1 caches.
88     */
89   def publicCache[G[_]: Temporal, F[_]](lifetime: Duration, http: Http[G, F]): Http[G, F] =
90     cache(
91       lifetime,
92       Either.left(CacheDirective.public),
93       Helpers.defaultMethodsToSetOn,
94       Helpers.defaultStatusToSetOn,
95       http,
96     )
97 
98   /** Publicly Cache a Response for the given lifetime.
99     *
100     * Note: If set to Duration.Inf, lifetime falls back to
101     * 10 years for support of Http1 caches.
102     */
103   def publicCacheResponse[G[_]](lifetime: Duration): PartiallyAppliedCache[G] =
104     cacheResponse(lifetime, Either.left(CacheDirective.public))
105 
106   /** Sets headers for response to be privately cached for the specified duration.
107     *
108     * Note: If set to Duration.Inf, lifetime falls back to
109     * 10 years for support of Http1 caches.
110     */
111   def privateCache[G[_]: Temporal, F[_]](
112       lifetime: Duration,
113       http: Http[G, F],
114       fieldNames: List[CIString] = Nil,
115   ): Http[G, F] =
116     cache(
117       lifetime,
118       Either.right(CacheDirective.`private`(fieldNames)),
119       Helpers.defaultMethodsToSetOn,
120       Helpers.defaultStatusToSetOn,
121       http,
122     )
123 
124   /** Privately Caches A Response for the given lifetime.
125     *
126     * Note: If set to Duration.Inf, lifetime falls back to
127     * 10 years for support of Http1 caches.
128     */
129   def privateCacheResponse[G[_]](
130       lifetime: Duration,
131       fieldNames: List[CIString] = Nil,
132   ): PartiallyAppliedCache[G] =
133     cacheResponse(lifetime, Either.right(CacheDirective.`private`(fieldNames)))
134 
135   /** Construct a Middleware that will apply the appropriate caching headers.
136     *
137     * Helper functions for methodToSetOn and statusToSetOn can be found in [[Helpers]].
138     *
139     * Note: If set to Duration.Inf, lifetime falls back to
140     * 10 years for support of Http1 caches.
141     */
142   def cache[G[_]: Temporal, F[_]](
143       lifetime: Duration,
144       isPublic: Either[CacheDirective.public.type, CacheDirective.`private`],
145       methodToSetOn: Method => Boolean,
146       statusToSetOn: Status => Boolean,
147       http: Http[G, F],
148   ): Http[G, F] = {
149     val cacher = cacheResponse[G](lifetime, isPublic)
150     Kleisli { (req: Request[F]) =>
151       if (methodToSetOn(req.method)) {
152         http(req).flatMap { resp =>
153           if (statusToSetOn(resp.status)) cacher(resp) else resp.pure[G]
154         }
155       } else http(req)
156     }
157   }
158 
159   // Here as an optimization so we don't recreate durations
160   // in cacheResponse #TeamStatic
161   private val tenYearDuration: FiniteDuration = 315360000.seconds
162 
163   /**  Method in order to turn a generated Response into one that
164     * will be appropriately cached.
165     *
166     *  Note: If set to Duration.Inf, lifetime falls back to
167     * 10 years for support of Http1 caches.
168     */
169   def cacheResponse[G[_]](
170       lifetime: Duration,
171       isPublic: Either[CacheDirective.public.type, CacheDirective.`private`],
172   ): PartiallyAppliedCache[G] =
173     new PartiallyAppliedCache[G] {
174       val actualLifetime = lifetime match {
175         case finite: FiniteDuration => finite
176         case _ => tenYearDuration
177         // Http1 caches do not respect max-age headers, so to work globally it is recommended
178         // to explicitly set an Expire which requires some time interval to work
179       }
180       val cacheControl = {
181         val cacheDirective = isPublic.fold(identity, identity)
182         val maxAge = CacheDirective.`max-age`(actualLifetime)
183         `Cache-Control`(NonEmptyList.of(cacheDirective, maxAge))
184       }
185 
186       override def apply[F[_]](resp: Response[F])(implicit G: Temporal[G]): G[Response[F]] =
187         for {
188           now <- HttpDate.current[G]
189           eps = now.epochSecond + actualLifetime.toSeconds
190           expires <- HttpDate.fromEpochSecond(eps).liftTo[G]
191         } yield resp.putHeaders(cacheControl, HDate(now), Expires(expires))
192     }
193 
194   trait PartiallyAppliedCache[G[_]] {
195     def apply[F[_]](resp: Response[F])(implicit G: Temporal[G]): G[Response[F]]
196   }
197 
198   trait PartiallyAppliedNoStoreCache[G[_]] {
199     def apply[F[_]](resp: Response[F])(implicit G: Temporal[G]): G[Response[F]]
200   }
