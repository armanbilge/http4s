32 /** Build a response which will accept an HTTP websocket upgrade request and initiate a websocket connection using the
33   * supplied exchange to process and respond to websocket messages.
34   * @param headers Handshake response headers, such as such as:Sec-WebSocket-Protocol.
35   * @param onNonWebSocketRequest The status code to return to a client making a non-websocket HTTP request to this route.
36   *                              default: NotImplemented
37   * @param onHandshakeFailure The status code to return when failing to handle a websocket HTTP request to this route.
38   *                           default: BadRequest
39   */
40 sealed abstract class WebSocketBuilder2[F[_]: Applicative] private (
41     headers: Headers,
42     onNonWebSocketRequest: F[Response[F]],
43     onHandshakeFailure: F[Response[F]],
44     onClose: F[Unit],
45     filterPingPongs: Boolean,
46     webSocketKey: Key[WebSocketContext[F]],
47 ) {

92   private def buildResponse(webSocket: WebSocket[F]): F[Response[F]] =
93     onNonWebSocketRequest
94       .map(
95         _.withAttribute(

100             onHandshakeFailure,
101           ),
102         )

105   /** @param sendReceive The send-receive stream represents transforming of incoming messages to outgoing for a single websocket
106     *                    Once the stream have terminated, the server will initiate a close of the websocket connection.
107     *                    As defined in the websocket specification, this means the server
108     *                    will send a CloseFrame to the client and wait for a CloseFrame in response before closing the
109     *                    connection, this ensures that no messages are lost in flight. The server will shutdown the
110     *                    connection when it receives the `CloseFrame` message back from the client. The connection will also
111     *                    be closed if the client does not respond with a `CloseFrame` after some reasonable amount of
112     *                    time.
113     *                    Another way of closing the connection is by emitting a `CloseFrame` in the stream of messages
114     *                    heading to the client. This method allows one to attach a message to the `CloseFrame` as defined
115     *                    by the websocket protocol.
116     *                    Unfortunately the current implementation does not quite respect the description above, it violates
117     *                    the websocket protocol by terminating the connection immediately upon reception
118     *                    of a `CloseFrame`. This bug will be addressed soon in an upcoming release and this message will be
119     *                    removed.
120     *                    Currently, there is no way for the server to be notified when the connection is closed, neither in
121     *                    the case of a normal disconnection such as a Close handshake or due to a connection error. There
122     *                    are plans to address this limitation in the future.
123     * @return
124     */
125   def build(sendReceive: Pipe[F, WebSocketFrame, WebSocketFrame]): F[Response[F]] = {
126 
127     val finalSendReceive: Pipe[F, WebSocketFrame, WebSocketFrame] =
128       if (filterPingPongs)
129         sendReceive.compose(inputStream => inputStream.filterNot(isPingPong))
130       else
131         sendReceive
132 
133     buildResponse(WebSocketCombinedPipe(finalSendReceive, onClose))
134   }
135 
136   /** @param send     The send side of the Exchange represents the outgoing stream of messages that should be sent to the client
137     * @param receive  The receive side of the Exchange is a sink to which the framework will push the incoming websocket messages
138     *                 Once both streams have terminated, the server will initiate a close of the websocket connection.
139     *                 As defined in the websocket specification, this means the server
140     *                 will send a CloseFrame to the client and wait for a CloseFrame in response before closing the
141     *                 connection, this ensures that no messages are lost in flight. The server will shutdown the
142     *                 connection when it receives the `CloseFrame` message back from the client. The connection will also
143     *                 be closed if the client does not respond with a `CloseFrame` after some reasonable amount of
144     *                 time.
145     *                 Another way of closing the connection is by emitting a `CloseFrame` in the stream of messages
146     *                 heading to the client. This method allows one to attach a message to the `CloseFrame` as defined
147     *                 by the websocket protocol.
148     *                 Unfortunately the current implementation does not quite respect the description above, it violates
149     *                 the websocket protocol by terminating the connection immediately upon reception
150     *                 of a `CloseFrame`. This bug will be addressed soon in an upcoming release and this message will be
151     *                 removed.
152     *                 Currently, there is no way for the server to be notified when the connection is closed, neither in
153     *                 the case of a normal disconnection such as a Close handshake or due to a connection error. There
154     *                 are plans to address this limitation in the future.
155     * @return
156     */
157   def build(
158       send: Stream[F, WebSocketFrame],
159       receive: Pipe[F, WebSocketFrame, Unit],
160   ): F[Response[F]] = {
161 
162     val finalReceive: Pipe[F, WebSocketFrame, Unit] =
163       if (filterPingPongs)
164         _.filterNot(isPingPong).through(receive)
165       else
166         receive
167 
168     buildResponse(WebSocketSeparatePipe(send, finalReceive, onClose))
169   }
170 
171   private val isPingPong: WebSocketFrame => Boolean = {
172     case _: WebSocketFrame.Ping => true
173     case _: WebSocketFrame.Pong => true
174     case _ => false
175   }
176 
177 }
