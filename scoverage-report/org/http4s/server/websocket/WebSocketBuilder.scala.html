26 /** Build a response which will accept an HTTP websocket upgrade request and initiate a websocket connection using the
27   * supplied exchange to process and respond to websocket messages.
28   * @param headers Handshake response headers, such as such as:Sec-WebSocket-Protocol.
29   * @param onNonWebSocketRequest The status code to return to a client making a non-websocket HTTP request to this route.
30   *                              default: NotImplemented
31   * @param onHandshakeFailure The status code to return when failing to handle a websocket HTTP request to this route.
32   *                           default: BadRequest
33   */
34 @deprecated(
35   "Relies on an unsafe cast; instead obtain a WebSocketBuilder2 via .withHttpWebSocketApp on your server builder",
36   "0.23.5",
37 )
38 final case class WebSocketBuilder[F[_]: Applicative](
39     headers: Headers,
40     onNonWebSocketRequest: F[Response[F]],
41     onHandshakeFailure: F[Response[F]],
42     onClose: F[Unit],
43     filterPingPongs: Boolean,
44 ) {
45 
46   private lazy val delegate: WebSocketBuilder2[F] =
47     WebSocketBuilder2(websocketKey[F])
48       .withHeaders(headers)
49       .withOnNonWebSocketRequest(onNonWebSocketRequest)
50       .withOnHandshakeFailure(onHandshakeFailure)
51       .withOnClose(onClose)
52       .withFilterPingPongs(filterPingPongs)
53 
54   /** @param sendReceive The send-receive stream represents transforming of incoming messages to outgoing for a single websocket
55     *                    Once the stream have terminated, the server will initiate a close of the websocket connection.
56     *                    As defined in the websocket specification, this means the server
57     *                    will send a CloseFrame to the client and wait for a CloseFrame in response before closing the
58     *                    connection, this ensures that no messages are lost in flight. The server will shutdown the
59     *                    connection when it receives the `CloseFrame` message back from the client. The connection will also
60     *                    be closed if the client does not respond with a `CloseFrame` after some reasonable amount of
61     *                    time.
62     *                    Another way of closing the connection is by emitting a `CloseFrame` in the stream of messages
63     *                    heading to the client. This method allows one to attach a message to the `CloseFrame` as defined
64     *                    by the websocket protocol.
65     *                    Unfortunately the current implementation does not quite respect the description above, it violates
66     *                    the websocket protocol by terminating the connection immediately upon reception
67     *                    of a `CloseFrame`. This bug will be addressed soon in an upcoming release and this message will be
68     *                    removed.
69     *                    Currently, there is no way for the server to be notified when the connection is closed, neither in
70     *                    the case of a normal disconnection such as a Close handshake or due to a connection error. There
71     *                    are plans to address this limitation in the future.
72     * @return
73     */
74   def build(sendReceive: Pipe[F, WebSocketFrame, WebSocketFrame]): F[Response[F]] =
75     delegate.build(sendReceive)
76 
77   /** @param send     The send side of the Exchange represents the outgoing stream of messages that should be sent to the client
78     * @param receive  The receive side of the Exchange is a sink to which the framework will push the incoming websocket messages
79     *                 Once both streams have terminated, the server will initiate a close of the websocket connection.
80     *                 As defined in the websocket specification, this means the server
81     *                 will send a CloseFrame to the client and wait for a CloseFrame in response before closing the
82     *                 connection, this ensures that no messages are lost in flight. The server will shutdown the
83     *                 connection when it receives the `CloseFrame` message back from the client. The connection will also
84     *                 be closed if the client does not respond with a `CloseFrame` after some reasonable amount of
85     *                 time.
86     *                 Another way of closing the connection is by emitting a `CloseFrame` in the stream of messages
87     *                 heading to the client. This method allows one to attach a message to the `CloseFrame` as defined
88     *                 by the websocket protocol.
89     *                 Unfortunately the current implementation does not quite respect the description above, it violates
90     *                 the websocket protocol by terminating the connection immediately upon reception
91     *                 of a `CloseFrame`. This bug will be addressed soon in an upcoming release and this message will be
92     *                 removed.
93     *                 Currently, there is no way for the server to be notified when the connection is closed, neither in
94     *                 the case of a normal disconnection such as a Close handshake or due to a connection error. There
95     *                 are plans to address this limitation in the future.
96     * @return
97     */
98   def build(
99       send: Stream[F, WebSocketFrame],
100       receive: Pipe[F, WebSocketFrame, Unit],
101   ): F[Response[F]] =
102     delegate.build(send, receive)
103 
104 }
