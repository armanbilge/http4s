30 trait ServerBuilder[F[_]] extends BackendBuilder[F, Server] {
31   type Self <: ServerBuilder[F]
32 
33   implicit protected def F: Concurrent[F]
34 
35   def bindSocketAddress(socketAddress: InetSocketAddress): Self
36 
37   final def bindHttp(port: Int = defaults.HttpPort, host: String = defaults.IPv4Host): Self =
38     bindSocketAddress(InetSocketAddress.createUnresolved(host, port))
39 
40   final def bindLocal(port: Int): Self = bindHttp(port, defaults.IPv4Host)
41 
42   final def bindAny(host: String = defaults.IPv4Host): Self = bindHttp(0, host)
43 
44   /** Sets the handler for errors thrown invoking the service.  Is not
45     * guaranteed to be invoked on errors on the server backend, such as
46     * parsing a request or handling a context timeout.
47     */
48   def withServiceErrorHandler(
49       serviceErrorHandler: Request[F] => PartialFunction[Throwable, F[Response[F]]]
50   ): Self
51 
52   /** Returns a Server resource.  The resource is not acquired until the
53     * server is started and ready to accept requests.
54     */
55   def resource: Resource[F, Server]
56 
57   /** Runs the server as a process that never emits.  Useful for a server
58     * that runs for the rest of the JVM's life.
59     */
60   final def serve: Stream[F, ExitCode] =
61     for {
62       signal <- Stream.eval(SignallingRef[F, Boolean](false))
63       exitCode <- Stream.eval(F.ref(ExitCode.Success))
64       serve <- serveWhile(signal, exitCode)
65     } yield serve
66 
67   /** Runs the server as a Stream that emits only when the terminated signal becomes true.
68     * Useful for servers with associated lifetime behaviors.
69     */
70   final def serveWhile(
71       terminateWhenTrue: Signal[F, Boolean],
72       exitWith: Ref[F, ExitCode],
73   ): Stream[F, ExitCode] =
74     Stream.resource(resource) *> (terminateWhenTrue.discrete
75       .takeWhile(_ === false)
76       .drain ++ Stream.eval(exitWith.get))
77 
78   /** Set the banner to display when the server starts up */
79   def withBanner(banner: immutable.Seq[String]): Self
80 
81   /** Disable the banner when the server starts up */
82   final def withoutBanner: Self = withBanner(immutable.Seq.empty)
83 }
