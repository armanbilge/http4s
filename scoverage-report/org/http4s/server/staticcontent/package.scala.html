23 /** Helpers for serving static content from http4s
24   *
25   * Note that these tools are relatively primitive and a dedicated server should be used
26   * for serious static content serving.
27   */
28 package object staticcontent {
29 
30   /** Make a new [[org.http4s.HttpRoutes]] that serves static files, possibly from the classpath. */
31   def resourceServiceBuilder[F[_]](basePath: String): ResourceServiceBuilder[F] =
32     ResourceServiceBuilder[F](basePath)
33 
34   /** Make a new [[org.http4s.HttpRoutes]] that serves static files. */
35   def fileService[F[_]: Async](config: FileService.Config[F]): HttpRoutes[F] =
36     FileService(config)
37 
38   /** Make a new [[org.http4s.HttpRoutes]] that serves static files from webjars */
39   def webjarServiceBuilder[F[_]]: WebjarServiceBuilder[F] =
40     WebjarServiceBuilder[F]
41 
42   private[staticcontent] val AcceptRangeHeader = `Accept-Ranges`(RangeUnit.Bytes)
43 
44   // Will strip the pathPrefix from the first part of the Uri, returning the remainder without a leading '/'
45   private[staticcontent] def getSubPath(uriPath: String, pathPrefix: String): String = {
46     val index = pathPrefix.length + {
47       if (
48         uriPath.length > pathPrefix.length &&
49         uriPath.charAt(pathPrefix.length) == '/'
50       ) 1
51       else 0
52     }
53 
54     uriPath.substring(index)
55   }
56 }
