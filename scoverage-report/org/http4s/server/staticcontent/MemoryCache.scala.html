29 /** [[CacheStrategy]] that will cache __all__ [[Response]] bodies in local memory
30   *
31   * This is useful when serving a very limited amount of static content and want
32   * to avoid disk access.
33   */
34 class MemoryCache[F[_]] extends CacheStrategy[F] {
35   private[this] val logger = getLogger
36   private val cacheMap = new ConcurrentHashMap[Uri.Path, Response[F]]()
37 
38   override def cache(uriPath: Uri.Path, resp: Response[F])(implicit
39       F: Concurrent[F]
40   ): F[Response[F]] =
41     if (resp.status == Status.Ok)
42       Option(cacheMap.get(uriPath)) match {
43         case Some(r) if r.headers.headers == resp.headers.headers =>
44           logger.debug(s"Cache hit: $resp")
45           F.pure(r)
46 
47         case _ =>
48           logger.debug(s"Cache miss: $resp")
49           collectResource(uriPath, resp) /* otherwise cache the response */
50       }
51     else F.pure(resp)
52 
53   // //////////// private methods //////////////////////////////////////////////
54 
55   private def collectResource(path: Uri.Path, resp: Response[F])(implicit
56       F: Concurrent[F]
57   ): F[Response[F]] =
58     resp
59       .as[Chunk[Byte]]
60       .map { chunk =>
61         val newResponse: Response[F] = resp.copy(body = Stream.chunk(chunk))
62         cacheMap.put(path, newResponse)
63         newResponse
64       }
65 }

68   def apply[F[_]](): MemoryCache[F] = new MemoryCache[F]
