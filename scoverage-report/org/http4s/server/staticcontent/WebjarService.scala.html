67   def toRoutes(implicit F: Async[F]): HttpRoutes[F] = {
68     object BadTraversal extends Exception with NoStackTrace
69     val Root = Paths.get("")
70     Kleisli {
71       // Intercepts the routes that match webjar asset names
72       case request if request.method == Method.GET =>
73         val segments = request.pathInfo.segments.map(_.decoded(plusIsSpace = true))
74         OptionT
75           .liftF(F.catchNonFatal {
76             segments.foldLeft(Root) {
77               case (_, "" | "." | "..") => throw BadTraversal
78               case (path, segment) =>
79                 path.resolve(segment)
80             }
81           })
82           .subflatMap(toWebjarAsset)
83           .filter(webjarAssetFilter)
84           .flatMap(serveWebjarAsset(cacheStrategy, classLoader, request, preferGzipped)(_))
85           .recover { case BadTraversal =>
86             Response(Status.BadRequest)
87           }
88       case _ => OptionT.none
89     }
90   }

92   /** Returns an Option(WebjarAsset) for a Request, or None if it couldn't be mapped
93     *
94     * @param p The request path without the prefix
95     * @return The WebjarAsset, or None if it couldn't be mapped
96     */
97   private def toWebjarAsset(p: Path): Option[WebjarAsset] = {
98     val count = p.getNameCount
99     if (count > 2) {
100       val library = p.getName(0).toString
101       val version = p.getName(1).toString
102       val asset = asScalaIterator(p.subpath(2, count).iterator()).mkString("/")
103       Some(WebjarAsset(library, version, asset))
104     } else
105       None
106   }

108   /** Creates a scala.Iterator from a java.util.Iterator.
109     *
110     * We're not using scala.jdk.CollectionConverters (which was added in 2.13)
111     * or scala.collection.convert.ImplicitConversion (which was deprecated in 2.13)
112     * to ease cross-building against multiple Scala versions.
113     */
114   private def asScalaIterator[A](underlying: java.util.Iterator[A]): Iterator[A] =
115     new Iterator[A] {
116       override def hasNext: Boolean = underlying.hasNext
117       override def next(): A = underlying.next()
118     }

137   /** Contains the information about an asset inside a webjar
138     *
139     * @param library The webjar's library name
140     * @param version The version of the webjar
141     * @param asset The asset name inside the webjar
142     */
143   final case class WebjarAsset(library: String, version: String, asset: String) {
144 
145     /** Constructs a full path for an asset inside a webjar asset
146       *
147       * @return The full name in the Webjar
148       */
149     private[staticcontent] lazy val pathInJar: String =
150       s"/META-INF/resources/webjars/$library/$version/$asset"
151   }
152 
153   /** Returns an asset that matched the request if it's found in the webjar path
154     *
155     * @param request The Request
156     * @param preferGzipped prefer gzip compression format?
157     * @param webjarAsset The WebjarAsset
158     * @return Either the the Asset, if it exist, or Pass
159     */
160   private def serveWebjarAsset[F[_]](
161       cacheStrategy: CacheStrategy[F],
162       classLoader: Option[ClassLoader],
163       request: Request[F],
164       preferGzipped: Boolean,
165   )(webjarAsset: WebjarAsset)(implicit F: Async[F]): OptionT[F, Response[F]] =
166     StaticFile
167       .fromResource(
168         webjarAsset.pathInJar,
169         Some(request),
170         classloader = classLoader,
171         preferGzipped = preferGzipped,
172       )
173       .semiflatMap(cacheStrategy.cache(request.pathInfo, _))
174 }
