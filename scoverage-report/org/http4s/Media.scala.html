26 trait Media[F[_]] {
27   def body: EntityBody[F]
28   def headers: Headers
29   def covary[F2[x] >: F[x]]: Media[F2]
30 
31   final def bodyText(implicit
32       RT: RaiseThrowable[F],
33       defaultCharset: Charset = `UTF-8`,
34   ): Stream[F, String] = {
35     val cs = charset.getOrElse(defaultCharset).nioCharset
36     body.through(decodeWithCharset(cs))
37   }
38 
39   final def contentType: Option[`Content-Type`] =
40     headers.get[`Content-Type`]
41 
42   final def contentLength: Option[Long] =
43     headers.get[`Content-Length`].map(_.length)
44 
45   final def charset: Option[Charset] =
46     contentType.flatMap(_.charset)
47 
48   // Decoding methods
49 
50   /** Decode the [[Media]] to the specified type
51     *
52     * @param decoder [[EntityDecoder]] used to decode the [[Media]]
53     * @tparam T type of the result
54     * @return the effect which will generate the `DecodeResult[T]`
55     */
56   final def attemptAs[T](implicit decoder: EntityDecoder[F, T]): DecodeResult[F, T] =
57     decoder.decode(this, strict = false)
58 
59   /** Decode the [[Media]] to the specified type
60     *
61     * If no valid [[Status]] has been described, allow Ok
62     *
63     * @param decoder [[EntityDecoder]] used to decode the [[Media]]
64     * @tparam A type of the result
65     * @return the effect which will generate the A
66     */
67   final def as[A](implicit F: MonadThrow[F], decoder: EntityDecoder[F, A]): F[A] =
68     F.rethrow(attemptAs.value)
69 }
