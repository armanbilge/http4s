87   /** Replace the body of this message with a new body
88     *
89     * @param b body to attach to this method
90     * @param w [[EntityEncoder]] with which to convert the body to an [[EntityBody]]
91     * @tparam T type of the Body
92     * @return a new message with the new body
93     */
94   def withEntity[T](b: T)(implicit w: EntityEncoder[F, T]): Self = {
95     val entity = w.toEntity(b)
96 
97     val hsBase = headers ++ w.headers
98     val cl: Option[`Content-Length`] = entity.length match {
99       case Some(l) =>
100         `Content-Length`
101           .fromLong(l)
102           .fold(
103             _ => {
104               Message.logger.warn(s"Attempt to provide a negative content length of $l")
105               None
106             },
107             Some(_),
108           )
109       case None => None
110     }
111 
112     change(body = entity.body, headers = cl.fold(hsBase)(hsBase.withContentLength))
113   }

251   /** Compiles the body stream to a single chunk and sets it as the
252     * body. Replaces any `Transfer-Encoding: chunked` with a
253     * `Content-Length` header. It is the caller's responsibility to
254     * assure there is enough memory to materialize the entity body and
255     * control the time limits of that materialization.
256     *
257     * @param maxBytes maximum length of the entity stream. If the stream
258     *                 exceeds the limit then processing fails with the [[Message.EntityStreamException]].
259     *                 Pass the [[scala.None]] if you don't want to limit the entity body.
260     */
261   def toStrict(maxBytes: Option[Long])(implicit F: Concurrent[F]): F[Self] = {
262     def withLimit(entityBody: EntityBody[F]) = maxBytes match {
263       case Some(limit) =>
264         entityBody.pull
265           .take(limit)
266           .flatMap {
267             case Some(_) =>
268               Pull.raiseError[F](EntityStreamException.createWithDefaultMsg(limit))
269             case None =>
270               Pull.done
271           }
272           .stream
273 
274       case None =>
275         entityBody
276     }
277 
278     withLimit(body).compile.to(Chunk).map { chunk =>
279       withBodyStream(Stream.chunk(chunk))
280         .transformHeaders(_.withContentLength(`Content-Length`.unsafeFromLong(chunk.size.toLong)))
281     }
282   }
