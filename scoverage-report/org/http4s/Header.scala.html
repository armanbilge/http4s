33 /** Typeclass representing an HTTP header, which all the http4s
34   * default headers satisfy.
35   * You can add modelled headers by providing an implicit instance of
36   * `Header[YourModelledHeader]`
37   */
38 trait Header[A, T <: Header.Type] {
39 
40   /** Name of the header. Not case sensitive.
41     */
42   def name: CIString
43 
44   /** Value of the header, which is represented as a String.
45     * Will be a comma separated String for headers with multiple values.
46     */
47   def value(a: A): String
48 
49   /** Parses the header from its String representation.
50     * Could be a comma separated String in case of a Header with
51     * multiple values.
52     */
53   def parse(headerValue: String): Either[ParseFailure, A]
54 }
55 
56 object Header {
57   final case class Raw(name: CIString, value: String) {
58     override def toString: String = s"${name}: ${value}"
59 
60     /** True if [[name]] is a valid field-name per RFC7230.  Where it
61       * is not, the header may be dropped by the backend.
62       */
63     def isNameValid: Boolean =
64       name.toString.nonEmpty && name.toString.forall(FieldNamePredicate)
65 
66     def sanitizedValue: String = {
67       val w = new StringWriter
68       w.sanitize(_ << value).result
69     }
70   }
71 
72   object Raw {
73     implicit lazy val catsInstancesForHttp4sHeaderRaw
74         : Order[Raw] with Hash[Raw] with Show[Raw] with Renderer[Raw] = new Order[Raw]
75       with Hash[Raw]
76       with Show[Raw]
77       with Renderer[Raw] {
78       def show(h: Raw): String = s"${h.name.show}: ${h.value}"
79       def hash(h: Raw): Int = h.hashCode
80 
81       def compare(x: Raw, y: Raw): Int =
82         x.name.compare(y.name) match {
83           case 0 => x.value.compare(y.value)
84           case c => c
85         }
86 
87       def render(writer: Writer, h: Raw): writer.type = {
88         writer << h.name << ':' << ' '
89         writer.sanitize(_ << h.value)
90       }
91     }
92   }
93 
94   /** Classifies modelled headers into `Single` headers, which can only
95     * appear once, and `Recurring` headers, which can appear multiple
96     * times.
97     */
98   sealed trait Type
99   case class Single() extends Type // scalafix:ok Http4sGeneralLinters; bincompat until 1.0
100   case class Recurring() extends Type // scalafix:ok Http4sGeneralLinters; bincompat until 1.0
101 
102   def apply[A](implicit ev: Header[A, _]): ev.type = ev
103 
104   @deprecated("use Header.Raw.apply", "0.22.0")
105   def apply(name: String, value: String): Header.Raw = Raw(CIString(name), value)
106 
107   def create[A, T <: Header.Type](
108       name_ : CIString,
109       value_ : A => String,
110       parse_ : String => Either[ParseFailure, A],
111   ): Header[A, T] = new Header[A, T] {
112     def name: CIString = name_
113     def value(a: A): String = value_(a)
114     def parse(s: String): Either[ParseFailure, A] = parse_(s)
115   }
116 
117   def createRendered[A, T <: Header.Type, B: Renderer](
118       name_ : CIString,
119       value_ : A => B,
120       parse_ : String => Either[ParseFailure, A],
121   ): Header[A, T] = new Header[A, T] {
122     def name: CIString = name_
123     def value(a: A): String = Renderer.renderString(value_(a))
124     def parse(s: String): Either[ParseFailure, A] = parse_(s)
125   }

127   /** Target for implicit conversions to Header.Raw from modelled
128     * headers and key-value pairs.
129     *
130     * A method taking variadic `ToRaw` arguments will allow taking
131     * heteregenous arguments, provided they are either:
132     *
133     * - A value of type `A`  which has a `Header[A]` in scope
134     * - A (name, value) pair of `String`, which is treated as a `Recurring` header
135     * - A `Header.Raw`
136     * - A `Foldable` (`List`, `Option`, etc) of the above.
137     *
138     * @see [[org.http4s.Headers.apply]]
139     */
140   sealed trait ToRaw {
141     def values: List[Header.Raw]
142   }
143   object ToRaw {
144     trait Primitive
145 
146     implicit def identityToRaw(h: Header.ToRaw): Header.ToRaw with Primitive = new Header.ToRaw
147     with Primitive {
148       val values: List[Raw] = h.values
149     }
150 
151     implicit def rawToRaw(h: Header.Raw): Header.ToRaw with Primitive =
152       new Header.ToRaw with Primitive {
153         val values = List(h)
154       }
155 
156     implicit def keyValuesToRaw(kv: (String, String)): Header.ToRaw with Primitive =
157       new Header.ToRaw with Primitive {
158         val values = List(Header.Raw(CIString(kv._1), kv._2))
159       }
160 
161     implicit def headersToRaw(h: Headers): Header.ToRaw =
162       new Header.ToRaw {
163         val values: List[Raw] = h.headers
164       }
165 
166     implicit def modelledHeadersToRaw[H](
167         h: H
168     )(implicit H: Header[H, _]): Header.ToRaw with Primitive =
169       new Header.ToRaw with Primitive {
170         val values = List(Header.Raw(H.name, H.value(h)))
171       }
172 
173     implicit def foldablesToRaw[F[_]: Foldable, H](
174         h: F[H]
175     )(implicit convert: H => ToRaw with Primitive): Header.ToRaw = new Header.ToRaw {
176       val values: List[Raw] = h.toList.foldMap(v => convert(v).values)
177     }
178 
179     // Required for 2.12 to convert variadic args.
180     implicit def scalaCollectionSeqToRaw[H](
181         h: collection.Seq[H]
182     )(implicit convert: H => ToRaw with Primitive): Header.ToRaw = new Header.ToRaw {
183       val values: List[Raw] = h.toList.foldMap(v => convert(v).values)
184     }
185   }

187   /** Abstracts over Single and Recurring Headers
188     */
189   sealed trait Select[A] {
190     type F[_]
191 
192     /** Transform this header into a [[Header.Raw]]
193       */
194     def toRaw1(a: A): Header.Raw
195 
196     /** Transform this (potentially repeating) header into a [[Header.Raw]] */
197     def toRaw(a: F[A]): NonEmptyList[Header.Raw]
198 
199     /** Selects this header from a list of [[Header.Raw]]
200       */
201     def from(headers: List[Header.Raw]): Option[Ior[NonEmptyList[ParseFailure], F[A]]]
202   }
203   trait LowPrio {
204     implicit def recurringHeadersNoMerge[A](implicit
205         h: Header[A, Header.Recurring]
206     ): Select[A] { type F[B] = NonEmptyList[B] } =
207       new Select[A] {
208         type F[B] = NonEmptyList[B]
209 
210         def toRaw1(a: A): Header.Raw =
211           Header.Raw(h.name, h.value(a))
212 
213         def toRaw(as: F[A]): NonEmptyList[Header.Raw] =
214           as.map(a => Header.Raw(h.name, h.value(a)))
215 
216         def from(headers: List[Raw]): Option[Ior[NonEmptyList[ParseFailure], NonEmptyList[A]]] =
217           headers.foldLeft(Option.empty[Ior[NonEmptyList[ParseFailure], NonEmptyList[A]]]) {
218             (a, raw) =>
219               Select.fromRaw(raw) match {
220                 case Some(aa) => a |+| aa.bimap(NonEmptyList.one, NonEmptyList.one).some
221                 case None => a
222               }
223           }
224       }
225   }
226   object Select extends LowPrio {
227     type Aux[A, G[_]] = Select[A] { type F[B] = G[B] }
228 
229     def fromRaw[A](h: Header.Raw)(implicit ev: Header[A, _]): Option[Ior[ParseFailure, A]] =
230       if (h.name == Header[A].name) Some(Header[A].parse(h.value).toIor) else None
231 
232     implicit def singleHeaders[A](implicit
233         h: Header[A, Header.Single]
234     ): Select[A] { type F[B] = B } =
235       new Select[A] {
236         type F[B] = B
237 
238         def toRaw1(a: A): Header.Raw =
239           Header.Raw(h.name, h.value(a))
240 
241         def toRaw(a: A): NonEmptyList[Header.Raw] =
242           NonEmptyList.one(toRaw1(a))
243 
244         def from(headers: List[Raw]): Option[Ior[NonEmptyList[ParseFailure], F[A]]] =
245           headers.collectFirst(Function.unlift(fromRaw(_).map(_.leftMap(NonEmptyList.one))))
246       }
247 
248     implicit def recurringHeadersWithMerge[A: Semigroup](implicit
249         h: Header[A, Header.Recurring]
250     ): Select[A] { type F[B] = B } =
251       new Select[A] {
252         type F[B] = B
253 
254         def toRaw1(a: A): Header.Raw =
255           Header.Raw(h.name, h.value(a))
256 
257         def toRaw(a: A): NonEmptyList[Header.Raw] =
258           NonEmptyList.one(toRaw1(a))
259 
260         def from(headers: List[Raw]): Option[Ior[NonEmptyList[ParseFailure], F[A]]] =
261           headers.foldLeft(Option.empty[Ior[NonEmptyList[ParseFailure], F[A]]]) { (a, raw) =>
262             fromRaw(raw) match {
263               case Some(aa) => a |+| aa.leftMap(NonEmptyList.one).some
264               case None => a
265             }
266           }
267       }
268   }
