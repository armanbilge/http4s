99     if (query.isEmpty) false

109   private def _removeQueryParam(name: QueryParameterKey): Self =
110     if (query.isEmpty) self
111     else {
112       val newQuery = query.filterNot { case (n, _) => n == name.value }
113       replaceQuery(newQuery)

116   /** Creates maybe a new `Self` with the specified parameters. The entire
117     * [[Query]] will be replaced with the given one.
118     */
119   def setQueryParams[K: QueryParamKeyLike, T: QueryParamEncoder](
120       params: Map[K, collection.Seq[T]]
121   ): Self = {
122     val penc = QueryParamKeyLike[K]
123     val venc = QueryParamEncoder[T]
124     val vec = params.foldLeft(query.toVector) {
125       case (m, (k, Seq())) => m :+ (penc.getKey(k).value -> None)
126       case (m, (k, vs)) =>
127         vs.foldLeft(m) { case (m, v) => m :+ (penc.getKey(k).value -> Some(venc.encode(v).value)) }
128     }
129     replaceQuery(Query.fromVector(vec))
130   }

164   /** Creates maybe a new `Self` with all the specified parameters in the
165     * [[Query]]. If any of the given parameters' keys already exists, the

173     }

188   private def _withQueryParam(
189       name: QueryParameterKey,
190       values: collection.Seq[QueryParameterValue],
191   ): Self = {
192     val q = if (query == Query.blank) Query.empty else query
193     val baseQuery = q.toVector.filter(_._1 != name.value)
194     val vec =
195       if (values.isEmpty) baseQuery :+ (name.value -> None)
196       else
197         values.toList.foldLeft(baseQuery) { case (vec, v) =>
198           vec :+ (name.value -> Some(v.value))
199         }
200 
201     replaceQuery(Query.fromVector(vec))
202   }
