80 final case class BasicCredentials(
81     username: String,
82     password: String,
83     charset: JavaCharset = StandardCharsets.UTF_8,
84 ) {
85   lazy val token: String = {
86     val userPass = username + ':' + password
87     val bytes = userPass.getBytes(charset)
88     Base64.getEncoder.encodeToString(bytes)
89   }
90 }
91 
92 object BasicCredentials {
93 
94   private val utf8Charset = StandardCharsets.UTF_8
95   private val utf8CharsetDecoder = utf8Charset.newDecoder
96   private val fallbackCharset = StandardCharsets.ISO_8859_1
97 
98   private def decode(bytes: Array[Byte], decoder: CharsetDecoder): Try[String] = {
99     val byteByffer = ByteBuffer.wrap(bytes)
100     Try(decoder.decode(byteByffer).toString)
101   }
102   private def decode(bytes: Array[Byte], charset: JavaCharset): String =
103     new String(bytes, charset)
104 
105   def apply(token: String): BasicCredentials = {
106     val bytes = Base64.getDecoder.decode(token)
107     val (userPass, charset) = decode(bytes, utf8CharsetDecoder)
108       .fold(_ => (decode(bytes, fallbackCharset), fallbackCharset), up => (up, utf8Charset))
109     userPass.indexOf(':') match {
110       case -1 => apply(userPass, "", charset)
111       case ix => apply(userPass.substring(0, ix), userPass.substring(ix + 1), charset)
112     }
113   }
114 
115   def unapply(creds: Credentials): Option[(String, String)] =
116     creds match {
117       case Credentials.Token(AuthScheme.Basic, token) =>
118         val basicCredentials = BasicCredentials(token)
119         Some((basicCredentials.username, basicCredentials.password))
120       case _ =>
121         None
122     }
123 }
