53   def fromResource[F[_]: Sync](
54       name: String,
55       req: Option[Request[F]] = None,
56       preferGzipped: Boolean = false,
57       classloader: Option[ClassLoader] = None,
58   ): OptionT[F, Response[F]] = {
59     val loader = classloader.getOrElse(getClass.getClassLoader)
60 
61     val acceptEncodingHeader: Option[`Accept-Encoding`] =
62       req.flatMap(_.headers.get[`Accept-Encoding`])
63     val tryGzipped =
64       preferGzipped && acceptEncodingHeader.exists { acceptEncoding =>
65         acceptEncoding.satisfiedBy(ContentCoding.gzip) || acceptEncoding.satisfiedBy(
66           ContentCoding.`x-gzip`
67         )
68       }
69     val normalizedName = name.split("/").filter(_.nonEmpty).mkString("/")
70 
71     def getResource(name: String) =
72       OptionT(Sync[F].blocking(Option(loader.getResource(name))))
73 
74     val gzUrl: OptionT[F, URL] =
75       if (tryGzipped) getResource(normalizedName + ".gz") else OptionT.none
76 
77     gzUrl
78       .flatMap { url =>
79         fromURL(url, req).map {
80           _.removeHeader[`Content-Type`]
81             .putHeaders(
82               `Content-Encoding`(ContentCoding.gzip),
83               nameToContentType(normalizedName), // Guess content type from the name without ".gz"
84             )
85         }
86       }
87       .orElse(
88         getResource(normalizedName)
89           .flatMap(fromURL(_, req))
90       )
91   }

93   def fromURL[F[_]](url: URL, req: Option[Request[F]] = None)(implicit
94       F: Sync[F]
95   ): OptionT[F, Response[F]] = {
96     val fileUrl = url.getFile()
97     val file = new File(fileUrl)
98     OptionT.apply(F.defer {
99       if (url.getProtocol === "file" && file.isDirectory())
100         F.pure(none[Response[F]])
101       else {
102         val urlConn = url.openConnection
103         val lastmod = HttpDate.fromEpochSecond(urlConn.getLastModified / 1000).toOption
104         val ifModifiedSince: Option[`If-Modified-Since`] =
105           req.flatMap(_.headers.get[`If-Modified-Since`])
106         val expired = (ifModifiedSince, lastmod).mapN(_.date < _).getOrElse(true)
107 
108         if (expired) {
109           val len = urlConn.getContentLengthLong
110           val headers = Headers(
111             lastmod.map(`Last-Modified`(_)),
112             nameToContentType(url.getPath),
113             if (len >= 0) `Content-Length`.unsafeFromLong(len)
114             else `Transfer-Encoding`(TransferCoding.chunked.pure[NonEmptyList]),
115           )
116 
117           F.blocking(urlConn.getInputStream)
118             .redeem(
119               recover = {
120                 case _: FileNotFoundException => None
121                 case other => throw other
122               },
123               f = { inputStream =>
124                 Some(
125                   Response(
126                     headers = headers,
127                     body = readInputStream[F](F.pure(inputStream), DefaultBufferSize),
128                   )
129                 )
130               },
131             )
132         } else
133           F.blocking(urlConn.getInputStream.close())
134             .handleError(_ => ())
135             .as(Some(Response(NotModified)))
136       }
137     })
138   }

243   def fromPath[F[_]: Files](

245       start: Long,
246       end: Long,
247       buffsize: Int,
248       req: Option[Request[F]],

250   )(implicit

252   ): OptionT[F, Response[F]] =
253     OptionT(for {
254       etagCalc <- etagCalculator(f).map(et => ETag(et))
255       res <- Files[F].isRegularFile(f).flatMap[Option[Response[F]]] { isFile =>
256         if (isFile) {
257           if (start >= 0 && end >= start && buffsize > 0) {
258             Files[F]
259               .getBasicFileAttributes(f, followLinks = true)
260               .flatMap { attr =>

262                   HttpDate.fromEpochSecond(attr.lastModifiedTime.toSeconds).toOption
263 
264                 F.pure(notModified(req, etagCalc, lastModified).orElse {
265                   val (body, contentLength) =
266                     if (attr.size < end) (Stream.empty, 0L)
267                     else (fileToBody[F](f, start, end), end - start)
268 
269                   val contentType = nameToContentType(f.fileName.toString)
270                   val hs =
271                     Headers(
272                       lastModified.map(`Last-Modified`(_)),
273                       `Content-Length`.fromLong(contentLength).toOption,
274                       contentType,
275                       etagCalc,
276                     )
277 
278                   val r = Response(
279                     headers = hs,
280                     body = body,
281                     attributes = Vault.empty.insert(staticPathKey, f),
282                   )
283 
284                   logger.trace(s"Static file generated response: $r")
285                   r.some
286                 })
287               }
288           } else {
289             F.raiseError[Option[Response[F]]](
290               new IllegalArgumentException(
291                 s"requirement failed: start: $start, end: $end, buffsize: $buffsize"

293             )
294           }
295 
296         } else {
297           F.pure(none[Response[F]])
298         }

303   private def notModified[F[_]](
304       req: Option[Request[F]],
305       etagCalc: ETag,
306       lastModified: Option[HttpDate],
307   ): Option[Response[F]] = {
308     implicit val conjunction: Semigroup[Boolean] = new Semigroup[Boolean] {
309       def combine(x: Boolean, y: Boolean): Boolean = x && y
310     }
311 
312     List(etagMatch(req, etagCalc), notModifiedSince(req, lastModified)).combineAll
313       .filter(identity)
314       .map(_ => Response[F](NotModified))
315   }

317   private def etagMatch[F[_]](req: Option[Request[F]], etagCalc: ETag) =
318     for {
319       r <- req
320       etagHeader <- r.headers.get[`If-None-Match`]
321       etagMatch = etagHeader.tags.exists(_.exists(_ == etagCalc.tag))
322       _ = logger.trace(
323         s"Matches `If-None-Match`: $etagMatch Previous ETag: ${etagHeader.value}, New ETag: $etagCalc"
324       )
325     } yield etagMatch
326 
327   private def notModifiedSince[F[_]](req: Option[Request[F]], lastModified: Option[HttpDate]) =
328     for {
329       r <- req
330       h <- r.headers.get[`If-Modified-Since`]
331       lm <- lastModified
332       notModified = h.date >= lm
333       _ = logger.trace(
334         s"Matches `If-Modified-Since`: $notModified. Request age: ${h.date}, Modified: $lm"
335       )
336     } yield notModified
