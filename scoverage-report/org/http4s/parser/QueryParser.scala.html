26 /** Split an encoded query string into unencoded key value pairs
27   * It always assumes any input is a  valid query, including "".
28   * If "" should be interpreted as no query that __MUST__ be
29   * checked beforehand.
30   */
31 private[http4s] class QueryParser(
32     codec: Codec,
33     colonSeparators: Boolean,
34     qChars: BitSet = QueryParser.ExtendedQChars,
35 ) {
36   import QueryParser._
37 
38   /** Decodes the input into key value pairs.
39     * `flush` signals that this is the last input
40     */
41   def decode(input: CharBuffer, flush: Boolean): ParseResult[Query] =
42     decodeVector(input, flush).map(Query.fromVector)
43 
44   /** Decodes the input into key value pairs.
45     * `flush` signals that this is the last input
46     */
47   def decodeVector(input: CharBuffer, flush: Boolean): ParseResult[Vector[Query.KeyValue]] = {
48     val acc: Builder[Query.KeyValue, Vector[Query.KeyValue]] = Vector.newBuilder
49     decodeBuffer(input, (k, v) => acc += ((k, v)), flush) match {
50       case Some(e) => ParseResult.fail("Decoding of url encoded data failed.", e)
51       case None => ParseResult.success(acc.result())
52     }
53   }
54 
55   // Some[String] represents an error message, None = success
56   private def decodeBuffer(
57       input: CharBuffer,
58       acc: (String, Option[String]) => Builder[Query.KeyValue, Vector[Query.KeyValue]],
59       flush: Boolean,
60   ): Option[String] = {
61     val valAcc = new StringBuilder(InitialBufferCapactiy)
62 
63     var error: String = null
64     var key: String = null
65     var state: State = KEY
66 
67     def appendValue(): Unit = {
68       if (state == KEY) {
69         val s = valAcc.result()
70         val k = decodeParam(s)
71         valAcc.clear()
72         acc(k, None)
73       } else {
74         val k = decodeParam(key)
75         key = null
76         val s = valAcc.result()
77         valAcc.clear()
78         val v = Some(decodeParam(s))
79         acc(k, v)
80       }
81       ()
82     }
83 
84     def endPair(): Unit = {
85       if (!flush) input.mark()
86       appendValue()
87       state = KEY
88     }
89 
90     if (!flush) input.mark()
91 
92     // begin iterating through the chars
93     while (error == null && input.hasRemaining) {
94       val c = input.get()
95       c match {
96         case '&' => endPair()
97 
98         case ';' if colonSeparators => endPair()
99 
100         case '=' =>
101           if (state == VALUE) valAcc.append('=')
102           else {
103             state = VALUE
104             key = valAcc.result()
105             valAcc.clear()
106           }
107 
108         case c if qChars.contains(c.toInt) => valAcc.append(c)
109 
110         case c => error = s"Invalid char while splitting key/value pairs: '$c'"
111       }
112     }
113     if (error != null) Some(error)
114     else {
115       if (flush) appendValue()
116       else input.reset() // rewind to the last mark position
117       None
118     }
119   }
120 
121   private def decodeParam(str: String): String =
122     try Uri.decode(str, codec.charSet, plusIsSpace = true)
123     catch {
124       case _: IllegalArgumentException => ""
125       case _: UnsupportedEncodingException => ""
126     }
127 }

129 private[http4s] object QueryParser {
130   private val InitialBufferCapactiy = 32
131 
132   def parseQueryString(queryString: String, codec: Codec = Codec.UTF8): ParseResult[Query] =
133     if (queryString.isEmpty) Right(Query.empty)
134     else new QueryParser(codec, true).decode(CharBuffer.wrap(queryString), true)
135 
136   def parseQueryStringVector(
137       queryString: String,
138       codec: Codec = Codec.UTF8,
139   ): ParseResult[Vector[Query.KeyValue]] =
140     if (queryString.isEmpty) Right(Vector.empty)
141     else new QueryParser(codec, true).decodeVector(CharBuffer.wrap(queryString), true)
142 
143   private sealed trait State
144   private case object KEY extends State
145   private case object VALUE extends State
146 
147   /** Defines the characters that are allowed unquoted within a query string as
148     * defined in RFC 3986
149     */
150   val QChars: BitSet = BitSet((Pchar ++ "/?".toSet - '&' - '=').map(_.toInt).toSeq: _*)
151 
152   /** PHP also includes square brackets ([ and ]) with query strings. This goes
153     * against the spec but due to PHP's widespread adoption it is necessary to
154     * support this extension.
155     */
156   val ExtendedQChars: BitSet = QChars ++ ("[]".map(_.toInt).toSet)
157   private def Pchar = Unreserved ++ SubDelims ++ ":@%".toSet
158   private def Unreserved = "-._~".toSet ++ AlphaNum
159   private def SubDelims = "!$&'()*+,;=".toSet
160   private def AlphaNum = (('a' to 'z') ++ ('A' to 'Z') ++ ('0' to '9')).toSet
161 }
