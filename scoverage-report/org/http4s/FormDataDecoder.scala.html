26 /** A decoder ware that uses [[QueryParamDecoder]] to decode values in [[org.http4s.UrlForm]]
27   *
28   * @example
29   * {{{
30   * scala> import cats.syntax.all._
31   * scala> import cats.data._
32   * scala> import org.http4s.FormDataDecoder._
33   * scala> import org.http4s.ParseFailure
34   * scala> case class Foo(a: String, b: Boolean)
35   * scala> case class Bar(fs: List[Foo], f: Foo, d: Boolean)
36   * scala>
37   * scala> implicit val fooMapper: FormDataDecoder[Foo] = (
38   *      |   field[String]("a"),
39   *      |   field[Boolean]("b")
40   *      | ).mapN(Foo.apply)
41   * scala>
42   * scala> val barMapper = (
43   *      |   list[Foo]("fs"),
44   *      |   nested[Foo]("f"),
45   *      |   field[Boolean]("d")
46   *      | ).mapN(Bar.apply)
47   * scala>
48   * scala> barMapper(
49   *      |   Map(
50   *      |    "fs[].a" -> Chain("a1", "a2"),
51   *      |    "fs[].b" -> Chain("true", "false"),
52   *      |    "f.a" -> Chain("fa"),
53   *      |    "f.b" -> Chain("false"),
54   *      |    "d" -> Chain("true"))
55   *      | )
56   * res1: ValidatedNel[ParseFailure, Bar] = Valid(Bar(List(Foo(a1,true), Foo(a2,false)),Foo(fa,false),true))
57   *
58   * }}}
59   *
60   * The companion object provides a [[EntityDecoder]] from
61   * HTML form parameters.
62   * @example
63   * {{{
64   *   import org.http4s.FormDataDecoder.formEntityDecoder
65   *   HttpRoutes
66   *    .of[F] {
67   *      case req @ POST -> Root =>
68   *        req.as[MyEntity].flatMap { entity =>
69   *          Ok()
70   *        }
71   *    }
72   * }}}
73   *
74   * For more examples, check the tests
75   * https://github.com/http4s/http4s/blob/main/tests/src/test/scala/org/http4s/FormDataDecoderSpec.scala
76   */
77 sealed trait FormDataDecoder[A] {
78   def apply(data: Map[String, Chain[String]]): ValidatedNel[ParseFailure, A]
79 
80   def mapValidated[B](f: A => ValidatedNel[ParseFailure, B]): FormDataDecoder[B] =
81     FormDataDecoder(this(_).andThen(f))
82 
83   /** Filters out empty strings including spaces before decoding
84     */
85   def sanitized: FormDataDecoder[A] =
86     FormDataDecoder { data =>
87       this(data.map { case (k, v) =>
88         (k, v.filter(_.trim.nonEmpty))
89       })
90     }
91 
92 }
93 
94 object FormDataDecoder {
95   type FormData = Map[String, Chain[String]]
96   type Result[A] = ValidatedNel[ParseFailure, A]
97 
98   def apply[A](f: FormData => Result[A]): FormDataDecoder[A] =
99     new FormDataDecoder[A] {
100       def apply(data: FormData): Result[A] = f(data)
101     }
102 
103   implicit def formEntityDecoder[F[_]: Concurrent, A](implicit
104       fdd: FormDataDecoder[A]
105   ): EntityDecoder[F, A] =
106     UrlForm.entityDecoder[F].flatMapR { d =>
107       fdd(d.values)
108         .leftMap(es => InvalidMessageBodyFailure(es.map(_.sanitized).mkString_("\n")))
109         .liftTo[DecodeResult[F, *]]
110     }
111 
112   private def apply[Data, A](
113       extract: FormData => Either[String, Data]
114   )(decode: Data => Result[A]): FormDataDecoder[Either[String, A]] =
115     new FormDataDecoder[Either[String, A]] {
116       def apply(data: FormData): Result[Either[String, A]] =
117         extract(data).traverse(decode(_))
118     }
119 
120   implicit class FormDataDecoderSyntax[A](private val decoder: FormDataDecoder[Either[String, A]])
121       extends AnyVal {
122 
123     def required: FormDataDecoder[A] =
124       decoder.mapValidated(_.fold(ParseFailure(_, "").invalidNel[A], Valid(_)))
125 
126     def optional: FormDataDecoder[Option[A]] =
127       decoder.map(_.map(Some(_)).getOrElse(None))
128 
129     /** Use a default value when the field is missing
130       * @param defaultValue
131       */
132     def default(defaultValue: A): FormDataDecoder[A] =
133       decoder.map(_.getOrElse(defaultValue))
134   }
135 
136   def fieldEither[A](
137       key: String
138   )(implicit
139       qpd: QueryParamDecoder[A]
140   ): FormDataDecoder[Either[String, A]] =
141     apply[String, A] { data =>
142       nonEmptyFields(data)
143         .get(key)
144         .flatMap(_.headOption)
145         .toRight(s"$key is missing")
146     }(v => qpd.decode(QueryParameterValue(v)))
147 
148   def field[A: QueryParamDecoder](key: String): FormDataDecoder[A] = fieldEither(key).required
149 
150   def fieldOptional[A: QueryParamDecoder](key: String): FormDataDecoder[Option[A]] = fieldEither(
151     key
152   ).optional
153 
154   /** For nested, this decoder assumes that the form parameter name use "." as deliminator for levels.
155     * E.g. For a field named "bar" inside a nested class under the field "foo",
156     * the parameter name is "foo.bar".
157     */
158   def nested[A: FormDataDecoder](key: String): FormDataDecoder[A] =
159     nestedEither(key).required
160 
161   /** For nested, this decoder assumes that the form parameter name use "." as deliminator for levels.
162     * E.g. For a field named "bar" inside a nested class under the field "foo",
163     * the parameter name is "foo.bar".
164     */
165   def nestedOptional[A: FormDataDecoder](key: String): FormDataDecoder[Option[A]] =
166     nestedEither(key).optional
167 
168   def nestedEither[A](
169       key: String
170   )(implicit
171       fdd: FormDataDecoder[A]
172   ): FormDataDecoder[Either[String, A]] =
173     apply[FormData, A](extractPrefix(key + "."))(fdd.apply)
174 
175   def chain[A: FormDataDecoder](key: String): FormDataDecoder[Chain[A]] =
176     chainEither(key).mapValidated(_.fold(_ => Valid(Chain.empty), Valid(_)))
177 
178   def chainOf[A](key: String)(qd: QueryParamDecoder[A]): FormDataDecoder[Chain[A]] =
179     apply(data =>
180       data
181         .get(key + "[]")
182         .orElse(data.get(key))
183         .getOrElse(Chain.empty)
184         .traverse(v => qd.decode(QueryParameterValue(v)))
185     )
186 
187   def chainEither[A](
188       key: String
189   )(implicit A: FormDataDecoder[A]): FormDataDecoder[Either[String, Chain[A]]] =
190     apply[FormData, Chain[A]](extractPrefix(key + "[]."))(
191       _.toList
192         .map { case (k, cv) =>
193           cv.map(v => List((k, Chain(v))))
194         }
195         .reduceOption { (left, right) =>
196           left.zipWith(right)(_ ++ _)
197         }
198         .getOrElse(Chain.empty)
199         .map(_.toMap)
200         .traverse(d => A(d))
201     )
202 
203   /** For repeated nested values, assuming that the form parameter name use "[]." as a suffix
204     * E.g. "foos[].bar"
205     */
206   def list[A: FormDataDecoder](key: String): FormDataDecoder[List[A]] =
207     chain(key).map(_.toList)
208 
209   /** For repeated primitive values, assuming that the form parameter name use "[]" as a suffix
210     * E.g. "foos[]"
211     */
212   def listOf[A](key: String)(implicit A: QueryParamDecoder[A]): FormDataDecoder[List[A]] =
213     chainOf(key)(A).map(_.toList)
214 
215   private def nonEmptyFields(data: FormData): FormData =
216     data.filter(_._2.nonEmpty)
217 
218   private def extractPrefix(
219       prefix: String
220   )(data: FormData): Either[String, FormData] = {
221     val extracted = nonEmptyFields(data).toList.mapFilter { case (k, v) =>
222       if (k.startsWith(prefix))
223         Some((k.stripPrefix(prefix), v))
224       else None
225     }.toMap
226 
227     if (extracted.isEmpty)
228       Left(s"There are no keys that starts with $prefix")
229     else Right(extracted)
230   }
231 
232   implicit val formDataDecoderInstances: Applicative[FormDataDecoder] =
233     new Applicative[FormDataDecoder] {
234 
235       def pure[A](a: A): FormDataDecoder[A] =
236         apply(_ => Valid(a))
237 
238       override def map[A, B](fa: FormDataDecoder[A])(f: A => B): FormDataDecoder[B] =
239         apply(data => fa(data).map(f))
240 
241       def ap[A, B](
242           ff: FormDataDecoder[A => B]
243       )(fa: FormDataDecoder[A]): FormDataDecoder[B] =
244         apply { data =>
245           fa(data).ap(ff(data))
246         }
247     }
