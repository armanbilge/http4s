62   // https://www.iana.org/assignments/http-parameters/http-parameters.xml#transfer-coding
63   val chunked: TransferCoding = new TransferCodingImpl("chunked")
64   val compress: TransferCoding = new TransferCodingImpl("compress")
65   val deflate: TransferCoding = new TransferCodingImpl("deflate")

67   val identity: TransferCoding = new TransferCodingImpl("identity")
68 
69   def parse(s: String): ParseResult[TransferCoding] =

71 
72   def parseList(s: String): ParseResult[NonEmptyList[TransferCoding]] =
73     ParseResult.fromParser(Rfc7230.headerRep1(parser), "Invalid transfer coding list")(s)
74 
75   private[http4s] val parser: Parser[TransferCoding] = {
76     import cats.parse.Parser.{ignoreCase, oneOf}
77     oneOf(
78       List(
79         ignoreCase("chunked").as(chunked),

81         ignoreCase("deflate").as(deflate),
82         ignoreCase("gzip").as(gzip),
83         ignoreCase("identity").as(identity),
84       )
85     )
86   }

98         writer << coding.coding
