22   /** Performs an operation using a resource, and then releases the resource,
23     * even if the operation throws an exception. This method behaves similarly
24     * to Java's try-with-resources.
25     * Ported from the Scala's 2.13 [[scala.util.Using.resource]].
26     *
27     * @param resource the resource
28     * @param body     the operation to perform with the resource
29     * @tparam R the type of the resource
30     * @tparam A the return type of the operation
31     * @return the result of the operation, if neither the operation nor
32     *         releasing the resource throws
33     */
34   private[http4s] def resource[R <: AutoCloseable, A](resource: R)(body: R => A): A = {
35     if (resource == null) throw new NullPointerException("null resource")
36 
37     var toThrow: Throwable = null
38 
39     try body(resource)
40     catch {
41       case t: Throwable =>
42         toThrow = t
43         null.asInstanceOf[A]
44     } finally
45       if (toThrow eq null) resource.close()
46       else {
47         try resource.close()
48         finally throw toThrow
49       }
50   }
51 }
