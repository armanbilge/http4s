40   /** Silences System.out and System.err streams for the duration of thunk.
41     * Restores the original streams before exiting.
42     */
43   def silenceOutputStreams[R](thunk: => R): R =
44     synchronized {
45       val originalOut = System.out
46       val originalErr = System.err
47 
48       // Redirect output to dummy stream
49       val fakeOutStream = new PrintStream(NullOutStream)
50       val fakeErrStream = new PrintStream(NullOutStream)
51       System.setOut(fakeOutStream)
52       System.setErr(fakeErrStream)
53       try thunk
54       finally {
55         // Set back the original streams
56         System.setOut(originalOut)
57         System.setErr(originalErr)
58       }
59     }
60 
61   /** Returns an ErrorHandler that does not log
62     */
63   def silentErrorHandler[F[_], G[_]](implicit
64       F: Monad[F]
65   ): Request[F] => PartialFunction[Throwable, F[Response[G]]] =
66     req => {
67       case mf: MessageFailure =>
68         mf.toHttpResponse[G](req.httpVersion).pure[F]
69       case NonFatal(_) =>
70         F.pure(
71           Response(
72             Status.InternalServerError,
73             req.httpVersion,
74             Headers(
75               Connection.close,
76               `Content-Length`.zero,
77             ),
78           )
79         )
80     }
81 
82   /** Silences `System.err`, only printing the output in case exceptions are
83     * thrown by the executed `thunk`.
84     */
85   def silenceSystemErr[A](thunk: => A): A =
86     synchronized {
87       // Silencing System.err
88       val oldErr = System.err
89       val outStream = new ByteArrayOutputStream()
90       val fakeErr = new PrintStream(outStream)
91       System.setErr(fakeErr)
92       try {
93         val result = thunk
94         System.setErr(oldErr)
95         result
96       } catch {
97         case NonFatal(e) =>
98           System.setErr(oldErr)
99           // In case of errors, print whatever was caught
100           fakeErr.close()
101           val out = outStream.toString("utf-8")
102           if (out.nonEmpty) oldErr.println(out)
103           throw e
104       }
105     }
106 
107   /** Catches `System.err` output, for testing purposes.
108     */
109   def catchSystemErr(thunk: => Unit): String =
110     synchronized {
111       val oldErr = System.err
112       val outStream = new ByteArrayOutputStream()
113       val fakeErr = new PrintStream(outStream)
114       System.setErr(fakeErr)
115       try thunk
116       finally {
117         System.setErr(oldErr)
118         fakeErr.close()
119       }
120       outStream.toString("utf-8")
121     }
