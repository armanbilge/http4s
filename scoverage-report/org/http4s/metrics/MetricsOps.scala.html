25 /** Describes an algebra capable of writing metrics to a metrics registry
26   */
27 trait MetricsOps[F[_]] {
28 
29   /** Increases the count of active requests
30     *
31     * @param classifier the classifier to apply
32     */
33   def increaseActiveRequests(classifier: Option[String]): F[Unit]
34 
35   /** Decreases the count of active requests
36     *
37     * @param classifier the classifier to apply
38     */
39   def decreaseActiveRequests(classifier: Option[String]): F[Unit]
40 
41   /** Records the time to receive the response headers
42     *
43     * @param method the http method of the request
44     * @param elapsed the time to record
45     * @param classifier the classifier to apply
46     */
47   def recordHeadersTime(method: Method, elapsed: Long, classifier: Option[String]): F[Unit]
48 
49   /** Records the time to fully consume the response, including the body
50     *
51     * @param method the http method of the request
52     * @param status the http status code of the response
53     * @param elapsed the time to record
54     * @param classifier the classifier to apply
55     */
56   def recordTotalTime(
57       method: Method,
58       status: Status,
59       elapsed: Long,
60       classifier: Option[String],
61   ): F[Unit]
62 
63   /** Record abnormal terminations, like errors, timeouts or just other abnormal terminations.
64     *
65     * @param elapsed the time to record
66     * @param terminationType the type of termination
67     * @param classifier the classifier to apply
68     */
69   def recordAbnormalTermination(
70       elapsed: Long,
71       terminationType: TerminationType,
72       classifier: Option[String],
73   ): F[Unit]
74 
75   /** Transform the effect of MetricOps using the supplied natural transformation
76     *
77     * @param fk natural transformation
78     * @tparam G the effect to transform to
79     * @return a new metric ops in the transformed effect
80     */
81   def mapK[G[_]](fk: F ~> G): MetricsOps[G] = {
82     val ops = this
83     new MetricsOps[G] {
84       override def increaseActiveRequests(classifier: Option[String]): G[Unit] = fk(
85         ops.increaseActiveRequests(classifier)
86       )
87       override def decreaseActiveRequests(classifier: Option[String]): G[Unit] = fk(
88         ops.decreaseActiveRequests(classifier)
89       )
90       override def recordHeadersTime(
91           method: Method,
92           elapsed: Long,
93           classifier: Option[String],
94       ): G[Unit] = fk(ops.recordHeadersTime(method, elapsed, classifier))
95       override def recordTotalTime(
96           method: Method,
97           status: Status,
98           elapsed: Long,
99           classifier: Option[String],
100       ): G[Unit] = fk(ops.recordTotalTime(method, status, elapsed, classifier))
101       override def recordAbnormalTermination(
102           elapsed: Long,
103           terminationType: TerminationType,
104           classifier: Option[String],
105       ): G[Unit] = fk(ops.recordAbnormalTermination(elapsed, terminationType, classifier))
106     }
107   }
108 }

110 object MetricsOps {
111 
112   /** Given an exclude function, return a 'classifier' function, i.e. for application in
113     * org.http4s.server/client.middleware.Metrics#apply.
114     *
115     * Let's say you want a classifier that excludes integers since your paths consist of:
116     *   * GET    /users/{integer} = GET_users_*
117     *   * POST   /users           = POST_users
118     *   * PUT    /users/{integer} = PUT_users_*
119     *   * DELETE /users/{integer} = DELETE_users_*
120     *
121     * In such a case, we could use:
122     *
123     * classifierFMethodWithOptionallyExcludedPath(
124     *   exclude          = { str: String => scala.util.Try(str.toInt).isSuccess },
125     *   excludedValue    = "*",
126     *   intercalateValue = "_"
127     * )
128     *
129     * Chris Davenport notes the following on performance considerations of exclude's function value:
130     *
131     * > It's worth noting that this runs on every segment of a path. So note that if an intermediate Throwables with
132     * > Stack traces is known and discarded, there may be a performance penalty, such as the above example with Try(str.toInt).
133     * > I benchmarked some approaches and regex matches should generally be preferred over Throwable's
134     * > in this position.
135     *
136     * @param exclude For a given String, namely a path value, determine whether the value gets excluded.
137     * @param excludedValue Indicates the String value to be supplied for an excluded path's field.
138     * @param pathSeparator Value to use for separating the metrics fields' values
139     * @return Request[F] => Option[String]
140     */
141   def classifierFMethodWithOptionallyExcludedPath[F[_]](
142       exclude: String => Boolean,
143       excludedValue: String = "*",
144       pathSeparator: String = "_",
145   ): Request[F] => Option[String] = { (request: Request[F]) =>
146     val initial: String = request.method.name
147 
148     val pathList: List[String] =
149       request.pathInfo.segments.map(_.decoded()).toList
150 
151     val minusExcluded: List[String] = pathList.map { (value: String) =>
152       if (exclude(value)) excludedValue else value
153     }
154 
155     val result: String =
156       minusExcluded match {
157         case Nil => initial
158         case nonEmpty @ _ :: _ =>
159           initial + pathSeparator + Foldable[List]
160             .intercalate(nonEmpty, pathSeparator)
161       }
162 
163     Some(result)
164   }
165 }

171   // scalafix:off Http4sGeneralLinters; bincompat until 1.0

173   /** Signals just a generic abnormal termination */

175 
176   /** Signals cancelation */
177   case object Canceled extends TerminationType
178
