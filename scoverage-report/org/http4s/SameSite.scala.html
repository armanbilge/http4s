23 /** RFC6265 SameSite cookie attribute values.
24   */
25 sealed trait SameSite extends Renderable {
26   override def render(writer: Writer): writer.type = {
27     val str = this match {
28       case Strict => "Strict"
29       case Lax => "Lax"
30       case None => "None"
31     }
32     writer.append(str)
33   }
34 }

37   case object Strict extends SameSite
38   case object Lax extends SameSite
39   case object None extends SameSite
