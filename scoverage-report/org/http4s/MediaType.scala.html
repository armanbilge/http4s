111   private[http4s] val mediaTypeExtensionParser: Parser[(String, String)] = {
112     import Parser.char
113     import org.http4s.internal.parsing.Rfc7230.{ows, quotedString, token}
114 
115     val escapedString = "\\\\"
116     val unescapedString = "\\"
117 
118     (char(';') *> ows *> token ~ (char('=') *> token.orElse(quotedString)).?).map {
119       case (s: String, s2: Option[String]) =>
120         (s, s2.map(_.replace(escapedString, unescapedString)).getOrElse(""))
121     }
122   }

124   private[http4s] val parser: Parser[MediaRange] = mediaRangeParser(getMediaRange)

126   private[http4s] val fullParser: Parser[MediaRange] = {
127     val extensions = MediaRange.mediaTypeExtensionParser.rep0
128 
129     (parser ~ extensions).map { case (mr, exts) =>
130       exts match {
131         case Nil => mr
132         case _ => mr.withExtensions(exts.toMap)
133       }
134     }
135   }

137   private[http4s] def renderExtensions(sb: Writer, mr: MediaRange): Unit =
138     mr.extensions.foreach { case (k, v) => sb << ';' << ' ' << k << '=' <<# v }
139 
140   private[http4s] def extensionsToString(mr: MediaRange): String = {
141     val sw = new StringWriter()
142     renderExtensions(sw, mr)
143     sw.result
144   }

146   private[http4s] def mediaRangeParser[A](builder: (String, String) => A): Parser[A] = {
147     import Parser.string
148     import org.http4s.internal.parsing.Rfc7230.token
149 
150     val anyStr1 = string("*")
151 
152     string("*/*")
153       .as(("*", "*"))
154       .orElse(
155         (token <* string("/")) ~ anyStr1.as("*").orElse(token)
156       )
157       .orElse(
158         anyStr1.as(("*", "*"))
159       )
160       .map { case (s1: String, s2: String) =>
161         builder(s1, s2)
162       }
163   }

199 sealed class MediaType(
200     mainType: String,
201     val subType: String,
202     val compressible: Boolean = false,
203     val binary: Boolean = false,
204     val fileExtensions: List[String] = Nil,
205     extensions: Map[String, String] = Map.empty,
206 ) extends MediaRange(mainType, extensions) {
207   override def withExtensions(ext: Map[String, String]): MediaType =
208     new MediaType(mainType, subType, compressible, binary, fileExtensions, ext)
209 
210   final def satisfies(mediaType: MediaType): Boolean = mediaType.satisfiedBy(this)
211 
212   override def satisfiedBy(mediaType: MediaRange): Boolean =
213     mediaType match {
214       case mediaType: MediaType =>
215         (this eq mediaType) ||
216         mainType === mediaType.mainType &&
217         subType === mediaType.subType
218 
219       case _ => false
220     }
221 
222   override def equals(obj: Any): Boolean =
223     obj match {
224       case x: MediaType =>
225         (this eq x) ||
226         mainType === x.mainType &&
227         subType === x.subType &&
228         extensions === x.extensions
229       case _ => false
230     }
231 
232   private[this] var hash = 0
233   override def hashCode(): Int = {
234     if (hash == 0)
235       hash = MurmurHash3.mixLast(
236         mainType.##,
237         MurmurHash3.mix(
238           subType.##,
239           MurmurHash3.mix(
240             compressible.##,
241             MurmurHash3.mix(binary.##, MurmurHash3.mix(fileExtensions.##, extensions.##)),
242           ),
243         ),
244       )
245     hash
246   }
247 
248   override def toString: String =
249     s"MediaType($mainType/$subType${MediaRange.extensionsToString(this)})"
