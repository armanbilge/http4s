37 /** A type that can be used to decode a [[Message]]
38   * EntityDecoder is used to attempt to decode a [[Message]] returning the
39   * entire resulting A. If an error occurs it will result in a failed effect.
40   * The default decoders provided here are not streaming, but one could implement
41   * a streaming decoder by having the value of A be some kind of streaming construct.
42   *
43   * @tparam T result type produced by the decoder
44   */
45 @implicitNotFound(
46   "Cannot decode into a value of type ${T}, because no EntityDecoder[${F}, ${T}] instance could be found."
47 )
48 trait EntityDecoder[F[_], T] { self =>
49 
50   /** Attempt to decode the body of the [[Message]] */
51   def decode(m: Media[F], strict: Boolean): DecodeResult[F, T]
52 
53   /** The [[MediaRange]]s this [[EntityDecoder]] knows how to handle */
54   def consumes: Set[MediaRange]
55 
56   /** Make a new [[EntityDecoder]] by mapping the output result */
57   def map[T2](f: T => T2)(implicit F: Functor[F]): EntityDecoder[F, T2] =
58     new EntityDecoder[F, T2] {
59       override def consumes: Set[MediaRange] = self.consumes
60 
61       override def decode(m: Media[F], strict: Boolean): DecodeResult[F, T2] =
62         self.decode(m, strict).map(f)
63     }
64 
65   def flatMapR[T2](f: T => DecodeResult[F, T2])(implicit F: Monad[F]): EntityDecoder[F, T2] =
66     new EntityDecoder[F, T2] {
67       override def decode(m: Media[F], strict: Boolean): DecodeResult[F, T2] =
68         self.decode(m, strict).flatMap(f)
69 
70       override def consumes: Set[MediaRange] = self.consumes
71     }
72 
73   def handleError(f: DecodeFailure => T)(implicit F: Functor[F]): EntityDecoder[F, T] =
74     transform {
75       case Left(e) => Right(f(e))
76       case r @ Right(_) => r
77     }
78 
79   def handleErrorWith(
80       f: DecodeFailure => DecodeResult[F, T]
81   )(implicit F: Monad[F]): EntityDecoder[F, T] =
82     transformWith {
83       case Left(e) => f(e)
84       case Right(r) => DecodeResult.successT(r)
85     }
86 
87   def bimap[T2](f: DecodeFailure => DecodeFailure, s: T => T2)(implicit
88       F: Functor[F]
89   ): EntityDecoder[F, T2] =
90     transform {
91       case Left(e) => Left(f(e))
92       case Right(r) => Right(s(r))
93     }
94 
95   def transform[T2](
96       t: Either[DecodeFailure, T] => Either[DecodeFailure, T2]
97   )(implicit F: Functor[F]): EntityDecoder[F, T2] =
98     new EntityDecoder[F, T2] {
99       override def consumes: Set[MediaRange] = self.consumes
100 
101       override def decode(m: Media[F], strict: Boolean): DecodeResult[F, T2] =
102         self.decode(m, strict).transform(t)
103     }
104 
105   def biflatMap[T2](f: DecodeFailure => DecodeResult[F, T2], s: T => DecodeResult[F, T2])(implicit
106       F: Monad[F]
107   ): EntityDecoder[F, T2] =
108     transformWith {
109       case Left(e) => f(e)
110       case Right(r) => s(r)
111     }
112 
113   def transformWith[T2](
114       f: Either[DecodeFailure, T] => DecodeResult[F, T2]
115   )(implicit F: Monad[F]): EntityDecoder[F, T2] =
116     new EntityDecoder[F, T2] {
117       override def consumes: Set[MediaRange] = self.consumes
118 
119       override def decode(m: Media[F], strict: Boolean): DecodeResult[F, T2] =
120         DecodeResult(
121           F.flatMap(self.decode(m, strict).value)(r => f(r).value)
122         )
123     }
124 
125   /** Combine two [[EntityDecoder]]'s
126     *
127     * The new [[EntityDecoder]] will first attempt to determine if it can perform the decode,
128     * and if not, defer to the second [[EntityDecoder]]
129     *
130     * @param other backup [[EntityDecoder]]
131     */
132   def orElse[T2 >: T](other: EntityDecoder[F, T2])(implicit F: Functor[F]): EntityDecoder[F, T2] =
133     widen[T2] <+> other
134 
135   /** true if this [[EntityDecoder]] knows how to decode the provided [[MediaType]] */
136   def matchesMediaType(mediaType: MediaType): Boolean =
137     consumes.exists(_.satisfiedBy(mediaType))
138 
139   def widen[T2 >: T]: EntityDecoder[F, T2] =
140     this.asInstanceOf[EntityDecoder[F, T2]]
141 }

143 /** EntityDecoder is used to attempt to decode an [[EntityBody]]
144   * This companion object provides a way to create `new EntityDecoder`s along
145   * with some commonly used instances which can be resolved implicitly.
146   */
147 object EntityDecoder {
148   // This is not a real media type but will still be matched by `*/*`
149   private val UndefinedMediaType = new MediaType("UNKNOWN", "UNKNOWN")
150 
151   /** summon an implicit [[EntityDecoder]] */
152   def apply[F[_], T](implicit ev: EntityDecoder[F, T]): EntityDecoder[F, T] = ev
153 
154   implicit def semigroupKForEntityDecoder[F[_]: Functor]: SemigroupK[EntityDecoder[F, *]] =
155     new SemigroupK[EntityDecoder[F, *]] {
156       override def combineK[T](
157           a: EntityDecoder[F, T],
158           b: EntityDecoder[F, T],
159       ): EntityDecoder[F, T] =
160         new EntityDecoder[F, T] {
161           override def decode(m: Media[F], strict: Boolean): DecodeResult[F, T] = {
162             val mediaType = m.contentType.fold(UndefinedMediaType)(_.mediaType)
163 
164             if (a.matchesMediaType(mediaType))
165               a.decode(m, strict)
166             else
167               b.decode(m, strict).leftMap {
168                 case MediaTypeMismatch(actual, expected) =>
169                   MediaTypeMismatch(actual, expected ++ a.consumes)
170                 case MediaTypeMissing(expected) =>
171                   MediaTypeMissing(expected ++ a.consumes)
172                 case other => other
173               }
174           }
175 
176           override def consumes: Set[MediaRange] = a.consumes ++ b.consumes
177         }
178     }
179 
180   /** Create a new [[EntityDecoder]]
181     *
182     * The new [[EntityDecoder]] will attempt to decode messages of type `T`
183     * only if the [[Message]] satisfies the provided [[MediaRange]].
184     *
185     * Exceptions thrown by `f` are not caught.  Care should be taken that
186     * recoverable errors are returned as a [[DecodeResult#failure]], or that
187     * system errors are raised in `F`.
188     */
189   def decodeBy[F[_]: Applicative, T](r1: MediaRange, rs: MediaRange*)(
190       f: Media[F] => DecodeResult[F, T]
191   ): EntityDecoder[F, T] =
192     new EntityDecoder[F, T] {
193       override def decode(m: Media[F], strict: Boolean): DecodeResult[F, T] =
194         if (strict)
195           m.contentType match {
196             case Some(c) if matchesMediaType(c.mediaType) => f(m)
197             case Some(c) => DecodeResult.failureT(MediaTypeMismatch(c.mediaType, consumes))
198             case None if matchesMediaType(UndefinedMediaType) => f(m)
199             case None => DecodeResult.failureT(MediaTypeMissing(consumes))
200           }
201         else
202           f(m)
203 
204       override val consumes: Set[MediaRange] = (r1 +: rs).toSet
205     }
206 
207   /** Helper method which simply gathers the body into a single Chunk */
208   def collectBinary[F[_]: Concurrent](m: Media[F]): DecodeResult[F, Chunk[Byte]] =
209     DecodeResult.success(m.body.chunks.compile.to(Chunk).map(_.flatten))
210 
211   /** Helper method which simply gathers the body into a single ByteVector */
212   private def collectByteVector[F[_]: Concurrent](m: Media[F]): DecodeResult[F, ByteVector] =
213     DecodeResult.success(m.body.compile.to(ByteVector))
214 
215   /** Decodes a message to a String */
216   def decodeText[F[_]](
217       m: Media[F]
218   )(implicit F: Concurrent[F], defaultCharset: Charset = `UTF-8`): F[String] =
219     m.bodyText.compile.string

223   /** Provides a mechanism to fail decoding */
224   def error[F[_], T](t: Throwable)(implicit F: Concurrent[F]): EntityDecoder[F, T] =
225     new EntityDecoder[F, T] {
226       override def decode(m: Media[F], strict: Boolean): DecodeResult[F, T] =
227         DecodeResult(m.body.compile.drain *> F.raiseError(t))
228       override def consumes: Set[MediaRange] = Set.empty
229     }

237   implicit def byteVector[F[_]: Concurrent]: EntityDecoder[F, ByteVector] =

240   implicit def text[F[_]](implicit
241       F: Concurrent[F],
242       defaultCharset: Charset = `UTF-8`,
243   ): EntityDecoder[F, String] =
244     EntityDecoder.decodeBy(MediaRange.`text/*`)(msg =>

274   implicit def multipart[F[_]: Concurrent]: EntityDecoder[F, Multipart[F]] =
275     MultipartDecoder.decoder

277   /** Multipart decoder that streams all parts past a threshold
278     * (anything above `maxSizeBeforeWrite`) into a temporary file.
279     * The decoder is only valid inside the `Resource` scope; once
280     * the `Resource` is released, all the created files are deleted.
281     *
282     * Note that no files are deleted until the `Resource` is released.
283     * Thus, sharing and reusing the resulting `EntityDecoder` is not
284     * recommended, and can lead to disk space leaks.
285     *
286     * The intended way to use this is as follows:
287     *
288     * {{{
289     * mixedMultipartResource[F]()
290     *   .flatTap(request.decodeWith(_, strict = true))
291     *   .use { multipart =>
292     *     // Use the decoded entity
293     *   }
294     * }}}
295     *
296     * @param headerLimit the max size for the headers, in bytes. This is required as
297     *                    headers are strictly evaluated and parsed.
298     * @param maxSizeBeforeWrite the maximum size of a particular part before writing to a file is triggered
299     * @param maxParts the maximum number of parts this decoder accepts. NOTE: this also may mean that a body that doesn't
300     *                 conform perfectly to the spec (i.e isn't terminated properly) but has a lot of parts might
301     *                 be parsed correctly, despite the total body being malformed due to not conforming to the multipart
302     *                 spec. You can control this by `failOnLimit`, by setting it to true if you want to raise
303     *                 an error if sending too many parts to a particular endpoint
304     * @param failOnLimit Fail if `maxParts` is exceeded _during_ multipart parsing.
305     * @param chunkSize the size of chunks created when reading data from temporary files.
306     * @return A supervised multipart decoder.
307     */
308   def mixedMultipartResource[F[_]: Concurrent: Files](
309       headerLimit: Int = 1024,
310       maxSizeBeforeWrite: Int = 52428800,
311       maxParts: Int = 50,
312       failOnLimit: Boolean = false,
313       chunkSize: Int = 8192,
314   ): Resource[F, EntityDecoder[F, Multipart[F]]] =
315     MultipartDecoder.mixedMultipartResource(
316       headerLimit,
317       maxSizeBeforeWrite,
318       maxParts,
319       failOnLimit,
320       chunkSize,
321     )
322 
323   /** Multipart decoder that streams all parts past a threshold
324     * (anything above maxSizeBeforeWrite) into a temporary file.
325     *
326     * Note: (BIG NOTE) Using this decoder for multipart decoding is good for the sake of
327     * not holding all information in memory, as it will never have more than
328     * `maxSizeBeforeWrite` in memory before writing to a temporary file. On top of this,
329     * you can gate the # of parts to further stop the quantity of parts you can have.
330     * That said, because after a threshold it writes into a temporary file, given
331     * bincompat reasons on 0.18.x, there is no way to make a distinction about which `Part[F]`
332     * is a stream reference to a file or not. Thus, consumers using this decoder
333     * should drain all `Part[F]` bodies if they were decoded correctly. That said,
334     * this decoder gives you more control about how many part bodies it parses in the first place, thus you can have
335     * more fine-grained control about how many parts you accept.
336     *
337     * @param headerLimit the max size for the headers, in bytes. This is required as
338     *                    headers are strictly evaluated and parsed.
339     * @param maxSizeBeforeWrite the maximum size of a particular part before writing to a file is triggered
340     * @param maxParts the maximum number of parts this decoder accepts. NOTE: this also may mean that a body that doesn't
341     *                 conform perfectly to the spec (i.e isn't terminated properly) but has a lot of parts might
342     *                 be parsed correctly, despite the total body being malformed due to not conforming to the multipart
343     *                 spec. You can control this by `failOnLimit`, by setting it to true if you want to raise
344     *                 an error if sending too many parts to a particular endpoint
345     * @param failOnLimit Fail if `maxParts` is exceeded _during_ multipart parsing.
346     * @return A multipart/form-data encoded vector of parts with some part bodies held in
347     *         temporary files.
348     */
349   @deprecated("Use mixedMultipartResource", "0.23")
350   def mixedMultipart[F[_]: Concurrent: Files](
351       headerLimit: Int = 1024,
352       maxSizeBeforeWrite: Int = 52428800,
353       maxParts: Int = 50,
354       failOnLimit: Boolean = false,
355   ): EntityDecoder[F, Multipart[F]] =
356     MultipartDecoder.mixedMultipart(headerLimit, maxSizeBeforeWrite, maxParts, failOnLimit)
