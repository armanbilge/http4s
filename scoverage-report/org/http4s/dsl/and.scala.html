19 /** A conjunction extractor.  Generally used as an infix operator.
20   *
21   * {{{
22   * scala> import org.http4s.dsl.&
23   * scala> object Even { def unapply(i: Int) = (i % 2) == 0 }
24   * scala> object Positive { def unapply(i: Int) = i > 0 }
25   * scala> def describe(i: Int) = i match {
26   *      |   case Even() & Positive() => "even and positive"
27   *      |   case Even() => "even but not positive"
28   *      |   case Positive() => "positive but not even"
29   *      |   case _ => "neither even nor positive"
30   *      | }
31   * scala> describe(-1)
32   * res0: String = neither even nor positive
33   * scala> describe(0)
34   * res1: String = even but not positive
35   * scala> describe(1)
36   * res2: String = positive but not even
37   * scala> describe(2)
38   * res3: String = even and positive
39   * }}}
40   */
41 object & {
42   def unapply[A](a: A): Some[(A, A)] = Some((a, a))
