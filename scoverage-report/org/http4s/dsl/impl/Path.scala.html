201 /** Matrix path variable extractor
202   * For an example see [[https://www.w3.org/DesignIssues/MatrixURIs.html MatrixURIs]]
203   * This is useful for representing a resource that may be addressed in multiple dimensions where order is unimportant
204   *
205   * {{{
206   *
207   *    object BoardVar extends MatrixVar("square", List("x", "y"))
208   *    Path("/board/square;x=5;y=3") match {
209   *      case Root / "board" / BoardVar(IntVar(x), IntVar(y)) => ...
210   *    }
211   * }}}
212   */
213 abstract class MatrixVar[F[_]: Foldable](name: String, domain: F[String]) {
214   private val domainList = domain.toList
215 
216   def unapplySeq(str: String): Option[Seq[String]] =
217     if (str.nonEmpty) {
218       val firstSemi = str.indexOf(';')
219       if (firstSemi < 0 && (domain.nonEmpty || name != str)) None
220       else if (firstSemi < 0 && name == str) Some(Seq.empty[String])
221       // Matrix segment didn't match the expected name
222       else if (str.substring(0, firstSemi) != name) None
223       else {
224         val assocListOpt =
225           if (firstSemi >= 0)
226             Monad[Option].tailRecM(MatrixVar.RecState(str, firstSemi + 1, List.empty))(toAssocList)
227           else Some(List.empty[(String, String)])
228         assocListOpt.flatMap { assocList =>
229           domainList.traverse(dom => assocList.find(_._1 == dom).map(_._2))
230         }
231       }
232     } else None
233 
234   private def toAssocList(
235       recState: MatrixVar.RecState
236   ): Option[Either[MatrixVar.RecState, List[(String, String)]]] =
237     // We can't extract anything else but there was a trailing ;
238     if (recState.position >= recState.str.length - 1)
239       Some(Right(recState.accumulated))
240     else {
241       val nextSplit = recState.str.indexOf(';', recState.position)
242       // This is the final ; delimited segment
243       if (nextSplit < 0)
244         toAssocListElem(recState.str, recState.position, recState.str.length)
245           .map(elem => Right(elem :: recState.accumulated))
246       // An internal empty ; delimited segment so just skip
247       else if (nextSplit == recState.position)
248         Some(Left(recState.copy(position = nextSplit + 1)))
249       else
250         toAssocListElem(recState.str, recState.position, nextSplit)
251           .map(elem =>
252             Left(
253               recState.copy(position = nextSplit + 1, accumulated = elem :: recState.accumulated)
254             )
255           )
256     }
257 
258   private def toAssocListElem(str: String, position: Int, end: Int): Option[(String, String)] = {
259     val delimSplit = str.indexOf('=', position)
260     val nextDelimSplit = str.indexOf('=', delimSplit + 1)
261     // if the segment does not contain an = inside then it is invalid
262     if (delimSplit < 0 || delimSplit === position || delimSplit >= end) None
263     // if the segment contains multiple = then it is invalid
264     else if (nextDelimSplit < end && nextDelimSplit >= 0) None
265     else Some(str.substring(position, delimSplit) -> str.substring(delimSplit + 1, end))
266   }
267 }
