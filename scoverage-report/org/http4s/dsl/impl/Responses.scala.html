225       extends AnyVal
226       with EmptyResponseGenerator[F, G] {
227     override def headers(header: Header.ToRaw, _headers: Header.ToRaw*)(implicit
228         F: Applicative[F]
229     ): F[Response[G]] =
230       F.pure(
231         Response(
232           ResetContent,
233           headers = Headers(`Content-Length`.zero) ++ Headers(header :: _headers.toList),
234         )
235       )
236 
237     override def apply()(implicit F: Applicative[F]): F[Response[G]] =
238       F.pure(Response[G](ResetContent, headers = Headers(List(`Content-Length`.zero))))
