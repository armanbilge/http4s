38 /**  Helper for the generation of a [[org.http4s.Response]] which will not contain a body
39   *
40   * While it is possible to for the [[org.http4s.Response]] manually, the EntityResponseGenerators
41   * offer shortcut syntax to make intention clear and concise.
42   *
43   * @example {{{
44   * val resp: F[Response] = Status.Continue()
45   * }}}
46   */
47 trait EmptyResponseGenerator[F[_], G[_]] extends Any with ResponseGenerator {
48   def apply()(implicit F: Applicative[F]): F[Response[G]] = F.pure(Response[G](status))
49 
50   def headers(header: Header.ToRaw, _headers: Header.ToRaw*)(implicit
51       F: Applicative[F]
52   ): F[Response[G]] =
53     F.pure(Response[G](status, headers = Headers(header :: _headers.toList)))
54 }

56 /** Helper for the generation of a [[org.http4s.Response]] which may contain a body
57   *
58   * While it is possible to construct the [[org.http4s.Response]]
59   * manually, the EntityResponseGenerators offer shortcut syntax to
60   * make intention clear and concise.
61   *
62   * @example {{{
63   * val resp: IO[Response] = Ok("Hello world!")
64   * }}}
65   */
66 trait EntityResponseGenerator[F[_], G[_]] extends Any with ResponseGenerator {
67   def liftG: FunctionK[G, F]
68 
69   def apply()(implicit F: Applicative[F]): F[Response[G]] =
70     F.pure(Response[G](status, headers = Headers(List(`Content-Length`.zero))))
71 
72   def headers(header: Header.ToRaw, _headers: Header.ToRaw*)(implicit
73       F: Applicative[F]
74   ): F[Response[G]] =
75     F.pure(
76       Response[G](
77         status,
78         headers = Headers(`Content-Length`.zero) ++ Headers(header :: _headers.toList),
79       )
80     )
81 
82   def apply[A](body: G[A])(implicit F: Monad[F], w: EntityEncoder[G, A]): F[Response[G]] =
83     F.flatMap(liftG(body))(apply[A](_))
84 
85   def apply[A](body: A, headers: Header.ToRaw*)(implicit
86       F: Applicative[F],
87       w: EntityEncoder[G, A],
88   ): F[Response[G]] = {
89     val h = w.headers |+| Headers(headers)
90     val entity = w.toEntity(body)
91     F.pure(Response[G](status = status, headers = addEntityLength(entity, h), body = entity.body))
92   }
93 }

95 /** Helper for the generation of a [[org.http4s.Response]] which may contain
96   * a Location header and may contain a body.
97   *
98   * A 300, 301, 302, 303, 307 and 308 status SHOULD contain a Location header, which
99   * distinguishes this from other `EntityResponseGenerator`s.
100   */
