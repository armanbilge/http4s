31   /** Multipart decoder that streams all parts past a threshold
32     * (anything above `maxSizeBeforeWrite`) into a temporary file.
33     * The decoder is only valid inside the `Resource` scope; once
34     * the `Resource` is released, all the created files are deleted.
35     *
36     * Note that no files are deleted until the `Resource` is released.
37     * Thus, sharing and reusing the resulting `EntityDecoder` is not
38     * recommended, and can lead to disk space leaks.
39     *
40     * The intended way to use this is as follows:
41     *
42     * {{{
43     * mixedMultipartResource[F]()
44     *   .flatTap(request.decodeWith(_, strict = true))
45     *   .use { multipart =>
46     *     // Use the decoded entity
47     *   }
48     * }}}
49     *
50     * @param headerLimit the max size for the headers, in bytes. This is required as
51     *                    headers are strictly evaluated and parsed.
52     * @param maxSizeBeforeWrite the maximum size of a particular part before writing to a file is triggered
53     * @param maxParts the maximum number of parts this decoder accepts. NOTE: this also may mean that a body that doesn't
54     *                 conform perfectly to the spec (i.e isn't terminated properly) but has a lot of parts might
55     *                 be parsed correctly, despite the total body being malformed due to not conforming to the multipart
56     *                 spec. You can control this by `failOnLimit`, by setting it to true if you want to raise
57     *                 an error if sending too many parts to a particular endpoint
58     * @param failOnLimit Fail if `maxParts` is exceeded _during_ multipart parsing.
59     * @param chunkSize the size of chunks created when reading data from temporary files.
60     * @return A multipart/form-data encoded vector of parts with some part bodies held in
61     *         temporary files.
62     */
63   def mixedMultipartResource[F[_]: Concurrent: Files](
64       headerLimit: Int = 1024,
65       maxSizeBeforeWrite: Int = 52428800,
66       maxParts: Int = 50,
67       failOnLimit: Boolean = false,
68       chunkSize: Int = 8192,
69   ): Resource[F, EntityDecoder[F, Multipart[F]]] =
70     Supervisor[F].map { supervisor =>
71       makeDecoder(
72         MultipartParser.parseToPartsSupervisedFile[F](
73           supervisor,
74           _,
75           headerLimit,
76           maxSizeBeforeWrite,
77           maxParts,
78           failOnLimit,
79           chunkSize,
80         )
81       )
82     }
83 
84   /** Multipart decoder that streams all parts past a threshold
85     * (anything above maxSizeBeforeWrite) into a temporary file.
86     *
87     * Note: (BIG NOTE) Using this decoder for multipart decoding is good for the sake of
88     * not holding all information in memory, as it will never have more than
89     * `maxSizeBeforeWrite` in memory before writing to a temporary file. On top of this,
90     * you can gate the # of parts to further stop the quantity of parts you can have.
91     * That said, because after a threshold it writes into a temporary file, given
92     * bincompat reasons on 0.18.x, there is no way to make a distinction about which `Part[F]`
93     * is a stream reference to a file or not. Thus, consumers using this decoder
94     * should drain all `Part[F]` bodies if they were decoded correctly. That said,
95     * this decoder gives you more control about how many part bodies it parses in the first place, thus you can have
96     * more fine-grained control about how many parts you accept.
97     *
98     * @param headerLimit the max size for the headers, in bytes. This is required as
99     *                    headers are strictly evaluated and parsed.
100     * @param maxSizeBeforeWrite the maximum size of a particular part before writing to a file is triggered
101     * @param maxParts the maximum number of parts this decoder accepts. NOTE: this also may mean that a body that doesn't
102     *                 conform perfectly to the spec (i.e isn't terminated properly) but has a lot of parts might
103     *                 be parsed correctly, despite the total body being malformed due to not conforming to the multipart
104     *                 spec. You can control this by `failOnLimit`, by setting it to true if you want to raise
105     *                 an error if sending too many parts to a particular endpoint
106     * @param failOnLimit Fail if `maxParts` is exceeded _during_ multipart parsing.
107     * @return A multipart/form-data encoded vector of parts with some part bodies held in
108     *         temporary files.
109     */
110   @deprecated("Use mixedMultipartResource", "0.23")
111   def mixedMultipart[F[_]: Concurrent: Files](
112       headerLimit: Int = 1024,
113       maxSizeBeforeWrite: Int = 52428800,
114       maxParts: Int = 50,
115       failOnLimit: Boolean = false,
116   ): EntityDecoder[F, Multipart[F]] =
117     makeDecoder(
118       MultipartParser.parseToPartsStreamedFile[F](
119         _,
120         headerLimit,
121         maxSizeBeforeWrite,
122         maxParts,
123         failOnLimit,
124       )
125     )
126 
127   private def makeDecoder[F[_]: Concurrent](
128       impl: Boundary => Pipe[F, Byte, Part[F]]
129   ): EntityDecoder[F, Multipart[F]] =
130     EntityDecoder.decodeBy(MediaRange.`multipart/*`) { msg =>
131       msg.contentType.flatMap(_.mediaType.extensions.get("boundary")) match {
132         case Some(boundary) =>
133           DecodeResult {
134             msg.body
135               .through(impl(Boundary(boundary)))
136               .compile
137               .toVector
138               .map[Either[DecodeFailure, Multipart[F]]](parts =>
139                 Right(Multipart(parts, Boundary(boundary)))
140               )
141               .handleError {
142                 case e: InvalidMessageBodyFailure => Left(e)
143                 case e => Left(InvalidMessageBodyFailure("Invalid multipart body", Some(e)))
144               }
145           }
146         case None =>
147           DecodeResult.failureT(
148             InvalidMessageBodyFailure("Missing boundary extension to Content-Type")
149           )
150       }
151     }
