34 final case class Part[F[_]](headers: Headers, body: Stream[F, Byte]) extends Media[F] {
35   def name: Option[String] = headers.get[`Content-Disposition`].flatMap(_.parameters.get(ci"name"))
36   def filename: Option[String] =
37     headers.get[`Content-Disposition`].flatMap(_.parameters.get(ci"filename"))
38 
39   override def covary[F2[x] >: F[x]]: Part[F2] = this.asInstanceOf[Part[F2]]
40 }
41 
42 object Part {
43   private val ChunkSize = 8192
44 
45   def formData[F[_]](name: String, value: String, headers: Header.ToRaw*): Part[F] =
46     Part(
47       Headers(`Content-Disposition`("form-data", Map(ci"name" -> name))).put(headers: _*),
48       Stream.emit(value).through(utf8.encode),
49     )
50 
51   @deprecated("Use overload with fs2.io.file.Path", "0.23.5")
52   def fileData[F[_]: Files](name: String, file: File, headers: Header.ToRaw*): Part[F] =
53     fileData(name, Path.fromNioPath(file.toPath), headers: _*)
54 
55   def fileData[F[_]: Files](name: String, path: Path, headers: Header.ToRaw*): Part[F] =
56     fileData(
57       name,
58       path.fileName.toString,
59       Files[F].readAll(path, ChunkSize, Flags.Read),
60       headers: _*
61     )
62 
63   def fileData[F[_]: Sync](name: String, resource: URL, headers: Header.ToRaw*): Part[F] =
64     fileData(name, resource.getPath.split("/").last, resource.openStream(), headers: _*)
65 
66   def fileData[F[_]](
67       name: String,
68       filename: String,
69       entityBody: EntityBody[F],
70       headers: Header.ToRaw*
71   ): Part[F] =
72     Part(
73       Headers(
74         `Content-Disposition`("form-data", Map(ci"name" -> name, ci"filename" -> filename)),
75         "Content-Transfer-Encoding" -> "binary",
76       ).put(headers: _*),
77       entityBody,
78     )
79 
80   // The InputStream is passed by name, and we open it in the by-name
81   // argument in callers, so we can avoid lifting into an effect.  Exposing
82   // this API publicly would invite unsafe use, and the `EntityBody` version
83   // should be safe.
84   private def fileData[F[_]](
85       name: String,
86       filename: String,
87       in: => InputStream,
88       headers: Header.ToRaw*
89   )(implicit F: Sync[F]): Part[F] =
90     fileData(name, filename, readInputStream(F.delay(in), ChunkSize), headers: _*)
91 }
