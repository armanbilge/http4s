25 private[http4s] class MultipartEncoder[F[_]] extends EntityEncoder[F, Multipart[F]] {
26   // TODO: Refactor encoders to create headers dependent on value.
27   def headers: Headers = Headers.empty
28 
29   def toEntity(mp: Multipart[F]): Entity[F] =
30     Entity(renderParts(mp.boundary)(mp.parts), None)
31 
32   val dash: String = "--"
33 
34   val dashBoundary: Boundary => String =
35     boundary => s"$dash${boundary.value}"
36 
37   val delimiter: Boundary => String =
38     boundary => s"${Boundary.CRLF}$dash${boundary.value}"
39 
40   // The close-delimiter does not require a trailing CRLF, but adding
41   // one makes it more robust with real implementations.  The wasted
42   // two bytes go into the "epilogue", which the recipient is to
43   // ignore.
44   val closeDelimiter: Boundary => String =
45     boundary => s"${delimiter(boundary)}$dash${Boundary.CRLF}"
46 
47   val start: Boundary => Chunk[Byte] = boundary =>
48     new ChunkWriter()
49       .append(dashBoundary(boundary))
50       .append(Boundary.CRLF)
51       .toChunk
52 
53   val end: Boundary => Chunk[Byte] = boundary =>
54     new ChunkWriter()
55       .append(closeDelimiter(boundary))
56       .toChunk
57 
58   /** encapsulation := delimiter CRLF body-part
59     */
60   val encapsulationWithoutBody: Boundary => String = boundary =>
61     s"${Boundary.CRLF}${dashBoundary(boundary)}${Boundary.CRLF}"
62 
63   val renderHeaders: Headers => Chunk[Byte] = headers =>
64     headers.headers
65       .foldLeft(new ChunkWriter()) { (chunkWriter, header) =>
66         chunkWriter
67           .append(header)
68           .append(Boundary.CRLF)
69       }
70       .toChunk
71 
72   def renderPart(prelude: Chunk[Byte])(part: Part[F]): Stream[F, Byte] =
73     Stream.chunk(prelude) ++
74       Stream.chunk(renderHeaders(part.headers)) ++
75       Stream.chunk(Chunk.array(Boundary.CRLF.getBytes(StandardCharsets.UTF_8))) ++
76       part.body
77 
78   def renderParts(boundary: Boundary)(parts: Vector[Part[F]]): Stream[F, Byte] =
79     if (parts.isEmpty)
80       Stream.empty
81     else
82       parts.tail
83         .foldLeft(renderPart(start(boundary))(parts.head)) { (acc, part) =>
84           acc ++
85             renderPart(
86               Chunk.array(encapsulationWithoutBody(boundary).getBytes(StandardCharsets.UTF_8))
87             )(part)
88         } ++ Stream.chunk(end(boundary))
89 }
