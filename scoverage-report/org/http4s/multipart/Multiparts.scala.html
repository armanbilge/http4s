24 /** An algebra for creating multipart values and boundaries.
25   *
26   * A single instance may be shared by the entire application.
27   */
28 trait Multiparts[F[_]] {
29 
30   /** Generate a random multipart boundary */
31   def boundary: F[Boundary]
32 
33   /** Generate a multipart value from Parts with a random multipart boundary */
34   def multipart(parts: Vector[Part[F]]): F[Multipart[F]]
35 }
36 
37 object Multiparts {
38 
39   /** Creates a `cats.effect.std.Random` and provides Multiparts around it.
40     * This instance can be shared, or is cheap enough to create closer
41     * to where Multipart values are generated.
42     */
43   def forSync[F[_]](implicit F: Sync[F]): F[Multiparts[F]] =
44     Random.scalaUtilRandom[F].map(fromRandom[F])
45 
46   def fromRandom[F[_]](random: Random[F])(implicit F: Functor[F]): Multiparts[F] =
47     new Multiparts[F] {
48       def boundary: F[Boundary] =
49         random.nextBytes(30).map(Boundary.unsafeFromBytes)
50 
51       def multipart(parts: Vector[Part[F]]): F[Multipart[F]] =
52         F.map(boundary)(Multipart(parts, _))
53     }
54 }
