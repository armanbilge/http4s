39 /** Collection representation of a query string
40   *
41   * It is a indexed sequence of key and maybe a value pairs which maps
42   * precisely to a query string, modulo
43   * [[https://datatracker.ietf.org/doc/html/rfc3986#section-2.1 percent-encoding]].
44   *
45   * When rendered, the resulting `String` will have the pairs separated
46   * by '&' while the key is separated from the value with '='
47   */
48 final class Query private (value: Either[Vector[KeyValue], String])
49     extends QueryOps
50     with Renderable {
51   private[this] var _pairs: Vector[KeyValue] = null
52 
53   def pairs: Vector[KeyValue] = {
54     if (_pairs == null) {
55       _pairs = value.fold(identity, Query.parse)
56     }
57     _pairs
58   }
59 
60   private def this(vec: Vector[KeyValue]) = this(Left(vec))
61 
62   def apply(idx: Int): KeyValue = pairs(idx)
63 
64   def length: Int = pairs.length
65 
66   def slice(from: Int, until: Int): Query = new Query(Left(pairs.slice(from, until)))
67 
68   def isEmpty: Boolean = pairs.isEmpty
69 
70   def nonEmpty: Boolean = pairs.nonEmpty
71 
72   def drop(n: Int): Query = new Query(Left(pairs.drop(n)))
73 
74   def dropRight(n: Int): Query = new Query(Left(pairs.dropRight(n)))
75 
76   def exists(f: KeyValue => Boolean): Boolean =
77     pairs.exists(f)
78 
79   def filterNot(f: KeyValue => Boolean): Query =
80     new Query(Left(pairs.filterNot(f)))
81 
82   def filter(f: KeyValue => Boolean): Query =
83     new Query(Left(pairs.filter(f)))
84 
85   def foreach(f: KeyValue => Unit): Unit =
86     pairs.foreach(f)
87 
88   def foldLeft[Z](z: Z)(f: (Z, KeyValue) => Z): Z =
89     pairs.foldLeft(z)(f)
90 
91   def foldRight[Z](z: Eval[Z])(f: (KeyValue, Eval[Z]) => Eval[Z]): Eval[Z] =

93 
94   def +:(elem: KeyValue): Query =
95     new Query(Left(elem +: pairs))
96 
97   def :+(elem: KeyValue): Query =

100   def ++(pairs: collection.Iterable[(String, Option[String])]): Query =

103   def toVector: Vector[(String, Option[String])] = pairs
104 
105   def toList: List[(String, Option[String])] = toVector.toList
106 
107   /** Render the Query as a `String`.
108     *
109     * Pairs are separated by '&' and keys are separated from values by '='
110     */
111   override def render(writer: Writer): writer.type = value.fold(
112     { pairs =>
113       var first = true
114 
115       def encode(s: String) =
116         UriCoding.encode(
117           s,
118           spaceIsPlus = false,
119           charset = StandardCharsets.UTF_8,
120           toSkip = UriCoding.QueryNoEncode,
121         )
122 
123       pairs.foreach {
124         case (n, None) =>
125           if (!first) writer.append('&')
126           else first = false
127           writer.append(encode(n))
128 
129         case (n, Some(v)) =>
130           if (!first) writer.append('&')
131           else first = false
132           writer
133             .append(encode(n))
134             .append("=")
135             .append(encode(v))
136       }
137       writer
138     },
139     raw => writer.append(raw),
140   )
141 
142   /** Map[String, String] representation of the [[Query]]
143     *
144     * If multiple values exist for a key, the first is returned. If
145     * none exist, the empty `String` "" is returned.
146     */
147   lazy val params: Map[String, String] =
148     CollectionCompat.mapValues(multiParams)(_.headOption.getOrElse(""))
149 
150   /** Map[String, Seq[String]] representation of the [[Query]]
151     *
152     * Params are represented as a `Seq[String]` and may be empty.
153     */
154   lazy val multiParams: Map[String, immutable.Seq[String]] =
155     if (toVector.isEmpty) Map.empty
156     else {
157       val m = mutable.Map.empty[String, ListBuffer[String]]
158       toVector.foreach {
159         case (k, None) => m.getOrElseUpdate(k, new ListBuffer)
160         case (k, Some(v)) => m.getOrElseUpdate(k, new ListBuffer) += v
161       }
162       CollectionCompat.mapValues(m.toMap)(_.toList)
163     }
164 
165   override def equals(that: Any): Boolean =
166     that match {
167       case that: Query => that.toVector == toVector
168       case _ => false
169     }

181 object Query {
182   type KeyValue = (String, Option[String])
183 
184   /** Represents the absence of a query string. */
185   val empty: Query = new Query(Vector.empty)
186 
187   /** Represents a query string with no keys or values: `?` */
188   val blank = new Query(Vector("" -> None))
189 
190   def apply(xs: (String, Option[String])*): Query =
191     new Query(xs.toVector)
192 
193   def fromVector(xs: Vector[(String, Option[String])]): Query =
194     new Query(xs)
195 
196   def fromPairs(xs: (String, String)*): Query =
197     new Query(
198       xs.toList.foldLeft(Vector.empty[KeyValue]) { case (m, (k, s)) =>
199         m :+ (k -> Some(s))
200       }
201     )
202 
203   /** Generate a [[Query]] from its `String` representation
204     *
205     * If parsing fails, the empty [[Query]] is returned
206     */
207   def unsafeFromString(query: String): Query =
208     if (query.isEmpty) new Query(Vector("" -> None))
209     else
210       QueryParser.parseQueryString(query) match {
211         case Right(query) => query
212         case Left(_) => Query.empty
213       }
214 
215   @deprecated(message = "Use unsafeFromString instead", since = "0.22.0-M6")
216   def fromString(query: String): Query =
217     unsafeFromString(query)
218 
219   /** Build a [[Query]] from the `Map` structure */
220   def fromMap(map: collection.Map[String, collection.Seq[String]]): Query =
221     new Query(map.foldLeft(Vector.empty[KeyValue]) {
222       case (m, (k, Seq())) => m :+ (k -> None)
223       case (m, (k, vs)) => vs.toList.foldLeft(m) { case (m, v) => m :+ (k -> Some(v)) }
224     })
225 
226   private def parse(query: String): Vector[KeyValue] =
227     if (query.isEmpty) blank.toVector
228     else
229       QueryParser.parseQueryStringVector(query) match {
230         case Right(query) => query
231         case Left(_) => Vector.empty
232       }
233 
234   /* query       = *( pchar / "/" / "?" )
235    *
236    * These are illegal, but common in the wild.  We will be
237    * "conservative in our sending behavior and liberal in our
238    * receiving behavior", and encode them.
239    */
240   private[http4s] lazy val parser: Parser0[Query] = {
241     import cats.parse.Parser.charIn
242     import Rfc3986.pchar
243 
244     pchar.orElse(charIn("/?[]")).rep0.string.map(pchars => new Query(Right(pchars)))
245   }

250         x.hashCode
251 
252       override def compare(x: Query, y: Query): Int =
253         x.toVector.compare(y.toVector)
254 
255       override def show(a: Query): String =
256         a.renderString
257     }
