25 object AuthedRoutes {
26 
27   /** Lifts a function into an [[AuthedRoutes]].  The application of `run`
28     * is suspended in `F` to permit more efficient combination of
29     * routes via `SemigroupK`.
30     *
31     * @tparam F the effect of the [[AuthedRoutes]]
32     * @tparam T the type of the auth info in the [[AuthedRequest]] accepted by the [[AuthedRoutes]]
33     * @param run the function to lift
34     * @return an [[AuthedRoutes]] that wraps `run`
35     */
36   def apply[T, F[_]](run: AuthedRequest[F, T] => OptionT[F, Response[F]])(implicit
37       F: Monad[F]
38   ): AuthedRoutes[T, F] =
39     Kleisli(req => OptionT(F.unit >> run(req).value))
40 
41   /** Lifts a partial function into an [[AuthedRoutes]].  The application of the
42     * partial function is suspended in `F` to permit more efficient combination
43     * of authed services via `SemigroupK`.
44     *
45     * @tparam F the base effect of the [[AuthedRoutes]]
46     * @param pf the partial function to lift
47     * @return An [[AuthedRoutes]] that returns some [[Response]] in an `OptionT[F, *]`
48     * wherever `pf` is defined, an `OptionT.none` wherever it is not
49     */
50   def of[T, F[_]](pf: PartialFunction[AuthedRequest[F, T], F[Response[F]]])(implicit
51       FA: Monad[F]
52   ): AuthedRoutes[T, F] =
53     Kleisli(req => OptionT(FA.unit >> pf.lift(req).sequence))
54 
55   /** The empty service (all requests fallthrough).
56     *
57     * @tparam T - ignored.
58     * @return
59     */
60   def empty[T, F[_]: Applicative]: AuthedRoutes[T, F] =
61     Kleisli.liftF(OptionT.none)
62 }
