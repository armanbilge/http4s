35 sealed trait CacheDirective extends Product with Renderable {
36   val name: CIString = CIString(productPrefix.replace("$minus", "-"))
37   def value: String = name.toString
38   override def toString: String = value
39   def render(writer: Writer): writer.type = writer.append(value)
40 }

42 /** A registry of cache-directives, as listed in
43   * http://www.iana.org/assignments/http-cache-directives/http-cache-directives.xhtml
44   */
45 object CacheDirective {
46   final case class `max-age`(deltaSeconds: Duration) extends CacheDirective {
47     override def value: String = name.toString + "=" + deltaSeconds.toSeconds
48   }
49 
50   final case class `max-stale`(deltaSeconds: Option[Duration] = None) extends CacheDirective {
51     override def value: String = name.toString + deltaSeconds.fold("")("=" + _.toSeconds)
52   }
53 
54   final case class `min-fresh`(deltaSeconds: Duration) extends CacheDirective {
55     override def value: String = name.toString + "=" + deltaSeconds.toSeconds
56   }
57 
58   case object `must-revalidate` extends CacheDirective
59 
60   final case class `no-cache`(fieldNames: List[CIString] = Nil) extends CacheDirective {
61     override def value: String =
62       name.toString + (if (fieldNames.isEmpty) "" else fieldNames.mkString("=\"", ",", "\""))
63   }
64 
65   case object `no-store` extends CacheDirective
66 
67   case object `no-transform` extends CacheDirective
68 
69   case object `only-if-cached` extends CacheDirective
70 
71   final case class `private`(fieldNames: List[CIString] = Nil) extends CacheDirective {
72     override def value: String =
73       name.toString + (if (fieldNames.isEmpty) "" else fieldNames.mkString("=\"", ",", "\""))
74   }
75 
76   case object `proxy-revalidate` extends CacheDirective
77 
78   case object public extends CacheDirective
79 
80   final case class `s-maxage`(deltaSeconds: Duration) extends CacheDirective {
81     override def value: String = name.toString + "=" + deltaSeconds.toSeconds
82   }
83 
84   final case class `stale-if-error`(deltaSeconds: Duration) extends CacheDirective {
85     override def value: String = name.toString + "=" + deltaSeconds.toSeconds
86   }
87 
88   final case class `stale-while-revalidate`(deltaSeconds: Duration) extends CacheDirective {
89     override def value: String = name.toString + "=" + deltaSeconds.toSeconds
90   }
91 
92   def apply(name: CIString, argument: Option[String] = None): CacheDirective =
93     new CustomCacheDirective(name, argument)
94 
95   def apply(name: String, argument: Option[String]): CacheDirective =
96     apply(CIString(name), argument)
97 
98   def apply(name: String): CacheDirective = apply(name, None)
99 
100   private final case class CustomCacheDirective(
101       override val name: CIString,
102       argument: Option[String] = None,
103   ) extends CacheDirective {
104     override def value: String = name.toString + argument.fold("")("=\"" + _ + '"')
105   }
