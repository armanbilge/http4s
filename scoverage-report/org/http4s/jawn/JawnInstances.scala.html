37   // some decoders may reuse it and avoid extra content negotiation
38   private[http4s] def jawnDecoderImpl[F[_]: Concurrent, J: Facade](
39       m: Media[F]
40   ): DecodeResult[F, J] =
41     DecodeResult {
42       m.body.chunks
43         .parseJson(AsyncParser.SingleValue)
44         .map(Either.right)
45         .handleErrorWith {
46           case pe: ParseException =>
47             Stream.emit(Left(jawnParseExceptionMessage(pe)))
48           case e =>
49             Stream.raiseError[F](e)
50         }
51         .compile
52         .last
53         .map(_.getOrElse(Left(jawnEmptyBodyMessage)))
54     }
55 }
