23 final case class ContextRequest[F[_], A](context: A, req: Request[F]) {
24   def mapK[G[_]](fk: F ~> G): ContextRequest[G, A] =
25     ContextRequest(context, req.mapK(fk))
26 
27   @deprecated("Use context instead", "0.21.0")
28   def authInfo: A = context
29 }

31 object ContextRequest {
32   def apply[F[_]: Functor, T](
33       getContext: Request[F] => F[T]
34   ): Kleisli[F, Request[F], ContextRequest[F, T]] =
35     Kleisli(request => getContext(request).map(ctx => ContextRequest(ctx, request)))
36 
37   implicit def contextRequestInstances[F[_]]: NonEmptyTraverse[ContextRequest[F, *]] =
38     new NonEmptyTraverse[ContextRequest[F, *]] {
39       override def foldLeft[A, B](fa: ContextRequest[F, A], b: B)(f: (B, A) => B): B =
40         f(b, fa.context)
41       override def foldRight[A, B](fa: ContextRequest[F, A], lb: Eval[B])(
42           f: (A, Eval[B]) => Eval[B]
43       ): Eval[B] =
44         f(fa.context, lb)
45       override def nonEmptyTraverse[G[_]: Apply, A, B](fa: ContextRequest[F, A])(
46           f: A => G[B]

48         f(fa.context).map(b => ContextRequest(b, fa.req))
49       def reduceLeftTo[A, B](fa: ContextRequest[F, A])(f: A => B)(g: (B, A) => B): B =
50         f(fa.context)
51       def reduceRightTo[A, B](fa: ContextRequest[F, A])(f: A => B)(
52           g: (A, Eval[B]) => Eval[B]
53       ): Eval[B] =
54         Eval.later(f(fa.context))
55     }
