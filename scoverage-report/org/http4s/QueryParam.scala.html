299 /** Type class defining how to decode a [[QueryParameterValue]] into a `T`
300   * @see QueryParamCodecLaws
301   */
302 trait QueryParamDecoder[T] { outer =>
303   def decode(value: QueryParameterValue): ValidatedNel[ParseFailure, T]
304 
305   /** QueryParamDecoder is a covariant functor. */
306   def map[U](f: T => U): QueryParamDecoder[U] =
307     new QueryParamDecoder[U] {
308       override def decode(value: QueryParameterValue): ValidatedNel[ParseFailure, U] =
309         outer.decode(value).map(f)
310     }
311 
312   /** Use another decoder if this one fails. */
313   def orElse[U >: T](qpd: QueryParamDecoder[U]): QueryParamDecoder[U] =
314     new QueryParamDecoder[U] {
315       override def decode(value: QueryParameterValue): ValidatedNel[ParseFailure, U] =
316         outer.decode(value).orElse(qpd.decode(value))
317     }
318 
319   /** Validate the currently parsed value a function to Either[ParseFailure, *]. */
320   def emap[U](f: T => Either[ParseFailure, U]): QueryParamDecoder[U] =
321     emapValidatedNel(f.andThen(_.toValidatedNel))
322 
323   /** Validate the currently parsed value using a function to ValidatedNel[ParseFailure, *]. */
324   def emapValidatedNel[U](f: T => ValidatedNel[ParseFailure, U]): QueryParamDecoder[U] =
325     new QueryParamDecoder[U] {
326       override def decode(value: QueryParameterValue): ValidatedNel[ParseFailure, U] =
327         outer.decode(value).andThen(f)
328     }
329 }

338   def fromUnsafeCast[T](cast: QueryParameterValue => T)(typeName: String): QueryParamDecoder[T] =
339     new QueryParamDecoder[T] {
340       def decode(value: QueryParameterValue): ValidatedNel[ParseFailure, T] =
341         Validated
342           .catchNonFatal(cast(value))
343           .leftMap(t => ParseFailure(s"Query decoding $typeName failed", t.getMessage))
344           .toValidatedNel
345     }
346 
347   /** QueryParamDecoder is a covariant functor. */
348   implicit val FunctorQueryParamDecoder: Functor[QueryParamDecoder] =
349     new Functor[QueryParamDecoder] {
350       override def map[A, B](fa: QueryParamDecoder[A])(f: A => B): QueryParamDecoder[B] =
351         fa.map(f)
352     }
353 
354   /** QueryParamDecoder is a MonoidK. */
355   implicit val PlusEmptyQueryParamDecoder: MonoidK[QueryParamDecoder] =
356     new MonoidK[QueryParamDecoder] {
357       def empty[A]: QueryParamDecoder[A] =
358         fail[A]("Decoding failed.", "Empty decoder (always fails).")
359       def combineK[A](a: QueryParamDecoder[A], b: QueryParamDecoder[A]): QueryParamDecoder[A] =
360         a.orElse(b)
361     }
362 
363   /** A decoder that always succeeds. */

462       formatter: DateTimeFormatter,
