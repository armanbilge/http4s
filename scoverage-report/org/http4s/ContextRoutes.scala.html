25 object ContextRoutes {
26 
27   /** Lifts a function into an [[ContextRoutes]].  The application of `run`
28     * is suspended in `F` to permit more efficient combination of
29     * routes via `SemigroupK`.
30     *
31     * @tparam F the effect of the [[ContextRoutes]]
32     * @tparam T the type of the auth info in the [[ContextRequest]] accepted by the [[ContextRoutes]]
33     * @param run the function to lift
34     * @return an [[ContextRoutes]] that wraps `run`
35     */
36   def apply[T, F[_]](run: ContextRequest[F, T] => OptionT[F, Response[F]])(implicit
37       F: Monad[F]
38   ): ContextRoutes[T, F] =
39     Kleisli(req => OptionT(F.unit >> run(req).value))
40 
41   /** Lifts a partial function into an [[ContextRoutes]].  The application of the
42     * partial function is suspended in `F` to permit more efficient combination
43     * of authed services via `SemigroupK`.
44     *
45     * @tparam F the base effect of the [[ContextRoutes]]
46     * @param pf the partial function to lift
47     * @return An [[ContextRoutes]] that returns some [[Response]] in an `OptionT[F, *]`
48     * wherever `pf` is defined, an `OptionT.none` wherever it is not
49     */
50   def of[T, F[_]](pf: PartialFunction[ContextRequest[F, T], F[Response[F]]])(implicit
51       F: Monad[F]
52   ): ContextRoutes[T, F] =
53     Kleisli(req => OptionT(Applicative[F].unit >> pf.lift(req).sequence))
54 
55   /** Lifts a partial function into an [[ContextRoutes]].  The application of the
56     * partial function is not suspended in `F`, unlike [[of]]. This allows for less
57     * constraints when not combining many routes.
58     *
59     * @tparam F the base effect of the [[ContextRoutes]]
60     * @param pf the partial function to lift
61     * @return A [[ContextRoutes]] that returns some [[Response]] in an `OptionT[F, *]`
62     * wherever `pf` is defined, an `OptionT.none` wherever it is not
63     */
64   def strict[T, F[_]: Applicative](
65       pf: PartialFunction[ContextRequest[F, T], F[Response[F]]]
66   ): ContextRoutes[T, F] =
67     Kleisli(req => OptionT(pf.lift(req).sequence))
68 
69   /** The empty service (all requests fallthrough).
70     *
71     * @tparam T - ignored.
72     * @return
73     */
74   def empty[T, F[_]: Applicative]: ContextRoutes[T, F] =
75     Kleisli.liftF(OptionT.none)
76 }
