28 /** HTTP's version number consists of two decimal digits separated by
29   * a "." (period or decimal point). The first digit ("major version")
30   * indicates the messaging syntax, whereas the second digit ("minor
31   * version") indicates the highest minor version within that major
32   * version to which the sender is conformant (able to understand for
33   * future communication).
34   *
35   * @param major The major version, `0` to `9` inclusive
36   *
37   * @param minor The minor version, `0` to `9` inclusive
38   *
39   * @see
40   * [[https://httpwg.org/http-core/draft-ietf-httpbis-semantics-latest.html#protocol.version
41   * HTTP Semantics, Protocol Versioning]]
42   */
43 // scalafix:off Http4sGeneralLinters.nonValidatingCopyConstructor; bincompat until 1.0
44 final case class HttpVersion private[HttpVersion] (major: Int, minor: Int)
45     extends Renderable
46     with Ordered[HttpVersion] {
47   // scalafix:on
48 
49   /** Renders as an HTTP/1.1 string
50     *
51     * {{{
52     * >>> HttpVersion.`HTTP/1.1`.renderString
53     * HTTP/1.1
54     * }}}
55     */
56   override def render(writer: Writer): writer.type = writer << "HTTP/" << major << '.' << minor
57 
58   /** Orders by major version ascending, then minor version ascending.
59     *
60     * {{{
61     * >>> List(HttpVersion.`HTTP/1.0`, HttpVersion.`HTTP/1.1`, HttpVersion.`HTTP/0.9`).sorted
62     * List(HTTP/0.9, HTTP/1.0, HTTP/1.1)
63     * }}}
64     */
65   override def compare(that: HttpVersion): Int =
66     (this.major, this.minor).compare((that.major, that.minor))
67 
68   @deprecated("Does not range check parameters. Will be removed from public API in 1.0.", "0.22.6")
69   def copy(major: Int = major, minor: Int = minor): HttpVersion =
70     new HttpVersion(major, minor)
71 }

75   /** HTTP/0.9 was first formalized in the HTTP/1.0 spec. `HTTP/0.9`
76     * does not literally appear in the HTTP/0.9 protocol.
77     *
78     * @see [[https://www.w3.org/Protocols/HTTP/AsImplemented.html The

93   /** HTTP/1.1 revises HTTP/1.0, and is currently defined by six RFCs.
94     *
95     * @see [[https://datatracker.ietf.org/doc/html/rfc208 Hypertext
96     * Transfer Protocol -- HTTP/1.1]] (obsolete)
97     *
98     * @see [[https://datatracker.ietf.org/doc/html/rfc2616 Hypertext
99     * Transfer Protocol -- HTTP/1.1]] (obsolete)
100     *
101     * @see [[https://datatracker.ietf.org/doc/html/rfc7230 Hypertext
102     * Transfer Protocol (HTTP/1.1): Message Syntax and Routing]]
103     *
104     * @see [[https://datatracker.ietf.org/doc/html/rfc7231 Hypertext
105     * Transfer Protocol (HTTP/1.1): Semantics and Content]]
106     *
107     * @see [[https://datatracker.ietf.org/doc/html/rfc7232 Hypertext

119     * @see

121     * HTTP/1.1]] (draft)
122     */
123   val `HTTP/1.1`: HttpVersion = new HttpVersion(1, 1)
124 
125   /** HTTP/2 is the second major version of HTTP.  It defines no minor
126     * versions, so minor version `0` is implied.

129     * Hypertext Transfer Protocol Version 2 (HTTP/2)]]
130     */

136   /** HTTP/3 is the third major version of HTTP.  It defines no minor
137     * versions, so minor version `0` is implied.
138     *
139     * @see [[https://quicwg.org/base-drafts/draft-ietf-quic-http.html
140     * Hypertext Transfer Protocol Version 3 (HTTP/3)]] (draft)

144   /** The set of HTTP versions that have a specification */
145   private[http4s] val specified: Set[HttpVersion] = Set(
146     `HTTP/0.9`,

153   private[this] val right_1_0 = Right(`HTTP/1.0`)
154   private[this] val right_1_1 = Right(`HTTP/1.1`)
155 
156   /** Returns an HTTP version from its HTTP/1 string representation.
157     *
158     * {{{
159     * >>> HttpVersion.fromString("HTTP/1.1")
160     * Right(HTTP/1.1)
161     * }}}
162     */
163   def fromString(s: String): ParseResult[HttpVersion] =
164     s match {
165       case "HTTP/1.1" => right_1_1
166       case "HTTP/1.0" => right_1_0
167       case _ =>
168         ParseResult.fromParser(parser, "HTTP version")(s)
169     }
170 
171   private val parser: P[HttpVersion] = {
172     // HTTP-name = %x48.54.54.50 ; HTTP
173     // HTTP-version = HTTP-name "/" DIGIT "." DIGIT
174     val httpVersion = P.string("HTTP/") *> digit ~ (P.char('.') *> digit)
175 
176     httpVersion.map { case (major, minor) =>
177       new HttpVersion(major - '0', minor - '0')
178     }
179   }
180 
181   /** Returns an HTTP version from a major and minor version.
182     *
183     * {{{
184     * >>> HttpVersion.fromVersion(1, 1)
185     * Right(HTTP/1.1)
186     *
187     * >>> HttpVersion.fromVersion(1, 10)
188     * Left(org.http4s.ParseFailure: Invalid HTTP version: major must be <= 9: 10)
189     * }}}
190     *
191     * @param major The major version, `0` to `9` inclusive
192     * @param minor The minor version, `0` to `9` inclusive
193     */
194   def fromVersion(major: Int, minor: Int): ParseResult[HttpVersion] =
195     if (major < 0) ParseResult.fail("Invalid HTTP version", s"major must be > 0: $major")
196     else if (major > 9) ParseResult.fail("Invalid HTTP version", s"major must be <= 9: $major")
197     else if (minor < 0) ParseResult.fail("Invalid HTTP version", s"major must be > 0: $minor")
198     else if (minor > 9) ParseResult.fail("Invalid HTTP version", s"major must be <= 9: $minor")
199     else ParseResult.success(new HttpVersion(major, minor))

219       case HttpVersion(9, 9) => None
220       case HttpVersion(major, minor) if fromVersion(major, minor).isLeft => None
221       case HttpVersion(major, 9) => Some(HttpVersion(major + 1, 0))
222       case HttpVersion(major, minor) => Some(HttpVersion(major, minor + 1))
223     }
224     override def partialPrevious(a: HttpVersion): Option[HttpVersion] = a match {
225       case HttpVersion(0, 0) => None
226       case HttpVersion(major, minor) if fromVersion(major, minor).isLeft => None
227       case HttpVersion(major, 0) => Some(HttpVersion(major - 1, 9))
228       case HttpVersion(major, minor) => Some(HttpVersion(major, minor - 1))
229     }
230     override def order: Order[HttpVersion] = self
231     override def minBound: HttpVersion = HttpVersion(0, 0)
232     override def maxBound: HttpVersion = HttpVersion(9, 9)
233   }
