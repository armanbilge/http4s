24 /** Functions for creating [[HttpRoutes]] kleislis. */
25 object HttpRoutes {
26 
27   /** Lifts a function into an [[HttpRoutes]].  The application of `run`
28     * is suspended in `F` to permit more efficient combination of
29     * routes via `SemigroupK`.
30     *
31     * @tparam F the effect of the [[HttpRoutes]]
32     * @param run the function to lift
33     * @return an [[HttpRoutes]] that wraps `run`
34     */
35   def apply[F[_]: Monad](run: Request[F] => OptionT[F, Response[F]]): HttpRoutes[F] =
36     Http(run)
37 
38   /** Lifts an effectful [[Response]] into an [[HttpRoutes]].
39     *
40     * @tparam F the effect of the [[HttpRoutes]]
41     * @param fr the effectful [[Response]] to lift
42     * @return an [[HttpRoutes]] that always returns `fr`
43     */
44   def liftF[F[_]](fr: OptionT[F, Response[F]]): HttpRoutes[F] =
45     Kleisli.liftF(fr)
46 
47   /** Lifts a [[Response]] into an [[HttpRoutes]].
48     *
49     * @tparam F the base effect of the [[HttpRoutes]]
50     * @param r the [[Response]] to lift
51     * @return an [[HttpRoutes]] that always returns `r` in effect `OptionT[F, *]`
52     */
53   def pure[F[_]](r: Response[F])(implicit FO: Applicative[OptionT[F, *]]): HttpRoutes[F] =
54     Kleisli.pure(r)
55 
56   /** Transforms an [[HttpRoutes]] on its input.  The application of the
57     * transformed function is suspended in `F` to permit more
58     * efficient combination of routes via `SemigroupK`.
59     *
60     * @tparam F the base effect of the [[HttpRoutes]]
61     * @param f a function to apply to the [[Request]]
62     * @param fa the [[HttpRoutes]] to transform
63     * @return An [[HttpRoutes]] whose input is transformed by `f` before
64     * being applied to `fa`
65     */
66   def local[F[_]: Monad](f: Request[F] => Request[F])(fa: HttpRoutes[F]): HttpRoutes[F] =
67     Http.local[OptionT[F, *], F](f)(fa)
68 
69   /** Lifts a partial function into an [[HttpRoutes]].  The application of the
70     * partial function is suspended in `F` to permit more efficient combination
71     * of routes via `SemigroupK`.
72     *
73     * @tparam F the base effect of the [[HttpRoutes]] - Defer suspends evaluation
74     * of routes, so only 1 section of routes is checked at a time.
75     * @param pf the partial function to lift
76     * @return An [[HttpRoutes]] that returns some [[Response]] in an `OptionT[F, *]`
77     * wherever `pf` is defined, an `OptionT.none` wherever it is not
78     */
79   def of[F[_]: Monad](pf: PartialFunction[Request[F], F[Response[F]]]): HttpRoutes[F] =
80     Kleisli(req => OptionT(Applicative[F].unit >> pf.lift(req).sequence))
81 
82   /** Lifts a partial function into an [[HttpRoutes]].  The application of the
83     * partial function is not suspended in `F`, unlike [[of]]. This allows for less
84     * constraints when not combining many routes.
85     *
86     * @tparam F the base effect of the [[HttpRoutes]]
87     * @param pf the partial function to lift
88     * @return An [[HttpRoutes]] that returns some [[Response]] in an `OptionT[F, *]`
89     * wherever `pf` is defined, an `OptionT.none` wherever it is not
90     */
91   def strict[F[_]: Applicative](pf: PartialFunction[Request[F], F[Response[F]]]): HttpRoutes[F] =
92     Kleisli(req => OptionT(pf.lift(req).sequence))
93 
94   /** An empty set of routes.  Always responds with `OptionT.none`.
95     *
96     * @tparam F the base effect of the [[HttpRoutes]]
97     */
98   def empty[F[_]: Applicative]: HttpRoutes[F] = liftF(OptionT.none)
99 }
