44 final case class LanguageTag(
45     primaryTag: String,
46     q: QValue = QValue.One,
47     subTags: List[String] = Nil,
48 ) extends Renderable {
49 
50   def withQValue(q: QValue): LanguageTag = copy(q = q)
51 
52   def render(writer: Writer): writer.type = {
53     writer.append(primaryTag)
54     subTags.foreach(s => writer.append('-').append(s))
55     q.render(writer)
56     writer
57   }
58 
59   @tailrec
60   private def checkLists(tags1: Seq[String], tags2: List[String]): Boolean =
61     if (tags1.isEmpty) true
62     else if (tags2.isEmpty || tags1.head != tags2.head) false
63     else checkLists(tags1.tail, tags2.tail)
64 
65   def matches(languageTag: LanguageTag): Boolean =
66     this.primaryTag == "*" || (this.primaryTag == languageTag.primaryTag &&
67       checkLists(subTags, languageTag.subTags))
