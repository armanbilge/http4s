39   private[http4s] val parser = {
40     sealed trait ValueChar

42     case class EncodedChar(a: Char, b: Char) extends ValueChar

44     val attrChar = Rfc3986.alpha
45       .orElse(Rfc3986.digit)
46       .orElse(Parser.charIn('!', '#', '$', '&', '+', '-', '.', '^', '_', '`', '|', '~'))

50     val pctEncoded = (Parser.string("%") *> Rfc3986.hexdig ~ Rfc3986.hexdig).map {

53     val valueChars = attrChar.orElse(pctEncoded).rep
54     val language = Parser.string(Rfc5234.alpha.rep) ~ (Parser.string("-") *> Rfc2616.token).rep0
55     val charset = Parser.ignoreCase("UTF-8").as(StandardCharsets.UTF_8)
56     val extValue = (Rfc5234.dquote *> Parser.charsWhile0(
57       CharPredicate.All -- '"'
58     ) <* Rfc5234.dquote) | (charset ~ (Parser.string("'") *> language.? <* Parser.string(
59       "'"
60     )) ~ valueChars).map { case ((charset, _), values) =>
61       values
62         .map {

64             val charByte = (Character.digit(a, 16) << 4) + Character.digit(b, 16)
65             new String(Array(charByte.toByte), charset)

68         .toList

70     }

72     val value = Rfc7230.token | Rfc7230.quotedString

74     val parameter = for {

91           def render(writer: Writer): writer.type = {
92             val renderExtFilename =
93               v.parameters.get(ci"filename").exists(!safeChars.matchesAll(_))
94             val withExtParams =
95               if (renderExtFilename && !v.parameters.contains(ci"filename*"))
96                 v.parameters + (ci"filename*" -> v.parameters(ci"filename"))
97               else v.parameters
98             val withExtParamsSorted =
99               if (withExtParams.contains(ci"filename") && withExtParams.contains(ci"filename*"))
