31   def parse(s: String): ParseResult[Cookie] =
32     ParseResult.fromParser(parser, "Invalid Cookie header")(s)
33 
34   private[http4s] val parser: Parser[Cookie] = {
35     import Parser.{char, string}
36 
37     /* cookie-string = cookie-pair *( ";" SP cookie-pair ) */
38     val cookieString = (RequestCookie.parser ~ (string("; ") *> RequestCookie.parser).rep0).map {
39       case (head, tail) =>
40         Cookie(NonEmptyList(head, tail))
41     }
42 
43     /* We also see trailing semi-colons in the wild, and grudgingly tolerate them
44      * here. */
45     cookieString <* char(';').?
46   }
47 
48   val name: CIString = ci"Cookie"
49 
50   implicit val headerInstance: Header[Cookie, Header.Recurring] =
51     Header.createRendered(
52       name,
53       h =>
54         new Renderable {
55           def render(writer: Writer): writer.type =
56             writer.addNel(h.values, sep = "; ")
57         },
58       parse,
59     )
60 
61   implicit val headerSemigroupInstance: cats.Semigroup[Cookie] =
62     (a, b) => Cookie(a.values.concatNel(b.values))
