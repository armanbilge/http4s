31 object Origin {
32   // An Origin header may be the string "null", representing an "opaque origin":
33   // https://stackoverflow.com/questions/42239643/when-does-firefox-set-the-origin-header-to-null-in-post-requests
34   case object Null extends Origin
35 
36   // If the Origin is not "null", it is a non-empty list of Hosts:
37   // https://datatracker.ietf.org/doc/html/rfc6454#section-7
38   final case class HostList(hosts: NonEmptyList[Host]) extends Origin
39 
40   // A host in an Origin header isn't a full URI.
41   // It only contains a scheme, a host, and an optional port.
42   // Hence we re-used parts of the Uri class here, but we don't use a whole Uri:
43   // https://datatracker.ietf.org/doc/html/rfc6454#section-7
44   final case class Host(scheme: Uri.Scheme, host: Uri.Host, port: Option[Int] = None)
45       extends Renderable {
46     def toUri: Uri =
47       Uri(scheme = Some(scheme), authority = Some(Uri.Authority(host = host, port = port)))
48 
49     def render(writer: Writer): writer.type =
50       toUri.render(writer)
51   }
52 
53   private[http4s] val parser: Parser0[Origin] = {
54     import Parser.{`end`, char, string, until}
55     import Rfc5234.{alpha, digit}
56 
57     val unknownScheme =
58       alpha ~ Parser.oneOf(List(alpha, digit, char('+'), char('-'), char('.'))).rep0
59     val http = string("http")
60     val https = string("https")
61     val scheme = List(https, http, unknownScheme)
62       .reduceLeft(_ orElse _)
63       .string
64       .map(Uri.Scheme.unsafeFromString)
65     val stringHost = until(char(':').orElse(`end`)).map(RegName.apply)
66     val bracketedIpv6 = char('[') *> Uri.Parser.ipv6Address <* char(']')
67     val host = List(bracketedIpv6, Uri.Parser.ipv4Address, stringHost).reduceLeft(_ orElse _)
68     val port = char(':') *> digit.rep.string.map(_.toInt)
69     val nullHost = (string("null") *> `end`).as(Origin.Null)
70 
71     val singleHost = ((scheme <* string("://")) ~ host ~ port.?).map { case ((sch, host), port) =>
72       Origin.Host(sch, host, port)
73     }
74 
75     nullHost.orElse(singleHost.repSep(char(' ')).map(hosts => Origin.HostList(hosts)))
76   }
77 
78   def parse(s: String): ParseResult[Origin] =
79     ParseResult.fromParser(parser, "Invalid Origin header")(s)
80 
81   implicit val headerInstance: Header[Origin, Header.Single] =
82     Header.createRendered(
83       ci"Origin",
84       v =>
85         new Renderable {
86           def render(writer: Writer): writer.type =
87             v match {
88               case HostList(hosts) =>
89                 writer << hosts.head
90                 hosts.tail.foreach { host =>
91                   writer << " "
92                   writer << host
93                 }
94                 writer
95               case Null => writer << "null"
96             }
97 
98         },
99       parse,
100     )
