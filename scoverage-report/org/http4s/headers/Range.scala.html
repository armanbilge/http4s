31 object Range {
32   def apply(unit: RangeUnit, r1: SubRange, rs: SubRange*): Range =
33     Range(unit, NonEmptyList.of(r1, rs: _*))
34 
35   def apply(r1: SubRange, rs: SubRange*): Range = apply(RangeUnit.Bytes, r1, rs: _*)
36 
37   def apply(begin: Long, end: Long): Range = apply(SubRange(begin, Some(end)))
38 
39   def apply(begin: Long): Range = apply(SubRange(begin, None))
40 
41   object SubRange {
42     def apply(first: Long): SubRange = SubRange(first, None)
43     def apply(first: Long, second: Long): SubRange = SubRange(first, Some(second))
44   }
45 
46   final case class SubRange(first: Long, second: Option[Long]) extends Renderable {
47 
48     /** Base method for rendering this object efficiently */
49     override def render(writer: Writer): writer.type = {
50       writer << first
51       // the trailing '-' is necessary unless this is a suffix range
52       if (first >= 0)
53         writer << '-'
54       second.foreach(writer << _)
55       writer
56     }
57   }
58 
59   val parser: P0[Range] = {
60     // https://datatracker.ietf.org/doc/html/rfc7233#section-3.1
61 
62     def toLong(s: String): Option[Long] =
63       try Some(s.toLong)
64       catch { case _: NumberFormatException => None }
65 
66     val nonNegativeLong = Numbers.digits
67       .mapFilter(toLong)
68 
69     val negativeLong = (P.char('-') ~ Numbers.digits).string
70       .mapFilter(toLong)
71 
72     // byte-range-spec = first-byte-pos "-" [ last-byte-pos ]
73     val byteRangeSpec = ((nonNegativeLong <* P.char('-')) ~ nonNegativeLong.?)
74       .map { case (first, last) => SubRange(first, last) }
75 
76     // suffix-byte-range-spec = "-" suffix-length
77     val suffixByteRangeSpec = negativeLong.map(SubRange(_))
78 
79     // byte-range-set  = 1#( byte-range-spec / suffix-byte-range-spec )
80     val byteRangeSet = Rfc7230.headerRep1(byteRangeSpec.orElse(suffixByteRangeSpec))
81 
82     // byte-ranges-specifier = bytes-unit "=" byte-range-set
83     val byteRangesSpecifier =
84       ((Rfc7230.token.map(RangeUnit(_)) <* P.char('=')) ~ byteRangeSet)
85         .map { case (unit, ranges) => Range(unit, ranges) }
86 
87     // Range = byte-ranges-specifier / other-ranges-specifier
88     // other types of ranges are not supported, `other-ranges-specifier` is ignored
89     byteRangesSpecifier
90   }
91 
92   def parse(s: String): ParseResult[Range] =
93     ParseResult.fromParser(parser, "Invalid Range header")(s)
94 
95   implicit val headerInstance: Header[Range, Header.Single] =
96     Header.createRendered(
97       ci"Range",
98       h =>
99         new Renderable {
100           def render(writer: Writer): writer.type = {
101             writer << h.unit << '=' << h.ranges.head
102             h.ranges.tail.foreach(writer << ',' << _)
103             writer
104           }
105         },
106       parse,
107     )
108 
109 }
