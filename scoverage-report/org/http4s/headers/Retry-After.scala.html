32   def fromLong(retry: Long): ParseResult[`Retry-After`] =
33     if (retry >= 0) ParseResult.success(new RetryAfterImpl(Right(retry)))
34     else
35       ParseResult.fail(
36         "Invalid retry value",
37         s"Retry param $retry must be more or equal than 0 seconds",

40   def unsafeFromLong(retry: Long): `Retry-After` =
41     fromLong(retry).fold(throw _, identity)

44     fromLong(retry.toSeconds).fold(throw _, identity)

46   def parse(s: String): ParseResult[`Retry-After`] =
47     ParseResult.fromParser(parser, "Invalid Retry-After header")(s)
48 
49   /* `Retry-After = HTTP-date / delay-seconds` */
50   private[http4s] val parser: Parser[`Retry-After`] = {
51     import Rfc5234.digit
52 
53     def httpDate = HttpDate.parser.map(apply)
54 
55     /* delay-seconds  = 1*DIGIT */
56     def delaySeconds = digit.rep.string.map(s => unsafeFromLong(s.toLong))
57 
58     httpDate.orElse(delaySeconds)
59   }
