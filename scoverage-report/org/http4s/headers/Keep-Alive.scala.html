35   private sealed trait Parameter
36   private final case class Timeout(timeoutSeconds: Long) extends Parameter

40   /*
41 Spec Version: https://datatracker.ietf.org/doc/html/draft-thomson-hybi-http-timeout-03 (Note:  Max is deprecated and gone)
42 Keep-Alive           = "Keep-Alive" ":" 1#keep-alive-info
43 keep-alive-info      =   "timeout" "=" delta-seconds
44                        / "max" "=" 1*DIGIT
45                        / keep-alive-extension
46 keep-alive-extension = token [ "=" ( token / quoted-string ) ]
47    */

49   def apply(
50       timeoutSeconds: Option[Long],
51       max: Option[Long],
52       extension: List[(String, Option[String])],
53   ): ParseResult[`Keep-Alive`] =
54     if (timeoutSeconds.isDefined || max.isDefined || extension.nonEmpty) {
55       val reservedTokens = List("token", "max")
56       if (extension.exists(p => reservedTokens.contains(p._1))) {
57         ParseResult.fail(
58           "Invalid Keep-Alive header",
59           s"Reserved token of list $reservedTokens was found in the extensions.",
60         )
61       } else {
62         val validatedTimeoutSeconds = timeoutSeconds.traverse(t => nonNegativeLong(t, "timeout"))
63         val validatedMax = max.traverse(m => nonNegativeLong(m, "max"))
64         (validatedTimeoutSeconds, validatedMax).mapN((t, m) => impl(t, m, extension))
65       }
66     } else {
67       ParseResult.fail("Invalid Keep-Alive header", "All fields of Keep-Alive were empty")
68     }

70   def unsafeApply(
71       timeoutSeconds: Option[Long],
72       max: Option[Long],
73       extension: List[(String, Option[String])],
74   ): `Keep-Alive` =
75     apply(timeoutSeconds, max, extension).fold(throw _, identity)

77   def parse(s: String): ParseResult[`Keep-Alive`] =

80   private def nonNegativeLong(l: Long, fieldName: String): ParseResult[Long] =
81     if (l >= 0) ParseResult.success(l)
82     else
83       ParseResult.fail(
84         s"Invalid long for $fieldName",
85         s"$fieldName which was $l must be greater than or equal to 0 seconds",
86       )

88   private def safeToLong(s: String): Option[Long] =
89     try Some(s.toLong)
90     catch {
91       case _: NumberFormatException => None
92     }
93   private val parser: Parser[`Keep-Alive`] = {
94     import Rfc7230.{headerRep1, quotedString, token}
95     import Numbers.digits
96 
97     // "timeout" "=" delta-seconds
98     val timeout: Parser[Timeout] =
99       Parser.string("timeout=") *> digits.mapFilter(s => safeToLong(s).map(Timeout.apply))
100 
101     // "max" "=" 1*DIGIT
102     val max: Parser[Max] =
103       Parser.string("max=") *> digits.mapFilter(s => safeToLong(s).map(Max.apply))
104 
105     // keep-alive-extension = token [ "=" ( token / quoted-string ) ]
106     val keepAliveExtension: Parser[Extension] =
107       (token ~ (Parser.char('=') *> token.orElse(quotedString)).?).map(Extension.apply)
108 
109     /*
110   keep-alive-info      = "timeout" "=" delta-seconds
111                        / "max" "=" 1*DIGIT
112                        / keep-alive-extension
113      */
114     val keepAliveInfo: Parser[Parameter] = timeout.orElse(max).orElse(keepAliveExtension)
115 
116     headerRep1(keepAliveInfo).map { nel =>
117       var timeoutSeconds: Option[Long] = None
118       var max: Option[Long] = None
119       val extension: ListBuffer[(String, Option[String])] = ListBuffer.empty
120       nel.foldLeft(()) { (_, ka) =>
121         ka match {
122           case Timeout(n) => if (timeoutSeconds.isEmpty) timeoutSeconds = Some(n) else ()
123           case Max(n) => if (max.isEmpty) max = Some(n) else ()
124           case Extension(p) => extension.append(p)
125         }
126       }
127       unsafeApply(timeoutSeconds, max, extension.toList)
128     }
129   }

133       max: Option[Long],
134       extension: List[(String, Option[String])],
135   ) =
