70 /** {{{
71   *   The "Content-Type" header field indicates the media type of the
72   *   associated representation: either the representation enclosed in the
73   *   message payload or the selected representation, as determined by the
74   *   message semantics.
75   * }}}
76   *
77   * [[https://datatracker.ietf.org/doc/html/rfc7231#section-3.1.1.5 RFC-7231]]
78   */
79 final case class `Content-Type` private (mediaType: MediaType, charset: Option[Charset]) { // scalafix:ok; private for API ergonomics, not correctness
80   def withMediaType(mediaType: MediaType): `Content-Type` =
81     if (mediaType != this.mediaType) copy(mediaType = mediaType) else this
82   def withCharset(charset: Charset): `Content-Type` =
83     if (noCharsetDefined || charset != this.charset.get) copy(charset = Some(charset)) else this
84   def withoutDefinedCharset: `Content-Type` =
85     if (isCharsetDefined) copy(charset = None) else this
86 
87   def isCharsetDefined: Boolean = charset.isDefined
88   def noCharsetDefined: Boolean = charset.isEmpty
89 }
