202     /** Parses host and optional port number from the given string according to RFC3986.

206         ParseFailure(s"invalid host '$s'", e.toString)

306   private val parser: P[Forwarded] = {
307     // https://datatracker.ietf.org/doc/html/rfc7239#section-4
308 
309     // A utility so that we can decode multiple pairs and join them in a single element
310     trait Pair {
311       def create: Element
312       def merge(e: Element): Element
313     }
314 
315     object Pair {
316       def apply(c: Element, m: Element => Element): Pair = new Pair {
317         override def create: Element = c
318         override def merge(e: Element): Element = m(e)
319       }
320     }
321 
322     def quoted[A](p: Parser0[A]): P[A] =
323       Rfc7230.token
324         .orElse(Rfc7230.quotedString)
325         .flatMap(str =>
326           p.parseAll(str)
327             .fold(_ => P.fail[A], P.pure)
328         ) // this looks not very good
329 
330     // forwarded-pair = token "=" value
331     // The syntax of a "by" value, after potential quoted-string unescaping
332     // conforms to the "node" ABNF
333     // The syntax of a "for" value, after potential quoted-string
334     // unescaping, conforms to the "node" ABNF
335     // The syntax for a "host" value, after potential quoted-string
336     // unescaping, MUST conform to the Host ABNF described in Section 5.4 of
337     // [RFC7230].
338     // The syntax of a "proto" value, after potential quoted-string
339     // unescaping, MUST conform to the URI scheme name as defined in Section 3.1 in
340     // [RFC3986]
341 
342     val host = Host.parser
343     val proto = Uri.Parser.scheme
344     val node = Node.parser
345 
346     val forwardedPair = P.oneOf(
347       List(
348         Rfc7230.token
349           .flatMap(tok =>
350             tok.toLowerCase(Locale.ROOT) match {
351               case "by" =>
352                 P.char('=') *> quoted(node).map(n => Pair(Element.fromBy(n), _.withBy(n)))
353               case "for" =>
354                 P.char('=') *> quoted(node).map(n => Pair(Element.fromFor(n), _.withFor(n)))
355               case "host" =>
356                 P.char('=') *> quoted(host).map(h => Pair(Element.fromHost(h), _.withHost(h)))
357               case "proto" =>
358                 P.char('=') *> quoted(proto).map(p => Pair(Element.fromProto(p), _.withProto(p)))
359               case other =>
360                 P.failWith(s"expected parameters: 'by', 'for', 'host' or 'proto', but got '$other'")
361             }
362           )
363       )
364     )
365 
366     // forwarded-element = [ forwarded-pair ] *( ";" [ forwarded-pair ] )
367     val forwardedElement =
368       P.repSep(forwardedPair, 1, P.char(';'))
369         .map(pairs => pairs.tail.foldLeft(pairs.head.create)((z, x) => x.merge(z)))
370 
371     // Forwarded = 1#forwarded-element
372     Rfc7230
373       .headerRep1(forwardedElement)
374       .map(Forwarded.apply)
375   }
