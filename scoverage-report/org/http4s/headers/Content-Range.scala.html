28 object `Content-Range` {
29   def apply(range: Range.SubRange, length: Option[Long] = None): `Content-Range` =
30     `Content-Range`(RangeUnit.Bytes, range, length)
31 
32   def apply(start: Long, end: Long): `Content-Range` = apply(Range.SubRange(start, Some(end)), None)
33 
34   def apply(start: Long): `Content-Range` =
35     apply(Range.SubRange(start, None), None)
36 
37   def parse(s: String): ParseResult[`Content-Range`] =
38     ParseResult.fromParser(parser, "Invalid Content-Range header")(s)
39 
40   val parser: P[`Content-Range`] = {
41 
42     val nonNegativeLong = Numbers.digits
43       .mapFilter { ds =>
44         try Some(ds.toLong)
45         catch { case _: NumberFormatException => None }
46       }
47 
48     // byte-range = first-byte-pos "-" last-byte-pos
49     val byteRange = ((nonNegativeLong <* P.char('-')) ~ nonNegativeLong)
50       .map { case (first, last) => SubRange(first, last) }
51 
52     // byte-range-resp = byte-range "/" ( complete-length / "*" )
53     val byteRangeResp =
54       (byteRange <* P.char('/')) ~ nonNegativeLong.map(Some(_)).orElse(P.char('*').as(None))
55 
56     // byte-content-range = bytes-unit SP ( byte-range-resp / unsatisfied-range )
57     // `unsatisfied-range` is not represented
58     val byteContentRange =
59       ((Rfc7230.token.map(RangeUnit(_)) <* P.char(' ')) ~ byteRangeResp)
60         .map { case (unit, (range, length)) => `Content-Range`(unit, range, length) }
61 
62     // Content-Range = byte-content-range / other-content-range
63     // other types of ranges are not supported, `other-content-range` is ignored
64     byteContentRange
65   }
66 
67   val name: CIString = ci"Content-Range"
68 
69   implicit val headerInstance: Header[`Content-Range`, Header.Single] =
70     Header.createRendered(
71       name,
72       h =>
73         new Renderable {
74           def render(writer: Writer): writer.type = {
75             writer << h.unit << ' ' << h.range << '/'
76             h.length match {
77               case Some(l) => writer << l
78               case None => writer << '*'
79             }
80           }
81         },
82       parse,
83     )
84 }
