23 /** Constructs a `Content-Length` header.
24   *
25   * The HTTP RFCs do not specify a maximum length.  We have decided that `Long.MaxValue`
26   * bytes ought to be good enough for anybody in order to avoid the irritations of `BigInt`.
27   *
28   * @param length the length
29   */
30 final case class `Content-Length`(length: Long) {
31   def modify(f: Long => Long): Option[`Content-Length`] =
32     `Content-Length`.fromLong(f(length)).toOption
33 }
34 
35 object `Content-Length` {
36   val zero: `Content-Length` = apply(0)
37 
38   def fromLong(length: Long): ParseResult[`Content-Length`] =
39     if (length >= 0L) ParseResult.success(apply(length))
40     else ParseResult.fail("Invalid Content-Length", length.toString)
41 
42   def unsafeFromLong(length: Long): `Content-Length` =
43     fromLong(length).fold(throw _, identity)
44 
45   def parse(s: String): ParseResult[`Content-Length`] =
46     ParseResult.fromParser(parser, "Invalid Content-Length header")(s)
47 
48   private[http4s] val parser = AdditionalRules.NonNegativeLong.map(fromLong).mapFilter(_.toOption)
49 
50   val name: CIString = ci"Content-Length"
51 
52   implicit val headerInstance: Header[`Content-Length`, Header.Single] =
53     Header.createRendered(
54       name,
55       _.length,
56       parse,
57     )
58 
59 }
