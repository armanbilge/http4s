28 /** Renderers for the [[Forwarded]] header models.
29   */
30 private[http4s] trait ForwardedRenderers {
31   import Forwarded._
32 
33   implicit val http4sForwardedNodeNameRenderer: Renderer[Node.Name] =
34     new Renderer[Node.Name] {
35       override def render(writer: Writer, nodeName: Node.Name): writer.type =
36         nodeName match {
37           case Node.Name.Ipv4(ipv4addr) => writer << ipv4addr.toUriString
38           case Node.Name.Ipv6(ipv6addr) => writer << ipv6addr.toUriString
39           case Node.Name.Unknown => writer << "unknown"
40           case Node.Obfuscated(str) => writer << str
41         }
42     }
43 
44   implicit val http4sForwardedNodePortRenderer: Renderer[Node.Port] = new Renderer[Node.Port] {
45     override def render(writer: Writer, nodePort: Node.Port): writer.type =
46       nodePort match {
47         case Node.Port.Numeric(num) => writer << num
48         case Node.Obfuscated(str) => writer << str
49       }
50   }
51 
52   implicit val http4sForwardedNodeRenderer: Renderer[Node] = new Renderer[Node] {
53     override def render(writer: Writer, node: Node): writer.type = {
54       writer << node.nodeName
55       node.nodePort.fold[writer.type](writer)(writer << ':' << _)
56     }
57   }
58 
59   implicit val http4sForwardedHostRenderer: Renderer[Host] = new Renderer[Host] {
60     // See in `Rfc3986Parser`: `RegName` -> `SubDelims`
61     private val RegNameChars = Uri.Unreserved ++ "!$&'()*+,;="
62 
63     override def render(writer: Writer, host: Host): writer.type = {
64       host.host match {
65         case Uri.RegName(name) =>
66           // TODO: A workaround for #1651, remove when the former issue gets fixed.
67           writer << Uri.encode(name.toString, StandardCharsets.ISO_8859_1, toSkip = RegNameChars)
68         case other =>
69           writer << other
70       }
71       host.port.fold[writer.type](writer)(writer << ':' << _)
72     }
73   }
74 
75   protected def renderElement(writer: Writer, elem: Element): writer.type = {
76 
77     def renderParamEval[A: Renderer](name: String, maybeValue: Option[A]) =
78       maybeValue.map { value =>
79         // Do not write it immediately since we're going to interleave existing parameters with ';'
80         Eval.always[writer.type] { // NOTE: not clear why the explicit type is necessary here
81           writer << name << '='
82 
83           val rendered = Renderer.renderString(value)
84           if (Rfc7230.token.parseAll(rendered).isRight)
85             writer << rendered
86           else
87             writer <<# rendered // quote non-token values
88         }
89       }.toList
90 
91     {
92       import elem._
93       renderParamEval("by", maybeBy) :::
94         renderParamEval("for", maybeFor) :::
95         renderParamEval("host", maybeHost) :::
96         renderParamEval("proto", maybeProto)
97     }.reduceLeft { (leftParamEval, rightParamEval) =>
98       // Interleave every couple of parameters with ';'
99       leftParamEval >> Eval.always(writer << ';') >> rightParamEval
100     }.value // all actual rendering happens here
101   }
102 }
