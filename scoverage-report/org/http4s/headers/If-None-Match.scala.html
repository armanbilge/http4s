26 /** {{{
27   *  The "If-None-Match" header field makes the request method conditional
28   *  on a recipient cache or origin server either not having any current
29   *  representation of the target resource, when the field-value is "*",
30   *  or having a selected representation with an entity-tag that does not
31   *  match any of those listed in the field-value.
32   * }}}
33   *
34   * From [[https://datatracker.ietf.org/doc/html/rfc7232#section-3.2 RFC-7232]]
35   */
36 object `If-None-Match` {
37 
38   /** Match any existing entity */
39   val `*` : `If-None-Match` = `If-None-Match`(None)
40 
41   def apply(first: EntityTag, rest: EntityTag*): `If-None-Match` =
42     `If-None-Match`(Some(NonEmptyList.of(first, rest: _*)))
43 
44   def parse(s: String): ParseResult[`If-None-Match`] =
45     ParseResult.fromParser(parser, "Invalid If-None-Match header")(s)
46 
47   private[http4s] val parser = Parser
48     .string("*")
49     .as(`*`)
50     .orElse(Rfc7230.headerRep1(EntityTag.parser).map { tags =>
51       `If-None-Match`(Some(tags))
52     })
53 
54   implicit val headerInstance: Header[`If-None-Match`, Header.Single] =
55     Header.create(
56       ci"If-None-Match",
57       _.tags match {
58         case None => "*"
59         case Some(tags) => tags.mkString_("", ",", "")
60       },
61       parse,
62     )
63 }
