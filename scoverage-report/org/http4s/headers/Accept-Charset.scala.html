34   private[http4s] val parser: Parser[`Accept-Charset`] = {
35     import cats.parse.Parser._
36     import org.http4s.internal.parsing.Rfc7230._
37 
38     val anyCharset = (char('*') *> QValue.parser)
39       .map(q => if (q != QValue.One) `*`.withQValue(q) else `*`)
40 
41     val fromToken = (token ~ QValue.parser).mapFilter { case (s, q) =>
42       // TODO handle tokens that aren't charsets
43       Charset
44         .fromString(s)
45         .toOption
46         .map { c =>
47           if (q != QValue.One) c.withQuality(q) else c.toRange
48         }
49     }
50 
51     val charsetRange = anyCharset.orElse(fromToken)
52 
53     headerRep1(charsetRange).map(xs => `Accept-Charset`(xs.head, xs.tail: _*))
54   }

67 /** {{{
68   *   The "Accept-Charset" header field can be sent by a user agent to
69   *   indicate what charsets are acceptable in textual response content.
70   *   This field allows user agents capable of understanding more
71   * }}}
72   *
73   * From [https://datatracker.ietf.org/doc/html/rfc7231#section-5.3.3 RFC-7231].
74   */
75 final case class `Accept-Charset`(values: NonEmptyList[CharsetRange]) {
76 
77   def qValue(charset: Charset): QValue = {
78     def specific =
79       values.collectFirst { case cs: Atom if cs.matches(charset) => cs.qValue }
80     def splatted =
81       values.collectFirst { case cs: CharsetRange.`*` => cs.qValue }
82 
83     specific.orElse(splatted).getOrElse(QValue.Zero)
84   }
85 
86   def satisfiedBy(charset: Charset): Boolean = qValue(charset) > QValue.Zero
87 
88   def map(f: CharsetRange => CharsetRange): `Accept-Charset` = `Accept-Charset`(values.map(f))
