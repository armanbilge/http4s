27 object DNT {
28   case object AllowTracking extends DNT("0")
29   case object DisallowTracking extends DNT("1")
30   case object NoPreference extends DNT("null")
31 
32   /*
33    * `DNT = 0 | 1 | null`
34    */
35   private[http4s] val parser: Parser[DNT] = {
36     val nullParser = string("null").as(NoPreference)
37     val allowTrackingParser = char('0').as(AllowTracking)
38     val disallowTrackingParser = char('1').as(DisallowTracking)
39     allowTrackingParser | disallowTrackingParser | nullParser
40   }
41 
42   def parse(s: String): ParseResult[DNT] =
43     ParseResult.fromParser(parser, "Invalid DNT header")(s)
44 
45   implicit val headerInstance: Header[DNT, Header.Single] =
46     Header.create(
47       ci"DNT",
48       _.value,
49       parse,
50     )
