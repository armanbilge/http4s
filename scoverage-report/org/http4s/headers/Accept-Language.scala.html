57 /** Request header used to indicate which natural language would be preferred for the response
58   * to be translated into.
59   *
60   * [[https://datatracker.ietf.org/doc/html/rfc7231#section-5.3.5 RFC-7231 Section 5.3.5]]
61   */
62 final case class `Accept-Language`(values: NonEmptyList[LanguageTag]) {
63   def qValue(languageTag: LanguageTag): QValue =
64     values.toList
65       .collect {
66         case tag: LanguageTag if tag.matches(languageTag) =>
67           if (tag.primaryTag == "*") (0, tag.q)
68           else (1 + tag.subTags.size, tag.q)
69       }
70       .sortBy(-_._1)
71       .headOption
72       .fold(QValue.Zero)(_._2)
73 
74   def satisfiedBy(languageTag: LanguageTag): Boolean = qValue(languageTag) > QValue.Zero
