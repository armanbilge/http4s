28 object Connection {
29   def apply(head: CIString, tail: CIString*): `Connection` =
30     apply(NonEmptyList(head, tail.toList))
31 
32   def parse(s: String): ParseResult[Connection] =
33     ParseResult.fromParser(parser, "Invalid Connection header")(s)
34 
35   val close: Connection = Connection(ci"close")
36 
37   private[http4s] val parser = Rfc7230.headerRep1(Rfc2616.token).map { (xs: NonEmptyList[String]) =>
38     Connection(CIString(xs.head), xs.tail.map(CIString(_)): _*)
39   }
40 
41   implicit val headerInstance: Header[Connection, Header.Recurring] =
42     Header.createRendered(
43       ci"Connection",
44       _.values,
45       parse,
46     )
47 
48   implicit val headerSemigroupInstance: cats.Semigroup[Connection] =
49     (a, b) => Connection(a.values.concatNel(b.values))
50 }
