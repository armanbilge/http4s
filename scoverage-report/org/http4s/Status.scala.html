25 /** Representation of the HTTP response code and reason
26   *
27   * '''Note: ''' the reason is not important to the protocol and is not considered in equality checks.
28   *
29   * @param code HTTP status code
30   * @param reason reason for the response. eg, OK
31   * @see [[https://datatracker.ietf.org/doc/html/rfc7231#section-6 RFC 7231, Section 6, Response Status Codes]]
32   * @see [[http://www.iana.org/assignments/http-status-codes/http-status-codes.xhtml IANA Status Code Registry]]
33   */
34 sealed abstract case class Status private (code: Int)(
35     val reason: String,
36     val isEntityAllowed: Boolean,
37 ) extends Ordered[Status]
38     with Renderable {
39 
40   val responseClass: ResponseClass =
41     if (code < 200) Status.Informational
42     else if (code < 300) Status.Successful
43     else if (code < 400) Status.Redirection
44     else if (code < 500) Status.ClientError
45     else Status.ServerError
46 
47   def compare(that: Status): Int = code - that.code
48 
49   def isSuccess: Boolean = responseClass.isSuccess
50 
51   @deprecated(
52     "Custom status phrases will be removed in 1.0. They are an optional feature, pose a security risk, and already unsupported on some backends.",
53     "0.22.6",
54   )
55   def withReason(reason: String): Status = Status(code, reason, isEntityAllowed)
56 
57   /** A sanitized [[reason]] phrase. Blank if reason is invalid per
58     * RFC7230, otherwise equivalent to reason.
59     */
60   def sanitizedReason: String = ""
61 
62   override def render(writer: org.http4s.util.Writer): writer.type =
63     writer << code << ' ' << sanitizedReason
64 
65   /** Helpers for for matching against a [[Response]] */
66   def unapply[F[_]](msg: Response[F]): Option[Response[F]] =
67     if (msg.status == this) Some(msg) else None
68 }

73   private val ReasonPhrasePredicate =
74     CharPredicate("\t ") ++ CharPredicate(0x21.toChar to 0x7e.toChar) ++ CharPredicate(

100   sealed trait ResponseClass {
101     def isSuccess: Boolean
102 
103     /** Match a [[Response]] based on [[Status]] category */
104     final def unapply[F[_]](resp: Response[F]): Option[Response[F]] =
105       resp match {
106         case Response(status, _, _, _, _) if status.responseClass == this => Some(resp)
107         case _ => None
108       }
109   }

111   case object Informational extends ResponseClass { val isSuccess = true }
112   case object Successful extends ResponseClass { val isSuccess = true }
113   case object Redirection extends ResponseClass { val isSuccess = true }
114   case object ClientError extends ResponseClass { val isSuccess = false }
115   case object ServerError extends ResponseClass { val isSuccess = false }

117   private[http4s] val MinCode = 100
118   private[http4s] val MaxCode = 599
119 
120   def fromInt(code: Int): ParseResult[Status] =
121     withRangeCheck(code) {
122       lookup(code) match {
123         case right: Right[_, _] => right
124         case _ => ParseResult.success(trust(code, ""))
125       }
126     }
127 
128   @deprecated(
129     "Use fromInt. Custom status phrases will be removed in 1.0. They are an optional feature, pose a security risk, and already unsupported on some backends.",
130     "0.22.6",
131   )
132   def fromIntAndReason(code: Int, reason: String): ParseResult[Status] =
133     withRangeCheck(code) {
134       lookup(code, reason) match {
135         case right: Right[_, _] => right
136         case _ => ParseResult.success(Status(code, reason))
137       }
138     }
139 
140   private def withRangeCheck(code: Int)(onSuccess: => ParseResult[Status]): ParseResult[Status] =
141     if (code >= MinCode && code <= MaxCode) onSuccess
142     else ParseResult.fail("Invalid code", s"$code is not between $MinCode and $MaxCode.")
143 
144   private[http4s] def registered: List[Status] = all
145 
146   private object Registry {
147     private val registry: Array[ParseResult[Status]] =
148       Array.fill[ParseResult[Status]](MaxCode + 1) {
149         ParseResult.fail("Unregistered", "Unregistered")
150       }
151 
152     def lookup(code: Int): ParseResult[Status] = registry(code)
153 
154     def lookup(code: Int, reason: String): ParseResult[Status] = {
155       val lookupResult = lookup(code)
156       lookupResult match {
157         case Right(r) if r.reason == reason => lookupResult
158         case _ => ParseResult.fail("Reason did not match", s"Nonstandard reason: $reason")
159       }
160     }
161 
162     def register(status: Status): Status = {
163       registry(status.code) = Right(status)
164       status
165     }
166 
167     def all: List[Status] = registry.collect { case Right(status) => status }.toList
168   }
