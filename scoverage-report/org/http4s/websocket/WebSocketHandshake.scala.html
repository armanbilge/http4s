32 private[http4s] object WebSocketHandshake {
33 
34   /** Creates a new [[ClientHandshaker]] */
35   def clientHandshaker(host: String): ClientHandshaker = new ClientHandshaker(host)
36 
37   /** Provides the initial headers and a 16 byte Base64 encoded random key for websocket connections */
38   class ClientHandshaker(host: String) {
39 
40     /** Randomly generated 16-byte key in Base64 encoded form */
41     val key: String = {
42       val bytes = new Array[Byte](16)
43       Random.nextBytes(bytes)
44       Base64.getEncoder.encodeToString(bytes)
45     }
46 
47     /** Initial headers to send to the server */
48     val initHeaders: List[(String, String)] =
49       ("Host", host) :: ("Sec-WebSocket-Key", key) :: clientBaseHeaders
50 
51     /** Check if the server response is a websocket handshake response */
52     def checkResponse(headers: Iterable[(String, String)]): Either[String, Unit] =
53       if (
54         !headers.exists { case (k, v) =>
55           k.equalsIgnoreCase("Connection") && valueContains("Upgrade", v)
56         }
57       )
58         Left("Bad Connection header")
59       else if (
60         !headers.exists { case (k, v) =>
61           k.equalsIgnoreCase("Upgrade") && v.equalsIgnoreCase("websocket")
62         }
63       )
64         Left("Bad Upgrade header")
65       else
66         headers
67           .find { case (k, _) => k.equalsIgnoreCase("Sec-WebSocket-Accept") }
68           .map {
69             case (_, v) if genAcceptKey(key) == v => Right(())
70             case (_, v) => Left(s"Invalid key: $v")
71           }
72           .getOrElse(Left("Missing Sec-WebSocket-Accept header"))
73   }
74 
75   /** Checks the headers received from the client and if they are valid, generates response headers */
76   def serverHandshake(
77       headers: Iterable[(String, String)]
78   ): Either[(Int, String), collection.Seq[(String, String)]] =
79     if (!headers.exists { case (k, _) => k.equalsIgnoreCase("Host") })
80       Left((-1, "Missing Host Header"))
81     else if (
82       !headers.exists { case (k, v) =>
83         k.equalsIgnoreCase("Connection") && valueContains("Upgrade", v)
84       }
85     )
86       Left((-1, "Bad Connection header"))
87     else if (
88       !headers.exists { case (k, v) =>
89         k.equalsIgnoreCase("Upgrade") && v.equalsIgnoreCase("websocket")
90       }
91     )
92       Left((-1, "Bad Upgrade header"))
93     else if (
94       !headers.exists { case (k, v) =>
95         k.equalsIgnoreCase("Sec-WebSocket-Version") && valueContains("13", v)
96       }
97     )
98       Left((-1, "Bad Websocket Version header"))
99     // we are past most of the 'just need them' headers
100     else
101       headers
102         .find { case (k, v) =>
103           k.equalsIgnoreCase("Sec-WebSocket-Key") && decodeLen(v) == 16
104         }
105         .map { case (_, v) =>
106           val respHeaders = collection.Seq(
107             ("Upgrade", "websocket"),
108             ("Connection", "Upgrade"),
109             ("Sec-WebSocket-Accept", genAcceptKey(v)),
110           )
111 
112           Right(respHeaders)
113         }
114         .getOrElse(Left((-1, "Bad Sec-WebSocket-Key header")))
115 
116   /** Check if the headers contain an 'Upgrade: websocket' header */
117   def isWebSocketRequest(headers: Iterable[(String, String)]): Boolean =
118     headers.exists { case (k, v) =>
119       k.equalsIgnoreCase("Upgrade") && v.equalsIgnoreCase("websocket")
120     }
121 
122   private def decodeLen(key: String): Int = Base64.getDecoder.decode(key).length
123 
124   private def genAcceptKey(str: String): String = (for {
125     data <- SyncIO.fromEither(ByteVector.encodeAscii(str))
126     digest <- Hash[SyncIO].digest(HashAlgorithm.SHA1, data ++ magicString)
127   } yield digest.toBase64).unsafeRunSync()
128 
129   private[websocket] def valueContains(key: String, value: String): Boolean = {
130     val parts = value.split(",").map(_.trim)
131     parts.foldLeft(false)((b, s) =>
132       b || {
133         s.equalsIgnoreCase(key) ||
134         s.length > 1 &&
135         s.startsWith("\"") &&
136         s.endsWith("\"") &&
137         s.substring(1, s.length - 1).equalsIgnoreCase(key)
138       }
139     )
140   }
141 
142   private val magicString =
143     ByteVector.view("258EAFA5-E914-47DA-95CA-C5AB0DC85B11".getBytes(US_ASCII))
144 
145   private val clientBaseHeaders = List(
146     ("Connection", "Upgrade"),
147     ("Upgrade", "websocket"),
148     ("Sec-WebSocket-Version", "13"),
149   )
