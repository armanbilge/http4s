32   // message op codes
33   private[websocket] val CONTINUATION = 0x0
34   private[websocket] val TEXT = 0x1
35   private[websocket] val BINARY = 0x2
36   private[websocket] val CLOSE = 0x8
37   private[websocket] val PING = 0x9
38   private[websocket] val PONG = 0xa
39 
40   // Type constructors
41   @throws[ProtocolException]
42   private[websocket] def makeFrame(opcode: Int, data: ByteVector, last: Boolean): WebSocketFrame =
43     (opcode: @switch) match {
44       case 0x0 => WebSocketFrame.Continuation(data, last)
45       case 0x1 => WebSocketFrame.Text(data, last)
46       case 0x2 => WebSocketFrame.Binary(data, last)
47       case 0x3 | 0x4 | 0x5 | 0x6 | 0x7 =>
48         // Reserved for future non-control frames.
49         throw new ReservedOpcodeException(opcode)
50       case 0x8 =>
51         if (data.length == 1)
52           throw new ProtocolException("Close frame must have 0 data bits or at least 2")
53         if (!last) throw new ProtocolException("Control frame cannot be fragmented: Close")
54         WebSocketFrame.Close(data)
55       case 0x9 =>
56         if (!last) throw new ProtocolException("Control frame cannot be fragmented: Ping")
57         else WebSocketFrame.Ping(data)
58       case 0xa =>
59         if (!last) throw new ProtocolException("Control frame cannot be fragmented: Pong")
60         else WebSocketFrame.Pong(data)
61       case 0xb | 0xc | 0xd | 0xe | 0xf =>
62         // Reserved for future control frames.
63         throw new ReservedOpcodeException(opcode)
64       case _ =>
65         throw new UnknownOpcodeException(opcode)
66     }
67 }
