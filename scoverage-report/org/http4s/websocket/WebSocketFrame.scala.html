24 abstract class WebSocketFrame {

29   final def length: Int = data.length.toInt

31   override def equals(obj: Any): Boolean =
32     obj match {
33       case wf: WebSocketFrame =>
34         this.opcode == wf.opcode &&
35         this.last == wf.last &&
36         this.data == wf.data
37       case _ => false
38     }

40   override def hashCode: Int = {
41     var hash = WebSocketFrame.hashSeed
42     hash = MurmurHash3.mix(hash, opcode.##)
43     hash = MurmurHash3.mix(hash, data.##)
44     hash = MurmurHash3.mixLast(hash, last.##)
45     hash
46   }
47 }

121   private def toUnsignedShort(x: Int) = Array[Byte](((x >> 8) & 0xff).toByte, (x & 0xff).toByte)
122 
123   private def reasonToBytes(reason: String) = {
124     val asBytes = ByteVector.view(reason.getBytes(UTF_8))
125     if (asBytes.length > 123)
126       Left(new ReasonTooLongException(reason))
127     else
128       Right(asBytes)
129   }
130 
131   private def closeCodeToBytes(code: Int): Either[InvalidCloseCodeException, ByteVector] =
132     if (code < 1000 || code > 4999) Left(new InvalidCloseCodeException(code))
133     else Right(ByteVector.view(toUnsignedShort(code)))
134 
135   object Close {
136     def apply(code: Int): Either[InvalidCloseDataException, Close] =
137       closeCodeToBytes(code).map(Close(_))
138 
139     def apply(code: Int, reason: String): Either[InvalidCloseDataException, Close] =
140       for {
141         c <- closeCodeToBytes(code): Either[InvalidCloseDataException, ByteVector]
142         r <- reasonToBytes(reason)
143       } yield Close(c ++ r)
144   }
