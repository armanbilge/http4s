26   private def decodeBinary(in: ByteBuffer, mask: Array[Byte]) = {
27     val data = new Array[Byte](in.remaining)
28     in.get(data)
29     if (mask != null) // We can use the charset decode
30       for (i <- data.indices)
31         data(i) = (data(i) ^ mask(i & 0x3)).toByte // i mod 4 is the same as i & 0x3 but slower
32     data
33   }
34 
35   private def lengthOffset(in: ByteBuffer) = {
36     val len = in.get(1) & LENGTH
37     if (len < 126) 2
38     else if (len == 126) 4
39     else if (len == 127) 10
40     else throw new FrameTranscoder.TranscodeError("Length error!")
41   }
42 
43   private def getMask(in: ByteBuffer): Array[Byte] = {
44     val m = new Array[Byte](4)
45     in.mark
46     in.position(lengthOffset(in))
47     in.get(m)
48     in.reset
49     m
50   }
51 
52   private def bodyLength(in: ByteBuffer) = {
53     val len = in.get(1) & LENGTH
54     if (len < 126) len
55     else if (len == 126) (in.get(2) << 8 & 0xff00) | (in.get(3) & 0xff)
56     else if (len == 127) {
57       val l = in.getLong(2)
58       if (l > Integer.MAX_VALUE) throw new FrameTranscoder.TranscodeError("Frame is too long")
59       else l.toInt
60     } else throw new FrameTranscoder.TranscodeError("Length error")
61   }
62 
63   private def getMsgLength(in: ByteBuffer) = {
64     var totalLen = 2
65     if ((in.get(1) & MASK) != 0) totalLen += 4
66 
67     val len = in.get(1) & LENGTH
68 
69     if (len == 126) totalLen += 2
70     if (len == 127) totalLen += 8
71 
72     if (in.remaining < totalLen)
73       -1
74     else {
75       totalLen += bodyLength(in)
76 
77       if (in.remaining < totalLen) -1
78       else totalLen
79     }
80   }
81 }

83 class FrameTranscoder(val isClient: Boolean) {
84   def frameToBuffer(in: WebSocketFrame): Array[ByteBuffer] = {
85     var size = 2
86 
87     if (isClient) size += 4 // for the mask
88 
89     if (in.length < 126) {
90       /* NOOP */
91     } else if (in.length <= 0xffff) size += 2
92     else size += 8
93 
94     val buff = ByteBuffer.allocate(if (isClient) size + in.length else size)
95 
96     val opcode = in.opcode
97 
98     if (in.length > 125 && (opcode == PING || opcode == PONG || opcode == CLOSE))
99       throw new FrameTranscoder.TranscodeError("Invalid PING frame: frame too long: " + in.length)
100 
101     // First byte. Finished, reserved, and OP CODE
102     val b1 = if (in.last) opcode | FINISHED else opcode
103 
104     buff.put(b1.byteValue)
105 
106     // Second byte. Mask bit and length
107     var b2 = 0x0
108 
109     if (isClient) b2 = MASK
110 
111     if (in.length < 126) b2 |= in.length
112     else if (in.length <= 0xffff) b2 |= 126
113     else b2 |= 127
114 
115     buff.put(b2.byteValue)
116 
117     // Put the length if we have an extended length packet
118     if (in.length > 125 && in.length <= 0xffff)
119       buff.put((in.length >>> 8 & 0xff).toByte).put((in.length & 0xff).toByte)
120     else if (in.length > 0xffff) buff.putLong(in.length.toLong)
121 
122     // If we are a client, we need to mask the data, else just wrap it in a buffer and done
123     if (isClient && in.length > 0) { // need to mask outgoing bytes
124       val mask = (Math.random * Integer.MAX_VALUE).toInt
125       val maskBits = Array(
126         ((mask >>> 24) & 0xff).toByte,
127         ((mask >>> 16) & 0xff).toByte,
128         ((mask >>> 8) & 0xff).toByte,
129         ((mask >>> 0) & 0xff).toByte,
130       )
131 
132       buff.put(maskBits)
133 
134       val data = in.data
135 
136       for (i <- 0 until in.length.toInt)
137         buff.put(
138           (data(i.toLong) ^ maskBits(i & 0x3)).toByte
139         ) // i & 0x3 is the same as i % 4 but faster
140       buff.flip
141       Array[ByteBuffer](buff)
142     } else {
143       buff.flip
144       Array[ByteBuffer](buff, in.data.toByteBuffer)
145     }
146   }
147 
148   /** Method that decodes ByteBuffers to objects. None reflects not enough data to decode a message
149     * Any unused data in the ByteBuffer will be recycled and available for the next read
150     *
151     * @param in ByteBuffer of immediately available data
152     * @return optional message if enough data was available
153     */
154   def bufferToFrame(in: ByteBuffer): WebSocketFrame =
155     if (in.remaining < 2 || FrameTranscoder.getMsgLength(in) < 0)
156       null
157     else {
158       val opcode = in.get(0) & OP_CODE
159       val finished = (in.get(0) & FINISHED) != 0
160       val masked = (in.get(1) & MASK) != 0
161 
162       if (masked && isClient)
163         throw new FrameTranscoder.TranscodeError("Client received a masked message")
164 
165       var bodyOffset = FrameTranscoder.lengthOffset(in)
166 
167       val m = if (masked) {
168         bodyOffset += 4
169         FrameTranscoder.getMask(in)
170       } else
171         null
172 
173       val oldLim = in.limit()
174       val bodylen = FrameTranscoder.bodyLength(in)
175 
176       in.position(bodyOffset)
177       in.limit(in.position() + bodylen)
178 
179       val slice = in.slice
180       in.position(in.limit)
181       in.limit(oldLim)
182 
183       makeFrame(opcode, ByteVector.view(FrameTranscoder.decodeBinary(slice, m)), finished)
184     }
