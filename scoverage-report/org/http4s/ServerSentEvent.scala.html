31 final case class ServerSentEvent(
32     data: Option[String] = None,
33     eventType: Option[String] = None,
34     id: Option[EventId] = None,
35     retry: Option[FiniteDuration] = None,
36     comment: Option[String] = None,
37 ) extends Renderable {
38   def render(writer: Writer): writer.type = {
39     data.foreach(writer << "data: " << _ << "\n")
40     comment.foreach(writer << ": " << _ << "\n")
41     eventType.foreach(writer << "event: " << _ << "\n")
42     id match {
43       case None =>
44       case Some(EventId.reset) => writer << "id\n"
45       case Some(EventId(id)) => writer << "id: " << id << "\n"
46     }
47     retry.foreach(writer << "retry: " << _.toMillis << "\n")
48     writer << "\n"
49   }
50 
51   override def toString: String =
52     s"ServerSentEvent($data,$eventType,$id,$retry)"
53 }
54 
55 object ServerSentEvent {
56   val empty: ServerSentEvent = ServerSentEvent()
57 
58   final case class EventId(value: String)
59 
60   object EventId {
61     val reset: EventId = EventId("")
62   }
63 
64   // Pattern instead of Regex, because Regex doesn't have split with limit. :(
65   private val FieldSeparator =
66     Pattern.compile(""": ?""")
67 
68   def decoder[F[_]]: Pipe[F, Byte, ServerSentEvent] = {
69     case class LineBuffer(lines: Chain[String] = Chain.empty) {
70       def append(line: String): LineBuffer =
71         // val scrubbed = if (line.endsWith("\n")) line.dropRight(1) else line
72         copy(lines = lines :+ line)
73       def reify: Option[String] =
74         if (lines.nonEmpty) {
75           Some(lines.iterator.mkString("\n"))
76         } else
77           None
78     }
79     val emptyBuffer = LineBuffer()
80     def go(
81         dataBuffer: LineBuffer,
82         eventType: Option[String],
83         id: Option[EventId],
84         retry: Option[Long],
85         commentBuffer: LineBuffer,
86         stream: Stream[F, String],
87     ): Pull[F, ServerSentEvent, Unit] = {
88       //      def dispatch(h: Handle[F, String]): Pull[F, ServerSentEvent, Nothing] =
89       def dispatch(stream: Stream[F, String]): Pull[F, ServerSentEvent, Unit] = {
90         val sse = ServerSentEvent(
91           dataBuffer.reify,
92           eventType,
93           id,
94           retry.map(FiniteDuration(_, TimeUnit.MILLISECONDS)),
95           commentBuffer.reify,
96         )
97         Pull.output1(sse) >> go(emptyBuffer, None, None, None, emptyBuffer, stream)
98       }
99 
100       def handleLine(
101           field: String,
102           value: String,
103           stream: Stream[F, String],
104       ): Pull[F, ServerSentEvent, Unit] =
105         field match {
106           case "" =>
107             go(dataBuffer, eventType, id, retry, commentBuffer.append(value), stream)
108           case "event" =>
109             go(dataBuffer, Some(value), id, retry, commentBuffer, stream)
110           case "data" =>
111             go(dataBuffer.append(value), eventType, id, retry, commentBuffer, stream)
112           case "id" =>
113             val newId = EventId(value)
114             go(dataBuffer, eventType, Some(newId), retry, commentBuffer, stream)
115           case "retry" =>
116             val newRetry = Try(value.toLong).toOption.orElse(retry)
117             go(dataBuffer, eventType, id, newRetry, commentBuffer, stream)
118           case _ =>
119             go(dataBuffer, eventType, id, retry, commentBuffer, stream)
120         }
121 
122       stream.pull.uncons1.flatMap {
123         case None =>
124           Pull.done
125         case Some(("", stream)) =>
126           dispatch(stream)
127         case Some((s, stream)) =>
128           FieldSeparator.split(s, 2) match {
129             case Array(field, value) =>
130               handleLine(field, value, stream)
131             case Array(line) =>
132               handleLine(line, "", stream)
133           }
134       }
135     }
136 
137     stream =>
138       go(
139         emptyBuffer,
140         None,
141         None,
142         None,
143         emptyBuffer,
144         stream.through(utf8.decode.andThen(text.lines)),
145       ).stream
146   }
147 
148   def encoder[F[_]]: Pipe[F, ServerSentEvent, Byte] =
149     _.map(_.renderString).through(utf8.encode)
