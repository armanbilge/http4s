23 /** Functions for creating [[Http]] kleislis. */
24 object Http {
25 
26   /** Lifts a function into an [[Http]] kleisli.  The application of
27     * `run` is suspended in `F` to permit more efficient combination
28     * of routes via `SemigroupK`.
29     *
30     * @tparam F the effect of the [[Response]] returned by the [[Http]]
31     * @tparam G the effect of the bodies of the [[Request]] and [[Response]]
32     * @param run the function to lift
33     * @return an [[Http]] that suspends `run`.
34     */
35   def apply[F[_], G[_]](run: Request[G] => F[Response[G]])(implicit F: Monad[F]): Http[F, G] =
36     Kleisli(req => F.unit >> run(req))
37 
38   /** Lifts an effectful [[Response]] into an [[Http]] kleisli.
39     *
40     * @tparam F the effect of the [[Response]] returned by the [[Http]]
41     * @tparam G the effect of the bodies of the [[Request]] and [[Response]]
42     * @param fr the effectful [[Response]] to lift
43     * @return an [[Http]] that always returns `fr`
44     */
45   def liftF[F[_], G[_]](fr: F[Response[G]]): Http[F, G] =
46     Kleisli.liftF(fr)
47 
48   /** Lifts a [[Response]] into an [[Http]] kleisli.
49     *
50     * @tparam F the effect of the [[Response]] returned by the [[Http]]
51     * @tparam G the effect of the bodies of the [[Request]] and [[Response]]
52     * @param r the [[Response]] to lift
53     * @return an [[Http]] that always returns `r` in effect `F`
54     */
55   def pure[F[_]: Applicative, G[_]](r: Response[G]): Http[F, G] =
56     Kleisli.pure(r)
57 
58   /** Transforms an [[Http]] on its input.  The application of the
59     * transformed function is suspended in `F` to permit more
60     * efficient combination of routes via `SemigroupK`.
61     *
62     * @tparam F the effect of the [[Response]] returned by the [[Http]]
63     * @tparam G the effect of the bodies of the [[Request]] and [[Response]]
64     * @param f a function to apply to the [[Request]]
65     * @param fa the [[Http]] to transform
66     * @return An [[Http]] whose input is transformed by `f` before
67     * being applied to `fa`
68     */
69   def local[F[_], G[_]](f: Request[G] => Request[G])(fa: Http[F, G])(implicit
70       F: Monad[F]
71   ): Http[F, G] =
72     Kleisli(req => F.unit >> fa.run(f(req)))
73 }
