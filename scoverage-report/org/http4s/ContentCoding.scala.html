38 class ContentCoding private (val coding: String, override val qValue: QValue = QValue.One)
39     extends HasQValue
40     with Ordered[ContentCoding]
41     with Renderable {
42   def withQValue(q: QValue): ContentCoding = new ContentCoding(coding, q)
43 
44   def matches(encoding: ContentCoding): Boolean =
45     this === ContentCoding.`*` || this.coding.equalsIgnoreCase(encoding.coding)
46 
47   override def equals(o: Any): Boolean =
48     o match {
49       case that: ContentCoding =>
50         this.coding.equalsIgnoreCase(that.coding) && this.qValue === that.qValue
51       case _ => false
52     }
53 
54   private[this] var hash = 0
55   override def hashCode(): Int = {
56     if (hash == 0)
57       hash = MurmurHash3.mixLast(coding.toLowerCase.##, qValue.##)
58     hash
59   }
60 
61   override def toString: String = s"ContentCoding(${coding.toLowerCase}, $qValue)"
62 
63   override def compare(other: ContentCoding): Int =
64     ContentCoding.http4sOrderForContentCoding.compare(this, other)
65 
66   override def render(writer: Writer): writer.type =
67     ContentCoding.http4sHttpCodecForContentCoding.render(writer, this)
68 }

70 object ContentCoding {
71   def unsafeFromString(coding: String): ContentCoding =
72     fromString(coding).valueOr(throw _)
73 
74   def fromString(coding: String): ParseResult[ContentCoding] =
75     parse(coding)
76 
77   val `*` : ContentCoding = new ContentCoding("*")
78 
79   // http://www.iana.org/assignments/http-parameters/http-parameters.xml#http-parameters-1
80   val aes128gcm = new ContentCoding("aes128gcm")
81   val br = new ContentCoding("br")
82   val compress = new ContentCoding("compress")
83   val deflate = new ContentCoding("deflate")
84   val exi = new ContentCoding("exi")
85   val gzip = new ContentCoding("gzip")
86   val identity = new ContentCoding("identity")
87   val `pack200-gzip` = new ContentCoding("pack200-gzip")
88   val zstd = new ContentCoding("zstd")
89 
90   // Legacy encodings defined by RFC2616 3.5.
91   val `x-compress`: ContentCoding = compress
92   val `x-gzip`: ContentCoding = gzip
93 
94   val standard: Map[String, ContentCoding] =
95     List(`*`, aes128gcm, br, compress, deflate, exi, gzip, identity, `pack200-gzip`, zstd)
96       .map(c => c.coding -> c)
97       .toMap
98 
99   private[http4s] val parser: Parser[ContentCoding] = {
100     import org.http4s.internal.parsing.Rfc7230.token
101 
102     val contentCoding = token.map(s => ContentCoding.standard.getOrElse(s, new ContentCoding(s)))
103 
104     (contentCoding ~ QValue.parser).map { case (coding, q) =>
105       if (q === QValue.One) coding
106       else coding.withQValue(q)
107     }
108   }
109 
110   /** Parse a Content Coding
111     */
112   def parse(s: String): ParseResult[ContentCoding] =
113     ParseResult.fromParser(parser, "content coding")(s)
114 
115   implicit val http4sOrderForContentCoding: Order[ContentCoding] =
116     Order.by(c => (c.coding.toLowerCase(ju.Locale.ENGLISH), c.qValue))
117 
118   implicit val http4sShowForContentCoding: Show[ContentCoding] =
119     Show.fromToString
120   implicit val http4sHttpCodecForContentCoding: HttpCodec[ContentCoding] =
121     new HttpCodec[ContentCoding] {
122       override def parse(s: String): ParseResult[ContentCoding] =
123         ContentCoding.parse(s)
124 
125       override def render(writer: Writer, coding: ContentCoding): writer.type =
126         writer << coding.coding.toLowerCase << coding.qValue
127     }
