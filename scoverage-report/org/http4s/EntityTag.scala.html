28 final case class EntityTag(tag: String, weakness: Weakness = Strong) extends Renderable {
29   override def toString(): String =
30     weakness match {
31       case EntityTag.Weak => "W/\"" + tag + '"'
32       case EntityTag.Strong => "\"" + tag + '"'
33     }
34 
35   override def render(writer: Writer): writer.type =
36     writer << toString()
37 }
38 
39 object EntityTag {
40   implicit val http4sShowForEntityTag: Show[EntityTag] =
41     Show.fromToString
42 
43   sealed trait Weakness extends Product with Serializable
44   case object Weak extends Weakness
45   case object Strong extends Weakness
46 
47   /*
48    * entity-tag = [ weak ] opaque-tag
49    *
50    * @see [[https://datatracker.ietf.org/doc/html/rfc7232#section-2.3]]
51    */
52   private[http4s] val parser: Parser[EntityTag] = {
53     import Parser.{charIn, string}
54     import Rfc5234.dquote
55     import Rfc7230.obsText
56 
57     // weak       = %x57.2F ; "W/", case-sensitive
58     val weak = string("W/").as(EntityTag.Weak)
59 
60     // etagc      = %x21 / %x23-7E / obs-text
61     //            ; VCHAR except double quotes, plus obs-text
62     val etagc = charIn(0x21.toChar, 0x23.toChar to 0x7e.toChar: _*).orElse(obsText)
63 
64     // opaque-tag = DQUOTE *etagc DQUOTE
65     val opaqueTag = dquote *> etagc.rep0.string <* dquote
66 
67     (weak.?.with1 ~ opaqueTag).map { case (weakness, tag) =>
68       new EntityTag(tag, weakness.getOrElse(EntityTag.Strong))
69     }
70   }
71 }
