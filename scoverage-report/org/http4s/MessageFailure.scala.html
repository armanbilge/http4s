28 /** Indicates a failure to handle an HTTP [[Message]]. */
29 trait MessageFailure extends RuntimeException {
30 
31   /** Provides a message appropriate for logging. */
32   def message: String
33 
34   /* Overridden for sensible logging of the failure */
35   override final def getMessage: String = message
36 
37   def cause: Option[Throwable]
38 
39   override final def getCause: Throwable = cause.orNull
40 
41   /** Provides a default rendering of this failure as a [[Response]]. */
42   def toHttpResponse[F[_]](httpVersion: HttpVersion): Response[F]
43 }
44 
45 /** Indicates an error parsing an HTTP [[Message]].
46   *
47   * @param sanitized May safely be displayed to a client to describe an error
48   *                  condition.  Should not echo any part of a Request.
49   * @param details Contains any relevant details omitted from the sanitized
50   *                version of the error.  This may freely echo a Request.
51   */
52 final case class ParseFailure(sanitized: String, details: String)
53     extends MessageFailure
54     with NoStackTrace {
55   def message: String =
56     if (sanitized.isEmpty) details
57     else if (details.isEmpty) sanitized
58     else s"$sanitized: $details"
59 
60   def cause: Option[Throwable] = None
61 
62   def toHttpResponse[F[_]](httpVersion: HttpVersion): Response[F] =
63     Response(Status.BadRequest, httpVersion)
64       .withEntity(sanitized)(EntityEncoder.stringEncoder[F])
65 }
66 
67 object ParseFailure {
68   implicit val eq: Eq[ParseFailure] = Eq.fromUniversalEquals[ParseFailure]
69 }
70 
71 object ParseResult {
72   def fail(sanitized: String, details: String): ParseResult[Nothing] =
73     Left(ParseFailure(sanitized, details))
74 
75   def success[A](a: A): ParseResult[A] =
76     Right(a)
77 
78   def fromTryCatchNonFatal[A](sanitized: String)(f: => A): ParseResult[A] =
79     try ParseResult.success(f)
80     catch {
81       case NonFatal(e) => Left(ParseFailure(sanitized, e.getMessage))
82     }
83 
84   private[http4s] def fromParser[A](parser: Parser0[A], errorMessage: => String)(
85       s: String
86   ): ParseResult[A] =
87     try parser.parseAll(s).leftMap(e => ParseFailure(errorMessage, e.toString))
88     catch { case p: ParseFailure => p.asLeft[A] }
89 
90   implicit val parseResultMonad: MonadError[ParseResult, ParseFailure] =
91     catsStdInstancesForEither[ParseFailure]
92 }

128 /** Indicates that a [[Message]] came with no supported [[MediaType]]. */
129 sealed abstract class UnsupportedMediaTypeFailure extends DecodeFailure with NoStackTrace {
130   def expected: Set[MediaRange]
131   def cause: Option[Throwable] = None
132 
133   protected def sanitizedResponsePrefix: String
134   protected def expectedMsg: String =
135     s"Expected one of the following media ranges: ${expected.map(_.show).mkString(", ")}"
136   protected def responseMsg: String = s"$sanitizedResponsePrefix. $expectedMsg"
137 
138   def toHttpResponse[F[_]](httpVersion: HttpVersion): Response[F] =
139     Response(Status.UnsupportedMediaType, httpVersion)
140       .withEntity(responseMsg)(EntityEncoder.stringEncoder[F])
141 }
142 
143 /** Indicates that a [[Message]] attempting to be decoded has no [[MediaType]] and no
144   * [[EntityDecoder]] was lenient enough to accept it.
145   */
146 final case class MediaTypeMissing(expected: Set[MediaRange]) extends UnsupportedMediaTypeFailure {
147   def sanitizedResponsePrefix: String = "No media type specified in Content-Type header"
148   def message: String = responseMsg
149 }
150 
151 /** Indicates that no [[EntityDecoder]] matches the [[MediaType]] of the [[Message]] being decoded */
152 final case class MediaTypeMismatch(messageType: MediaType, expected: Set[MediaRange])
153     extends UnsupportedMediaTypeFailure {
154   def sanitizedResponsePrefix: String =
155     "Media type supplied in Content-Type header is not supported"
156   def message: String = s"${messageType.show} is not a supported media type. $expectedMsg"
157 }
