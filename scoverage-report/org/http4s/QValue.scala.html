26 /** A Quality Value.  Represented as thousandths for an exact representation rounded to three
27   * decimal places.
28   *
29   * @param thousandths between 0 (for q=0) and 1000 (for q=1)
30   * @see [[http://www.w3.org/Protocols/rfc2616/rfc2616-sec3.html#sec3.9 RFC 2616, Section 3.9, Quality Values]]
31   */
32 final class QValue private (val thousandths: Int) extends AnyVal with Ordered[QValue] {
33   def toDouble: Double = 0.001 * thousandths
34 
35   def isAcceptable: Boolean = thousandths > 0
36 
37   override def toString: String = s"QValue(${0.001 * thousandths})"
38 
39   override def compare(that: QValue): Int = thousandths - that.thousandths
40 
41   def render(writer: Writer): writer.type =
42     if (thousandths == 1000) writer
43     else {
44       writer.append(";q=")
45       formatq(writer)
46     }
47 
48   // Assumes that q is in the proper bounds, otherwise you get an exception!
49   private def formatq(b: Writer): b.type =
50     // Skip the rest of the stuff if we are 1.0
51     if (thousandths == 1000) b.append("1.0")
52     else if (thousandths == 0) b.append('0')
53     else {
54       // Need to start appending stuff
55       b.append("0.")
56 
57       @inline
58       def convert(i: Int): Char = ('0' + i).toChar
59 
60       val mod100 = thousandths % 100
61 
62       if (thousandths > 99) {
63         b.append(convert(thousandths / 100))
64         if (mod100 == 0) return b // First digit only
65       } else b.append('0')
66 
67       val mod10 = thousandths % 10
68 
69       if (thousandths > 9) {
70         b.append(convert(mod100 / 10))
71         if (mod10 == 0) return b // Second digit only
72       } else b.append('0')
73 
74       b.append(convert(mod10)) // Last digit
75     }
76 
77 }
