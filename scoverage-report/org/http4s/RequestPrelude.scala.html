23 /** A projection of a [[Request]] without the body.
24   *
25   * @note The [[Request#attributes]] are omitted in this encoding because they
26   *       do not (and can not) have a [[cats.kernel.Order]] instance. If they
27   *       were included here, then we could not write a [[cats.kernel.Order]]
28   *       instance for [[RequestPrelude]], limiting some of its utility, e.g. it
29   *       could not be used in a [[cats.data.NonEmptySet]].
30   */
31 sealed abstract class RequestPrelude extends Product with Serializable {
32   def headers: Headers
33   def httpVersion: HttpVersion
34   def method: Method
35   def uri: Uri
36 
37   def withHeaders(value: Headers): RequestPrelude
38   def withHttpVersion(value: HttpVersion): RequestPrelude
39   def withMethod(value: Method): RequestPrelude
40   def withUri(value: Uri): RequestPrelude
41 
42   // final //
43 
44   final def mapHeaders(f: Headers => Headers): RequestPrelude =
45     withHeaders(f(headers))
46 
47   final def mapHttpVersion(f: HttpVersion => HttpVersion): RequestPrelude =
48     withHttpVersion(f(httpVersion))
49 
50   final def mapMethod(f: Method => Method): RequestPrelude =
51     withMethod(f(method))
52 
53   final def mapUri(f: Uri => Uri): RequestPrelude =
54     withUri(f(uri))
55 
56   override final def toString: String =
57     s"RequestPrelude(headers = ${headers
58         .redactSensitive()}, httpVersion = ${httpVersion}, method = ${method}, uri = ${uri})"
59 }
60 
61 object RequestPrelude {
62   private[this] final case class RequestPreludeImpl(
63       override final val headers: Headers,
64       override final val httpVersion: HttpVersion,
65       override final val method: Method,
66       override final val uri: Uri,
67   ) extends RequestPrelude {
68     override final def withHeaders(value: Headers): RequestPrelude =
69       this.copy(headers = value)
70 
71     override final def withHttpVersion(value: HttpVersion): RequestPrelude =
72       this.copy(httpVersion = value)
73 
74     override final def withMethod(value: Method): RequestPrelude =
75       this.copy(method = value)
76 
77     override final def withUri(value: Uri): RequestPrelude =
78       this.copy(uri = value)
79   }
80 
81   def apply(
82       headers: Headers,
83       httpVersion: HttpVersion,
84       method: Method,
85       uri: Uri,
86   ): RequestPrelude =
87     RequestPreludeImpl(
88       headers,
89       httpVersion,
90       method,
91       uri,
92     )
93 
94   def fromRequest[F[_]](value: Request[F]): RequestPrelude =
95     RequestPreludeImpl(
96       value.headers,
97       value.httpVersion,
98       value.method,
99       value.uri,
100     )
101 
102   implicit val catsHashAndOrderForRequestPrelude: Hash[RequestPrelude] with Order[RequestPrelude] =
103     new Hash[RequestPrelude] with Order[RequestPrelude] {
104       override def hash(x: RequestPrelude): Int = x.hashCode
105 
106       override def compare(x: RequestPrelude, y: RequestPrelude): Int =
107         reduceComparisons(
108           x.headers.compare(y.headers),
109           Eval.later(x.httpVersion.compare(y.httpVersion)),
110           Eval.later(x.method.compare(y.method)),
111           Eval.later(x.uri.compare(y.uri)),
112         )
113     }
114 
115   implicit val catsShowForRequestPrelude: Show[RequestPrelude] =
116     Show.fromToString
117 
118   implicit def stdLibOrdering: Ordering[RequestPrelude] =
119     catsHashAndOrderForRequestPrelude.toOrdering
120 }
