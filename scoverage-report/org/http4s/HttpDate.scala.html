34 /** An HTTP-date value represents time as an instance of Coordinated Universal
35   * Time (UTC). It expresses time at a resolution of one second.  By using it
36   * over java.time.Instant in the model, we assure that if two headers render
37   * equally, their values are equal.
38   *
39   * @see [[https://datatracker.ietf.org/doc/html/rfc7231#section-7.1.1 RFC 7231, Section 7.1.1, Origination Date]]
40   */
41 class HttpDate private (val epochSecond: Long) extends Renderable with Ordered[HttpDate] {
42   def compare(that: HttpDate): Int =
43     this.epochSecond.compare(that.epochSecond)
44 
45   def toInstant: Instant =
46     Instant.ofEpochSecond(epochSecond)
47 
48   def render(writer: Writer): writer.type =
49     writer << toInstant
50 
51   override def equals(o: Any): Boolean =
52     o match {
53       case that: HttpDate => this.epochSecond == that.epochSecond
54       case _ => false
55     }
56 
57   override def hashCode(): Int =
58     epochSecond.##
59 }

73   /** The earliest value reprsentable as an HTTP-date, `Mon, 01 Jan 1900 00:00:00 GMT`.
74     *
75     * The minimum year is specified by RFC5322 as 1900.
76     *
77     * @see [[https://datatracker.ietf.org/doc/html/rfc7231#section-7.1.1 RFC 7231, Section 7.1.1, Origination Date]]
78     * @see [[https://datatracker.ietf.org/doc/html/rfc5322#section-3.3 RFC 5322, Section 3.3, Date and Time Specification]]
79     */
80   val MinValue: HttpDate = HttpDate.unsafeFromEpochSecond(MinEpochSecond)
81 
82   /** The latest value reprsentable by RFC1123, `Fri, 31 Dec 9999 23:59:59 GMT`. */
83   val MaxValue: HttpDate = HttpDate.unsafeFromEpochSecond(MaxEpochSecond)

85   /** Constructs an `HttpDate` from the current time. Starting on January 1,n
86     * 10000, this will throw an exception. The author intends to leave this
87     * problem for future generations.
88     */
89   @deprecated("0.20.16", "Use HttpDate.current instead, this breaks referential transparency")
90   def now: HttpDate =
91     unsafeFromInstant(Instant.now)

93   /** Constructs an [[HttpDate]] from the current time. Starting on January 1,n
94     * 10000, this will throw an exception. The author intends to leave this
95     * problem for future generations.
96     */
97   def current[F[_]: Functor: Clock]: F[HttpDate] =
98     Clock[F].realTime.map(v => unsafeFromEpochSecond(v.toSeconds))

100   /** The `HttpDate` equal to `Thu, Jan 01 1970 00:00:00 GMT` */

104   /** Parses a date according to RFC7231, Section 7.1.1.1
105     *
106     * @see [[https://datatracker.ietf.org/doc/html/rfc7231#section-7.1.1 RFC 7231, Section 7.1.1, Origination Date]]
107     */
108   def fromString(s: String): ParseResult[HttpDate] =
109     ParseResult.fromParser(parser, "Invalid HTTP date")(s)
110 
111   /** Like `fromString`, but throws on invalid input */
112   def unsafeFromString(s: String): HttpDate =
113     fromString(s).fold(throw _, identity)
114 
115   /** Constructs a date from the seconds since the [[Epoch]]. If out of range,
116     *  returns a ParseFailure.
117     */
118   def fromEpochSecond(epochSecond: Long): ParseResult[HttpDate] =
119     if (epochSecond < MinEpochSecond || epochSecond > MaxEpochSecond)
120       ParseResult.fail(
121         "Invalid HTTP date",
122         s"${epochSecond} out of range for HTTP date. Must be between ${MinEpochSecond} and ${MaxEpochSecond}, inclusive",
123       )
124     else
125       ParseResult.success(new HttpDate(epochSecond))
126 
127   /** Like `fromEpochSecond`, but throws any parse failures */
128   def unsafeFromEpochSecond(epochSecond: Long): HttpDate =
129     fromEpochSecond(epochSecond).fold(throw _, identity)
130 
131   /** Constructs a date from an instant, truncating to the most recent second. If
132     *  out of range, returns a ParseFailure.
133     */
134   def fromInstant(instant: Instant): ParseResult[HttpDate] =
135     fromEpochSecond(instant.toEpochMilli / 1000)
136 
137   /** Like `fromInstant`, but throws any parse failures */
138   def unsafeFromInstant(instant: Instant): HttpDate =
139     unsafeFromEpochSecond(instant.toEpochMilli / 1000)

141   /** Constructs a date from an zoned date-time, truncating to the most recent
142     *  second. If out of range, returns a ParseFailure.
143     */
144   def fromZonedDateTime(dateTime: ZonedDateTime): ParseResult[HttpDate] =
145     fromInstant(dateTime.toInstant)
146 
147   /** Like `fromZonedDateTime`, but throws any parse failures */
148   def unsafeFromZonedDateTime(dateTime: ZonedDateTime): HttpDate =

151   /** `HTTP-date = IMF-fixdate / obs-date` */
152   private[http4s] val parser: Parser[HttpDate] = {
153     import Parser.{char, string}
154     import Rfc5234.{digit, sp}
155 
156     def mkHttpDate(
157         year: Int,
158         month: Int,
159         day: Int,
160         hour: Int,
161         min: Int,
162         sec: Int,
163     ): Option[HttpDate] =
164       try {
165         val dt = ZonedDateTime.of(year, month, day, hour, min, sec, 0, ZoneOffset.UTC)
166         Some(org.http4s.HttpDate.unsafeFromZonedDateTime(dt))
167       } catch {
168         case _: DateTimeException =>
169           None
170       }
171 
172     /* day-name     = %x4D.6F.6E ; "Mon", case-sensitive
173      *              / %x54.75.65 ; "Tue", case-sensitive
174      *              / %x57.65.64 ; "Wed", case-sensitive
175      *              / %x54.68.75 ; "Thu", case-sensitive
176      *              / %x46.72.69 ; "Fri", case-sensitive
177      *              / %x53.61.74 ; "Sat", case-sensitive
178      *              / %x53.75.6E ; "Sun", case-sensitive
179      */
180     val dayName =
181       List("Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun")
182         .map(string)
183         .zipWithIndex
184         .map { case (s, i) => string(s).as(i) }
185         .reduceLeft(_.orElse(_))
186         .soft
187 
188     /* day          = 2DIGIT */
189     val day = (digit ~ digit).string.map(_.toInt)
190 
191     /* month        = %x4A.61.6E ; "Jan", case-sensitive
192      *              / %x46.65.62 ; "Feb", case-sensitive
193      *              / %x4D.61.72 ; "Mar", case-sensitive
194      *              / %x41.70.72 ; "Apr", case-sensitive
195      *              / %x4D.61.79 ; "May", case-sensitive
196      *              / %x4A.75.6E ; "Jun", case-sensitive
197      *              / %x4A.75.6C ; "Jul", case-sensitive
198      *              / %x41.75.67 ; "Aug", case-sensitive
199      *              / %x53.65.70 ; "Sep", case-sensitive
200      *              / %x4F.63.74 ; "Oct", case-sensitive
201      *              / %x4E.6F.76 ; "Nov", case-sensitive
202      *              / %x44.65.63 ; "Dec", case-sensitive
203      */
204     val month =
205       List(
206         "Jan",
207         "Feb",
208         "Mar",
209         "Apr",
210         "May",
211         "Jun",
212         "Jul",
213         "Aug",
214         "Sep",
215         "Oct",
216         "Nov",
217         "Dec",
218       ).zipWithIndex
219         .map { case (s, i) => string(s).as(i + 1) }
220         .reduceLeft(_.orElse(_))
221 
222     /* year         = 4DIGIT */
223     val year = (digit ~ digit ~ digit ~ digit).string.map(_.toInt)
224 
225     /* date1        = day SP month SP year
226      *              ; e.g., 02 Jun 1982
227      */
228     val date1 = (day <* sp) ~ (month <* sp) ~ year
229 
230     /* hour         = 2DIGIT */
231     val hour = (digit ~ digit).string.map(_.toInt)
232 
233     /* minute       = 2DIGIT */
234     val minute = (digit ~ digit).string.map(_.toInt)
235 
236     /* second       = 2DIGIT */
237     val second = (digit ~ digit).string.map(_.toInt)
238 
239     val colon = char(':')
240     /* time-of-day  = hour ":" minute ":" second
241      *              ; 00:00:00 - 23:59:60 (leap second)
242      */
243     val timeOfDay = (hour <* colon) ~ (minute <* colon) ~ second
244 
245     /* IMF-fixdate  = day-name "," SP date1 SP time-of-day SP GMT
246      * ; fixed length/zone/capitalization subset of the format
247      * ; see Section 3.3 of [RFC5322]
248      *
249      * GMT          = %x47.4D.54 ; "GMT", case-sensitive
250      */
251     val imfFixdate =
252       ((dayName <* string(", ")) ~ (date1 <* sp) ~ (timeOfDay <* string(" GMT"))).mapFilter {
253         case ((_, ((day, month), year)), ((hour, min), sec)) =>
254           mkHttpDate(year, month, day, hour, min, sec)
255       }
256 
257     val twoDigit = (digit ~ digit).string.map(_.toInt)
258 
259     /* date2        = day "-" month "-" 2DIGIT
260      *              ; e.g., 02-Jun-82
261      */
262     val date2 = (day <* char('-')) ~ (month <* char('-')) ~ twoDigit
263 
264     /* day-name-l   = %x4D.6F.6E.64.61.79    ; "Monday", case-sensitive
265      *              / %x54.75.65.73.64.61.79       ; "Tuesday", case-sensitive
266      *              / %x57.65.64.6E.65.73.64.61.79 ; "Wednesday", case-sensitive
267      *              / %x54.68.75.72.73.64.61.79    ; "Thursday", case-sensitive
268      *              / %x46.72.69.64.61.79          ; "Friday", case-sensitive
269      *              / %x53.61.74.75.72.64.61.79    ; "Saturday", case-sensitive
270      *              / %x53.75.6E.64.61.79          ; "Sunday", case-sensitive
271      */
272     val dayNameL =
273       List("Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday")
274         .map(string)
275         .reduceLeft(_.orElse(_))
276 
277     /* rfc850-date  = day-name-l "," SP date2 SP time-of-day SP GMT
278      *
279      * "Recipients of a timestamp value in rfc850-date format, which uses a
280      * two-digit year, MUST interpret a timestamp that appears to be more
281      * than 50 years in the future as representing the most recent year in
282      * the past that had the same last two digits."
283      *
284      * Following that rule would make the parser impure for a case
285      * rarely seen in the wild anymore.  We're going to observe it
286      * from 2020 and hope our descendants are smarter than us.
287      */
288     val rfc850Date =
289       ((dayNameL <* string(", ")) ~ (date2 <* sp) ~ (timeOfDay <* string(" GMT"))).mapFilter {
290         case ((_, ((day, month), year)), ((hour, min), sec)) =>
291           val wrapYear = if (year < 70) year + 2000 else year + 1900
292           mkHttpDate(wrapYear, month, day, hour, min, sec)
293       }
294 
295     val oneDigit = digit.map(_ - '0')
296 
297     /* date3        = month SP ( 2DIGIT / ( SP 1DIGIT )) */
298     val date3 = (month <* sp) ~ (twoDigit.orElse(sp *> oneDigit))
299 
300     /* asctime-date = day-name SP date3 SP time-of-day SP year
301      *              ; e.g., Jun  2
302      */
303     val asctimeDate = ((dayName <* sp) ~ (date3 <* sp) ~ (timeOfDay <* sp) ~ year).mapFilter {
304       case (((_, (month, day)), ((hour, min), sec)), year) =>
305         mkHttpDate(year, month, day, hour, min, sec)
306     }
307 
308     /* obs-date     = rfc850-date / asctime-date */
309     val obsDate = rfc850Date.orElse(asctimeDate)
310 
311     imfFixdate.orElse(obsDate)
312   }
313 
314   implicit val stdLibOrderingInstance: Ordering[HttpDate] =
