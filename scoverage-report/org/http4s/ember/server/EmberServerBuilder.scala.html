176   def build: Resource[F, Server] =
177     for {
178       sg <- sgOpt.getOrElse(Network[F]).pure[Resource[F, *]]
179       ready <- Resource.eval(Deferred[F, Either[Throwable, SocketAddress[IpAddress]]])
180       shutdown <- Resource.eval(Shutdown[F](shutdownTimeout))
181       wsKey <- Resource.eval(Key.newKey[F, WebSocketContext[F]])
182       _ <- unixSocketConfig.fold(
183         Concurrent[F].background(
184           ServerHelpers
185             .server(
186               host,
187               port,
188               additionalSocketOptions,
189               sg,
190               httpApp(WebSocketBuilder2(wsKey)),
191               tlsInfoOpt,
192               ready,
193               shutdown,
194               errorHandler,
195               onWriteFailure,
196               maxConnections,
197               receiveBufferSize,
198               maxHeaderSize,
199               requestHeaderReceiveTimeout,
200               idleTimeout,
201               logger,
202               wsKey,
203               enableHttp2,
204             )
205             .compile
206             .drain
207         )
208       ) { case (unixSockets, unixSocketAddress, deleteIfExists, deleteOnClose) =>
209         ServerHelpers
210           .unixSocketServer(
211             unixSockets,
212             unixSocketAddress,
213             deleteIfExists,
214             deleteOnClose,
215             httpApp(WebSocketBuilder2(wsKey)),
216             tlsInfoOpt,
217             ready,
218             shutdown,
219             errorHandler,
220             onWriteFailure,
221             maxConnections,
222             receiveBufferSize,
223             maxHeaderSize,
224             requestHeaderReceiveTimeout,
225             idleTimeout,
226             logger,
227             wsKey,
228             enableHttp2,
229           )
230           .compile
231           .drain
232           .background
233       }
234       _ <- Resource.onFinalize(shutdown.await)
235       bindAddress <- Resource.eval(ready.get.rethrow)
236       _ <- Resource.eval(logger.info(s"Ember-Server service bound to address: ${bindAddress}"))
237     } yield new Ip4sServer {
238       def ip4sAddress: SocketAddress[IpAddress] = bindAddress
239       def isSecure: Boolean = tlsInfoOpt.isDefined
240     }
