29   /** forking has similar semantics to parJoin, but there are two key differences.
30     * The first is that inner stream outputs are not shuffled to the forked stream.
31     * The second is that the outer stream may terminate and finalize before inner
32     * streams complete. This is generally unsafe, because inner streams are lexically
33     * scoped within the outer stream and accordingly has resources bound to the outer
34     * stream available in scope. However, network servers built on top of fs2.io can
35     * safely utilize this because inner streams are created fresh from socket Resources
36     * that don't close over any resources from the outer stream.
37     *
38     * One important semantic that the previous prefetch trick didn't have was backpressuring
39     * the outer stream from continuing if the max concurrency is reached, which has been
40     * recovered here.
41     */
42   def forking[F[_], O](streams: Stream[F, Stream[F, O]], maxConcurrency: Int = Int.MaxValue)(
43       implicit F: Concurrent[F]
44   ): Stream[F, INothing] = {
45     val fstream = for {
46       done <- SignallingRef[F, Option[Option[Throwable]]](None)
47       available <- Semaphore[F](maxConcurrency.toLong)
48       running <- SignallingRef[F, Long](1)
49     } yield {
50       val incrementRunning: F[Unit] = running.update(_ + 1)
51       val decrementRunning: F[Unit] = running.update(_ - 1)
52       val awaitWhileRunning: F[Unit] = running.discrete.dropWhile(_ > 0).take(1).compile.drain
53 
54       val stop: F[Unit] = done.update(_.orElse(Some(None)))
55       val stopSignal: Signal[F, Boolean] = done.map(_.nonEmpty)
56       def stopFailed(err: Throwable): F[Unit] =
57         done.update(_.orElse(Some(Some(err))))
58 
59       def runInner(inner: Stream[F, O]): F[Unit] = {
60         val fa = inner
61           .interruptWhen(stopSignal)
62           .compile
63           .drain
64           .handleErrorWith(stopFailed) >> available.release >> decrementRunning
65 
66         available.acquire >> incrementRunning >> F.start(fa).void
67       }
68 
69       val runOuter: F[Unit] =
70         streams
71           .evalMap(runInner(_))
72           .interruptWhen(stopSignal)
73           .compile
74           .drain
75           .handleErrorWith(stopFailed) >> decrementRunning
76 
77       val signalResult: F[Unit] =
78         done.get.flatMap {
79           case Some(Some(err)) => F.raiseError(err)
80           case _ => F.unit
81         }
82 
83       Stream.bracket(F.start(runOuter))(_ => stop >> awaitWhileRunning >> signalResult) >>
84         Stream.eval(awaitWhileRunning).drain
85     }
86 
87     Stream.eval(fstream).flatten
88   }
