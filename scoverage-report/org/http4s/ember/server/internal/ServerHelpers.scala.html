158   def serverInternal[F[_]: Async](
159       server: Stream[F, Socket[F]],
160       httpApp: HttpApp[F],
161       tlsInfoOpt: Option[(TLSContext[F], TLSParameters)],
162       shutdown: Shutdown[F],
163       // Defaults
164       errorHandler: Throwable => F[Response[F]],
165       onWriteFailure: (Option[Request[F]], Response[F], Throwable) => F[Unit],
166       maxConnections: Int,
167       receiveBufferSize: Int,
168       maxHeaderSize: Int,
169       requestHeaderReceiveTimeout: Duration,
170       idleTimeout: Duration,
171       logger: Logger[F],
172       createRequestVault: Boolean,
173       webSocketKey: Key[WebSocketContext[F]],
174       enableHttp2: Boolean,
175   ): Stream[F, Nothing] = {
176     val streams: Stream[F, Stream[F, Nothing]] = server
177       .interruptWhen(shutdown.signal.attempt)
178       .map { connect =>
179         val handler: Stream[F, fs2.INothing] = shutdown.trackConnection >>
180           Stream
181             .resource(upgradeSocket(connect, tlsInfoOpt, logger, enableHttp2))
182             .flatMap {
183               case (socket, Some("h2")) =>
184                 // ALPN H2 Strategy
185                 Stream.exec(H2Server.requireConnectionPreface(socket)) ++
186                   Stream
187                     .resource(
188                       H2Server
189                         .fromSocket[F](
190                           socket,
191                           httpApp,
192                           H2Frame.Settings.ConnectionSettings.default,
193                           logger,
194                         )
195                     )
196                     .drain
197               case (socket, Some(_)) =>
198                 // SSL Connection, not h2, will be http/1.1 but thats not how types align
199                 // Prior Knowledge is only allowed over clear where application
200                 // protocol has not been agreed via handshake
201                 runConnection(
202                   socket,
203                   logger,
204                   idleTimeout,
205                   receiveBufferSize,
206                   maxHeaderSize,
207                   requestHeaderReceiveTimeout,
208                   httpApp,
209                   errorHandler,
210                   onWriteFailure,
211                   createRequestVault,
212                   webSocketKey,
213                   ByteVector.empty,
214                   enableHttp2,
215                 ).drain
216               case (socket, None) => // Cleartext Protocol
217                 enableHttp2 match {
218                   case true =>
219                     // Http2 Prior Knowledge Check, if prelude is first bytes received tread as http2
220                     // Otherwise this is now http1
221                     Stream.eval(H2Server.checkConnectionPreface(socket)).flatMap {
222                       case Left(bv) =>
223                         runConnection(
224                           socket,
225                           logger,
226                           idleTimeout,
227                           receiveBufferSize,
228                           maxHeaderSize,
229                           requestHeaderReceiveTimeout,
230                           httpApp,
231                           errorHandler,
232                           onWriteFailure,
233                           createRequestVault,
234                           webSocketKey,
235                           bv, // Pass read bytes we thought might be the prelude
236                           enableHttp2,
237                         ).drain
238                       case Right(_) =>
239                         Stream
240                           .resource(
241                             H2Server.fromSocket[F](
242                               socket,
243                               httpApp,
244                               H2Frame.Settings.ConnectionSettings.default,
245                               logger,
246                             )
247                           )
248                           .drain
249                     }
250                   // Since its not enabled, run connection normally.
251                   case false =>
252                     runConnection(
253                       socket,
254                       logger,
255                       idleTimeout,
256                       receiveBufferSize,
257                       maxHeaderSize,
258                       requestHeaderReceiveTimeout,
259                       httpApp,
260                       errorHandler,
261                       onWriteFailure,
262                       createRequestVault,
263                       webSocketKey,
264                       ByteVector.empty,
265                       enableHttp2,
266                     ).drain
267                 }
268             }
269 
270         handler.handleErrorWith { t =>
271           Stream.eval(logger.error(t)("Request handler failed with exception")).drain
272         }
273       }
274 
275     streams.parJoin(
276       maxConnections
277     ) // TODO: replace with forking after we fix serverResource upstream
278     // StreamForking.forking(streams, maxConnections)
279   }

292       socketInit: Socket[F],
293       tlsInfoOpt: Option[(TLSContext[F], TLSParameters)],
294       logger: Logger[F],
295       enableHttp2: Boolean,
296   ): Resource[F, (Socket[F], Option[String])] =
297     tlsInfoOpt.fold((socketInit, Option.empty[String]).pure[Resource[F, *]]) {

299         val newParams = if (enableHttp2) {

316   private[internal] def runApp[F[_]](
317       head: Array[Byte],
318       read: Read[F],
319       maxHeaderSize: Int,
320       requestHeaderReceiveTimeout: Duration,
321       httpApp: HttpApp[F],
322       errorHandler: Throwable => F[Response[F]],
323       socket: Socket[F],
324       createRequestVault: Boolean,
325   )(implicit F: Temporal[F], D: Defer[F]): F[(Request[F], Response[F], Drain[F])] = {
326 
327     val parse = Parser.Request.parser(maxHeaderSize)(head, read)
328     val parseWithHeaderTimeout = timeoutToMaybe(
329       parse,
330       requestHeaderReceiveTimeout,
331       D.defer(
332         F.raiseError[(Request[F], F[Option[Array[Byte]]])](
333           EmberException.RequestHeadersTimeout(requestHeaderReceiveTimeout)
334         )
335       ),
336     )
337 
338     for {
339       tmp <- parseWithHeaderTimeout
340       (req, drain) = tmp
341       requestVault <- if (createRequestVault) mkRequestVault(socket) else Vault.empty.pure[F]
342       resp <- httpApp
343         .run(req.withAttributes(requestVault))
344         .handleErrorWith(errorHandler)
345         .handleError(_ => serverFailure.covary[F])
346     } yield (req, resp, drain)
347   }

349   private[internal] def send[F[_]: Temporal](socket: Socket[F])(
350       request: Option[Request[F]],
351       resp: Response[F],
352       idleTimeout: Duration,
353       onWriteFailure: (Option[Request[F]], Response[F], Throwable) => F[Unit],
354   ): F[Unit] =
355     Encoder
356       .respToBytes[F](resp)
357       .through(_.chunks.foreach(c => timeoutMaybe(socket.write(c), idleTimeout)))
358       .compile
359       .drain
360       .attempt
361       .flatMap {
362         case Left(err) => onWriteFailure(request, resp, err)
363         case Right(()) => Applicative[F].unit
364       }

376   private[internal] def runConnection[F[_]: Async](
377       socket: Socket[F],
378       logger: Logger[F],
379       idleTimeout: Duration,
380       receiveBufferSize: Int,
381       maxHeaderSize: Int,
382       requestHeaderReceiveTimeout: Duration,
383       httpApp: HttpApp[F],
384       errorHandler: Throwable => F[org.http4s.Response[F]],
385       onWriteFailure: (Option[Request[F]], Response[F], Throwable) => F[Unit],
386       createRequestVault: Boolean,
387       webSocketKey: Key[WebSocketContext[F]],
388       initialBuffer: ByteVector,
389       enableHttp2: Boolean,
390   ): Stream[F, Nothing] = {
391     type State = (Array[Byte], Boolean)
392     val finalApp = if (enableHttp2) H2Server.h2cUpgradeMiddleware(httpApp) else httpApp
393     val read: Read[F] = timeoutMaybe(socket.read(receiveBufferSize), idleTimeout)
394       .adaptError {
395         // TODO MERGE: Replace with TimeoutException on series/0.23+.
396         case _: TimeoutException => EmberException.ReadTimeout(idleTimeout)
397       }
398     Stream
399       .unfoldEval[F, State, (Request[F], Response[F])](initialBuffer.toArray -> false) {
400         case (buffer, reuse) =>
401           val initRead: F[Array[Byte]] = if (buffer.nonEmpty) {
402             // next request has already been (partially) received
403             buffer.pure[F]
404           } else if (reuse) {
405             // the connection is keep-alive, but we don't have any bytes.
406             // we want to be on the idle timeout until the next request is received.
407             read
408               .flatMap {
409                 case Some(chunk) => chunk.toArray.pure[F]
410                 case None => Concurrent[F].raiseError(EmberException.EmptyStream())
411               }
412           } else {
413             // first request begins immediately
414             Array.emptyByteArray.pure[F]
415           }
416 
417           val result = initRead.flatMap { initBuffer =>
418             runApp(
419               initBuffer,
420               read,
421               maxHeaderSize,
422               requestHeaderReceiveTimeout,
423               finalApp,
424               errorHandler,
425               socket,
426               createRequestVault,
427             )
428           }
429 
430           result.attempt.flatMap {
431             case Right((req, resp, drain)) =>
432               // TODO: Should we pay this cost for every HTTP request?
433               // Intercept the response for various upgrade paths
434               resp.attributes.lookup(webSocketKey) match {
435                 case Some(ctx) =>
436                   drain.flatMap {
437                     case Some(buffer) =>
438                       WebSocketHelpers
439                         .upgrade(
440                           socket,
441                           req,
442                           ctx,
443                           buffer,
444                           receiveBufferSize,
445                           idleTimeout,
446                           onWriteFailure,
447                           errorHandler,
448                           logger,
449                         )
450                         .as(None)
451                     case None =>
452                       Applicative[F].pure(None)
453                   }
454                 case None =>
455                   resp.attributes.lookup(H2Keys.H2cUpgrade) match {
456                     // Http1.1
457                     case None =>
458                       for {
459                         nextResp <- postProcessResponse(req, resp)
460                         _ <- send(socket)(Some(req), nextResp, idleTimeout, onWriteFailure)
461                         nextBuffer <- drain
462                       } yield nextBuffer.map(buffer => ((req, nextResp), (buffer, true)))
463                     // h2c escalation of the connection
464                     case Some((settings, newReq)) =>
465                       for {
466                         nextResp <- postProcessResponse(req, resp)
467                         _ <- send(socket)(Some(req), nextResp, idleTimeout, onWriteFailure)
468                         _ <- H2Server.requireConnectionPreface(socket)
469                         out <- H2Server
470                           .fromSocket(
471                             socket,
472                             httpApp,
473                             H2Frame.Settings.ConnectionSettings.default,
474                             logger,
475                             settings,
476                             newReq.some,
477                           )
478                           .use(_ => Async[F].never[Unit])
479                           .as(None)
480                       } yield out
481                   }
482               }
483             case Left(err) =>
484               err match {
485                 case EmberException.EmptyStream() | EmberException.RequestHeadersTimeout(_) |
486                     EmberException.ReadTimeout(_) =>
487                   Applicative[F].pure(None)
488                 case err =>
489                   errorHandler(err)
490                     .handleError(_ => serverFailure.covary[F])
491                     .flatMap(send(socket)(None, _, idleTimeout, onWriteFailure))
492                     .as(None)
493               }
494           }
495       }
496       .takeWhile { case (_, resp) =>
497         resp.headers.get[Connection].exists(_.hasKeepAlive)
498       }
499       .drain
500   }

512           secure = socket.isInstanceOf[TLSSocket[F]],

524         Vault.empty.pure[F]
