26 final class EmberClient[F[_]] private[client] (
27     private val client: Client[F],
28     private val pool: KeyPool[F, RequestKey, EmberConnection[F]],
29 )(implicit F: MonadCancelThrow[F])
30     extends DefaultClient[F] {
31 
32   /** The reason for this extra class. This allows you to see the present state
33     * of the underlying Pool, without having access to the pool itself.
34     *
35     * The first element represents total connections in the pool, the second
36     * is a mapping between the number of connections in the pool for each requestKey.
37     */
38   def state: F[(Int, Map[RequestKey, Int])] = pool.state
39 
40   def run(req: Request[F]): Resource[F, Response[F]] = client.run(req)
41 
42   override def defaultOnError(req: Request[F])(resp: Response[F])(implicit
43       G: Applicative[F]
44   ): F[Throwable] = F match {
45     case concurrentF: Concurrent[F @unchecked] =>
46       implicit val C = concurrentF
47       resp.body.compile.drain.as(UnexpectedStatus(resp.status, req.method, req.uri))
48 
49     case _ =>
50       F.pure(UnexpectedStatus(resp.status, req.method, req.uri))
51   }
52 }
