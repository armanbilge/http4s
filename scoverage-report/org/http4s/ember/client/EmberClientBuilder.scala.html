198   /** Push promises are implemented via responding with a PushPromise frame
199     * which is effectively a request headers frame for a request that wasn't
200     * sent by the client.
201     *
202     * The second param is the response once it is available that you can wait
203     * for OR you can cancel the Outcome to send a termination signal to
204     * ask the remote server to stop sending additional data from this data stream.
205     * If you want to handle these the outcome can just be outcome successful. But
206     * you can save significant data by canceling requests you don't want.
207     *
208     * Push promises are very useful to get all the data necessary to render a page in parallel
209     * to the actual data for that page leading to much faster render times, or sending
210     * additional cache enriching information.

225   def build: Resource[F, Client[F]] =
226     for {
227       sg <- Resource.pure(sgOpt.getOrElse(Network[F]))
228       tlsContextOptWithDefault <- Resource.eval(
229         tlsContextOpt.fold(Network[F].tlsContext.system.attempt.map(_.toOption))(_.some.pure[F])
230       )
231       builder =
232         KeyPool.Builder
233           .apply[F, RequestKey, EmberConnection[F]](
234             (requestKey: RequestKey) =>
235               EmberConnection(
236                 org.http4s.ember.client.internal.ClientHelpers
237                   .requestKeyToSocketWithKey[F](
238                     requestKey,
239                     tlsContextOptWithDefault,
240                     checkEndpointIdentification,
241                     sg,
242                     additionalSocketOptions,
243                   )
244               ) <* logger.trace(s"Created Connection - RequestKey: ${requestKey}"),
245             (connection: EmberConnection[F]) =>
246               logger.trace(
247                 s"Shutting Down Connection - RequestKey: ${connection.keySocket.requestKey}"
248               ) >>
249                 connection.cleanup,
250           )
251           .withDefaultReuseState(Reusable.DontReuse)
252           .withIdleTimeAllowedInPool(idleTimeInPool)
253           .withMaxPerKey(maxPerKey)
254           .withMaxTotal(maxTotal)
255           .withOnReaperException(_ => Applicative[F].unit)
256       pool <- builder.build
257       optH2 <- (if (enableHttp2) tlsContextOptWithDefault else None).traverse { context =>
258         H2Client.impl[F](
259           pushPromiseSupport.getOrElse { case (_, _) => Applicative[F].pure(Outcome.canceled) },
260           context,
261           logger,
262           if (pushPromiseSupport.isDefined) default
263           else
264             default.copy(enablePush = H2Frame.Settings.SettingsEnablePush(false)),
265         )
266       }
267     } yield {
268       def webClient(request: Request[F]): Resource[F, Response[F]] =
269         for {
270           managed <- ClientHelpers.getValidManaged(pool, request)
271           _ <- Resource.eval(
272             pool.state.flatMap { poolState =>
273               logger.trace(
274                 s"Connection Taken - Key: ${managed.value.keySocket.requestKey} - Reused: ${managed.isReused} - PoolState: $poolState"
275               )
276             }
277           )
278           responseResource <- Resource.makeCaseFull((poll: Poll[F]) =>
279             poll(
280               ClientHelpers
281                 .request[F](
282                   request,
283                   managed.value,
284                   chunkSize,
285                   maxResponseHeaderSize,
286                   idleConnectionTime,
287                   timeout,
288                   userAgent,
289                 )
290             )
291           ) { case ((response, drain), exitCase) =>
292             exitCase match {
293               case Resource.ExitCase.Succeeded =>
294                 ClientHelpers.postProcessResponse(
295                   request,
296                   response,
297                   drain,
298                   managed.value.nextBytes,
299                   managed.canBeReused,
300                 )
301               case _ => Applicative[F].unit
302             }
303           }
304         } yield responseResource._1
305 
306       def unixSocketClient(
307           request: Request[F],
308           address: UnixSocketAddress,
309       ): Resource[F, Response[F]] =
310         Resource
311           .eval(ApplicativeThrow[F].catchNonFatal(unixSockets.getOrElse(UnixSockets.forAsync[F])))
312           .flatMap(unixSockets =>
313             Resource
314               .make(
315                 EmberConnection(
316                   ClientHelpers.unixSocket(request, unixSockets, address, tlsContextOpt)
317                 )
318               )(ec => ec.shutdown)
319           )
320           .flatMap(connection =>
321             Resource.eval(
322               ClientHelpers
323                 .request[F](
324                   request,
325                   connection,
326                   chunkSize,
327                   maxResponseHeaderSize,
328                   idleConnectionTime,
329                   timeout,
330                   userAgent,
331                 )
332                 .map(_._1)
333             )
334           )
335       val client = Client[F] { request =>
336         request.attributes
337           .lookup(Request.Keys.UnixSocketAddress)
338           .fold(webClient(request)) { (address: UnixSocketAddress) =>
339             unixSocketClient(request, address)
340           }
341       }
342       val stackClient = Retry.create(retryPolicy, logRetries = false)(client)
343       val iClient = new EmberClient[F](stackClient, pool)
344 
345       optH2.fold(iClient) { h2 =>
346         val h2Client = Client(h2(iClient.run))
347         new EmberClient(h2Client, pool)
348       }
349     }
