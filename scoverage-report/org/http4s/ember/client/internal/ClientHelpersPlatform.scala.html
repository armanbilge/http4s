31   private[internal] def mkTLSParameters(
32       address: Option[SocketAddress[Host]],
33       enableEndpointValidation: Boolean,
34   ): TLSParameters =
35     TLSParameters(
36       serverNames = address.map(a => List(extractHostname(a.host))),
37       endpointIdentificationAlgorithm = if (enableEndpointValidation) Some("HTTPS") else None,
38     )
39 
40   @tailrec
41   private def extractHostname(from: Host): SNIHostName = from match {
42     case hostname: Hostname => new SNIHostName(hostname.normalized.toString)

44     case idn: IDN => extractHostname(idn.hostname)
45   }
