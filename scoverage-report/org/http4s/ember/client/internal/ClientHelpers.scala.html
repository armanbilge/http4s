132   def request[F[_]: Async](
133       request: Request[F],
134       connection: EmberConnection[F],
135       chunkSize: Int,
136       maxResponseHeaderSize: Int,
137       idleTimeout: Duration,
138       timeout: Duration,
139       userAgent: Option[`User-Agent`],
140   ): F[(Response[F], F[Option[Array[Byte]]])] = {
141 
142     def writeRequestToSocket(req: Request[F], socket: Socket[F]): F[Unit] =
143       Encoder
144         .reqToBytes(req)
145         .through(_.chunks.foreach(c => timeoutMaybe(socket.write(c), idleTimeout)))
146         .compile
147         .drain
148 
149     def writeRead(req: Request[F]): F[(Response[F], F[Option[Array[Byte]]])] =
150       writeRequestToSocket(req, connection.keySocket.socket) >>
151         connection.nextBytes.getAndSet(Array.emptyByteArray).flatMap { head =>
152           val parse = Parser.Response.parser(maxResponseHeaderSize)(
153             head,
154             timeoutMaybe(connection.keySocket.socket.read(chunkSize), idleTimeout),
155           )
156           timeoutToMaybe(
157             parse,
158             timeout,
159             Defer[F].defer(
160               ApplicativeThrow[F].raiseError(
161                 new java.util.concurrent.TimeoutException(
162                   s"Timed Out on EmberClient Header Receive Timeout: $timeout"
163                 )
164               )
165             ),
166           )
167         }
168 
169     for {
170       processedReq <- preprocessRequest(request, userAgent)
171       res <- writeRead(processedReq)
172     } yield res
173   }.adaptError { case e: EmberException.EmptyStream =>
174     new ClosedChannelException() {
175       initCause(e)
176 
177       override def getMessage(): String =
178         "Remote Disconnect: Received zero bytes after sending request"
179     }
180   }

182   private[internal] def preprocessRequest[F[_]: Monad: Clock](
183       req: Request[F],
184       userAgent: Option[`User-Agent`],
185   ): F[Request[F]] = {
186     val connection = req.headers
187       .get[Connection]
188       .fold(Connection(NonEmptyList.of(ci"keep-alive")))(identity)
189     val userAgentHeader: Option[`User-Agent`] = req.headers.get[`User-Agent`].orElse(userAgent)
190     for {
191       date <- req.headers.get[Date].fold(HttpDate.current[F].map(Date(_)))(_.pure[F])
192     } yield req
193       .putHeaders(date, connection)
194       .putHeaders(userAgentHeader)
195   }

197   private[ember] def postProcessResponse[F[_]](
198       req: Request[F],
199       resp: Response[F],
200       drain: F[Option[Array[Byte]]],
201       nextBytes: Ref[F, Array[Byte]],
202       canBeReused: Ref[F, Reusable],
203   )(implicit F: Concurrent[F]): F[Unit] =
204     drain.flatMap {
205       case Some(bytes) =>
206         val requestClose = connectionFor(req.httpVersion, req.headers).hasClose
207         val responseClose = connectionFor(resp.httpVersion, resp.headers).hasClose
208 
209         if (requestClose || responseClose) F.unit
210         else nextBytes.set(bytes) >> canBeReused.set(Reusable.Reuse)
211       case None => F.unit
212     }

226   private[client] def getValidManaged[F[_]: Sync](
227       pool: KeyPool[F, RequestKey, EmberConnection[F]],
228       request: Request[F],
229   ): Resource[F, Managed[F, EmberConnection[F]]] =
230     pool.take(RequestKey.fromRequest(request)).flatMap { managed =>
231       Resource
232         .eval(managed.value.keySocket.socket.isOpen)
233         .ifM(
234           managed.pure[Resource[F, *]],
235           // Already Closed,
236           // The Resource Scopes Aren't doing us anything
237           // if we have max removed from pool we will need to revisit
238           if (managed.isReused) {
239             Resource.eval(managed.canBeReused.set(Reusable.DontReuse)) >>
240               getValidManaged(pool, request)
241           } else
242             Resource.eval(
243               Sync[F].raiseError(
244                 new fs2.io.net.SocketException("Fresh connection from pool was not open")
245               )
246             ),
247         )
248     }

250   private[ember] object RetryLogic {
251     private val retryNow = 0.seconds.some
252     def retryUntilFresh[F[_]]: RetryPolicy[F] = { (req, result, retries) =>
253       if (emberDeadFromPoolPolicy(req, result) && retries <= 2) retryNow
254       else None
255     }
256 
257     def emberDeadFromPoolPolicy[F[_]](
258         req: Request[F],
259         result: Either[Throwable, Response[F]],
260     ): Boolean =
261       req.isIdempotent && isRetryableError(result)
262 
263     def isRetryableError[F[_]](result: Either[Throwable, Response[F]]): Boolean =
264       result match {
265         case Right(_) => false
266         case Left(_: ClosedChannelException) => true
267         case Left(ex: IOException) =>
268           val msg = ex.getMessage()
269           msg == "Connection reset by peer" || msg == "Broken pipe"
270         case _ => false
271       }
272   }
