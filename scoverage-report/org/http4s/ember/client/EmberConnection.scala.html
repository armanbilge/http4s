25 private[ember] final case class EmberConnection[F[_]](
26     keySocket: RequestKeySocket[F],
27     shutdown: F[Unit],
28     nextBytes: Ref[F, Array[Byte]],
29 )(implicit F: MonadThrow[F]) {
30   def cleanup: F[Unit] =
31     nextBytes.set(Array.emptyByteArray) >>
32       keySocket.socket.endOfInput.attempt.void >>
33       keySocket.socket.endOfOutput.attempt.void >>
34       shutdown.attempt.void
35 }
