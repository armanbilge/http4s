43   /** From fs2-http
44     * decodes from the HTTP chunked encoding. After last chunk this terminates. Allows to specify max header size, after which this terminates
45     * Please see https://en.wikipedia.org/wiki/Chunked_transfer_encoding for details
46     */
47   def decode[F[_]](
48       head: Array[Byte],
49       read: F[Option[Chunk[Byte]]],
50       maxHeaderSize: Int,
51       maxChunkHeaderSize: Int,
52       trailers: Deferred[F, Headers],
53       rest: Ref[F, Option[Array[Byte]]],
54   )(implicit F: MonadThrow[F]): Stream[F, Byte] = {
55     // on left reading the header of chunk (acting as buffer)
56     // on right reading the chunk itself, and storing remaining bytes of the chunk
57     def go(expect: Either[ByteVector, Long], head: ByteVector): Pull[F, Byte, Unit] = {
58       val nextBytes =
59         if (head.nonEmpty)
60           Pull.pure(Some(head))
61         else
62           Pull.eval(read).map(_.map(_.toByteVector))
63 
64       nextBytes.flatMap {
65         case None =>
66           Pull.raiseError(EmberException.ReachedEndOfStream())
67 
68         case Some(bv) =>
69           expect match {
70             case Left(header) =>
71               val nh = header ++ bv
72               val endOfHeader = nh.indexOfSlice(backslashrbackslashn)
73               if (endOfHeader == 0)
74                 // strip any leading crlf on header, as this starts with /r/n
75                 go(expect, nh.drop(backslashrbackslashn.size))
76               else if (endOfHeader < 0 && nh.size > maxChunkHeaderSize)
77                 Pull.raiseError[F](
78                   EmberException.ChunkedEncodingError(
79                     s"Failed to get Chunk header. Size exceeds max($maxChunkHeaderSize) : ${nh.size} ${nh.decodeUtf8}"
80                   )
81                 )
82               else if (endOfHeader < 0) go(Left(nh), ByteVector.empty)
83               else {
84                 val (hdr, rem) = nh.splitAt(endOfHeader + backslashrbackslashn.size)
85                 readChunkedHeader(hdr.dropRight(backslashrbackslashn.size)) match {
86                   case None =>
87                     Pull.raiseError[F](
88                       EmberException.ChunkedEncodingError(
89                         s"Failed to parse chunked header : ${hdr.decodeUtf8}"
90                       )
91                     )
92                   case Some(0) =>
93                     // Done With Message, Now Parse Trailers
94                     Pull.eval(
95                       parseTrailers[F](maxHeaderSize)(rem.toArray, read)
96                         .flatMap { t =>
97                           trailers.complete(t.headers) >> rest.set(Some(t.rest))
98                         }
99                     ) >> Pull.done
100                   case Some(sz) => go(Right(sz), rem)
101                 }
102               }
103 
104             case Right(remains) =>
105               if (remains > bv.size)
106                 Pull.output(Chunk.byteVector(bv)) >> go(Right(remains - bv.size), ByteVector.empty)
107               else {
108                 val (out, next) = bv.splitAt(remains.toLong)
109                 Pull.output(Chunk.byteVector(out)) >> go(Left(ByteVector.empty), next)
110               }
111           }
112       }
113     }
114 
115     go(Left(ByteVector.empty), ByteVector.view(head)).stream
116   }
117 
118   final case class Trailers(headers: Headers, rest: Array[Byte])
119 
120   private def parseTrailers[F[_]: MonadThrow](
121       maxHeaderSize: Int
122   )(buffer: Array[Byte], read: F[Option[Chunk[Byte]]]): F[Trailers] =
123     if (buffer.startsWith(Shared.backslashrbackslashn.toArray)) {
124       Trailers(Headers.empty, buffer.drop(backslashrbackslashn.size.toInt)).pure[F]
125     } else if (buffer.length < 2) {
126       read.flatMap {
127         case None =>
128           MonadThrow[F].raiseError(EmberException.ReachedEndOfStream())
129         case Some(chunk) =>
130           parseTrailers(maxHeaderSize)(buffer ++ chunk.toArray[Byte], read)
131       }
132     } else {
133       Parser.MessageP
134         .recurseFind(buffer, read, maxHeaderSize)(buffer =>
135           Parser.HeaderP.parseHeaders(buffer, 0, maxHeaderSize)
136         )(_.idx)
137         .map { case (headerP, rest) => Trailers(headerP.headers, rest) }
138     }
139 
140   private val lastChunk: Chunk[Byte] =
141     Chunk.byteVector((ByteVector('0') ++ backslashrbackslashn ++ backslashrbackslashn).compact)
142 
143   /** Encodes chunk of bytes to http chunked encoding.
144     */
145   def encode[F[_]]: Pipe[F, Byte, Byte] = {
146     def encodeChunk(bv: ByteVector): Chunk[Byte] =
147       if (bv.isEmpty) Chunk.empty
148       else
149         Chunk.byteVector(
150           ByteVector.view(bv.size.toHexString.toUpperCase.getBytes) ++ backslashrbackslashn ++ bv ++ backslashrbackslashn
151         )
152     _.mapChunks { ch =>
153       encodeChunk(ch.toByteVector)
154     } ++ Stream.chunk(lastChunk)
155   }
156 
157   /** yields to size of header in case the chunked header was succesfully parsed, else yields to None */
158   private def readChunkedHeader(hdr: ByteVector): Option[Long] =
159     hdr.decodeUtf8.toOption.flatMap { s =>
160       val parts = s.split(';') // lets ignore any extensions
161       if (parts.isEmpty) None
162       else
163         try Some(java.lang.Long.parseLong(parts(0).trim, 16))
164         catch { case NonFatal(_) => None }
165     }
