36   def respToBytes[F[_]](resp: Response[F], writeBufferSize: Int = 32 * 1024): Stream[F, Byte] = {
37     var chunked = resp.isChunked
38     // resp.status.isEntityAllowed TODO
39     val initSection = {
40       var appliedContentLength = false
41       val stringBuilder = new StringBuilder()
42 
43       // Response Prelude: HTTP-Version SP STATUS CRLF
44       stringBuilder
45         .append(resp.httpVersion.renderString)
46         .append(SPACE)
47         .append(resp.status.renderString)
48         .append(CRLF)
49 
50       resp.headers
51         .get[`Content-Length`]
52         .foreach { _ =>
53           appliedContentLength = true
54         }
55 
56       // Apply each header followed by a CRLF
57       resp.headers.foreach { h =>
58         if (h.isNameValid) {
59           stringBuilder
60             .append(h.name)
61             .append(": ")
62           appendSanitized(stringBuilder, h.value)
63           stringBuilder.append(CRLF)
64           ()
65         }
66       }
67       if (!appliedContentLength && resp.body == EmptyBody && resp.status.isEntityAllowed) {
68         stringBuilder.append(zeroContentLengthRaw).append(CRLF)
69         chunked = false
70         ()
71       } else if (!chunked && !appliedContentLength && resp.status.isEntityAllowed) {
72         stringBuilder.append(chunkedTransferEncodingHeaderRaw).append(CRLF)
73         chunked = true
74         ()
75       }
76       // Final CRLF terminates headers and signals body to follow.
77       stringBuilder.append(CRLF)
78       stringBuilder.toString.getBytes(StandardCharsets.ISO_8859_1)
79     }
80 
81     if (chunked)
82       Stream.chunk(Chunk.array(initSection)) ++ resp.body.through(ChunkedEncoding.encode[F])
83     else
84       (Stream.chunk(Chunk.array(initSection)) ++ resp.body)
85         .chunkMin(writeBufferSize)
86         .flatMap(Stream.chunk)
87   }

89   private val NoPayloadMethods: Set[Method] =
90     Set(Method.GET, Method.DELETE, Method.CONNECT, Method.TRACE)

92   def reqToBytes[F[_]: ApplicativeThrow](
93       req: Request[F],
94       writeBufferSize: Int = 32 * 1024,
95   ): Stream[F, Byte] = {
96     val uriOriginFormString = req.uri.toOriginForm.renderString
97 
98     if (uriOriginFormString.exists(ForbiddenUriCharacters)) {
99       Stream.raiseError(new IllegalArgumentException(s"Invalid URI: ${uriOriginFormString}"))
100     } else {
101       var chunked = req.isChunked
102       val initSection = {
103         var appliedContentLength = false
104         val stringBuilder = new StringBuilder()
105 
106         // Request-Line   = Method SP Request-URI SP HTTP-Version CRLF
107         stringBuilder
108           .append(req.method.renderString)
109           .append(SPACE)
110           .append(uriOriginFormString)
111           .append(SPACE)
112           .append(req.httpVersion.renderString)
113           .append(CRLF)
114 
115         // Host From Uri Becomes Header if not already present in headers
116         if (!req.headers.contains[Host])
117           req.uri.authority.foreach { auth =>
118             stringBuilder
119               .append("Host: ")
120               .append(auth.renderString)
121               .append(CRLF)
122           }
123 
124         req.headers
125           .get[`Content-Length`]
126           .foreach { _ =>
127             appliedContentLength = true
128           }
129 
130         // Apply each header followed by a CRLF
131         req.headers.foreach { h =>
132           if (h.isNameValid) {
133             stringBuilder
134               .append(h.name)
135               .append(": ")
136             appendSanitized(stringBuilder, h.value)
137             stringBuilder.append(CRLF)
138             ()
139           }
140         }
141 
142         if (!chunked && !appliedContentLength && !NoPayloadMethods.contains(req.method)) {
143           stringBuilder.append(chunkedTransferEncodingHeaderRaw).append(CRLF)
144           chunked = true
145           ()
146         }
147 
148         // Final CRLF terminates headers and signals body to follow.
149         stringBuilder.append(CRLF)
150         stringBuilder.toString.getBytes(StandardCharsets.ISO_8859_1)
151       }
152       if (chunked)
153         Stream.chunk(Chunk.array(initSection)) ++ req.body.through(ChunkedEncoding.encode[F])
154       else
155         (Stream.chunk(Chunk.array(initSection)) ++ req.body)
156           .chunkMin(writeBufferSize)
157           .flatMap(Stream.chunk)
158     }
159   }
160 
161   private val ForbiddenUriCharacters = CharPredicate(0x0.toChar, '\r', '\n')
