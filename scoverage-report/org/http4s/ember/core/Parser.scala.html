36     def recurseFind[F[_], A](currentBuffer: Array[Byte], read: Read[F], maxHeaderSize: Int)(
37         f: Array[Byte] => F[Either[Unit, A]]
38     )(idx: A => Int)(implicit F: MonadThrow[F]): F[(A, Array[Byte])] =
39       f(currentBuffer).flatMap {
40         case Left(_) =>
41           if (currentBuffer.length > maxHeaderSize)
42             F.raiseError(EmberException.MessageTooLong(maxHeaderSize))
43           else {
44             read.flatMap {
45               case Some(chunk) =>
46                 val nextBuffer = combineArrays(currentBuffer, chunk.toArray)
47                 recurseFind(nextBuffer, read, maxHeaderSize)(f)(idx)
48               case None if currentBuffer.length > 0 =>
49                 F.raiseError(EmberException.ReachedEndOfStream())
50               case _ => F.raiseError(EmberException.EmptyStream())
51             }
52           }
53         case Right(out) =>
54           (out, currentBuffer.drop(idx(out))).pure[F]
55       }

73     def parseHeaders[F[_]](message: Array[Byte], initIndex: Int, maxHeaderSize: Int)(implicit
74         F: MonadThrow[F]
75     ): F[Either[Unit, HeaderP]] = {
76       import scala.collection.mutable.ListBuffer
77       var idx = initIndex
78       var state = false
79       var throwable: Throwable = null
80       var complete = false
81       var chunked: Boolean = false
82       var contentLength: Option[Long] = None
83 
84       val headers: ListBuffer[Header.Raw] = ListBuffer()
85       var name: String = null
86       var start = initIndex
87       val upperBound = Math.min(message.size - 1, maxHeaderSize)
88 
89       while (!complete && idx <= upperBound) {
90         if (!state) {
91           val current = message(idx)
92           // if current index is colon our name is complete
93           if (current == colon) {
94             state = true // set state to check for header value
95             name = new String(message, start, idx - start) // extract name string
96             start = idx + 1 // advance past colon for next start
97 
98             // TODO: This if clause may not be necessary since the header value parser trims
99             if (message.size > idx + 1 && message(idx + 1) == space) {
100               start += 1 // if colon is followed by space advance again
101               idx += 1 // double advance index here to skip the space
102             }
103             // double CRLF condition - Termination of headers
104           } else if (current == lf && (idx > 0 && message(idx - 1) == cr)) {
105             complete = true // completed terminate loop
106           }
107         } else {
108           val current = message(idx)
109           // If crlf is next we have completed the header value
110           if (current == lf && (idx > 0 && message(idx - 1) == cr)) {
111             // extract header value, trim leading and trailing whitespace
112             val hValue = new String(message, start, idx - start - 1).trim
113 
114             val hName = name // copy var to val
115             name = null // set name back to null
116             val newHeader = Header.Raw(CIString(hName), hValue) // create header
117             if (hName.equalsIgnoreCase(contentLengthS)) { // Check if this is content-length.
118               try contentLength = hValue.toLong.some
119               catch {
120                 case scala.util.control.NonFatal(e) =>
121                   throwable = e
122                   complete = true
123               }
124             } else if (
125               hName
126                 .equalsIgnoreCase(transferEncodingS)
127             ) { // Check if this is Transfer-encoding
128               chunked = hValue.contains(chunkedS)
129             }
130             start = idx + 1 // Next Start is after the CRLF
131             headers += newHeader // Add Header
132             state = false // Go back to Looking for HeaderName or Termination
133           }
134         }
135         idx += 1 // Single Advance Every Iteration
136       }
137 
138       if (throwable != null) {
139         F.raiseError(ParseHeadersError(throwable))
140       } else if (!complete) {
141         ().asLeft.pure[F]
142       } else {
143         HeaderP(Headers(headers.toList), chunked, contentLength, idx).asRight.pure[F]
144       }
145     }

164       def parsePrelude[F[_]](message: Array[Byte], maxHeaderSize: Int)(implicit
165           F: MonadThrow[F]
166       ): F[Either[Unit, ReqPrelude]] = {
167         var idx = 0
168         var state: Byte = 0
169         var complete = false
170 
171         var throwable: Throwable = null
172         var method: Method = null
173         var uri: Uri = null
174         var httpVersion: HttpVersion = null
175         val upperBound = Math.min(message.size - 1, maxHeaderSize)
176 
177         var start = 0
178         while (!complete && idx <= upperBound) {
179           val value = message(idx)
180           (state: @switch) match {
181             case 0 =>
182               if (value == space) {
183                 Method.fromString(new String(message, start, idx - start)) match {
184                   case Left(e) =>
185                     throwable = e
186                     complete = true
187                   case Right(m) =>
188                     method = m
189                 }
190                 start = idx + 1
191                 state = 1
192               }
193             case 1 =>
194               if (value == space) {
195                 Uri.fromString(new String(message, start, idx - start)) match {
196                   case Left(e) =>
197                     throwable = e
198                     complete = true
199                   case Right(u) =>
200                     uri = u
201                 }
202                 start = idx + 1
203                 state = 2
204               }
205             case 2 =>
206               if (value == lf && (idx > 0 && message(idx - 1) == cr)) {
207                 HttpVersion.fromString(new String(message, start, idx - start - 1)) match {
208                   case Left(e) =>
209                     throwable = e
210                     complete = true
211                   case Right(h) =>
212                     httpVersion = h
213                 }
214                 complete = true
215               }
216           }
217           idx += 1
218         }
219 
220         if (throwable != null)
221           F.raiseError(
222             ParsePreludeError(
223               throwable.getMessage(),
224               Option(throwable),
225               Option(method),
226               Option(uri),
227               Option(httpVersion),
228             )
229           )
230         else if (method == null || uri == null || httpVersion == null)
231           ().asLeft.pure[F]
232         else
233           ReqPrelude(method, uri, httpVersion, idx).asRight.pure[F]
234       }

248     def parser[F[_]](maxHeaderSize: Int)(
249         buffer: Array[Byte],
250         read: Read[F],
251     )(implicit F: Concurrent[F]): F[(Request[F], Drain[F])] =
252       for {
253         t <- MessageP.recurseFind(buffer, read, maxHeaderSize)(ibuffer =>
254           EitherT(ReqPrelude.parsePrelude[F](ibuffer, maxHeaderSize))
255             .flatMap(prelude =>
256               EitherT(HeaderP.parseHeaders(ibuffer, prelude.nextIndex, maxHeaderSize))
257                 .tupleLeft(prelude)
258             )
259             .value
260         ) { case (_, headerP) => headerP.idx }
261         ((prelude, headerP), finalBuffer) = t
262 
263         baseReq = org.http4s.Request[F](
264           method = prelude.method,
265           uri = prelude.uri,
266           httpVersion = prelude.version,
267           headers = headerP.headers,
268         )
269 
270         request <-
271           if (headerP.chunked) {
272             Ref.of[F, Option[Array[Byte]]](None).product(Deferred[F, Headers]).map {
273               case (rest, trailers) =>
274                 (
275                   baseReq
276                     .withAttribute(Message.Keys.TrailerHeaders[F], trailers.get)
277                     .withBodyStream(
278                       ChunkedEncoding
279                         .decode(finalBuffer, read, maxHeaderSize, maxHeaderSize, trailers, rest)
280                     ),
281                   rest.get,
282                 )
283             }
284           } else {
285             Body.parseFixedBody(headerP.contentLength.getOrElse(0L), finalBuffer, read).map {
286               case (bodyStream, drain) =>
287                 (baseReq.withBodyStream(bodyStream), drain)
288             }
289           }
290       } yield request
291   }

296         buffer: Array[Byte],
297         read: Read[F],
298     ): F[(Response[F], Drain[F])] = {
299       // per https://httpwg.org/specs/rfc7230.html#rfc.section.3.3.3
300       def expectNoBody(status: Status): Boolean =
301         status == Status.NoContent ||
302           status == Status.NotModified ||
303           status.responseClass == Status.Informational
304 
305       for {
306         t <- MessageP.recurseFind(buffer, read, maxHeaderSize)(ibuffer =>
307           EitherT(RespPrelude.parsePrelude[F](ibuffer, maxHeaderSize))
308             .flatMap(prelude =>
309               EitherT(HeaderP.parseHeaders(ibuffer, prelude.nextIndex, maxHeaderSize))
310                 .tupleLeft(prelude)
311             )
312             .value
313         ) { case (_, headerP) => headerP.idx }
314         ((prelude, headerP), finalBuffer) = t
315 
316         baseResp = org.http4s.Response[F](
317           httpVersion = prelude.version,
318           status = prelude.status,
319           headers = headerP.headers,
320         )
321 
322         resp <-
323           if (headerP.chunked) {
324             Ref.of[F, Option[Array[Byte]]](None).product(Deferred[F, Headers]).map {
325               case (rest, trailers) =>
326                 baseResp
327                   .withAttribute(Message.Keys.TrailerHeaders[F], trailers.get)
328                   .withBodyStream(
329                     ChunkedEncoding
330                       .decode(finalBuffer, read, maxHeaderSize, maxHeaderSize, trailers, rest)
331                   ) ->
332                   rest.get
333             }
334           } else if (expectNoBody(prelude.status)) {
335             (baseResp -> (Some(finalBuffer): Option[Array[Byte]]).pure[F]).pure[F]
336           } else {
337             headerP.contentLength
338               .fold(Body.parseUnknownBody(finalBuffer, read))(
339                 Body.parseFixedBody(_, finalBuffer, read)
340               )
341               .map { case (bodyStream, drain) =>
342                 baseResp.withBodyStream(bodyStream) -> drain
343               }
344           }
345       } yield resp
346     }

352       // HTTP/1.1 200 OK
353       // Status-Line = HTTP-Version SP Status-Code SP Reason-Phrase CRLF
354       def parsePrelude[F[_]](buffer: Array[Byte], maxHeaderSize: Int)(implicit
355           F: MonadThrow[F]
356       ): F[Either[Unit, RespPrelude]] = {
357         var complete = false
358         var idx = 0
359         var throwable: Throwable = null
360         var httpVersion: HttpVersion = null
361 
362         var codeS: String = null
363         // val reason: String = null
364         var status: Status = null
365         var start = 0
366         var state = 0 // 0 Is for HttpVersion, 1 for Status Code, 2 For Reason Phrase
367         val upperBound = Math.min(buffer.size - 1, maxHeaderSize)
368 
369         while (!complete && idx <= upperBound) {
370           val value = buffer(idx)
371           (state: @switch) match {
372             case 0 =>
373               if (value == space) {
374                 val s = new String(buffer, start, idx - start)
375                 HttpVersion.fromString(s) match {
376                   case Left(e) =>
377                     throwable = e
378                     complete = true
379                   case Right(h) =>
380                     httpVersion = h
381                 }
382                 start = idx + 1
383                 state = 1
384               }
385             case 1 =>
386               if (value == space) {
387                 codeS = new String(buffer, start, idx - start)
388                 state = 2
389                 start = idx + 1
390               }
391             case 2 =>
392               if (value == lf && (idx > 0 && buffer(idx - 1) == cr)) {
393                 try {
394                   val codeInt = codeS.toInt
395                   Status.fromInt(codeInt) match {
396                     case Left(e) =>
397                       throw e
398                     case Right(s) =>
399                       status = s
400                       complete = true
401                   }
402                 } catch {
403                   case scala.util.control.NonFatal(e) =>
404                     throwable = e
405                     complete = true
406                 }
407               }
408           }
409           idx += 1
410         }
411 
412         if (throwable != null)
413           F.raiseError(RespPreludeError("Encountered Error parsing", Option(throwable)))
414         else if (httpVersion == null || status == null)
415           ().asLeft.pure[F]
416         else
417           RespPrelude(httpVersion, status, idx).asRight.pure[F]
418       }
419 
420       case class RespPreludeError(message: String, cause: Option[Throwable])
421           extends Exception(
422             s"Received Error while parsing prelude - Message: $message - ${cause.map(_.getMessage)}",
423             cause.orNull,
424           ) // scalafix:ok Http4sGeneralLinters.noCaseClassWithoutAccessModifier; bincompat until 1.0

429     def parseFixedBody[F[_]: Concurrent](
430         contentLength: Long,
431         buffer: Array[Byte],
432         read: Read[F],
433     ): F[(EntityBody[F], Drain[F])] =
434       if (contentLength > 0) {
435         if (buffer.length >= contentLength) {
436           val (body, rest) = buffer.splitAt(contentLength.toInt)
437           (
438             Stream.chunk(Chunk.byteVector(ByteVector(body))).covary[F],
439             (Some(rest): Option[Array[Byte]]).pure[F],
440           )
441             .pure[F]
442         } else {
443           val unread = contentLength - buffer.length
444           Ref.of[F, Option[Array[Byte]]](None).map { state =>
445             val body = decode(unread, buffer, state, read)
446             // If the remaining bytes for the body have not yet been read, close the connection.
447             // followup: Check if there are bytes immediately available without blocking
448             val drain: Drain[F] = state.get
449 
450             (body, drain)
451           }
452         }
453       } else {
454         (EmptyBody.covary[F], (Some(buffer): Option[Array[Byte]]).pure[F]).pure[F]
455       }

457     def parseUnknownBody[F[_]: Concurrent](
458         buffer: Array[Byte],
459         read: Read[F],
460     ): F[(EntityBody[F], Drain[F])] =
461       Ref[F].of(false).map { consumed =>
462         lazy val readAll: Pull[F, Byte, Unit] =
463           Pull.eval(read).flatMap {
464             case Some(c) => Pull.output(c) >> readAll
465             case None => Pull.eval(consumed.set(true)).void
466           }
467 
468         val body =
469           Pull
470             .eval(consumed.get)
471             .flatMap {
472               case true => Pull.raiseError(BodyAlreadyConsumedError())
473               case false => Pull.output(Chunk.array(buffer)) >> readAll
474             }
475             .stream
476 
477         val drain: Drain[F] = (None: Option[Array[Byte]]).pure[F]
478 
479         (body, drain)
480       }
481 
482     final case class BodyAlreadyConsumedError()
483         extends Exception("Body Has Been Consumed Completely Already")
484 
485     def decode[F[_]: Concurrent](
486         unread: Long,
487         buffer: Array[Byte],
488         nextBuffer: Ref[F, Option[Array[Byte]]],
489         read: Read[F],
490     ): Stream[F, Byte] = {
491       def go(remaining: Long): Pull[F, Byte, Unit] =
492         Pull.eval(read).flatMap {
493           case Some(chunk) =>
494             if (chunk.size >= remaining) {
495               val (rest, after) = chunk.splitAt(remaining.toInt)
496               Pull.eval(nextBuffer.set(Some(after.toArray))) >> Pull.output(rest) >> Pull.done
497             } else {
498               Pull.output(chunk) >> go(remaining - chunk.size)
499             }
500           case None => Pull.raiseError(EmberException.ReachedEndOfStream())
501         }
502 
503       // TODO: This doesn't forbid concurrent reads of the body stream, only sequential reads.
504       val pull = Pull.eval(nextBuffer.get).flatMap {
505         case Some(_) =>
506           Pull.raiseError(BodyAlreadyConsumedError())
507         case None =>
508           Pull.output(Chunk.array(buffer)) >> go(unread)
509       }
510 
511       pull.stream
512     }
