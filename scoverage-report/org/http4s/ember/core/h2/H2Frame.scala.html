28   /*
29   All frames begin with a fixed 9-octet header followed by a variable-
30   length payload.
31 
32     +-----------------------------------------------+
33     |                 Length (24)                   |
34     +---------------+---------------+---------------+
35     |   Type (8)    |   Flags (8)   |
36     +-+-------------+---------------+-------------------------------+
37     |R|                 Stream Identifier (31)                      |
38     +=+=============================================================+
39     |                   Frame Payload (0...)                      ...
40     +---------------------------------------------------------------+
41 
42   Chris:
43   To Unpack this each from should consist of
44   Length 3 bytes
45   Type 1 Byte
46   Flags 1 Byte
47   1 bit that must be sent 0x0, and must be ignored
48   31 bit unsigned Integer
49   FramePayload that consists of Length
50 
51    */
52 
53   final case class RawFrame(
54       length: Int, // 3 bytes is within int range -- 16,777,216 16 mb max frame, bigger isn't necessarily better
55       `type`: Byte,
56       flags: Byte,
57       identifier: Int, // 31 bit Integer
58       payload: ByteVector,
59   )

63     def fromByteVector(bv: ByteVector): Option[(RawFrame, ByteVector)] =
64       if (bv.length >= 9) {
65         val length = (bv(2) & 0xff) | ((bv(1) & 0xff) << 8) | ((bv(0) & 0xff) << 16)
66         if (bv.length >= 9 + length) {
67           val `type` = bv(3)
68           val flags = bv(4)
69           val s0 = bv(5) & 0xff
70           val s1 = (bv(6) & 0xff) << 16
71           val s2 = (bv(7) & 0xff) << 8
72           val s3 = (bv(8) & 0xff) << 0
73           val modS0 = (s0 & ~(1 << 7)) << 24
74           val identifier = modS0 | s1 | s2 | s3
75           // val identifier =  (bv(8) & 0xFF) | ((bv(7) & 0xFF) << 8) | ((bv(6) & 0xFF) << 16) | ((bv(5) & 0xFF) << 24)
76           (
77             RawFrame(
78               length,
79               `type`,
80               flags,
81               identifier,
82               bv.drop(9).take(length),
83             ),
84             bv.drop(9 + length),
85           ).some
86         } else None
87       } else None
88 
89     // Network Byte Order is Big Endian, so Highest Identifier is First
90     def toByteVector(raw: RawFrame): ByteVector = {
91       // 3
92       val zero = ((raw.length >> 16) & 0xff).toByte
93       val one = ((raw.length >> 8) & 0xff).toByte
94       val two = ((raw.length >> 0) & 0xff).toByte
95 
96       // 2
97       val t = raw.`type`
98       val f = raw.flags
99 
100       // 4
101       val iZero = ((raw.identifier >> 24) & 0xff).toByte
102       val iOne = ((raw.identifier >> 16) & 0xff).toByte
103       val iTwo = ((raw.identifier >> 8) & 0xff).toByte
104       val iThree = ((raw.identifier) & 0xff).toByte
105 
106       ByteVector(
107         zero,
108         one,
109         two,
110         t,
111         f,
112         iZero,
113         iOne,
114         iTwo,
115         iThree,
116       ) ++ raw.payload
117     }
118   }

172   object Data {
173     val `type`: Byte = 0x0
174     // 2 flags
175     // EndStream = Bit 0 indicates this is the last frame this will send
176     // Padded = Bit 3 indicates
177     def fromRaw(rawFrame: RawFrame): Either[H2Error, Data] =
178       if (rawFrame.`type` === `type`) {
179         val endStream = (rawFrame.flags & (0x01 << 0)) != 0
180         val padded = (rawFrame.flags & (0x01 << 3)) != 0
181         if (padded) {
182           val padLength = rawFrame.payload(0)
183           val baseSize = rawFrame.payload.size.toInt - 1
184           val dataSize = baseSize - padLength
185           val data = rawFrame.payload.drop(1).take(dataSize)
186           val pad = rawFrame.payload.drop(1).drop(dataSize)
187           if (pad.length == padLength && data.size == dataSize) {
188             Data(rawFrame.identifier, data, Some(pad), endStream).asRight
189           } else Either.left(H2Error.ProtocolError)
190         } else {
191           Data(rawFrame.identifier, rawFrame.payload, None, endStream).asRight
192         }
193       } else Either.left(H2Error.ProtocolError)
194 
195     def toRaw(data: Data): RawFrame = {
196       val payload = data.pad
197         .map(p =>
198           ByteVector(p.length.toByte) ++
199             data.data ++
200             p
201         )
202         .getOrElse(data.data)
203       val flags: Byte = {
204         val init: Int = 0x0
205         val endStreamBitSet: Int = if (data.endStream) init | (1 << 0) else init
206         val paddedSet: Int = if (data.pad.isDefined) endStreamBitSet | (1 << 3) else endStreamBitSet
207         paddedSet.toByte
208       }
209       RawFrame(
210         payload.size.toInt,
211         `type`,
212         flags,
213         data.identifier,
214         payload,
215       )
216     }
217   }

233       identifier: Int,
234       dependency: Option[Headers.StreamDependency],
235       endStream: Boolean, // No Body Follows
236       endHeaders: Boolean, // Whether or not to Expect Continuation Frame (if false, continuation must directly follow)
237       headerBlock: ByteVector,
238       padding: Option[ByteVector],

243   object Headers {
244     val `type`: Byte = 0x1
245 
246     final case class StreamDependency(exclusive: Boolean, dependency: Int, weight: Byte)
247 
248     def fromRaw(rawFrame: RawFrame): Either[H2Error, Headers] =
249       rawFrame.`type` match {
250         case `type` =>
251           val endStream = (rawFrame.flags & (0x01 << 0)) != 0
252           val endHeaders = (rawFrame.flags & (0x01 << 2)) != 0
253           val padded = (rawFrame.flags & (0x01 << 3)) != 0
254           val priority = (rawFrame.flags & (0x01 << 5)) != 0
255 
256           (priority, padded) match {
257             case (false, false) =>
258               Headers(
259                 rawFrame.identifier,
260                 None,
261                 endStream,
262                 endHeaders,
263                 rawFrame.payload,
264                 None,
265               ).asRight
266             case (true, true) =>
267               // This hurts. And is SO inefficient
268               val bv = rawFrame.payload
269               val padLength = bv.get(0)
270               val pad = bv.takeRight(padLength)
271               val rest = bv.dropRight(padLength).drop(1)
272               val s0 = rest.get(0)
273               val s1 = rest.get(1)
274               val s2 = rest.get(2)
275               val s3 = rest.get(3)
276               val weight = rest.get(4)
277               val mod0 = s0 & ~(1 << 7)
278               val dependsOnStream = (mod0 << 24) + (s1 << 16) + (s2 << 8) + (s3 << 0)
279               val exclusive = (s0 & (0x01 << 7)) != 0
280               val payload = rest.drop(5)
281               Headers(
282                 rawFrame.identifier,
283                 Some(StreamDependency(exclusive, dependsOnStream, weight)),
284                 endStream,
285                 endHeaders,
286                 payload,
287                 Some(pad),
288               ).asRight
289             case (true, false) =>
290               val rest = rawFrame.payload
291               val s0 = rest.get(0)
292               val s1 = rest.get(1)
293               val s2 = rest.get(2)
294               val s3 = rest.get(3)
295               val weight = rest.get(4)
296               val mod0 = s0 & ~(1 << 7)
297               val dependsOnStream = (mod0 << 24) + (s1 << 16) + (s2 << 8) + (s3 << 0)
298               val exclusive = (s0 & (0x01 << 7)) != 0
299               val payload = rest.drop(5)
300 
301               Headers(
302                 rawFrame.identifier,
303                 Some(StreamDependency(exclusive, dependsOnStream, weight)),
304                 endStream,
305                 endHeaders,
306                 payload,
307                 None,
308               ).asRight
309 
310             case (false, true) =>
311               val bv = rawFrame.payload
312               val padLength = bv.get(0)
313               val pad = bv.takeRight(padLength)
314               val payload = bv.dropRight(padLength).drop(1)
315               Headers(
316                 rawFrame.identifier,
317                 None,
318                 endStream,
319                 endHeaders,
320                 payload,
321                 Some(pad),
322               ).asRight
323           }
324         case _ => Either.left(H2Error.InternalError)
325       }
326     def toRaw(headers: Headers): RawFrame = {
327       val flags = {
328         var init = 0
329         if (headers.endStream) init = init | (1 << 0)
330         if (headers.endHeaders) init = init | (1 << 2)
331         if (headers.padding.isDefined) init = init | (1 << 3)
332         if (headers.dependency.isDefined) init = init | (1 << 5)
333         init
334       }.toByte
335 
336       val body = (headers.padding, headers.dependency) match {
337         case (None, None) => headers.headerBlock
338         case (Some(pad), None) =>
339           ByteVector(pad.length.toByte) ++ headers.headerBlock ++
340             pad
341         case (padO, Some(dependency)) =>
342           val dep0 = ((dependency.dependency >> 24) & 0xff).toByte
343           val dep1 = ((dependency.dependency >> 16) & 0xff).toByte
344           val dep2 = ((dependency.dependency >> 8) & 0xff).toByte
345           val dep3 = ((dependency.dependency >> 0) & 0xff).toByte
346           val modDep0 = (if (dependency.exclusive) dep0 | (1 << 7) else dep0 & ~(1 << 7)).toByte
347           val base = ByteVector(modDep0, dep1, dep2, dep3, dependency.weight) ++ headers.headerBlock
348           padO match {
349             case None => base
350             case Some(pad) =>
351               ByteVector(pad.length.toByte) ++ base ++
352                 pad
353           }
354       }
355 
356       RawFrame(
357         body.size.toInt,
358         `type`,
359         flags,
360         headers.identifier,
361         body,
362       )
363     }
364   }

377       weight: Byte,

379   object Priority {
380     val `type`: Byte = 0x2
381     def fromRaw(raw: RawFrame): Either[H2Error, Priority] =
382       if (raw.`type` == `type`) {
383         if (raw.length === 5) {
384           val s0 = raw.payload(0)
385           val s1 = raw.payload(1)
386           val s2 = raw.payload(2)
387           val s3 = raw.payload(3)
388           val mod0 = s0 & ~(1 << 7)
389           val exclusive = (s0 & (0x01 << 7)) != 0
390           val dependency = (mod0 << 24) + (s1 << 16) + (s2 << 8) + (s3 << 0)
391           val weight = raw.payload(4)
392 
393           Priority(raw.identifier, exclusive, dependency, weight).asRight
394         } else H2Error.FrameSizeError.asLeft
395       } else Either.left(H2Error.InternalError)
396 
397     def toRaw(priority: Priority): RawFrame = {
398       val payload = {
399         val dep0 = ((priority.streamDependency >> 24) & 0xff).toByte
400         val dep1 = ((priority.streamDependency >> 16) & 0xff).toByte
401         val dep2 = ((priority.streamDependency >> 8) & 0xff).toByte
402         val dep3 = ((priority.streamDependency >> 0) & 0xff).toByte
403         val modDep0 = (if (priority.exclusive) dep0 | (1 << 7) else dep0 & ~(1 << 7)).toByte
404         ByteVector(modDep0, dep1, dep2, dep3, priority.weight)
405       }
406       RawFrame(payload.size.toInt, `type`, 0, priority.identifier, payload)
407     }

534     // Effect?
535     def fromRaw(raw: RawFrame): Either[H2Error, Settings] =
536       if (raw.`type` == `type`) {
537         val ack = (raw.flags & (0x01 << 0)) != 0
538         if (ack && raw.payload.nonEmpty) H2Error.FrameSizeError.asLeft
539         else if (ack) Settings(raw.identifier, ack, List.empty).asRight
540         else fromPayload(raw.payload, raw.identifier, ack)
541       } else Either.left(H2Error.InternalError)
542     def fromPayload(payload: ByteVector, identifier: Int, ack: Boolean): Either[H2Error, Settings] =
543       if (payload.size % 6 != 0) H2Error.FrameSizeError.asLeft
544       else {
545         val settings = for {
546           i <- 0 to (payload.size.toInt - 5) by 6
547         } yield {
548           val s = (payload(i) << 8) + (payload(i + 1) << 0)
549           val v0 = (payload(i + 2) & 0xff) << 24
550           val v1 = (payload(i + 3) & 0xff) << 16
551           val v2 = (payload(i + 4) & 0xff) << 8
552           val v3 = (payload(i + 5) & 0xff) << 0
553           val value = v0 | v1 | v2 | v3
554           Setting(s.toShort, value)
555         }
556         settings.toList.sequence.map(s => Settings(identifier, ack, s.toList))
557       }

559     def toRaw(settings: Settings): RawFrame = {
560       val payload = settings.list.foldRight(ByteVector.empty) { case (next, bv) =>
561         val s0 = ((next.identifier >> 8) & 0xff).toByte
562         val s1 = ((next.identifier >> 0) & 0xff).toByte
563 
564         val v0 = ((next.value >> 24) & 0xff).toByte
565         val v1 = ((next.value >> 16) & 0xff).toByte
566         val v2 = ((next.value >> 8) & 0xff).toByte
567         val v3 = ((next.value >> 0) & 0xff).toByte
568         ByteVector(s0, s1, v0, v1, v2, v3) ++ bv
569       }
570       val flag: Byte = (if (settings.ack) 0 | (1 << 0) else 0).toByte
571       RawFrame(payload.size.toInt, `type`, flag, settings.identifier, payload)
572     }

574     sealed abstract class Setting(val identifier: Short, val value: Integer)
575     object Setting {
576       def apply(identifier: Short, value: Integer): Either[H2Error, Setting] = identifier match {
577         case 0x1 => SettingsHeaderTableSize(value).asRight
578         case 0x2 =>
579           value.toInt match {
580             case 1 => SettingsEnablePush(true).asRight
581             case 0 => SettingsEnablePush(false).asRight
582             case _ => H2Error.ProtocolError.asLeft
583           }
584         case 0x3 => SettingsMaxConcurrentStreams(value).asRight
585         case 0x4 => SettingsInitialWindowSize.fromInt(value)
586         case 0x5 => SettingsMaxFrameSize.fromInt(value)
587         case 0x6 => SettingsMaxHeaderListSize(value).asRight
588         case _ => Unknown(identifier, value).asRight
589       }
590     }
591     // The initial value is 4,096 octets
592     final case class SettingsHeaderTableSize(size: Integer) extends Setting(0x1, size)
593     // Default true

596     // Unbounded
597     final case class SettingsMaxConcurrentStreams(maxConcurrency: Integer)
598         extends Setting(0x3, maxConcurrency)
