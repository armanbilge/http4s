24 /** HTTP/2 pseudo headers */
25 private[h2] object PseudoHeaders {
26   // Request pseudo headers
27   val METHOD = ":method"
28   val SCHEME = ":scheme"
29   val PATH = ":path"
30   val AUTHORITY = ":authority"
31   val requestPsedo: Set[String] = Set(
32     METHOD,
33     SCHEME,
34     PATH,
35     AUTHORITY,
36   )

38   import org.http4s.Request
39   def requestToHeaders[F[_]](req: Request[F]): NonEmptyList[(String, String, Boolean)] = {
40     val path = {
41       val s = req.uri.path.renderString
42       if (s.isEmpty) "/"
43       else s
44     }
45     val l = NonEmptyList.of(
46       (METHOD, req.method.toString, false),
47       (SCHEME, req.uri.scheme.map(_.value).getOrElse("https"), false) ::
48         (PATH, path, false) ::
49         (AUTHORITY, req.uri.authority.map(_.toString).getOrElse(""), false) ::
50         req.headers.headers.map(raw =>
51           (
52             raw.name.toString.toLowerCase(),
53             raw.value,
54             org.http4s.Headers.SensitiveHeaders.contains(raw.name),
55           )
56         ): _*
57     )
58     l
59   }

61   def headersToRequestNoBody(hI: NonEmptyList[(String, String)]): Option[Request[fs2.Pure]] = {
62     // TODO This can be a 1 pass operation. This is not...
63     val headers: List[(String, String)] = hI.toList
64     val pseudos = headers.takeWhile(_._1.startsWith(":"))
65     val method = findWithNoDuplicates(pseudos)(_._1 === METHOD)
66       .map(_._2)
67       .flatMap(Method.fromString(_).toOption)
68     val scheme =
69       findWithNoDuplicates(pseudos)(_._1 === SCHEME).map(_._2).map(Uri.Scheme.unsafeFromString(_))
70     val path = findWithNoDuplicates(pseudos)(_._1 === PATH).map(_._2)
71     val authority = extractAuthority(pseudos)
72 
73     val noOtherPseudo = pseudos
74       .filterNot(t => requestPsedo.contains(t._1))
75       .isEmpty
76 
77     val teIsCorrect = headers.find(_._1 === "te").headOption.fold(true)(_._2 === "trailers")
78     val connectionIsAbsent = headers.find(_._1 === "connection").isEmpty
79 
80     val reqHeaders = headers.dropWhile(_._1.startsWith(":"))
81 
82     val noOtherPseudos = !reqHeaders.exists(_._1.startsWith(":"))
83     val noUppercaseHeaders = reqHeaders.map(_._1).forall(s => s === s.toLowerCase)
84 
85     val h = Headers(
86       reqHeaders
87         .map(t => Header.Raw(org.typelevel.ci.CIString(t._1), t._2))
88         .map(Header.ToRaw.rawToRaw): _*
89     )
90     for {
91       _ <- Alternative[Option].guard(
92         noOtherPseudo && teIsCorrect && connectionIsAbsent && noOtherPseudos && noUppercaseHeaders
93       )
94       m <- method
95       p <- path
96       _ <- Alternative[Option].guard(p.nonEmpty) // Not Allowed to be empty in http/2
97 
98       u <- Uri.fromString(p).toOption
99       s <- scheme // Required
100     } yield Request(m, u.copy(scheme = s.some, authority = authority), HttpVersion.`HTTP/2`, h)
101   }
102 
103   def extractAuthority(headers: List[(String, String)]): Option[Uri.Authority] =
104     headers.collectFirstSome {
105       case (PseudoHeaders.AUTHORITY, value) =>
106         val index = value.indexOf(":")
107         if (index > 0 && index < value.length) {
108           Option(
109             Uri.Authority(
110               userInfo = None,
111               host = Uri.RegName(value.take(index)),
112               port = value.drop(index + 1).toInt.some,
113             )
114           )
115         } else Option.empty
116       case (_, _) => None
117     }

120   val STATUS = ":status"

122   def responseToHeaders[F[_]](response: Response[F]): NonEmptyList[(String, String, Boolean)] =
123     NonEmptyList(
124       (STATUS, response.status.code.toString, false),
125       response.headers.headers
126         .map(raw =>
127           (
128             raw.name.toString.toLowerCase,
129             raw.value,
130             org.http4s.Headers.SensitiveHeaders.contains(raw.name),
131           )
132         ),
133     )
134 
135   def headersToResponseNoBody(
136       headers: NonEmptyList[(String, String)]
137   ): Option[Response[fs2.Pure]] = {
138     // TODO Duplicate Check
139     val statusO =
140       findWithNoDuplicates(headers.toList)(_._1 === PseudoHeaders.STATUS).map(_._2).flatMap {
141         value =>
142           Status.fromInt(value.toInt).toOption
143       }
144     val h = Headers(
145       headers
146         .filterNot(t => t._1 == PseudoHeaders.STATUS)
147         .map(t => Header.Raw(org.typelevel.ci.CIString(t._1), t._2))
148         .map(Header.ToRaw.rawToRaw): _*
149     )
150     statusO.map(s => Response(status = s, httpVersion = HttpVersion.`HTTP/2`, headers = h))
151   }
152 
153   def findWithNoDuplicates[A](l: List[A])(bool: A => Boolean): Option[A] =
154     l.foldLeft(Either.right[Unit, Option[A]](None)) {
155       case (Left(e), _) => Left(e)
156       case (Right(Some(a)), next) =>
157         if (bool(next)) Left(())
158         else Right(Some(a))
159       case (Right(None), next) =>
160         if (bool(next)) Right(Some(next))
161         else Right(None)
162     }.toOption
163       .flatten
