35 private[h2] object Hpack extends HpackPlatform {
36   def create[F[_]: Async]: F[Hpack[F]] = for {
37     eLock <- Semaphore[F](1)
38     dLock <- Semaphore[F](1)
39     e <- Sync[F].delay(new Encoder(4096))
40     d <- Sync[F].delay(new Decoder(65536, 4096))
41   } yield new Impl(eLock, e, dLock, d)
42 
43   private class Impl[F[_]: Async](
44       encodeLock: Semaphore[F],
45       tEncoder: Encoder,
46       decodeLock: Semaphore[F],
47       tDecoder: Decoder,
48   ) extends Hpack[F] {
49     def encodeHeaders(headers: NonEmptyList[(String, String, Boolean)]): F[ByteVector] =
50       encodeLock.permit.use(_ => Hpack.encodeHeaders[F](tEncoder, headers.toList))
51     def decodeHeaders(bv: ByteVector): F[NonEmptyList[(String, String)]] =
52       decodeLock.permit.use(_ => Hpack.decodeHeaders[F](tDecoder, bv))
53 
54   }
55 
56   def decodeHeaders[F[_]: Sync](
57       tDecoder: Decoder,
58       bv: ByteVector,
59   ): F[NonEmptyList[(String, String)]] = Sync[F].delay {
60     val buffer = new ListBuffer[(String, String)]
61     val is = new ByteArrayInputStream(bv.toArray)
62     val listener = new HeaderListener {
63       def addHeader(name: Array[Byte], value: Array[Byte], sensitive: Boolean): Unit = {
64         buffer.+=(
65           new String(name, StandardCharsets.ISO_8859_1) -> new String(
66             value,
67             StandardCharsets.ISO_8859_1,
68           )
69         )
70         ()
71       }
72     }
73 
74     tDecoder.decode(is, listener)
75     tDecoder.endHeaderBlock()
76 
77     val decoded = buffer.toList
78     NonEmptyList.fromList(decoded).toRight(new NoSuchElementException("Header List Was Empty"))
79   }.rethrow
80 
81   def encodeHeaders[F[_]: Sync](
82       tEncoder: Encoder,
83       headers: List[(String, String, Boolean)],
84   ): F[ByteVector] = Sync[F].delay {
85     val os = new ByteArrayOutputStream(1024)
86     headers.foreach { h =>
87       tEncoder.encodeHeader(
88         os,
89         h._1.getBytes(StandardCharsets.ISO_8859_1),
90         h._2.getBytes(StandardCharsets.ISO_8859_1),
91         h._3,
92       )
93     }
94     ByteVector.view(os.toByteArray())
95   }
