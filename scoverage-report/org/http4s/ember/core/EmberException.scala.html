25 sealed trait EmberException extends RuntimeException with Product with Serializable

27 object EmberException {

29     override def getMessage: String =
30       show"Timeout Occured - Started: ${started.toString}, Timed Out: ${timedOut.toString}"
31   }
32   final case class IncompleteClientRequest(missing: String)
33       extends IllegalArgumentException
34       with EmberException {
35     override def getMessage: String = show"Incomplete Client Request: Mising $missing"
36   }
37 
38   final case class ParseError(message: String) extends EmberException {
39     override def getMessage: String = message
40   }
41 
42   final case class ChunkedEncodingError(message: String) extends EmberException {
43     override def getMessage: String = message
44   }
45 
46   final case class EmptyStream() extends EmberException {
47     override def getMessage: String = "Cannot Parse Empty Stream"
48   }
49 
50   final case class ReachedEndOfStream() extends EmberException {
51     override def getMessage: String = "Reached End Of Stream While Reading"
52   }
53 
54   final case class MessageTooLong(maxHeaderSize: Int) extends EmberException {
55     override def getMessage: String = s"HTTP Header Section Exceeds Max Size: $maxHeaderSize Bytes"
56   }
57 
58   final case class ReadTimeout(duration: Duration) extends EmberException {
