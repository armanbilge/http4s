47   /** The issue with a normal http body is that there is no termination character,
48     * thus unless you have content-length and the client still has their input side open,
49     * the server cannot know whether more data follows or not
50     * This means this Stream MUST be infinite and additional parsing is required.
51     * To know how much client input to consume
52     *
53     * Function if timeout reads via socket read and then incrementally lowers
54     * the remaining time after each read.
55     * By setting the timeout signal outside this after the
56     * headers have been read it triggers this function
57     * to then not timeout on the remaining body.
58     */
59   def readWithTimeout[F[_]](
60       socket: Socket[F],
61       started: Long,
62       timeout: FiniteDuration,
63       shallTimeout: F[Boolean],
64       chunkSize: Int,
65   )(implicit F: ApplicativeThrow[F], C: Clock[F]): Stream[F, Byte] = {
66     def whenWontTimeout: Stream[F, Byte] =
67       socket.reads
68     def whenMayTimeout(remains: FiniteDuration): Stream[F, Byte] =
69       if (remains <= 0.millis)
70         streamCurrentTimeMillis(C)
71           .flatMap(now =>
72             Stream.raiseError[F](
73               EmberException.Timeout(Instant.ofEpochMilli(started), Instant.ofEpochMilli(now))
74             )
75           )
76       else
77         for {
78           start <- streamCurrentTimeMillis(C)
79           read <- Stream.eval(socket.read(chunkSize)) //  Each Read Yields
80           end <- streamCurrentTimeMillis(C)
81           out <- read.fold[Stream[F, Byte]](
82             Stream.empty
83           )(
84             Stream.chunk(_) ++ go(remains - (end - start).millis)
85           )
86         } yield out
87     def go(remains: FiniteDuration): Stream[F, Byte] =
88       Stream
89         .eval(shallTimeout)
90         .ifM(
91           whenMayTimeout(remains),
92           whenWontTimeout,
93         )
94     go(timeout)
95   }

102   def timeoutMaybe[F[_], A](fa: F[A], d: Duration)(implicit F: Temporal[F]): F[A] =
103     d match {
104       case fd: FiniteDuration => F.timeout(fa, fd)

108   def timeoutToMaybe[F[_], A](fa: F[A], d: Duration, ft: F[A])(implicit F: Temporal[F]): F[A] =
109     d match {
110       case fd: FiniteDuration => F.timeoutTo(fa, fd, ft)

112     }

114   def connectionFor(httpVersion: HttpVersion, headers: Headers): Connection =
115     if (isKeepAlive(httpVersion, headers)) keepAlive
116     else close
117 
118   def isKeepAlive(httpVersion: HttpVersion, headers: Headers): Boolean = {
119     // TODO: the problem is that any string that contains `expected` is admissible
120     def hasConnection(expected: String): Boolean =
121       headers.headers.exists { case Header.Raw(name, value) =>
122         name == connectionCi && value.toLowerCase(Locale.ROOT).contains(expected)
123       }
124 
125     httpVersion match {
126       case HttpVersion.`HTTP/1.0` => hasConnection(keepAliveCi.toString)
127       case HttpVersion.`HTTP/1.1` => !hasConnection(closeCi.toString)
128       case _ => sys.error("unsupported http version")
129     }
130   }
