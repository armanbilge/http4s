31 /** Common rules defined in RFC7230
32   *
33   * @see [[https://datatracker.ietf.org/doc/html/rfc7230#appendix-B]]
34   */
35 private[http4s] object Rfc7230 {
36   /* `tchar = "!" / "#" / "$" / "%" / "&" / "'" / "*" / "+" / "-" / "." /
37    *  "^" / "_" / "`" / "|" / "~" / DIGIT / ALPHA`
38    */
39   val tchar: Parser[Char] = charIn("!#$%&'*+-.^_`|~").orElse(digit).orElse(alpha)
40 
41   /* `token = 1*tchar` */
42   val token: Parser[String] = tchar.rep.string
43 
44   /* `obs-text = %x80-FF` */
45   val obsText: Parser[Char] = charIn(0x80.toChar to 0xff.toChar)
46 
47   /* `OWS = *( SP / HTAB )` */
48   val ows: Parser0[Unit] = sp.orElse(htab).rep0.void
49 
50   /*https://datatracker.ietf.org/doc/html/rfc7230#section-3.2.3 last paragraph */
51   val bws: Parser0[Unit] = ows
52 
53   /*   qdtext         = HTAB / SP /%x21 / %x23-5B / %x5D-7E / obs-text */
54   val qdText: Parser[Char] =
55     charIn('\t', ' ', 0x21.toChar)
56       .orElse(charIn(0x23.toChar to 0x5b.toChar))
57       .orElse(charIn(0x5d.toChar to 0x7e.toChar))
58       .orElse(obsText)
59 
60   val qdPairChar: Parser[Char] = charIn('\t', ' ').orElse(vchar).orElse(obsText)
61 
62   /* quoted-pair    = "\" ( HTAB / SP / VCHAR / obs-text ) */
63   val quotedPair: Parser[Char] = char('\\') *> qdPairChar
64 
65   private def surroundedBy[A](a: Parser0[A], b: Parser[Any]): Parser[A] =
66     b *> a <* b
67   private def between[A](a: Parser[Any], b: Parser0[A], c: Parser[Any]): Parser[A] =
68     a *> b <* c
69 
70   /*quoted-string  = DQUOTE *( qdtext / quoted-pair ) DQUOTE*/
71   val quotedString: Parser[String] =
72     surroundedBy(qdText.orElse(quotedPair).rep0.string, dquote)
73 
74   /* HTAB / SP / %x21-27 / %x2A-5B / %x5D-7E / obs-text */
75   val cText: Parser[Char] =
76     charIn('\t', ' ', 0x21.toChar)
77       .orElse(charIn(0x21.toChar to 0x27.toChar))
78       .orElse(charIn(0x2a.toChar to 0x5b.toChar))
79       .orElse(charIn(0x5d.toChar to 0x7e.toChar))
80       .orElse(obsText)
81 
82   /* "(" *( ctext / quoted-pair / comment ) ")" */
83   val comment: Parser[String] = Parser.recursive[String] { (comment: Parser[String]) =>
84     between(char('('), cText.orElse(quotedPair).orElse(comment).rep0.string, char(')'))
85   }
86 
87   def headerRep[A](element: Parser[A]): Parser0[List[A]] =
88     headerRep1(element).?.map(_.fold(List.empty[A])(_.toList))
89 
90   /* `1#element => *( "," OWS ) element *( OWS "," [ OWS element ] )` */
91   def headerRep1[A](element: Parser[A]): Parser[NonEmptyList[A]] = {
92     val prelude = (char(',') <* ows).rep0
93     val tailOpt = (ows.with1 *> char(',') *> (ows.with1 *> element).?).rep0
94     val tail = tailOpt.map(_.collect { case Some(x) => x })
95 
96     (prelude.with1 *> element ~ tail).map { case (h, t) =>
97       NonEmptyList(h, t)
98     }
99   }
100 
101   def listSep: Parser[Unit] = Parser.char(',').surroundedBy(ows)
102 }
