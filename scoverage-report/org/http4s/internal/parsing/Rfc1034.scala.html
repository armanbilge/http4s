24 /** Common rules defined in RFC1034, as amended by RFC1123.
25   *
26   * @see [[https://datatracker.ietf.org/doc/html/rfc1034] RFC1034, Domain Names: Concepts and Facilities]]
27   * @see [[https://datatracker.ietf.org/doc/html/rfc1123] RFC1123, Requirements for Internet Hosts -- Application and Support]]
28   */
29 private[http4s] object Rfc1034 {
30   def subdomain: Parser[CIString] = {
31     /* <letter> ::= any one of the 52 alphabetic characters A through Z in
32      * upper case and a through z in lower case */
33     val letter = charIn('A' to 'Z').orElse(charIn('a' to 'z'))
34 
35     /* <digit> ::= any one of the ten digits 0 through 9 */
36     val digit = charIn('0' to '9')
37 
38     /* <let-dig> ::= <letter> | <digit> */
39     val letDig = letter.orElse(digit)
40 
41     /* <let-dig-hyp> ::= <let-dig> | "-"
42      * <ldh-str> ::= <let-dig-hyp> | <let-dig-hyp> <ldh-str>
43      * <label> ::= <letter> [ [ <ldh-str> ] <let-dig> ]
44      *
45      * RFC1123: One aspect of host name syntax is hereby changed: the
46      * restriction on the first character is relaxed to allow either a
47      * letter or a digit.  Host software MUST support this more liberal
48      * syntax.
49      */
50     val label = letDig.rep.repSep(char('-'))
51 
52     /* <subdomain> ::= <label> | <subdomain> "." <label> */
53     label.repSep(char('.')).string.map(CIString(_))
54   }
55 }
