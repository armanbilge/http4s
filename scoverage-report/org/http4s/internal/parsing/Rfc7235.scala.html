38    *  *"="
39    * */

49   // auth-scheme = token

54     OWS "," [ OWS auth-param ] ) ] ) ]
55    */
56   val challenge: Parser[Challenge] =

58       .repSep0(char(',').surroundedBy(ows))

69     ((scheme <* sp) ~ headerRep1(authParam.backtrack).map(Right(_)).orElse(token68.map(Left(_))))
