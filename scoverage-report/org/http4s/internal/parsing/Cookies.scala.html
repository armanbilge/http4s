24 /** Common rules defined in RFC6265
25   *
26   * @see [[https://datatracker.ietf.org/doc/html/rfc6265] RFC6265, HTTP State Management Mechanism]
27   */
28 private[http4s] abstract class Cookies(cookieOctet: Parser[Char]) {
29   /* token             = <token, defined in [RFC2616], Section 2.2> */
30   def token: Parser[String] = Rfc2616.token
31 
32   /* cookie-name       = token */
33   def cookieName: Parser[String] = token
34 
35   /* cookie-value      = *cookie-octet */
36   val cookieValue: Parser0[String] = cookieOctet.rep0.string
37 
38   /* cookie-pair       = cookie-name "=" cookie-value */
39   val cookiePair: Parser[(String, String)] = (cookieName <* char('=')) ~ cookieValue
40 
41 }
42 
43 /* This is the strict Rfc6265 implementation. */
44 private[http4s] object Rfc6265
45     extends Cookies(
46       /* cookie-octet      = %x21 / %x23-2B / %x2D-3A / %x3C-5B / %x5D-7E
47        *                   ; US-ASCII characters excluding CTLs,
48        *                   ; whitespace DQUOTE, comma, semicolon,
49        *                   ; and backslash
50        */
51       cookieOctet = charIn(
52         Set(0x21.toChar) ++
53           Set(0x23.toChar to 0x2b.toChar: _*) ++
54           Set(0x2d.toChar to 0x3a.toChar: _*) ++
55           Set(0x3c.toChar to 0x5b.toChar: _*) ++
56           Set(0x5d.toChar to 0x7e.toChar: _*)
57       )
58     )
59 
60 /* This is a relaxed implementation based on spec implemented by Chromium
61  * US-ASCII characters excluding CTLs and semicolon
62  *  See https://groups.google.com/a/chromium.org/g/chromium-discuss/c/T3oHExJMr0M
63  * */
64 private[http4s] object RelaxedCookies
65     extends Cookies(
66       cookieOctet = charIn(
67         Set(0x20.toChar to 0xff.toChar: _*) - 0x3b.toChar
68       )
69     )
