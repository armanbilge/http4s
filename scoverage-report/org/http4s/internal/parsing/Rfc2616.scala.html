26 /** Common rules defined in RFC2616.  This RFC is now obsolete,
27   * but some other active ones still refer to it.
28   *
29   * @see [[https://datatracker.ietf.org/doc/html/rfc2616#section-2.2] RFC2616, Basic Rules]
30   */
31 private[http4s] object Rfc2616 {
32   /* CHAR           = <any US-ASCII character (octets 0 - 127)>
33    * CTL            = <any US-ASCII control character
34    *                  (octets 0 - 31) and DEL (127)>
35    * separators     = "(" | ")" | "<" | ">" | "@"
36    *                | "," | ";" | ":" | "\" | <">
37    *                | "/" | "[" | "]" | "?" | "="
38    *                | "{" | "}" | SP  | HT
39    * token          = 1*<any CHAR except CTLs or separators>
40    */
41   val token: Parser[String] = {
42     val ascii = Set(0.toChar to 127.toChar: _*)
43     val ctl = Set(0.toChar to 31.toChar: _*) + 127.toChar
44     val separators = Set("()<>@,;:\\\"/[]?={} \t".toSeq: _*)
45 
46     val tchar = charIn(ascii -- ctl -- separators)
47     tchar.rep.string
48   }
49 
50   final case class Rfc1123Date(
51       wkday: Int,
52       day: Int,
53       month: Int,
54       year: Int,
55       hour: Int,
56       min: Int,
57       sec: Int,
58   )
59 
60   val rfc1123Date: Parser[Rfc1123Date] = {
61     /* wkday        = "Mon" | "Tue" | "Wed"
62      *              | "Thu" | "Fri" | "Sat" | "Sun"
63      */
64     val wkday = List("Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun").zipWithIndex
65       .map { case (s, i) => string(s).as(i) }
66       .reduceLeft(_.orElse(_))
67 
68     val twoDigit = (digit ~ digit).string.map(_.toInt)
69 
70     val fourDigit = (digit ~ digit ~ digit ~ digit).string.map(_.toInt)
71 
72     /* month        = "Jan" | "Feb" | "Mar" | "Apr"
73      *              | "May" | "Jun" | "Jul" | "Aug"
74      *              | "Sep" | "Oct" | "Nov" | "Dec"
75      */
76     val month =
77       List(
78         "Jan",
79         "Feb",
80         "Mar",
81         "Apr",
82         "May",
83         "Jun",
84         "Jul",
85         "Aug",
86         "Sep",
87         "Oct",
88         "Nov",
89         "Dec",
90       ).zipWithIndex
91         .map { case (s, i) => string(s).as(i + 1) }
92         .reduceLeft(_.orElse(_))
93 
94     /* date1        = 2DIGIT SP month SP 4DIGIT
95      *              ; day month year (e.g., 02 Jun 1982)
96      */
97     val date1 = (twoDigit <* sp) ~ (month <* sp) ~ fourDigit
98 
99     /* time         = 2DIGIT ":" 2DIGIT ":" 2DIGIT
100      *              ; 00:00:00 - 23:59:59
101      */
102     val time = (twoDigit <* char(':')) ~ (twoDigit <* char(':')) ~ twoDigit
103 
104     /* rfc1123-date = wkday "," SP date1 SP time SP "GMT" */
105     ((wkday <* string(", ")) ~ (date1 <* sp) ~ (time <* string(" GMT"))).map {
106       case ((wkday, ((day, month), year)), ((hour, min), sec)) =>
107         Rfc1123Date(wkday, day, month, year, hour, min, sec)
108     }
109   }
