22 /** Common rules defined in RFC2234.
23   *
24   * @see [[https://datatracker.ietf.org/doc/html/rfc2234] RFC2234, Augmented BNF for Syntax Specifications: ABNF]
25   */
26 private[http4s] object Rfc2234 {
27   /* ALPHA          =  %x41-5A / %x61-7A   ; A-Z / a-z */
28   val alpha: Parser[Char] =
29     charIn(0x41.toChar to 0x5a.toChar).orElse(charIn(0x61.toChar to 0x7a.toChar))
30 
31   /* DIGIT          =  %x30-39
32    *                       ; 0-9 */
33   val digit: Parser[Char] =
34     charIn(0x30.toChar to 0x39.toChar)
35 }
