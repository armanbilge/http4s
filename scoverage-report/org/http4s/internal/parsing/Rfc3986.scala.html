30 /** Common rules defined in Rfc3986
31   *
32   * @see [[https://datatracker.ietf.org/doc/html/rfc3986]]
33   */
34 private[http4s] object Rfc3986 {
35   def alpha: Parser[Char] = Rfc2234.alpha
36 
37   def digit: Parser[Char] = Rfc2234.digit
38 
39   /* The spec references RFC2234, which is 0-9A-F, but it also
40    * explicitly permits lowercase. */
41   val hexdig: Parser[Char] =
42     digit.orElse(charIn("ABCDEFabcdef"))
43 
44   /* unreserved    = ALPHA / DIGIT / "-" / "." / "_" / "~" */
45   val unreserved: Parser[Char] =
46     alpha.orElse(digit).orElse(charIn("-._~"))
47 
48   /* pct-encoded   = "%" HEXDIG HEXDIG */
49   val pctEncoded: Parser[Char] =
50     (char('%') *> hexdig ~ hexdig).map { case (a, b) =>
51       def toInt(c: Char) = c match {
52         case c if c >= '0' && c <= '9' => c - '0'
53         case c if c >= 'a' && c <= 'f' => c - 'a' + 10
54         case c if c >= 'A' && c <= 'F' => c - 'A' + 10
55       }
56       (toInt(a) << 4 + toInt(b)).toChar
57     }
58 
59   /* sub-delims    = "!" / "$" / "&" / "'" / "(" / ")"
60    *               / "*" / "+" / "," / ";" / "=" */
61   val subDelims: Parser[Char] =
62     charIn("!$&'()*+,;=")
63 
64   /* pchar         = unreserved / pct-encoded / sub-delims / ":" / "@" */
65   val pchar: Parser[Char] =
66     unreserved.orElse(pctEncoded).orElse(subDelims).orElse(charIn(":@"))
67 
68   private val ipv4Bytes: Parser[(Byte, Byte, Byte, Byte)] = {
69     val decOctet = (char('1') ~ digit ~ digit).backtrack
70       .orElse(char('2') ~ charIn('0' to '4') ~ digit)
71       .backtrack
72       .orElse(string("25") ~ charIn('0' to '5'))
73       .backtrack
74       .orElse(charIn('1' to '9') ~ digit)
75       .backtrack
76       .orElse(digit)
77       .string
78       .map(_.toInt.toByte)
79       .backtrack
80 
81     val dot = char('.')
82     val decOctDot = decOctet <* dot
83     (decOctDot, decOctDot, decOctDot, decOctet).tupled
84   }
85 
86   val ipv4Address: Parser[Ipv4Address] =
87     ipv4Bytes.map { case (a, b, c, d) =>
88       Ipv4Address.fromBytes(a.toInt, b.toInt, c.toInt, d.toInt)
89     }.backtrack
90 
91   val ipv6Address: Parser[Ipv6Address] = {
92     import cats.parse.{Parser => P}
93     import cats.parse.Parser.{char, string}
94 
95     def toIpv6(lefts: collection.Seq[Short], rights: collection.Seq[Short]): Ipv6Address =
96       lefts ++ collection.Seq.fill(8 - lefts.size - rights.size)(0.toShort) ++ rights match {
97         case collection.Seq(a, b, c, d, e, f, g, h) =>
98           val bb = ByteBuffer.allocate(16)
99           bb.putShort(a)
100           bb.putShort(b)
101           bb.putShort(c)
102           bb.putShort(d)
103           bb.putShort(e)
104           bb.putShort(f)
105           bb.putShort(g)
106           bb.putShort(h)
107           Ipv6Address.fromBytes(bb.array).get
108       }
109 
110     val h16: P[Short] =
111       (hexdig ~ hexdig.? ~ hexdig.? ~ hexdig.?).string.map { (s: String) =>
112         java.lang.Integer.parseInt(s, 16).toShort
113       }
114 
115     val colon = char(':')
116     val doubleColon = string("::").void
117     val h16Colon = h16 <* colon
118 
119     val parsedIpv4Bytes = ipv4Bytes.map { case (a: Byte, b: Byte, c: Byte, d: Byte) =>
120       List(((a << 8) | b).toShort, ((c << 8) | d).toShort)
121     }
122 
123     def rightsWithIpv4(n: Int) = (1 to n)
124       .map { i =>
125         (h16Colon.repExactlyAs[List[Short]](i) ~ parsedIpv4Bytes).backtrack.map { case (l, r) =>
126           l ++ r
127         }
128       }
129       .foldLeft(parsedIpv4Bytes.backtrack)(_ | _)
130 
131     val ls32: P[List[Short]] = {
132       val option1 = ((h16 <* colon.void) ~ h16).map(t => List(t._1, t._2))
133       option1.backtrack.orElse(parsedIpv4Bytes)
134     }
135 
136     val fullIpv6WihtOptionalIpv4 = (h16Colon.repExactlyAs[List[Short]](6) ~ ls32)
137       .map { case (ls: List[Short], rs) => toIpv6(ls, rs) }
138 
139     val shortIpv6WithIpv4 = for {
140       lefts <- h16.repSep0(0, 5, colon).with1 <* doubleColon
141       rights <- rightsWithIpv4(4 - lefts.size)
142     } yield toIpv6(lefts, rights)
143 
144     val shortIpv6 = for {
145       lefts <- h16.repSep0(0, 7, colon).with1 <* doubleColon
146       rights <-
147         if (6 - lefts.size > 0) h16.repSep0(0, 6 - lefts.size, colon) else Parser.pure(Nil)
148     } yield toIpv6(lefts, rights)
149 
150     fullIpv6WihtOptionalIpv4.backtrack.orElse(shortIpv6WithIpv4.backtrack).orElse(shortIpv6)
151   }
152 }
