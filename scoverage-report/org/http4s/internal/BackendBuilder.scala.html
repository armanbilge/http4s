19 import cats.effect.MonadCancelThrow
20 import cats.effect.Resource
21 import fs2.Stream
22 
23 private[http4s] trait BackendBuilder[F[_], A] {
24   implicit protected def F: MonadCancelThrow[F]
25 
26   /** Returns the backend as a resource.  Resource acquire waits
27     * until the backend is ready to process requests.
28     */
29   def resource: Resource[F, A]
30 
31   /** Returns the backend as a single-element stream.  The stream
32     * does not emit until the backend is ready to process requests.
33     * The backend is shut down when the stream is finalized.
34     */
35   def stream: Stream[F, A] = Stream.resource(resource)
36 
37   /** Returns an effect that allocates a backend and an `F[Unit]` to
38     * release it.  The returned `F` waits until the backend is ready
39     * to process requests.  The second element of the tuple shuts
40     * down the backend when run.
41     *
42     * Unlike [[resource]] and [[stream]], there is no automatic
43     * release of the backend.  This function is intended for REPL
44     * sessions, tests, and other situations where composing a
45     * [[cats.effect.Resource]] or [[fs2.Stream]] is not tenable.
46     * [[resource]] or [[stream]] is recommended wherever possible.
47     */
48   def allocated: F[(A, F[Unit])] = resource.allocated
49 }
