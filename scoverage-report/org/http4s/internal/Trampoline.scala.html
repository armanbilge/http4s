28   def execute(runnable: Runnable): Unit = {
29     var queue = local.get()
30     if (queue == null) {
31       queue = new ThreadLocalTrampoline
32       local.set(queue)
33     }
34

36   }
37 
38   def reportFailure(t: Throwable): Unit = ExecutionContext.defaultReporter(t)

40   // Only safe to use from a single thread
41   private final class ThreadLocalTrampoline extends ExecutionContext {
42     private var running = false
43     private var r0, r1, r2: Runnable = null
44     private var rest: ArrayDeque[Runnable] = null
45 
46     override def execute(runnable: Runnable): Unit = {
47       if (r0 == null) r0 = runnable
48       else if (r1 == null) r1 = runnable
49       else if (r2 == null) r2 = runnable
50       else {
51         if (rest == null) rest = new ArrayDeque[Runnable]()
52         rest.add(runnable)
53       }
54 
55       if (!running) {
56         running = true
57         run()
58       }
59     }
60 
61     override def reportFailure(cause: Throwable): Unit = ExecutionContext.defaultReporter(cause)
62 
63     @tailrec
64     private def run(): Unit = {
65       val r = next()
66       if (r == null) {
67         rest = null // don't want a memory leak from potentially large array buffers
68         running = false
69       } else {
70         try r.run()
71         catch { case e: Throwable => reportFailure(e) }
72         run()
73       }
74     }
75 
76     private def next(): Runnable = {
77       val r = r0
78       r0 = r1
79       r1 = r2
80       r2 = if (rest != null) rest.pollFirst() else null
81       r
82     }
83   }
