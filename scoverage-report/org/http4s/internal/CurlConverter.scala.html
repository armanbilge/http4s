19 import org.http4s.Headers

23 import org.typelevel.ci.CIString

25 private[http4s] object CurlConverter {
26 
27   // escapes characters that are used in the curl-command, such as '
28   private def escapeQuotationMarks(s: String) = s.replaceAll("'", """'\\''""")
29 
30   private def newline: String = " \\\n  "
31 
32   private def prepareMethodName(method: Method): String =
33     s"$newline--request ${method.name}"
34 
35   private def prepareUri(uri: Uri): String =
36     s"$newline--url '${escapeQuotationMarks(uri.renderString)}'"
37 
38   private def prepareHeaders(headers: Headers, redactHeadersWhen: CIString => Boolean): String = {
39     val preparedHeaders = headers
40       .redactSensitive(redactHeadersWhen)
41       .headers
42       .map { header =>
43         s"""--header '${escapeQuotationMarks(s"${header.name}: ${header.value}")}'"""
44       }
45       .mkString(newline)
46 
47     if (preparedHeaders.isEmpty) "" else newline + preparedHeaders
48   }
49 
50   def requestToCurlWithoutBody[F[_]](
51       request: Request[F],
52       redactHeadersWhen: CIString => Boolean,
53   ): String = {
54     val params = List(
55       prepareMethodName(request.method),
56       prepareUri(request.uri),
57       prepareHeaders(request.headers, redactHeadersWhen),
58     ).mkString
59 
60     s"curl$params"
61   }
62 }
