33   def threadFactory(
34       name: Long => String = { l =>
35         s"http4s-$l"
36       },
37       daemon: Boolean = false,
38       priority: ThreadPriority = ThreadPriority.Norm,
39       uncaughtExceptionHandler: PartialFunction[(Thread, Throwable), Unit] = PartialFunction.empty,
40       backingThreadFactory: ThreadFactory = Executors.defaultThreadFactory,
41   ): ThreadFactory =
42     new ThreadFactory {
43       val count = new AtomicLong(0)
44       override def newThread(r: Runnable): Thread = {
45         val thread = backingThreadFactory.newThread(r)
46         thread.setName(name(count.getAndIncrement))
47         thread.setDaemon(daemon)
48         thread.setPriority(priority.toInt)
49         thread.setUncaughtExceptionHandler(new UncaughtExceptionHandler {
50           override def uncaughtException(t: Thread, e: Throwable): Unit =
51             uncaughtExceptionHandler.orElse(fallthrough)((t, e))
52           val fallthrough: PartialFunction[(Thread, Throwable), Unit] = {
53             case (t: Thread, e: Throwable) =>
54               Option(t.getThreadGroup)
55                 .getOrElse(Thread.getDefaultUncaughtExceptionHandler)
56                 .uncaughtException(t, e)
57           }
58         })
59         thread
60       }
61     }
62 
63   def newDaemonPool(
64       name: String,
65       min: Int = 4,
66       cpuFactor: Double = 3.0,
67       timeout: Boolean = false,
68   ): ThreadPoolExecutor = {
69     val cpus = Runtime.getRuntime.availableProcessors
70     val exec = new ThreadPoolExecutor(
71       math.max(min, cpus),
72       math.max(min, (cpus * cpuFactor).ceil.toInt),
73       10L,
74       TimeUnit.SECONDS,
75       new LinkedBlockingQueue[Runnable],
76       threadFactory(i => s"$name-$i", daemon = true),
77     )
78     exec.allowCoreThreadTimeOut(timeout)
79     exec
80   }
81 
82   def newDaemonPoolExecutionContext(
83       name: String,
84       min: Int = 4,
85       cpuFactor: Double = 3.0,
86       timeout: Boolean = false,
87   ): ExecutionContext =
88     ExecutionContext.fromExecutorService(newDaemonPool(name, min, cpuFactor, timeout))
89 
90   def newBlockingPool(name: String): ExecutorService = {
91     val CorePoolSize = 0
92     val MaxPoolSize = Int.MaxValue
93     val KeepAliveTime = 1.minute
94 
95     new ThreadPoolExecutor(
96       CorePoolSize,
97       MaxPoolSize,
98       KeepAliveTime.toSeconds,
99       TimeUnit.SECONDS,
100       new SynchronousQueue[Runnable](false),
101       threadFactory(i => s"$name-$i", daemon = true),
102     )
103   }
