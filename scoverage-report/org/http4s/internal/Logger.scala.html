33   def defaultLogHeaders[F[_], A <: Message[F]](message: A)(
34       logHeaders: Boolean,
35       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
36   ): String =
37     if (logHeaders)

39     else ""
40 
41   def defaultLogBody[F[_]: Concurrent, A <: Message[F]](
42       message: A
43   )(logBody: Boolean): Option[F[String]] =
44     if (logBody) {
45       val isBinary = message.contentType.exists(_.mediaType.binary)
46       val isJson = message.contentType.exists(mT =>
47         mT.mediaType == MediaType.application.json || mT.mediaType.subType.endsWith("+json")
48       )
49       val bodyStream = if (!isBinary || isJson) {
50         message.bodyText(implicitly, message.charset.getOrElse(Charset.`UTF-8`))
51       } else {
52         message.body.map(b => java.lang.Integer.toHexString(b & 0xff))
53       }
54       Some(bodyStream.compile.string)
55     } else None
56 
57   def logMessage[F[_], A <: Message[F]](message: A)(
58       logHeaders: Boolean,
59       logBody: Boolean,
60       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
61   )(log: String => F[Unit])(implicit F: Concurrent[F]): F[Unit] = {
62     val logBodyText = (_: Stream[F, Byte]) => defaultLogBody[F, A](message)(logBody)
63 
64     logMessageWithBodyText[F, A](message)(logHeaders, logBodyText, redactHeadersWhen)(log)
65   }
66 
67   def logMessageWithBodyText[F[_], A <: Message[F]](message: A)(
68       logHeaders: Boolean,
69       logBodyText: Stream[F, Byte] => Option[F[String]],
70       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
71   )(log: String => F[Unit])(implicit F: Monad[F]): F[Unit] = {
72     def prelude =
73       message match {
74         case req: Request[_] => s"${req.httpVersion} ${req.method} ${req.uri}"
75         case resp: Response[_] => s"${resp.httpVersion} ${resp.status}"
76       }
77 
78     val headers: String = defaultLogHeaders[F, A](message)(logHeaders, redactHeadersWhen)
79 
80     val bodyText: F[String] =
81       logBodyText(message.body) match {
82         case Some(textF) => textF.map(text => s"""body="$text"""")
83         case None => F.pure("")
84       }
85 
86     def spaced(x: String): String = if (x.isEmpty) x else s" $x"
87 
88     bodyText
89       .map(body => s"$prelude${spaced(headers)}${spaced(body)}")
90       .flatMap(log)
91   }
