64   /** Encode a string to a Hexadecimal string representation
65     * Adapted from apache commons Hex.encodeHex
66     */
67   private[http4s] final def encodeHex(data: Array[Byte]): Array[Char] = {
68     val l = data.length
69     val out = new Array[Char](l << 1)
70     // two characters form the hex value.
71     def iterateData(out: Array[Char], l: Int): Array[Char] = {
72       def innerEncode(l: Int, i: Int, j: Int): Array[Char] =
73         i match {
74           case k if k < l =>
75             out(j) = Digits((0xf0 & data(k)) >>> 4)
76             out(j + 1) = Digits(0x0f & data(k))
77             innerEncode(l, k + 1, j + 2)
78           case _ => out
79         }
80       innerEncode(l, 0, 0)
81     }
82     iterateData(out, l)
83   }

90   /** Dirty, optimized hex decoding based off of apache
91     * common hex decoding, ported over to scala
92     *
93     * @param data
94     * @return
95     */
96   private[http4s] final def decodeHex(data: Array[Char]): Option[Array[Byte]] = {
97     def toDigit(ch: Char): Int = {
98       val digit = Character.digit(ch, 16)
99       if (digit == -1)
100         throw HexDecodeException
101       else
102         digit
103     }
104 
105     val len = data.length
106     if ((len & 0x01) != 0) None
107     val out = new Array[Byte](len >> 1)
108     var f: Int = -1
109     // two characters form the hex value.
110     try {
111       var i = 0
112       var j = 0
113       while (j < len) {
114         f = toDigit(data(j)) << 4
115         j += 1
116         f = f | toDigit(data(j))
117         j += 1
118         out(i) = (f & 0xff).toByte
119 
120         i += 1
121       }
122       Some(out)
123     } catch {
124       case HexDecodeException => None
125     }
126   }

128   private[http4s] def fromCompletionStage[F[_], CF[x] <: CompletionStage[x], A](fcs: F[CF[A]])(
129       implicit
130       // Concurrent is intentional, see https://github.com/http4s/http4s/pull/3255#discussion_r395719880
131       F: Async[F]
132   ): F[A] =
133     fcs.flatMap { cs =>
134       F.async_ { cb =>
135         cs.handle[Unit] {
136           { (result: A, err: Throwable) =>
137             err match {
138               case null => cb(Right(result))
139               case _: CancellationException => ()
140               case ex: CompletionException if ex.getCause ne null => cb(Left(ex.getCause))
141               case ex => cb(Left(ex))
142             }
143           }: java.util.function.BiFunction[A, Throwable, Unit]
144         }
145         ()
146       }
147     }

166   // TODO Remove in 1.0. We can do better with MurmurHash3.
167   private[http4s] def hashLower(s: String): Int = {
168     var h = 0
169     var i = 0
170     val len = s.length
171     while (i < len) {
172       // Strings are equal igoring case if either their uppercase or lowercase
173       // forms are equal. Equality of one does not imply the other, so we need
174       // to go in both directions. A character is not guaranteed to make this
175       // round trip, but it doesn't matter as long as all equal characters
176       // hash the same.
177       h = h * 31 + Character.toLowerCase(Character.toUpperCase(s.charAt(i)))
178       i += 1
179     }
180     h
181   }

183   @deprecated("Use fs2.text.decodeWithCharset", "0.23.5")
184   def decode[F[_]: RaiseThrowable](charset: Charset): Pipe[F, Byte, String] = { in =>
185     val decoder = charset.nioCharset.newDecoder
186     val byteBufferSize = 16
187     val byteBuffer = ByteBuffer.allocate(byteBufferSize)
188     val charBuffer =
189       CharBuffer.allocate(math.ceil(byteBufferSize.toDouble * decoder.averageCharsPerByte).toInt)
190 
191     def skipByteOrderMark(chunk: Chunk[Byte]): Chunk[Byte] =
192       if (chunk.size >= 3 && chunk.take(3) == utf8Bom)
193         chunk.drop(3)
194       else chunk
195 
196     def out = {
197       val s = charBuffer.flip().toString()
198       charBuffer.clear()
199       if (s.isEmpty) Pull.done else Pull.output1(s)
200     }
201 
202     Pull
203       .loop[F, String, Stream[F, Byte]] { stream =>
204         stream.pull.unconsN(byteBuffer.remaining(), allowFewer = true).flatMap {
205           case None =>
206             byteBuffer.flip()
207             val result = decoder.decode(byteBuffer, charBuffer, true)
208             byteBuffer.compact()
209             result match {
210               case _ if result.isUnderflow =>
211                 def flushLoop: Pull[F, String, Unit] =
212                   decoder.flush(charBuffer) match {
213                     case result if result.isUnderflow =>
214                       out
215                     case result if result.isOverflow =>
216                       out >> flushLoop
217                   }
218                 flushLoop.as(None)
219               case _ if result.isOverflow =>
220                 out.as(Some(Stream.empty))
221               case _ if result.isMalformed =>
222                 Pull.raiseError(new MalformedInputException(result.length()))
223               case _ if result.isUnmappable =>
224                 Pull.raiseError(new UnmappableCharacterException(result.length()))
225             }
226           case Some((chunk, stream)) =>
227             val chunkWithoutBom = skipByteOrderMark(chunk)
228             byteBuffer.put(chunkWithoutBom.toArray)
229             byteBuffer.flip()
230             val result = decoder.decode(byteBuffer, charBuffer, false)
231             byteBuffer.compact()
232             result match {
233               case _ if result.isUnderflow || result.isOverflow =>
234                 out.as(Some(stream))
235               case _ if result.isMalformed =>
236                 Pull.raiseError(new MalformedInputException(result.length()))
237               case _ if result.isUnmappable =>
238                 Pull.raiseError(new UnmappableCharacterException(result.length()))
239             }
240         }
241       }(in)
242       .void
243       .stream
244   }

255   /** This is the same as `Order.by(f).compare(a, b)`, but with the parameters
256     * re-arraigned to make it easier to partially apply the function to two
257     * instances of a type before supplying the `A => B`.
258     *
259     * The intended use case is that `f: A => B` will extract out a single
260     * field from two instances of some Product type and then compare the value
261     * of the field. This can then be done in turn for each field of a Product,
262     * significantly reducing the amount of code needed to write an `Order`
263     * instance for a Product with many fields.
264     *
265     * See the `Order` instance for `Uri` for an example of this usage.
266     */
267   private[http4s] def compareField[A, B: Order](
268       a: A,
269       b: A,
270       f: A => B,
271   ): Int =
272     Order.by[A, B](f).compare(a, b)

274   /** Given at least one `Int` intended to represent the result of a comparison
275     * of two fields of some Product type, reduce the result to the first
276     * non-zero value, or return 0 if all comparisons are 0.
277     *
278     * The intended use case for this function is to reduce the amount of code
279     * needed to write an `Order` instance for Product types. One can use
280     * [[compareField]] to generate a comparison for each field in a product
281     * type, then apply this function to get a ordering for the entire Product
282     * type.
283     *
284     * See the `Order` instance for `Uri` for an example of this usage.
285     *
286     * @note The values of the `NonEmptyChain` are encoded `F[Int]`, where `F`
287     *       is some `Comonad`. The primary Comonads with which we are
288     *       concerned are `Eval` and `Id`. `Eval` will give lazy evaluation of
289     *       the Ordering, stopping as soon as the result is known, and `Id`
290     *       will give strict evaluation, in the case where the caller has good
291     *       reason to believe that evaluating the thunks will be slower than
292     *       strictly evaluating the result over all fields.
293     */
294   private[http4s] def reduceComparisons_[F[_]: Comonad](
295       comparisons: NonEmptyChain[F[Int]]
296   ): Int = {
297     val extractComparison: F[Int] => Option[Int] =
298       _.extract match {
299         case 0 => None
300         case otherwise => Some(otherwise)
301       }
302 
303     comparisons
304       .reduceLeftTo(extractComparison) {
305         case (None, next) => extractComparison(next)
306         case (otherwise, _) => otherwise
307       }
308       .getOrElse(0)
309   }
310 
311   /** Similar to [[reduceComparisons_]] but with the `F` type forced to `Eval`
312     * for every comparison other than the first one. This encodes the commonly
313     * desired use case of only evaluating the minimum number of comparisons
314     * required to determine the ordering.
315     */
316   private[http4s] def reduceComparisons(
317       head: Int,
318       tail: Eval[Int]*
319   ): Int =
320     reduceComparisons_(NonEmptyChain(Eval.now(head), tail: _*))

322   private[http4s] def appendSanitized(sb: StringBuilder, s: String): Unit = {
