28   /** Return X509 certificates for the session.
29     *
30     * @param sslSession Session from which certificate to be read
31     * @return Empty array if no certificates can be read from {{{sslSession}}}
32     */
33   def getCertChain(sslSession: SSLSession): List[X509Certificate] =
34     Try {
35       val cf = CertificateFactory.getInstance("X.509")
36       sslSession.getPeerCertificates.map { certificate =>
37         val stream = new ByteArrayInputStream(certificate.getEncoded)
38         cf.generateCertificate(stream).asInstanceOf[X509Certificate]
39       }
40     }.toOption.getOrElse(Array.empty[X509Certificate]).toList

42   /** Given the name of a TLS/SSL cipher suite, return an int representing it effective stream
43     * cipher key strength. i.e. How much entropy material is in the key material being fed into the
44     * encryption routines.
45     *
46     * The following specs and resources were consulted for the implementation of this function:
47     * IANA TLS Cipher Suites registry (IANA https://www.iana.org/assignments/tls-parameters/tls-parameters.txt),
48     * RFC 2246 (The TLS Protocol Version 1.0, Appendix C), RFC 4346 (The Transport Layer Security
49     * (TLS) Protocol Version 1.1, Appendix C), RFC 5246 (The Transport Layer Security (TLS) Protocol
50     * Version 1.2, Appendix C), RFC 8446 (The Transport Layer Security (TLS) Protocol Version 1.3,
51     * Appendix B), RFC 7539 (ChaCha20 and Poly1305 for IETF Protocols), RFC 3713 (A Description of the
52     * Camellia Encryption Algorithm), RFC 5794 (A Description of the ARIA Encryption Algorithm),
53     * RFC 4269 (The SEED Encryption Algorithm), draft-crypto-sm4-00 (The SM4 Block Cipher Algorithm
54     * And Its Modes Of Operations), draft-smyshlyaev-tls12-gost-suites (GOST Cipher Suites for
55     * Transport Layer Security (TLS) Protocol Version 1.2), and draft-smyshlyaev-tls13-gost-suites
56     * (GOST Cipher Suites for Transport Layer Security (TLS) Protocol Version).
57     * The following table summarizes the relevant information from the sources listed above:
58     * <pre>
59     *                         Effective
60     *     Cipher       Type    Key Bits
61     *
62     *     NULL       *  Stream     0
63     *     IDEA_CBC      Block    128
64     *     RC2_CBC_40 *  Block     40
65     *     RC4_40     *  Stream    40
66     *     RC4_128       Stream   128
67     *     DES40_CBC  *  Block     40
68     *     DES_CBC       Block     56
69     *     3DES_EDE_CBC  Block    168
70     *     AES_128       Block    128
71     *     AES_256       Block    256
72     *     ChaCha20      Stream   256
73     *     Camellia_128  Block    128
74     *     Camellia_256  Block    256
75     *     Aria_128      Block    128
76     *     Aria_256      Block    256
77     *     SEED          Block    128
78     *     SM4           Block    128
79     *     Kuznyechik    Block    256
80     *     Magma         Block    256
81     *     GOST 28147-89 Block    256
82     * </pre>
83     *
84     * @param cipherSuite String name of the TLS cipher suite.
85     * @return int indicating the effective key entropy bit-length.
86     */
87   def deduceKeyLength(cipherSuite: String): Int =
88     if (cipherSuite == null) 0
89     else if (cipherSuite.contains("_AES_256_")) 256
90     else if (cipherSuite.contains("WITH_RC4_128_")) 128
91     else if (cipherSuite.contains("_AES_128_")) 128
92     else if (cipherSuite.contains("WITH_RC4_40_")) 40
93     else if (cipherSuite.contains("WITH_3DES_EDE_CBC_")) 168
94     else if (cipherSuite.contains("WITH_IDEA_CBC_")) 128
95     else if (cipherSuite.contains("WITH_RC2_CBC_40_")) 40
96     else if (cipherSuite.contains("WITH_DES40_CBC_")) 40
97     else if (cipherSuite.contains("WITH_DES_CBC_40")) 40
98     else if (cipherSuite.contains("WITH_DES_CBC_")) 56
99     else if (cipherSuite.contains("_CHACHA20_")) 256
100     else if (cipherSuite.contains("_WITH_CAMELLIA_128_")) 128
101     else if (cipherSuite.contains("_WITH_CAMELLIA_256_")) 256
102     else if (cipherSuite.contains("_WITH_ARIA_128_")) 128
103     else if (cipherSuite.contains("_WITH_ARIA_256_")) 256
104     else if (cipherSuite.contains("_WITH_SEED_")) 128
105     else if (cipherSuite.contains("_SM4_")) 128
106     else if (cipherSuite.contains("_WITH_KUZNYECHIK_")) 256
107     else if (cipherSuite.contains("_WITH_MAGMA_")) 256
108     else if (cipherSuite.contains("_WITH_28147_")) 256
109     else 0
