19 import javax.net.ssl.SSLContext
20 import scala.util.control.NonFatal

22 /** Indicates how to resolve SSLContext.
23   *  * NoSSL                = do not use SSL/HTTPS
24   *  * TryDefaultSSLContext = `SSLContext.getDefault()`, or `None` on systems where the default is unavailable
25   *  * Provided             = use the explicitly passed SSLContext
26   */
27 private[http4s] sealed trait SSLContextOption extends Product with Serializable
28 private[http4s] object SSLContextOption {
29   case object NoSSL extends SSLContextOption
30   case object TryDefaultSSLContext extends SSLContextOption
31   final case class Provided(sslContext: SSLContext) extends SSLContextOption
32 
33   def toMaybeSSLContext(sco: SSLContextOption): Option[SSLContext] =
34     sco match {
35       case SSLContextOption.NoSSL => None
36       case SSLContextOption.TryDefaultSSLContext => tryDefaultSslContext
37       case SSLContextOption.Provided(context) => Some(context)
38     }
39 
40   def tryDefaultSslContext: Option[SSLContext] =
41     try Some(SSLContext.getDefault())
42     catch {
43       case NonFatal(_) => None
44     }
45 }
