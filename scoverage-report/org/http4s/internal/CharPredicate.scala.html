153   // efficient handling of 7bit-ASCII chars
154   final case class MaskBased private[CharPredicate] (lowMask: Long, highMask: Long)
155       extends CharPredicate {
156 
157     def apply(c: Char): Boolean = {
158       val mask = if (c < 64) lowMask else highMask
159       ((1L << c.toLong) & ((c - 128) >> 31) & mask) != 0L // branchless for `(c < 128) && (mask & (1L << c) != 0)`
160     }
161 
162     def ++(that: CharPredicate): CharPredicate =
163       that match {
164         case Empty => this
165         case _ if this == Empty => that
166         case MaskBased(low, high) => MaskBased(lowMask | low, highMask | high)
167         case _ => this.or(that)
168       }
169 
170     def ++(chars: Seq[Char]): CharPredicate =
171       chars.foldLeft(this: CharPredicate) {
172         case (_: MaskBased, c) if unmaskable(c) =>
173           new ArrayBased(chars.toArray) ++ new ArrayBased(toArray)
174         case (MaskBased(low, high), c) if c < 64 => MaskBased(low | 1L << c.toLong, high)
175         case (MaskBased(low, high), c) => MaskBased(low, high | 1L << c.toLong)
176         case (x, _) => x // once the fold acc is not a MaskBased we are done
177       }
178 
179     def --(that: CharPredicate): CharPredicate =
180       that match {
181         case Empty => this
182         case _ if this == Empty => this
183         case MaskBased(low, high) => MaskBased(lowMask & ~low, highMask & ~high)
184         case _ => this.andNot(that)
185       }
186 
187     def --(chars: Seq[Char]): CharPredicate =
188       if (this != Empty)
189         chars.foldLeft(this: CharPredicate) {
190           case (_: MaskBased, c) if unmaskable(c) => this.andNot(new ArrayBased(chars.toArray))
191           case (MaskBased(low, high), c) if c < 64 => MaskBased(low & ~(1L << c.toLong), high)
192           case (MaskBased(low, high), c) => MaskBased(low, high & ~(1L << c.toLong))
193           case (x, _) => x // once the fold acc is not a MaskBased we are done
194         }
195       else this
196 
197     def intersect(that: CharPredicate): CharPredicate =
198       that match {
199         case Empty => Empty
200         case _ if this == Empty => Empty
201         case MaskBased(low, high) => MaskBased(lowMask & low, highMask & high)
202         case _ => this.and(that)
203       }
204 
205     def size: Int = java.lang.Long.bitCount(lowMask) + java.lang.Long.bitCount(highMask)
206 
207     def toArray: Array[Char] = {
208       val array = new Array[Char](size)
209       getChars(array, 0)
210       array
211     }
212 
213     def getChars(array: Array[Char], startIx: Int): Unit = {
214       @tailrec def rec(mask: Long, offset: Int, bit: Int, ix: Int): Int =
215         if (bit < 64 && ix < array.length)
216           if ((mask & (1L << bit)) > 0) {
217             array(ix) = (offset + bit).toChar
218             rec(mask, offset, bit + 1, ix + 1)
219           } else rec(mask, offset, bit + 1, ix)
220         else ix
221       rec(
222         highMask,
223         64,
224         java.lang.Long.numberOfTrailingZeros(highMask),
225         rec(lowMask, 0, java.lang.Long.numberOfTrailingZeros(lowMask), startIx),
226       )
227       ()
228     }
229 
230     override def toString(): String = "CharPredicate.MaskBased(" + new String(toArray) + ')'
231   }
