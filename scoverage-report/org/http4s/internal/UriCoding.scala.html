24 /* Exists to work around circular dependencies */
25 private[http4s] object UriCoding {
26   val Unreserved: CharPredicate = AlphaNum ++ "-_.~"
27   val QueryNoEncode: CharPredicate = Unreserved ++ "?/"
28 
29   def encode(
30       toEncode: String,
31       charset: JCharset,
32       spaceIsPlus: Boolean,
33       toSkip: Char => Boolean,
34   ): String = {
35     val in = charset.encode(toEncode)
36     val out = CharBuffer.allocate((in.remaining() * 3).toInt)
37     while (in.hasRemaining) {
38       val c = in.get().toChar
39       if (toSkip(c)) {
40         out.put(c)
41       } else if (c == ' ' && spaceIsPlus) {
42         out.put('+')
43       } else {
44         out.put('%')
45         out.put(HexUpperCaseChars((c >> 4) & 0xf))
46         out.put(HexUpperCaseChars(c & 0xf))
47       }
48     }
49     out.flip()
50     out.toString
51   }
52 
53   private val HexUpperCaseChars = (0 until 16).map { i =>
54     Character.toUpperCase(Character.forDigit(i, 16))
55   }
