23 /** A projection of a [[Response]] without the body.
24   *
25   * @note The [[Response#attributes]] are omitted in this encoding because they
26   *       do not (and can not) have a [[cats.kernel.Order]] instance. If they
27   *       were included here, then we could not write a [[cats.kernel.Order]]
28   *       instance for [[ResponsePrelude]], limiting some of its utility, e.g. it
29   *       could not be used in a [[cats.data.NonEmptySet]].
30   */
31 sealed abstract class ResponsePrelude extends Product with Serializable {
32   def headers: Headers
33   def httpVersion: HttpVersion
34   def status: Status
35 
36   def withHeaders(value: Headers): ResponsePrelude
37   def withHttpVersion(value: HttpVersion): ResponsePrelude
38   def withStatus(value: Status): ResponsePrelude
39 
40   // final //
41 
42   final def mapHeaders(f: Headers => Headers): ResponsePrelude =
43     withHeaders(f(headers))
44 
45   final def mapHttpVersion(f: HttpVersion => HttpVersion): ResponsePrelude =
46     withHttpVersion(f(httpVersion))
47 
48   final def mapStatus(f: Status => Status): ResponsePrelude =
49     withStatus(f(status))
50 
51   override final def toString: String =
52     s"ResponsePrelude(headers = ${headers.redactSensitive()}, httpVersion = ${httpVersion}, status = ${status})"
53 }
54 
55 object ResponsePrelude {
56   private[this] final case class ResponsePreludeImpl(
57       override final val headers: Headers,
58       override final val httpVersion: HttpVersion,
59       override final val status: Status,
60   ) extends ResponsePrelude {
61     override final def withHeaders(value: Headers): ResponsePrelude =
62       this.copy(headers = value)
63 
64     override final def withHttpVersion(value: HttpVersion): ResponsePrelude =
65       this.copy(httpVersion = value)
66 
67     override final def withStatus(value: Status): ResponsePrelude =
68       this.copy(status = value)
69   }
70 
71   def apply(
72       headers: Headers,
73       httpVersion: HttpVersion,
74       status: Status,
75   ): ResponsePrelude =
76     ResponsePreludeImpl(
77       headers,
78       httpVersion,
79       status,
80     )
81 
82   def fromResponse[F[_]](value: Response[F]): ResponsePrelude =
83     ResponsePreludeImpl(
84       value.headers,
85       value.httpVersion,
86       value.status,
87     )
88 
89   implicit val catsHashAndOrderForResponsePrelude
90       : Hash[ResponsePrelude] with Order[ResponsePrelude] =
91     new Hash[ResponsePrelude] with Order[ResponsePrelude] {
92       override def hash(x: ResponsePrelude): Int = x.hashCode
93 
94       override def compare(x: ResponsePrelude, y: ResponsePrelude): Int =
95         reduceComparisons(
96           x.headers.compare(y.headers),
97           Eval.later(x.httpVersion.compare(y.httpVersion)),
98           Eval.later(x.status.compare(y.status)),
99         )
100     }
101 
102   implicit val catsShowForResponsePrelude: Show[ResponsePrelude] =
103     Show.fromToString
104 
105   implicit def stdLibOrdering: Ordering[ResponsePrelude] =
106     catsHashAndOrderForResponsePrelude.toOrdering
107 }
