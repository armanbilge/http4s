31   def accurateContentLengthIfDefined(a: A): IsEq[F[Boolean]] =
32     (for {
33       entity <- F.pure(encoder.toEntity(a))
34       body <- entity.body.compile.toVector
35       bodyLength = body.size.toLong
36       contentLength = entity.length
37     } yield contentLength.fold(true)(_ === bodyLength)) <-> F.pure(true)
38 
39   def noContentLengthInStaticHeaders: Boolean =
40     !encoder.headers.contains[`Content-Length`]
41 
42   def noTransferEncodingInStaticHeaders: Boolean =
43     !encoder.headers.contains[`Transfer-Encoding`]
44 }
