39 final case class Charset private (nioCharset: NioCharset) extends Renderable {
40   def withQuality(q: QValue): CharsetRange.Atom = CharsetRange.Atom(this, q)
41   def toRange: CharsetRange.Atom = withQuality(QValue.One)
42 
43   def render(writer: Writer): writer.type = writer << nioCharset.name
44 }
45 
46 object Charset extends CharsetCompanionPlatform {
47 
48   implicit val catsInstancesForHttp4sCharset: Hash[Charset] with Order[Charset] =
49     new Hash[Charset] with Order[Charset] {
50       override def hash(x: Charset): Int =
51         x.hashCode
52 
53       override def compare(x: Charset, y: Charset): Int =
54         // Using Comparable
55         x.nioCharset.compareTo(y.nioCharset)
56     }
57 
58   val `US-ASCII`: Charset = Charset(StandardCharsets.US_ASCII)
59   val `ISO-8859-1`: Charset = Charset(StandardCharsets.ISO_8859_1)
60   val `UTF-8`: Charset = Charset(StandardCharsets.UTF_8)
61   val `UTF-16`: Charset = Charset(StandardCharsets.UTF_16)
62   val `UTF-16BE`: Charset = Charset(StandardCharsets.UTF_16BE)
63   val `UTF-16LE`: Charset = Charset(StandardCharsets.UTF_16LE)
64 
65   // Charset.forName caches a whopping two values and then
66   // synchronizes.  We can prevent this by pre-caching all the lookups
67   // that will succeed.
68   private val cache: HashMap[String, NioCharset] = {
69     val map = new HashMap[String, NioCharset]
70     for {
71       cs <- availableCharsets
72       name <- cs.name :: cs.aliases.asScala.toList
73     } map.put(name.toLowerCase(Locale.ROOT), cs)
74     map
75   }
76 
77   def fromNioCharset(nioCharset: NioCharset): Charset = Charset(nioCharset)
78 
79   def fromString(name: String): ParseResult[Charset] =
80     cache.get(name.toLowerCase(Locale.ROOT)) match {
81       case nioCharset: NioCharset => Right(Charset.apply(nioCharset))
82       case null => Left(ParseFailure("Invalid charset", s"$name is not a supported Charset"))
83     }
84 }
