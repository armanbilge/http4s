39 @implicitNotFound(
40   "Cannot convert from ${A} to an Entity, because no EntityEncoder[${F}, ${A}] instance could be found."
41 )
42 trait EntityEncoder[F[_], A] { self =>
43 
44   /** Convert the type `A` to an [[Entity]] in the effect type `F` */
45   def toEntity(a: A): Entity[F]
46 
47   /** Headers that may be added to a [[Message]]
48     *
49     * Examples of such headers would be Content-Type.
50     * __NOTE:__ The Content-Length header will be generated from the resulting Entity and thus should not be added.
51     */
52   def headers: Headers
53 
54   /** Make a new [[EntityEncoder]] using this type as a foundation */
55   def contramap[B](f: B => A): EntityEncoder[F, B] =
56     new EntityEncoder[F, B] {
57       override def toEntity(a: B): Entity[F] = self.toEntity(f(a))
58       override def headers: Headers = self.headers
59     }
60 
61   /** Get the [[org.http4s.headers.Content-Type]] of the body encoded by this [[EntityEncoder]],
62     * if defined the headers
63     */
64   def contentType: Option[`Content-Type`] = headers.get[`Content-Type`]
65 
66   /** Get the [[Charset]] of the body encoded by this [[EntityEncoder]], if defined the headers */
67   def charset: Option[Charset] = headers.get[`Content-Type`].flatMap(_.charset)
68 
69   /** Generate a new EntityEncoder that will contain the `Content-Type` header */
70   def withContentType(tpe: `Content-Type`): EntityEncoder[F, A] =
71     new EntityEncoder[F, A] {
72       override def toEntity(a: A): Entity[F] = self.toEntity(a)
73       override val headers: Headers = self.headers.put(tpe)
74     }
75 }

77 object EntityEncoder {
78   private val DefaultChunkSize = 4096
79 
80   /** summon an implicit [[EntityEncoder]] */
81   def apply[F[_], A](implicit ev: EntityEncoder[F, A]): EntityEncoder[F, A] = ev
82 
83   /** Create a new [[EntityEncoder]] */
84   def encodeBy[F[_], A](hs: Headers)(f: A => Entity[F]): EntityEncoder[F, A] =
85     new EntityEncoder[F, A] {
86       override def toEntity(a: A): Entity[F] = f(a)
87       override def headers: Headers = hs
88     }
89 
90   /** Create a new [[EntityEncoder]] */
91   def encodeBy[F[_], A](hs: Header.ToRaw*)(f: A => Entity[F]): EntityEncoder[F, A] = {
92     val hdrs = if (hs.nonEmpty) Headers(hs: _*) else Headers.empty
93     encodeBy(hdrs)(f)
94   }
95 
96   /** Create a new [[EntityEncoder]]
97     *
98     * This constructor is a helper for types that can be serialized synchronously, for example a String.
99     */
100   def simple[F[_], A](hs: Header.ToRaw*)(toChunk: A => Chunk[Byte]): EntityEncoder[F, A] =
101     encodeBy(hs: _*) { a =>
102       val c = toChunk(a)
103       Entity[F](Stream.chunk(c), Some(c.size.toLong))
104     }
105 
106   /** Encodes a value from its Show instance.  Too broad to be implicit, too useful to not exist. */
107   def showEncoder[F[_], A](implicit
108       charset: Charset = `UTF-8`,
109       show: Show[A],
110   ): EntityEncoder[F, A] = {
111     val hdr = `Content-Type`(MediaType.text.plain).withCharset(charset)
112     simple[F, A](hdr)(a => Chunk.array(show.show(a).getBytes(charset.nioCharset)))
113   }
114 
115   def emptyEncoder[F[_], A]: EntityEncoder[F, A] =
116     new EntityEncoder[F, A] {
117       def toEntity(a: A): Entity[F] = Entity.empty
118       def headers: Headers = Headers.empty
119     }
120 
121   /** A stream encoder is intended for streaming, and does not calculate its
122     * bodies in advance.  As such, it does not calculate the Content-Length in
123     * advance.  This is for use with chunked transfer encoding.
124     */
125   implicit def streamEncoder[F[_], A](implicit
126       W: EntityEncoder[F, A]
127   ): EntityEncoder[F, Stream[F, A]] =
128     new EntityEncoder[F, Stream[F, A]] {
129       override def toEntity(a: Stream[F, A]): Entity[F] =
130         Entity(a.flatMap(W.toEntity(_).body))
131 
132       override def headers: Headers =
133         W.headers.get[`Transfer-Encoding`] match {
134           case Some(transferCoding) if transferCoding.hasChunked =>
135             W.headers
136           case _ =>
137             W.headers.add(`Transfer-Encoding`(TransferCoding.chunked.pure[NonEmptyList]))
138         }
139     }
140 
141   implicit def unitEncoder[F[_]]: EntityEncoder[F, Unit] =
142     emptyEncoder[F, Unit]
143 
144   implicit def stringEncoder[F[_]](implicit
145       charset: Charset = `UTF-8`
146   ): EntityEncoder[F, String] = {
147     val hdr = `Content-Type`(MediaType.text.plain).withCharset(charset)
148     simple(hdr)(s => Chunk.array(s.getBytes(charset.nioCharset)))
149   }
150 
151   implicit def charArrayEncoder[F[_]](implicit
152       charset: Charset = `UTF-8`
153   ): EntityEncoder[F, Array[Char]] =
154     stringEncoder[F].contramap(new String(_))
155 
156   implicit def chunkEncoder[F[_]]: EntityEncoder[F, Chunk[Byte]] =
157     simple(`Content-Type`(MediaType.application.`octet-stream`))(identity)
158 
159   implicit def byteArrayEncoder[F[_]]: EntityEncoder[F, Array[Byte]] =
160     chunkEncoder[F].contramap(Chunk.array[Byte])
161 
162   implicit def byteVectorEncoder[F[_]]: EntityEncoder[F, ByteVector] =
163     chunkEncoder[F].contramap(Chunk.byteVector)
164 
165   /** Encodes an entity body.  Chunking of the stream is preserved.  A
166     * `Transfer-Encoding: chunked` header is set, as we cannot know
167     * the content length without running the stream.
168     */
169   implicit def entityBodyEncoder[F[_]]: EntityEncoder[F, EntityBody[F]] =
170     encodeBy(`Transfer-Encoding`(TransferCoding.chunked.pure[NonEmptyList])) { body =>
171       Entity(body, None)
172     }
173

184   // TODO if Header moves to Entity, can add a Content-Disposition with the filename
185   implicit def pathEncoder[F[_]: Files]: EntityEncoder[F, fs2.io.file.Path] =
186     encodeBy[F, fs2.io.file.Path](`Transfer-Encoding`(TransferCoding.chunked)) { p =>
187       Entity(Files[F].readAll(p))
188     }
189 
190   implicit def inputStreamEncoder[F[_]: Sync, IS <: InputStream]: EntityEncoder[F, F[IS]] =
191     entityBodyEncoder[F].contramap { (in: F[IS]) =>
192       readInputStream[F](in.widen[InputStream], DefaultChunkSize)
193     }
194 
195   // TODO parameterize chunk size
196   implicit def readerEncoder[F[_], R <: Reader](implicit
197       F: Sync[F],
198       charset: Charset = `UTF-8`,
199   ): EntityEncoder[F, F[R]] =
200     entityBodyEncoder[F].contramap { (fr: F[R]) =>
201       // Shared buffer
202       val charBuffer = CharBuffer.allocate(DefaultChunkSize)
203       def readToBytes(r: Reader): F[Option[Chunk[Byte]]] =
204         for {
205           // Read into the buffer
206           readChars <- F.blocking(r.read(charBuffer))
207         } yield {
208           // Flip to read
209           charBuffer.flip()
210 
211           if (readChars < 0) None
212           else if (readChars == 0) Some(Chunk.empty)
213           else {
214             // Encode to bytes according to the charset
215             val bb = charset.nioCharset.encode(charBuffer)
216             // Read into a Chunk
217             val b = new Array[Byte](bb.remaining())
218             bb.get(b)
219             Some(Chunk.array(b))
220           }
221         }
222 
223       def useReader(r: Reader) =
224         Stream
225           .eval(readToBytes(r))
226           .repeat
227           .unNoneTerminate
228           .flatMap(Stream.chunk[F, Byte])
229 
230       // The reader is closed at the end like InputStream
231       Stream.bracket(fr)(r => F.delay(r.close())).flatMap(useReader)
232     }
233 
234   implicit def multipartEncoder[F[_]]: EntityEncoder[F, Multipart[F]] =
235     new MultipartEncoder[F]

237   implicit def entityEncoderContravariant[F[_]]: Contravariant[EntityEncoder[F, *]] =
