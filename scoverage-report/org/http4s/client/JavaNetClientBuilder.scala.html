41 /** Builder for a [[Client]] backed by on `java.net.HttpUrlConnection`.
42   *
43   * The java.net client adds no dependencies beyond `http4s-client`.
44   * This client is generally not production grade, but convenient for
45   * exploration in a REPL.
46   *
47   * All I/O operations in this client are blocking.
48   *
49   * @define WHYNOSHUTDOWN Creation of the client allocates no
50   * resources, and any resources allocated while using this client
51   * are reclaimed by the JVM at its own leisure.
52   */
53 sealed abstract class JavaNetClientBuilder[F[_]] private (
54     val connectTimeout: Duration,
55     val readTimeout: Duration,
56     val proxy: Option[Proxy],
57     val hostnameVerifier: Option[HostnameVerifier],
58     val sslSocketFactory: Option[SSLSocketFactory],
59 )(implicit protected val F: Async[F])
60     extends BackendBuilder[F, Client[F]] {
61   private def copy(
62       connectTimeout: Duration = connectTimeout,
63       readTimeout: Duration = readTimeout,
64       proxy: Option[Proxy] = proxy,
65       hostnameVerifier: Option[HostnameVerifier] = hostnameVerifier,
66       sslSocketFactory: Option[SSLSocketFactory] = sslSocketFactory,
67   ): JavaNetClientBuilder[F] =
68     new JavaNetClientBuilder[F](
69       connectTimeout = connectTimeout,
70       readTimeout = readTimeout,
71       proxy = proxy,
72       hostnameVerifier = hostnameVerifier,
73       sslSocketFactory = sslSocketFactory,
74     ) {}
75 
76   def withConnectTimeout(connectTimeout: Duration): JavaNetClientBuilder[F] =
77     copy(connectTimeout = connectTimeout)
78 
79   def withReadTimeout(readTimeout: Duration): JavaNetClientBuilder[F] =
80     copy(readTimeout = readTimeout)
81 
82   def withProxyOption(proxy: Option[Proxy]): JavaNetClientBuilder[F] =
83     copy(proxy = proxy)
84   def withProxy(proxy: Proxy): JavaNetClientBuilder[F] =
85     withProxyOption(Some(proxy))
86   def withoutProxy: JavaNetClientBuilder[F] =
87     withProxyOption(None)
88 
89   def withHostnameVerifierOption(
90       hostnameVerifier: Option[HostnameVerifier]
91   ): JavaNetClientBuilder[F] =
92     copy(hostnameVerifier = hostnameVerifier)
93   def withHostnameVerifier(hostnameVerifier: HostnameVerifier): JavaNetClientBuilder[F] =
94     withHostnameVerifierOption(Some(hostnameVerifier))
95   def withoutHostnameVerifier: JavaNetClientBuilder[F] =
96     withHostnameVerifierOption(None)
97 
98   def withSslSocketFactoryOption(
99       sslSocketFactory: Option[SSLSocketFactory]
100   ): JavaNetClientBuilder[F] =
101     copy(sslSocketFactory = sslSocketFactory)
102   def withSslSocketFactory(sslSocketFactory: SSLSocketFactory): JavaNetClientBuilder[F] =
103     withSslSocketFactoryOption(Some(sslSocketFactory))
104   def withoutSslSocketFactory: JavaNetClientBuilder[F] =
105     withSslSocketFactoryOption(None)
106 
107   /** Creates a [[Client]].
108     *
109     * The shutdown of this client is a no-op. $WHYNOSHUTDOWN
110     */
111   def create: Client[F] =
112     Client { (req: Request[F]) =>
113       def respond(conn: HttpURLConnection): F[Response[F]] =
114         for {
115           _ <- configureSsl(conn)
116           _ <- F.delay(conn.setConnectTimeout(timeoutMillis(connectTimeout)))
117           _ <- F.delay(conn.setReadTimeout(timeoutMillis(readTimeout)))
118           _ <- F.delay(conn.setRequestMethod(req.method.renderString))
119           _ <- F.delay(req.headers.foreach { case Header.Raw(name, value) =>
120             conn.setRequestProperty(name.toString, value)
121           })
122           _ <- F.delay(conn.setInstanceFollowRedirects(false))
123           _ <- F.delay(conn.setDoInput(true))
124           // TODO: fix the blocking here
125           resp <- fetchResponse(req, conn)
126         } yield resp
127 
128       for {
129         url <- Resource.eval(F.delay(new URL(req.uri.toString)))
130         conn <- Resource.make(openConnection(url)) { conn =>
131           F.delay(conn.getInputStream().close()).recoverWith { case _: IOException =>
132             F.delay(Option(conn.getErrorStream()).foreach(_.close()))
133           }
134         }
135         resp <- Resource.eval(respond(conn))
136       } yield resp
137     }
138 
139   def resource: Resource[F, Client[F]] =
140     Resource.eval(F.delay(create))
141 
142   private def fetchResponse(req: Request[F], conn: HttpURLConnection) =
143     for {
144       _ <- writeBody(req, conn)
145       code <- F.blocking(conn.getResponseCode)
146       status <- F.fromEither(Status.fromInt(code))
147       headers <- F.blocking(
148         Headers(
149           conn.getHeaderFields.asScala
150             .filter(_._1 != null)
151             .flatMap { case (k, vs) => vs.asScala.map(Header.Raw(CIString(k), _)) }
152             .toList
153         )
154       )
155     } yield Response(status = status, headers = headers, body = readBody(conn))
156 
157   private def timeoutMillis(d: Duration): Int =
158     d match {
159       case d: FiniteDuration if d > Duration.Zero => d.toMillis.max(0).min(Int.MaxValue).toInt
160       case _ => 0
161     }
162 
163   private def openConnection(url: URL)(implicit F: Sync[F]) =
164     proxy match {
165       case Some(p) =>
166         F.delay(url.openConnection(p).asInstanceOf[HttpURLConnection])
167       case None =>
168         F.delay(url.openConnection().asInstanceOf[HttpURLConnection])
169     }
170 
171   private def writeBody(req: Request[F], conn: HttpURLConnection): F[Unit] =
172     if (req.isChunked)
173       F.delay(conn.setDoOutput(true)) *>
174         F.delay(conn.setChunkedStreamingMode(4096)) *>
175         req.body
176           .through(writeOutputStream(F.delay(conn.getOutputStream), false))
177           .compile
178           .drain
179     else
180       req.contentLength match {
181         case Some(len) if len >= 0L =>
182           F.delay(conn.setDoOutput(true)) *>
183             F.delay(conn.setFixedLengthStreamingMode(len)) *>
184             req.body
185               .through(writeOutputStream(F.delay(conn.getOutputStream), false))
186               .compile
187               .drain
188         case _ =>
189           F.delay(conn.setDoOutput(false))
190       }
191 
192   private def readBody(conn: HttpURLConnection): Stream[F, Byte] = {
193     def inputStream =
194       F.delay(Option(conn.getInputStream)).recoverWith {
195         case _: IOException if conn.getResponseCode > 0 =>
196           F.delay(Option(conn.getErrorStream))
197       }
198     Stream.eval(inputStream).flatMap {
199       case Some(in) => readInputStream(F.pure(in), 4096, false)
200       case None => Stream.empty
201     }
202   }
203 
204   private def configureSsl(conn: HttpURLConnection) =
205     conn match {
206       case connSsl: HttpsURLConnection =>
207         for {
208           _ <- hostnameVerifier.fold(F.unit)(hv => F.delay(connSsl.setHostnameVerifier(hv)))
209           _ <- sslSocketFactory.fold(F.unit)(sslf => F.delay(connSsl.setSSLSocketFactory(sslf)))
210         } yield ()
211       case _ => F.unit
212     }
213 }
