37   def getPaths(implicit F: Temporal[IO]): Map[String, IO[Response[IO]]] =

49       NoContentPath -> Response[IO](NoContent).pure[IO],

51       EmptyNotFoundPath -> Response[IO](NotFound).pure[IO],
