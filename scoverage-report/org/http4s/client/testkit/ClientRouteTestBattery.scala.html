195   private def checkResponse(rec: Response[IO], expected: Response[IO]): IO[Boolean] = {
196     // This isn't a generically safe normalization for all header, but
197     // it's close enough for our purposes
198     def normalizeHeaderValue(h: Header.Raw) =
199       Header.Raw(h.name, h.value.replace("; ", ";").toLowerCase(Locale.ROOT))
200 
201     for {
202       _ <- IO(rec.status).assertEquals(expected.status)
203       body <- rec.body.compile.to(Array)
204       expBody <- expected.body.compile.to(Array)
205       _ <- IO(body).map(Arrays.equals(_, expBody)).assert
206       headers = rec.headers.headers.map(normalizeHeaderValue)
207       expectedHeaders = expected.headers.headers.map(normalizeHeaderValue)
208       _ <- IO(expectedHeaders.diff(headers)).assertEquals(Nil)
209       _ <- IO(rec.httpVersion).assertEquals(expected.httpVersion)
210     } yield true
