41   // high-level API
42   def apply[F[_]](num: Int, secure: Boolean, routes: HttpRoutes[F])(implicit
43       F: Async[F]
44   ): Resource[F, ServerScaffold[F]] =
45     for {
46       dispatcher <- Dispatcher[F]
47       scaffold <- apply(num, secure, RoutesToNettyAdapter[F](routes, dispatcher))
48     } yield scaffold

50   // mid-level API

52       F: Async[F]
53   ): Resource[F, ServerScaffold[F]] =
54     apply[F](num, secure, HandlersToNettyAdapter(handlers))
55 
56   // low-level API
57   def apply[F[_]](num: Int, secure: Boolean, makeHandler: F[ChannelInboundHandler])(implicit
58       F: Async[F]
59   ): Resource[F, ServerScaffold[F]] =
60     for {
61       dispatcher <- Dispatcher[F]
62       maybeSsl <-
63         if (secure) Resource.eval[F, SSLContext](makeSslContext[F]).map(Some(_))
64         else Resource.pure[F, Option[SSLContext]](None)
65       servers <- NettyTestServer[F](port = 0, makeHandler, maybeSsl, dispatcher).replicateA(num)
66     } yield new ServerScaffold(servers.toVector)
67 
68   private def makeSslContext[F[_]](implicit F: Sync[F]): F[SSLContext] = F.delay {
69     val ksStream = this.getClass.getResourceAsStream("/server.jks")
70     val ks = KeyStore.getInstance("JKS")
71     ks.load(ksStream, "password".toCharArray)
72     ksStream.close()
73 
74     val kmf = KeyManagerFactory.getInstance(
75       Option(Security.getProperty("ssl.KeyManagerFactory.algorithm"))
76         .getOrElse(KeyManagerFactory.getDefaultAlgorithm)
77     )
78 
79     kmf.init(ks, "secure".toCharArray)
80 
81     val sslContext = SSLContext.getInstance("TLS")
82     sslContext.init(kmf.getKeyManagers, null, null)
83     sslContext
84   }
