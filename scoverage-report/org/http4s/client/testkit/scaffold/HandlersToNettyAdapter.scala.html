32 private[http4s] object HandlersToNettyAdapter {
33   val defaultFallbackHandler: Handler = (ctx: ChannelHandlerContext, _: HttpRequest) => {

45 private[http4s] class HandlersToNettyAdapter private (
46     handlers: Map[(HttpMethod, String), Handler],
47     fallbackHandler: Handler,
48 ) extends SimpleChannelInboundHandler[HttpObject] {
49 
50   private val logger = getLogger(this.getClass)
51 
52   private var currentRequest: HttpRequest = null
53   private var currentHandler: Handler = null
54 
55   override def channelRead0(ctx: ChannelHandlerContext, msg: HttpObject): Unit = {
56 
57     msg match {
58       case request: HttpRequest =>
59         logger.trace(
60           s"Recieved [${request.method()}] [${request.uri()}] request from [${ctx.channel.remoteAddress()}]."
61         )
62         if (HttpUtil.is100ContinueExpected(request)) {
63           ctx.write(new DefaultFullHttpResponse(HTTP_1_1, CONTINUE))
64         } else {
65           currentRequest = request
66           currentHandler = handlers.getOrElse(
67             (request.method(), new URI(request.uri()).getPath()),
68             fallbackHandler,
69           )
70           currentHandler.onRequestStart(ctx, currentRequest)
71         }
72       case _ =>
73     }
74 
75     msg match {
76       case content: HttpContent =>
77         logger.trace("Recieved content.")
78         currentHandler.onContent(ctx, currentRequest, content)
79       case _ =>
80     }
81 
82     msg match {
83       case _: LastHttpContent =>
84         logger.trace("Request finished.")
85         currentHandler.onRequestEnd(ctx, currentRequest)
86         currentRequest = null
87         currentHandler = null
88       case _ =>
89     }
90   }
91 
92   override def channelReadComplete(ctx: ChannelHandlerContext): Unit = {
93     ctx.flush()
94     ()
95   }
96 
97   override def exceptionCaught(ctx: ChannelHandlerContext, cause: Throwable): Unit = {
98     logger.warn(cause)("")
99     ctx.close()
100     ()
101   }
102 
103 }

122 private[http4s] object HandlerHelpers {
123 
124   def sendResponse(
125       ctx: ChannelHandlerContext,
126       status: HttpResponseStatus,
127       content: ByteBuf = Unpooled.buffer(0),
128       headers: HttpHeaders = EmptyHttpHeaders.INSTANCE,
129       closeConnection: Boolean = false,
130   ): ChannelFuture = {
131     val response = new DefaultFullHttpResponse(HTTP_1_1, status, content)
132     response.headers().setAll(headers)
133     response.headers().set(CONTENT_LENGTH, response.content().readableBytes())
134     response.headers().set(CONNECTION, HttpHeaderValues.KEEP_ALIVE)
135     if (closeConnection) {
136       // disconnect sends FIN.
137       ctx.writeAndFlush(response).addListener { (f: ChannelFuture) =>
138         f.channel.disconnect()
139         ()
140       }
141     } else {
142       ctx.writeAndFlush(response)
143     }
144   }
145 
146   def sendChunkedResponseHead(
147       ctx: ChannelHandlerContext,
148       status: HttpResponseStatus,
149       headers: HttpHeaders = EmptyHttpHeaders.INSTANCE,
150   ): ChannelFuture = {
151     val response = new DefaultHttpResponse(HTTP_1_1, status)
152     response.headers().setAll(headers)
153     response.headers().set(CONNECTION, HttpHeaderValues.KEEP_ALIVE)
154     ctx.writeAndFlush(response)
155   }
156 
157   def sendChunk(ctx: ChannelHandlerContext, content: ByteBuf): ChannelFuture =
158     ctx.writeAndFlush(new DefaultHttpContent(content))
159 
160   def sendEmptyLastChunk(ctx: ChannelHandlerContext): ChannelFuture =
161     ctx.writeAndFlush(LastHttpContent.EMPTY_LAST_CONTENT)
162 
163   def utf8Text(s: String): ByteBuf =
164     Unpooled.copiedBuffer(s, CharsetUtil.UTF_8)
