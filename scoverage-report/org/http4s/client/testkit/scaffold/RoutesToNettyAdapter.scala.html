58 private[http4s] class RoutesToHandlerAdapter[F[_]](
59     routes: HttpRoutes[F],
60     dispatcher: Dispatcher[F],
61     requestBodyQueue: Ref[F, Queue[F, Option[Chunk[Byte]]]],
62 )(implicit F: Async[F])
63     extends Handler {
64 
65   override def onRequestStart(ctx: ChannelHandlerContext, request: HttpRequest): Unit =
66     dispatcher.unsafeRunSync(
67       for {
68         method <- Method.fromString(request.method().name()).liftTo[F]
69         uri <- http4s.Uri.fromString(request.uri()).liftTo[F]
70         headers = http4s.Headers(request.headers().names().asScala.toVector.flatMap { k =>
71           val vs = request.headers().getAll(k)
72           vs.asScala.toVector.map(v => (k -> v): http4s.Header.ToRaw)
73         }): @nowarn("cat=deprecation")
74         bodyQueue <- Queue.unbounded[F, Option[Chunk[Byte]]]
75         _ <- requestBodyQueue.set(bodyQueue)
76         body = fs2.Stream.fromQueueNoneTerminatedChunk(bodyQueue)
77         http4sRequest = http4s.Request(method, uri, headers = headers, body = body)
78         _ <- processRequest(ctx, http4sRequest).start
79       } yield ()
80     )
81 
82   private def processRequest(
83       ctx: ChannelHandlerContext,
84       http4sRequest: http4s.Request[F],
85   ): F[Unit] =
86     for {
87       response <- routes.run(http4sRequest).value
88       _ <- response.fold(
89         F.delay(HandlerHelpers.sendResponse(ctx, NOT_FOUND)).liftToF
90       )(response => sendResponse(ctx, response))
91     } yield ()
92 
93   private def sendResponse(ctx: ChannelHandlerContext, response: Response[F]): F[Unit] =
94     response.headers.get[`Transfer-Encoding`] match {
95       case None => sendStrictResponse(ctx, response)
96       case Some(codings) if codings.hasChunked => sendChunkedResponse(ctx, response)
97       case Some(_) => F.delay(HandlerHelpers.sendResponse(ctx, BAD_REQUEST)).liftToF
98     }
99 
100   private def sendStrictResponse(ctx: ChannelHandlerContext, response: Response[F]): F[Unit] =
101     response.body.compile.to(Array).flatMap { bodyBytes =>
102       F.delay(
103         HandlerHelpers.sendResponse(
104           ctx,
105           HttpResponseStatus.valueOf(response.status.code),
106           Unpooled.copiedBuffer(bodyBytes),
107           makeNettyHeaders(response.headers),
108         )
109       ).liftToF
110     }
111 
112   private def sendChunkedResponse(ctx: ChannelHandlerContext, response: Response[F]): F[Unit] =
113     for {
114       _ <- F
115         .delay(
116           HandlerHelpers.sendChunkedResponseHead(
117             ctx,
118             HttpResponseStatus.valueOf(response.status.code),
119             makeNettyHeaders(response.headers),
120           )
121         )
122         .liftToF
123       _ <- response.body.chunks
124         .evalMap(chunk =>
125           F.delay(HandlerHelpers.sendChunk(ctx, Unpooled.copiedBuffer(chunk.toArray))).liftToF
126         )
127         .compile
128         .drain
129       _ <- F.delay(HandlerHelpers.sendEmptyLastChunk(ctx)).liftToF
130     } yield ()
131 
132   private def makeNettyHeaders(http4sHeaders: Headers): DefaultHttpHeaders = {
133     val nettyHeaders = new DefaultHttpHeaders()
134     http4sHeaders.foreach { h =>
135       nettyHeaders.add(h.name.toString, h.value)
136       ()
137     }
138     nettyHeaders
139   }
140 
141   override def onContent(
142       ctx: ChannelHandlerContext,
143       request: HttpRequest,
144       content: HttpContent,
145   ): Unit = {
146     val bytes = new Array[Byte](content.content().readableBytes())
147     content.content().readBytes(bytes)
148     dispatcher.unsafeRunSync(requestBodyQueue.get.flatMap(_.offer(Some(Chunk.array(bytes)))))
149   }
150 
151   override def onRequestEnd(ctx: ChannelHandlerContext, request: HttpRequest): Unit =
152     dispatcher.unsafeRunSync(requestBodyQueue.get.flatMap(_.offer(None)))
153
