21 import org.http4s.Uri.Scheme

23 /** Represents a key for requests that can conceivably share a [[Connection]]. */
24 final case class RequestKey(scheme: Scheme, authority: Authority) {
25   override def toString: String = s"${scheme.value}://${authority}"
26 }
27 
28 object RequestKey {
29   def fromRequest[F[_]](request: Request[F]): RequestKey = {
30     val uri = request.uri
31     RequestKey(uri.scheme.getOrElse(Scheme.http), uri.authority.getOrElse(Authority()))
32   }
33 }
