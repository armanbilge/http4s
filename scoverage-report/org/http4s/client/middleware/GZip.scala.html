34 /** Client middleware for enabling gzip.
35   */
36 object GZip {
37   private val supportedCompressions =
38     NonEmptyList.of(ContentCoding.gzip, ContentCoding.deflate)
39 
40   def apply[F[_]](bufferSize: Int = 32 * 1024)(client: Client[F])(implicit F: Async[F]): Client[F] =
41     Client[F] { req =>
42       val reqWithEncoding = addHeaders(req)
43       val responseResource = client.run(reqWithEncoding)
44 
45       responseResource.map { actualResponse =>
46         decompress(bufferSize, actualResponse)
47       }
48     }
49 
50   private def addHeaders[F[_]](req: Request[F]): Request[F] =
51     req.headers.get[`Accept-Encoding`] match {
52       case Some(_) =>
53         req
54       case _ =>
55         req.putHeaders(`Accept-Encoding`(supportedCompressions))
56     }
57 
58   private def decompress[F[_]](bufferSize: Int, response: Response[F])(implicit
59       F: Async[F]
60   ): Response[F] =
61     response.headers.get[`Content-Encoding`] match {
62       case Some(header)
63           if header.contentCoding == ContentCoding.gzip || header.contentCoding == ContentCoding.`x-gzip` =>
64         val gunzip: Pipe[F, Byte, Byte] =
65           _.through(Compression[F].gunzip(bufferSize)).flatMap(_.content)
66         response
67           .filterHeaders(nonCompressionHeader)
68           .pipeBodyThrough(decompressWith(gunzip))
69 
70       case Some(header) if header.contentCoding == ContentCoding.deflate =>
71         val deflate: Pipe[F, Byte, Byte] = Compression[F].deflate(DeflateParams(bufferSize))
72         response
73           .filterHeaders(nonCompressionHeader)
74           .pipeBodyThrough(decompressWith(deflate))
75 
76       case _ =>
77         response
78     }
79 
80   private def decompressWith[F[_]](
81       decompressor: Pipe[F, Byte, Byte]
82   )(implicit F: Async[F]): Pipe[F, Byte, Byte] =
83     _.pull.peek1
84       .flatMap {
85         case None => Pull.raiseError(EmptyBodyException)
86         case Some((_, fullStream)) => Pull.output1(fullStream)
87       }
88       .stream
89       .flatten
90       .through(decompressor)
91       .handleErrorWith {
92         case EmptyBodyException => Stream.empty
93         case error => Stream.raiseError(error)
94       }
95 
96   private def nonCompressionHeader(header: Header.Raw): Boolean =
97     header.name != ci"Content-Encoding" &&
98       header.name != ci"Content-Length"
99 
100   private object EmptyBodyException extends Throwable with NoStackTrace
101 }
