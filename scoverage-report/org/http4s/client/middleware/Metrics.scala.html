34 /** Client middleware to record metrics for the http4s client.
35   *
36   * This middleware will record:
37   * - Number of active requests
38   * - Time duration to receive the response headers
39   * - Time duration to process the whole response body
40   * - Time duration of errors, timeouts and other abnormal terminations
41   *
42   * This middleware can be extended to support any metrics ecosystem by
43   * implementing the [[org.http4s.metrics.MetricsOps]] type
44   */
45 object Metrics {
46 
47   /** Wraps a [[Client]] with a middleware capable of recording metrics
48     *
49     * @param ops a algebra describing the metrics operations
50     * @param classifierF a function that allows to add a classifier that can be customized per request
51     * @param client the [[Client]] to gather metrics from
52     * @return the metrics middleware wrapping the [[Client]]
53     */
54   def apply[F[_]](
55       ops: MetricsOps[F],
56       classifierF: Request[F] => Option[String] = { (_: Request[F]) =>
57         None
58       },
59   )(client: Client[F])(implicit F: Clock[F], C: Concurrent[F]): Client[F] =
60     effect(ops, classifierF.andThen(_.pure[F]))(client)
61 
62   /** Wraps a [[Client]] with a middleware capable of recording metrics
63     *
64     * Same as [[apply]], but can classify requests effectually, e.g. performing side-effects or examining the body.
65     * Failed attempt to classify the request (e.g. failing with `F.raiseError`) leads to not recording metrics for that request.
66     *
67     * @note Compiling the request body in `classifierF` is unsafe, unless you are using some caching middleware.
68     *
69     * @param ops a algebra describing the metrics operations
70     * @param classifierF a function that allows to add a classifier that can be customized per request
71     * @param client the [[Client]] to gather metrics from
72     * @return the metrics middleware wrapping the [[Client]]
73     */
74   def effect[F[_]](ops: MetricsOps[F], classifierF: Request[F] => F[Option[String]])(
75       client: Client[F]
76   )(implicit F: Clock[F], C: Concurrent[F]): Client[F] =
77     Client(withMetrics(client, ops, classifierF))
78 
79   private def withMetrics[F[_]](
80       client: Client[F],
81       ops: MetricsOps[F],
82       classifierF: Request[F] => F[Option[String]],
83   )(req: Request[F])(implicit F: Clock[F], C: Concurrent[F]): Resource[F, Response[F]] =
84     for {
85       statusRef <- Resource.eval(C.ref[Option[Status]](None))
86       start <- Resource.eval(F.monotonic)
87       resp <- executeRequestAndRecordMetrics(
88         client,
89         ops,
90         classifierF,
91         req,
92         statusRef,
93         start.toNanos,
94       )
95     } yield resp
96 
97   private def executeRequestAndRecordMetrics[F[_]](
98       client: Client[F],
99       ops: MetricsOps[F],
100       classifierF: Request[F] => F[Option[String]],
101       req: Request[F],
102       statusRef: Ref[F, Option[Status]],
103       start: Long,
104   )(implicit F: Clock[F], C: Concurrent[F]): Resource[F, Response[F]] =
105     (for {
106       classifier <- Resource.eval(classifierF(req))
107       _ <- Resource.make(ops.increaseActiveRequests(classifier))(_ =>
108         ops.decreaseActiveRequests(classifier)
109       )
110       _ <- Resource.onFinalize(
111         F.monotonic
112           .flatMap(now =>
113             statusRef.get.flatMap(oStatus =>
114               oStatus.traverse_(status =>
115                 ops.recordTotalTime(req.method, status, now.toNanos - start, classifier)
116               )
117             )
118           )
119       )
120       resp <- client.run(req)
121       _ <- Resource.eval(statusRef.set(Some(resp.status)))
122       end <- Resource.eval(F.monotonic)
123       _ <- Resource.eval(ops.recordHeadersTime(req.method, end.toNanos - start, classifier))
124     } yield resp).handleErrorWith { (e: Throwable) =>
125       Resource.eval(
126         classifierF(req).flatMap(registerError(start, ops, _)(e)) *> C.raiseError[Response[F]](e)
127       )
128     }
129 
130   private def registerError[F[_]](start: Long, ops: MetricsOps[F], classifier: Option[String])(
131       e: Throwable
132   )(implicit F: Clock[F], C: Concurrent[F]): F[Unit] =
133     F.monotonic
134       .flatMap { now =>
135         if (e.isInstanceOf[TimeoutException])
136           ops.recordAbnormalTermination(now.toNanos - start, Timeout, classifier)
137         else
138           ops.recordAbnormalTermination(now.toNanos - start, Error(e), classifier)
139       }
140 }
