29 /** Client middleware to follow redirect responses.
30   *
31   * A 301 or 302 response is followed by:
32   * - a GET if the request was GET or POST
33   * - a HEAD if the request was a HEAD
34   * - the original request method and body if the body had no effects
35   * - the redirect is not followed otherwise
36   *
37   * A 303 response is followed by:
38   * - a HEAD if the request was a HEAD
39   * - a GET for all other methods
40   *
41   * A 307 or 308 response is followed by:
42   * - the original request method and body, if the body had no effects
43   * - the redirect is not followed otherwise
44   *
45   * Whenever we follow with a GET or HEAD, an empty body is sent, and
46   * all payload headers defined in https://datatracker.ietf.org/doc/html/rfc7231#section-3.3
47   * are stripped.
48   *
49   * If the response does not contain a valid Location header, the redirect is
50   * not followed.
51   *
52   * Headers whose names match `sensitiveHeaderFilter` are not exposed when
53   * redirecting to a different authority.
54   */
55 object FollowRedirect {
56   def apply[F[_]](
57       maxRedirects: Int,
58       sensitiveHeaderFilter: CIString => Boolean = Headers.SensitiveHeaders,
59   )(client: Client[F])(implicit F: Concurrent[F]): Client[F] = {
60     def nextRequest(
61         req: Request[F],
62         uri: Uri,
63         method: Method,
64         cookies: List[ResponseCookie],
65     ): Request[F] = {
66       // https://datatracker.ietf.org/doc/html/rfc7231#section-7.1
67       val nextUri = uri.copy(
68         scheme = uri.scheme.orElse(req.uri.scheme),
69         authority = uri.authority.orElse(req.uri.authority),
70         fragment = uri.fragment.orElse(req.uri.fragment),
71       )
72 
73       def stripSensitiveHeaders(req: Request[F]): Request[F] =
74         if (req.uri.authority != nextUri.authority)
75           req.transformHeaders(hs =>
76             Headers(hs.headers.filterNot(h => sensitiveHeaderFilter(h.name)))
77           )
78         else
79           req
80 
81       def propagateCookies(req: Request[F]): Request[F] =
82         if (req.uri.authority == nextUri.authority)
83           cookies.foldLeft(req) { case (nextReq, cookie) =>
84             nextReq.addCookie(cookie.name, cookie.content)
85           }
86         else
87           req
88 
89       def clearBodyFromGetHead(req: Request[F]): Request[F] =
90         method match {
91           case GET | HEAD => req.withEmptyBody
92           case _ => req
93         }
94 
95       clearBodyFromGetHead(
96         propagateCookies(stripSensitiveHeaders(req)).withMethod(method).withUri(nextUri)
97       )
98     }
99 
100     def redirectLoop(
101         req: Request[F],
102         redirects: Int,
103         hotswap: Hotswap[F, Response[F]],
104     ): F[Response[F]] =
105       hotswap.clear *> // Release the prior connection before allocating a new
106         hotswap.swap(client.run(req)).flatMap { resp =>
107           val l: Option[Location] = resp.headers.get[Location]
108           (methodForRedirect(req, resp), l) match {
109             case (Some(method), Some(loc)) if redirects < maxRedirects =>
110               val nextReq = nextRequest(req, loc.uri, method, resp.cookies)
111               redirectLoop(nextReq, redirects + 1, hotswap)
112                 .map(res => res.withAttribute(redirectUrisKey, nextReq.uri +: getRedirectUris(res)))
113             case _ =>
114               // IF the response is missing the Location header, OR there is no method to redirect,
115               // OR we have exceeded max number of redirections, THEN we redirect no more
116               resp.pure[F]
117           }
118         }
119 
120     Client { req =>
121       Hotswap.create[F, Response[F]].flatMap { case hotswap =>
122         Resource.eval(redirectLoop(req, 0, hotswap))
123       }
124     }
125   }
126 
127   private def methodForRedirect[F[_]](req: Request[F], resp: Response[F]): Option[Method] =
128     resp.status.code match {
129       case 301 | 302 =>
130         req.method match {
131           case POST =>
132             // "For historical reasons, a user agent MAY change the request method
133             // from POST to GET for the subsequent request." -- RFC 7231
134             //
135             // This is common practice, so we do.
136             //
137             // TODO In a future version, configure this behavior through a
138             // redirect config.
139             Some(GET)
140 
141           case m =>
142             Some(m)
143         }
144 
145       case 303 =>
146         // "303 (See Other) status code indicates that the server is
147         // redirecting the user agent to a different resource, as indicated
148         // by a URI in the Location header field, which is intended to
149         // provide an indirect response to the original request.  A user
150         // agent can perform a retrieval request targeting that URI (a GET
151         // or HEAD request if using HTTP)" -- RFC 7231
152         req.method match {
153           case HEAD => Some(HEAD)
154           case _ => Some(GET)
155         }
156 
157       case 307 | 308 =>
158         // "Note: This status code is similar to 302 (Found), except that
159         // it does not allow changing the request method from POST to GET.
160         // This specification defines no equivalent counterpart for 301
161         // (Moved Permanently) ([RFC7238], however, defines the status code
162         // 308 (Permanent Redirect) for this purpose). These status codes
163         // may not change the method." -- RFC 7231
164         //
165         Some(req.method)
166 
167       case _ =>
168         None
169     }
170 
171   private val redirectUrisKey = Key.newKey[SyncIO, List[Uri]].unsafeRunSync()
172 
173   /** Get the redirection URIs for a `response`.
174     * Excludes the initial request URI
175     */
176   def getRedirectUris[F[_]](response: Response[F]): List[Uri] =
177     response.attributes.lookup(redirectUrisKey).getOrElse(Nil)
178 }
