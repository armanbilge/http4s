25 /** Algebra for Interfacing with the Cookie Jar.
26   * Allows manual intervention and eviction.
27   */
28 trait CookieJar[F[_]] {
29 
30   /** Default Expiration Approach, Removes Expired Cookies
31     */
32   def evictExpired: F[Unit]
33 
34   /** Available for Use To Relieve Memory Pressure
35     */
36   def evictAll: F[Unit]
37 
38   /** Add Cookie to the cookie jar
39     */
40   def addCookie(c: ResponseCookie, uri: Uri): F[Unit] =
41     addCookies(List((c, uri)))
42 
43   /** Like addCookie but puts several in at once
44     */
45   def addCookies[G[_]: Foldable](cookies: G[(ResponseCookie, Uri)]): F[Unit]
46 
47   /** Enrich a Request with the cookies available
48     */
49   def enrichRequest[G[_]](r: Request[G]): F[Request[G]]
50 }
51 
52 /** Cookie Jar Companion Object
53   * Contains constructors for client middleware or raw
54   * jar creation, as well as the middleware
55   */
56 object CookieJar {
57 
58   /** Middleware Constructor Using a Provided [[CookieJar]].
59     */
60   def apply[F[_]: Async](
61       alg: CookieJar[F]
62   )(
63       client: Client[F]
64   ): Client[F] =
65     Client { req =>
66       for {
67         _ <- Resource.eval(alg.evictExpired)
68         modRequest <- Resource.eval(alg.enrichRequest(req))
69         out <- client.run(modRequest)
70         _ <- Resource.eval(
71           out.cookies
72             .map(r => r.domain.fold(r.copy(domain = req.uri.host.map(_.value)))(_ => r))
73             .traverse_(alg.addCookie(_, req.uri))
74         )
75       } yield out
76     }
77 
78   /** Constructor which builds a non-exposed CookieJar
79     * and applies it to the client.
80     */
81   def impl[F[_]: Async](c: Client[F]): F[Client[F]] =
82     in[F, F](c)
83 
84   /** Like [[impl]] except it allows the creation of the middleware in a
85     * different HKT than the client is in.
86     */
87   def in[F[_]: Async, G[_]: Sync](c: Client[F]): G[Client[F]] =
88     jarIn[F, G].map(apply(_)(c))
89 
90   /** Jar Constructor
91     */
92   def jarImpl[F[_]: Async]: F[CookieJar[F]] =
93     jarIn[F, F]
94 
95   /** Like [[jarImpl]] except it allows the creation of the CookieJar in a
96     * different HKT than the client is in.
97     */
98   def jarIn[F[_]: Async, G[_]: Sync]: G[CookieJar[F]] =
99     Ref.in[G, F, Map[CookieKey, CookieValue]](Map.empty).map { ref =>
100       new CookieJarRefImpl[F](ref) {}
101     }
102 
103   private[CookieJar] class CookieJarRefImpl[F[_]: Async](
104       ref: Ref[F, Map[CookieKey, CookieValue]]
105   ) extends CookieJar[F] {
106     override def evictExpired: F[Unit] =
107       for {
108         now <- HttpDate.current[F]
109         out <- ref.update(
110           _.filter { t =>
111             now <= t._2.expiresAt
112           }
113         )
114       } yield out
115 
116     override def evictAll: F[Unit] = ref.set(Map.empty)
117 
118     override def addCookies[G[_]: Foldable](cookies: G[(ResponseCookie, Uri)]): F[Unit] =
119       for {
120         now <- HttpDate.current[F]
121         out <- ref.update(extractFromResponseCookies(_)(cookies, now))
122       } yield out
123 
124     override def enrichRequest[N[_]](r: Request[N]): F[Request[N]] =
125       for {
126         cookies <- ref.get.map(_.map(_._2.cookie).toList)
127       } yield cookiesForRequest(r, cookies)
128         .foldLeft(r) { case (req, cookie) => req.addCookie(cookie) }
129   }
130 
131   private[middleware] final case class CookieKey(
132       name: String,
133       domain: String,
134       path: Option[String],
135   )
136 
137   private[middleware] final class CookieValue(
138       val setAt: HttpDate,
139       val expiresAt: HttpDate,
140       val cookie: ResponseCookie,
141   ) {
142     override def equals(obj: Any): Boolean =
143       obj match {
144         case c: CookieValue =>
145           setAt == c.setAt &&
146           expiresAt == c.expiresAt &&
147           cookie == c.cookie
148         case _ => false
149       }
150   }
151 
152   private[middleware] object CookieValue {
153     def apply(
154         setAt: HttpDate,
155         expiresAt: HttpDate,
156         cookie: ResponseCookie,
157     ): CookieValue = new CookieValue(setAt, expiresAt, cookie)
158   }
159 
160   private[middleware] def expiresAt(
161       now: HttpDate,
162       c: ResponseCookie,
163       default: HttpDate,
164   ): HttpDate =
165     c.expires
166       .orElse(
167         c.maxAge.flatMap(seconds => HttpDate.fromEpochSecond(now.epochSecond + seconds).toOption)
168       )
169       .getOrElse(default)
170 
171   private[middleware] def extractFromResponseCookies[G[_]: Foldable](
172       m: Map[CookieKey, CookieValue]
173   )(
174       cookies: G[(ResponseCookie, Uri)],
175       httpDate: HttpDate,
176   ): Map[CookieKey, CookieValue] =
177     cookies
178       .foldRight(Eval.now(m)) { case ((rc, uri), eM) =>
179         eM.map(m => extractFromResponseCookie(m)(rc, httpDate, uri))
180       }
181       .value
182 
183   private[middleware] def extractFromResponseCookie(
184       m: Map[CookieKey, CookieValue]
185   )(c: ResponseCookie, httpDate: HttpDate, uri: Uri): Map[CookieKey, CookieValue] =
186     c.domain.orElse(uri.host.map(_.value)) match {
187       case Some(domainS) =>
188         val key = CookieKey(c.name, domainS, c.path)
189         val newCookie = c.copy(domain = domainS.some)
190         val expires: HttpDate = expiresAt(httpDate, c, HttpDate.MaxValue)
191         val value = CookieValue(httpDate, expires, newCookie)
192         m + (key -> value)
193       case None => // Ignore Cookies We Can't get a domain for
194         m
195     }
196 
197   private[middleware] def responseCookieToRequestCookie(r: ResponseCookie): RequestCookie =
198     RequestCookie(r.name, r.content)
199 
200   private[middleware] def cookieAppliesToRequest[N[_]](
201       r: Request[N],
202       c: ResponseCookie,
203   ): Boolean = {
204     val domainApplies = c.domain.exists(s =>
205       r.uri.host.forall { authority =>
206         authority.renderString.contains(s)
207       }
208     )
209     val pathApplies = c.path.forall(s => r.uri.path.renderString.contains(s))
210 
211     val secureSatisfied =
212       if (c.secure)
213         r.uri.scheme.exists { scheme =>
214           scheme === Uri.Scheme.https
215         }
216       else true
217 
218     domainApplies && pathApplies && secureSatisfied
219   }
220 
221   private[middleware] def cookiesForRequest[N[_]](
222       r: Request[N],
223       l: List[ResponseCookie],
224   ): List[RequestCookie] =
225     l.foldLeft(List.empty[RequestCookie]) { case (list, cookie) =>
226       if (cookieAppliesToRequest(r, cookie)) responseCookieToRequestCookie(cookie) :: list
227       else list
228     }
229 }
