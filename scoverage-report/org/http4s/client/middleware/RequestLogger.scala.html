29 /** Simple Middleware for Logging Requests As They Are Processed
30   */
31 object RequestLogger {
32   private[this] val logger = getLogger
33 
34   private def defaultLogAction[F[_]: Sync](s: String): F[Unit] = Sync[F].delay(logger.info(s))
35 
36   def apply[F[_]: Async](
37       logHeaders: Boolean,
38       logBody: Boolean,
39       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
40       logAction: Option[String => F[Unit]] = None,
41   )(client: Client[F]): Client[F] =
42     impl(client, logBody) { request =>
43       Logger.logMessage[F, Request[F]](request)(
44         logHeaders,
45         logBody,
46         redactHeadersWhen,
47       )(logAction.getOrElse(defaultLogAction[F]))
48     }
49 
50   def logBodyText[F[_]: Async](
51       logHeaders: Boolean,
52       logBody: Stream[F, Byte] => Option[F[String]],
53       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
54       logAction: Option[String => F[Unit]] = None,
55   )(client: Client[F]): Client[F] =
56     impl(client, logBody = true) { request =>
57       InternalLogger.logMessageWithBodyText[F, Request[F]](request)(
58         logHeaders,
59         logBody,
60         redactHeadersWhen,
61       )(logAction.getOrElse(defaultLogAction[F]))
62     }
63 
64   def customized[F[_]: Async](
65       client: Client[F],
66       logBody: Boolean = true,
67       logAction: Option[String => F[Unit]] = None,
68   )(requestToText: Request[F] => F[String]): Client[F] =
69     impl(client, logBody) { request =>
70       val log = logAction.getOrElse(defaultLogAction[F] _)
71       requestToText(request).flatMap(log)
72     }
73 
74   private def impl[F[_]](client: Client[F], logBody: Boolean)(
75       logMessage: Request[F] => F[Unit]
76   )(implicit F: Async[F]): Client[F] =
77     Client { req =>
78       if (!logBody)
79         Resource.eval(logMessage(req)) *> client.run(req)
80       else
81         Resource.suspend {
82           Ref[F].of(Vector.empty[Chunk[Byte]]).map { vec =>
83             val newBody = Stream.eval(vec.get).flatMap(v => Stream.emits(v)).unchunks
84 
85             val logAtEnd: F[Unit] =
86               logMessage(req.withBodyStream(newBody)).handleErrorWith { case t =>
87                 F.delay(logger.error(t)("Error logging request body"))
88               }
89 
90             // Cannot Be Done Asynchronously - Otherwise All Chunks May Not Be Appended Previous to Finalization
91             val logPipe: Pipe[F, Byte, Byte] =
92               _.observe(_.chunks.flatMap(s => Stream.exec(vec.update(_ :+ s))))
93                 .onFinalizeWeak(logAtEnd)
94 
95             client.run(req.pipeBodyThrough(logPipe))
96           }
97         }
98     }
99 
100   val defaultRequestColor: String = Console.BLUE
101 
102   def colored[F[_]](
103       logHeaders: Boolean,
104       logBody: Boolean,
105       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
106       color: String = defaultRequestColor,
107       logAction: Option[String => F[Unit]] = None,
108   )(client: Client[F])(implicit F: Async[F]): Client[F] =
109     customized(client, logBody, logAction) { request =>
110       import Console._
111       val methodColor =
112         if (request.method.isSafe) color
113         else YELLOW
114 
115       val prelude =
116         s"${request.httpVersion} $methodColor${request.method}$RESET$color $BOLD${request.uri}$RESET$color"
117 
118       val headers: String =
119         InternalLogger.defaultLogHeaders[F, Request[F]](request)(logHeaders, redactHeadersWhen)
120 
121       val bodyText: F[String] =
122         InternalLogger.defaultLogBody[F, Request[F]](request)(logBody) match {
123           case Some(textF) => textF.map(text => s"""body="$text"""")
124           case None => Sync[F].pure("")
125         }
126 
127       def spaced(x: String): String = if (x.isEmpty) x else s" $x"
128 
129       bodyText
130         .map(body => s"$color$prelude${spaced(headers)}${spaced(body)}$RESET")
131     }
132
