29 /** Client middlewares that logs the HTTP responses it receives as soon as they are received locally.
30   *
31   * The "logging" is represented as an effectful action `String => F[Unit]`
32   */
33 object ResponseLogger {
34   private[this] val logger = getLogger
35 
36   private def defaultLogAction[F[_]: Sync](s: String): F[Unit] = Sync[F].delay(logger.info(s))
37 
38   def apply[F[_]: Async](
39       logHeaders: Boolean,
40       logBody: Boolean,
41       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
42       logAction: Option[String => F[Unit]] = None,
43   )(client: Client[F]): Client[F] =
44     impl(client, logBody) { response =>
45       Logger.logMessage[F, Response[F]](response)(
46         logHeaders,
47         logBody,
48         redactHeadersWhen,
49       )(logAction.getOrElse(defaultLogAction[F]))
50     }
51 
52   def logBodyText[F[_]: Async](
53       logHeaders: Boolean,
54       logBody: Stream[F, Byte] => Option[F[String]],
55       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
56       logAction: Option[String => F[Unit]] = None,
57   )(client: Client[F]): Client[F] =
58     impl(client, logBody = true) { response =>
59       InternalLogger.logMessageWithBodyText[F, Response[F]](response)(
60         logHeaders,
61         logBody,
62         redactHeadersWhen,
63       )(logAction.getOrElse(defaultLogAction[F]))
64     }
65 
66   def customized[F[_]: Async](
67       client: Client[F],
68       logBody: Boolean = true,
69       logAction: Option[String => F[Unit]] = None,
70   )(responseToText: Response[F] => F[String]): Client[F] =
71     impl(client, logBody) { response =>
72       val log = logAction.getOrElse(defaultLogAction[F] _)
73       responseToText(response).flatMap(log)
74     }
75 
76   private def impl[F[_]](client: Client[F], logBody: Boolean)(
77       logMessage: Response[F] => F[Unit]
78   )(implicit F: Async[F]): Client[F] = {
79     def logResponse(response: Response[F]): Resource[F, Response[F]] =
80       if (!logBody)
81         Resource.eval(logMessage(response) *> F.delay(response))
82       else
83         Resource.suspend {
84           Ref[F].of(Vector.empty[Chunk[Byte]]).map { vec =>
85             val dumpChunksToVec: Pipe[F, Byte, Nothing] =
86               _.chunks.flatMap(s => Stream.exec(vec.update(_ :+ s)))
87 
88             Resource.make(
89               // Cannot Be Done Asynchronously - Otherwise All Chunks May Not Be Appended before Finalization
90               F.pure(response.pipeBodyThrough(_.observe(dumpChunksToVec)))
91             ) { _ =>
92               val newBody = Stream.eval(vec.get).flatMap(Stream.emits).unchunks
93               logMessage(response.withBodyStream(newBody))
94                 .handleErrorWith(t => F.delay(logger.error(t)("Error logging response body")))
95             }
96           }
97         }
98 
99     Client(req => client.run(req).flatMap(logResponse))
100   }
101 
102   def defaultResponseColor[F[_]](response: Response[F]): String =
103     response.status.responseClass match {
104       case Status.Informational | Status.Successful | Status.Redirection => Console.GREEN
105       case Status.ClientError => Console.YELLOW
106       case Status.ServerError => Console.RED
107     }
108 
109   def colored[F[_]: Async](
110       logHeaders: Boolean,
111       logBody: Boolean,
112       redactHeadersWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
113       color: Response[F] => String = defaultResponseColor _,
114       logAction: Option[String => F[Unit]] = None,
115   )(client: Client[F]): Client[F] =
116     customized(client, logBody, logAction) { response =>
117       val prelude = s"${response.httpVersion} ${response.status}"
118 
119       val headers: String =
120         InternalLogger.defaultLogHeaders[F, Response[F]](response)(logHeaders, redactHeadersWhen)
121 
122       val bodyText: F[String] =
123         InternalLogger.defaultLogBody[F, Response[F]](response)(logBody) match {
124           case Some(textF) => textF.map(text => s"""body="$text"""")
125           case None => Sync[F].pure("")
126         }
127 
128       def spaced(x: String): String = if (x.isEmpty) x else s" $x"
129 
130       bodyText
131         .map(body => s"${color(response)}$prelude${spaced(headers)}${spaced(body)}${Console.RESET}")
132     }
133 
134 }
