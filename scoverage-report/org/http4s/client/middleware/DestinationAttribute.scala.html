24 /** Client middleware that sets the destination attribute of every request to the specified value.
25   */
26 object DestinationAttribute {
27   def apply[F[_]: Async](client: Client[F], destination: String): Client[F] =
28     Client { req =>
29       client.run(req.withAttribute(Destination, destination))
30     }
31 
32   /** The returned function can be used as classifier function when creating the [[Metrics]] middleware, to use the destination
33     * attribute from the request as classifier.
34     *
35     * @return the classifier function
36     */
37   def getDestination[F[_]](): Request[F] => Option[String] = _.attributes.lookup(Destination)
38 
39   val Destination: Key[String] = Key.newKey[SyncIO, String].unsafeRunSync()
40 
41   val EmptyDestination: String = ""
42 }
