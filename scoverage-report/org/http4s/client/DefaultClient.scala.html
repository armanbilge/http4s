30 private[http4s] abstract class DefaultClient[F[_]](implicit F: MonadCancelThrow[F])
31     extends Client[F] {
32   def run(req: Request[F]): Resource[F, Response[F]]
33 
34   /** Submits a request, and provides a callback to process the response.
35     *
36     * @param req The request to submit
37     * @param f   A callback for the response to req.  The underlying HTTP connection
38     *            is disposed when the returned task completes.  Attempts to read the
39     *            response body afterward will result in an error.
40     * @return The result of applying f to the response to req
41     */
42   def fetch[A](req: Request[F])(f: Response[F] => F[A]): F[A] =
43     run(req).use(f)
44 
45   /** Submits a request, and provides a callback to process the response.
46     *
47     * @param req An effect of the request to submit
48     * @param f A callback for the response to req.  The underlying HTTP connection
49     *          is disposed when the returned task completes.  Attempts to read the
50     *          response body afterward will result in an error.
51     * @return The result of applying f to the response to req
52     */
53   def fetch[A](req: F[Request[F]])(f: Response[F] => F[A]): F[A] =
54     req.flatMap(run(_).use(f))
55 
56   /** Returns this client as a [[cats.data.Kleisli]].  All connections created
57     * by this service are disposed on completion of callback task f.
58     *
59     * This method effectively reverses the arguments to `run` followed by `use`, and is
60     * preferred when an HTTP client is composed into a larger Kleisli function,
61     * or when a common response callback is used by many call sites.
62     */
63   def toKleisli[A](f: Response[F] => F[A]): Kleisli[F, Request[F], A] =
64     Kleisli(run(_).use(f))
65 
66   /** Returns this client as an [[HttpApp]].  It is the responsibility of
67     * callers of this service to run the response body to dispose of the
68     * underlying HTTP connection.
69     *
70     * This is intended for use in proxy servers.  `run`, `fetchAs`,
71     * [[toKleisli]] and [[stream]] signatures guarantee disposal of the
72     * HTTP connection.
73     */
74   def toHttpApp: HttpApp[F] =
75     Kleisli { req =>
76       F.map(run(req).allocated) { case (resp, release) =>
77         resp.pipeBodyThrough(_.onFinalizeWeak(release))
78       }
79     }
80 
81   def stream(req: Request[F]): Stream[F, Response[F]] =
82     Stream.resource(run(req))
83 
84   def streaming[A](req: Request[F])(f: Response[F] => Stream[F, A]): Stream[F, A] =
85     stream(req).flatMap(f)
86 
87   def streaming[A](req: F[Request[F]])(f: Response[F] => Stream[F, A]): Stream[F, A] =
88     Stream.eval(req).flatMap(stream).flatMap(f)

90   def expectOr[A](
91       req: Request[F]
92   )(onError: Response[F] => F[Throwable])(implicit d: EntityDecoder[F, A]): F[A] = {
93     val r = if (d.consumes.nonEmpty) {
94       val m = d.consumes.toList
95       req.addHeader(Accept(MediaRangeAndQValue(m.head), m.tail.map(MediaRangeAndQValue(_)): _*))
96     } else req
97 
98     run(r).use {
99       case Successful(resp) =>
100         d.decode(resp, strict = false).leftWiden[Throwable].rethrowT
101       case failedResponse =>
102         onError(failedResponse).flatMap(F.raiseError)
103     }
104   }
105 
106   /** Submits a request and decodes the response on success.  On failure, the
107     * status code is returned.  The underlying HTTP connection is closed at the
108     * completion of the decoding.
109     */
110   def expect[A](req: Request[F])(implicit d: EntityDecoder[F, A]): F[A] =
111     expectOr(req)(defaultOnError(req))

133   def expectOr[A](s: String)(onError: Response[F] => F[Throwable])(implicit

138   /** Submits a GET request to the URI specified by the String and decodes the
139     * response on success.  On failure, the status code is returned.  The
140     * underlying HTTP connection is closed at the completion of the decoding.

142   def expect[A](s: String)(implicit d: EntityDecoder[F, A]): F[A] =

145   def expectOptionOr[A](
146       req: Request[F]
147   )(onError: Response[F] => F[Throwable])(implicit d: EntityDecoder[F, A]): F[Option[A]] = {
148     val r = if (d.consumes.nonEmpty) {
149       val m = d.consumes.toList
150       req.addHeader(Accept(MediaRangeAndQValue(m.head), m.tail.map(MediaRangeAndQValue(_)): _*))
151     } else req
152 
153     run(r).use {
154       case Successful(resp) =>
155         d.decode(resp, strict = false).leftWiden[Throwable].rethrowT.map(_.some)
156       case failedResponse =>
157         failedResponse.status match {
158           case Status.NotFound => Option.empty[A].pure[F]
159           case Status.Gone => Option.empty[A].pure[F]
160           case _ => onError(failedResponse).flatMap(F.raiseError)
161         }
162     }
163   }

218   /** Submits a GET request, and provides a callback to process the response.
219     *
220     * @param uri The URI to GET
221     * @param f A callback for the response to a GET on uri.  The underlying HTTP connection
222     *          is disposed when the returned task completes.  Attempts to read the
223     *          response body afterward will result in an error.
224     * @return The result of applying f to the response to req
225     */

227     run(Request[F](Method.GET, uri)).use(f)

229   /** Submits a request and decodes the response on success.  On failure, the

231     * completion of the decoding.

234     Uri.fromString(s).fold(F.raiseError, uri => get(uri)(f))
