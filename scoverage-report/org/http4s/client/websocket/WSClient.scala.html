34 /** A websocket request.
35   *
36   * @param uri
37   *   The URI.
38   * @param headers
39   *   The headers to send. Put your `Sec-Websocket-Protocol` headers here if needed. Some websocket
40   *   clients reject other WS-specific headers.
41   * @param method
42   *   The method of the intial HTTP request. Ignored by some clients.
43   */
44 sealed abstract class WSRequest {
45   def uri: Uri
46   def headers: Headers
47   def method: Method
48 
49   def withUri(uri: Uri): WSRequest
50   def withHeaders(headers: Headers): WSRequest
51   def withMethod(method: Method): WSRequest
52 }
53 
54 object WSRequest {
55   def apply(uri: Uri): WSRequest = apply(uri, Headers.empty, Method.GET)
56 
57   def apply(uri: Uri, headers: Headers, method: Method): WSRequest =
58     WSRequestImpl(uri, headers, method)
59 
60   private[this] final case class WSRequestImpl(
61       override val uri: Uri,
62       override val headers: Headers,
63       override val method: Method,
64   ) extends WSRequest {
65     def withUri(uri: Uri): WSRequestImpl = copy(uri = uri)
66     def withHeaders(headers: Headers): WSRequestImpl = copy(headers = headers)
67     def withMethod(method: Method): WSRequestImpl = copy(method = method)
68   }
69 
70   implicit val catsHashAndOrderForWSRequest: Hash[WSRequest] with Order[WSRequest] =
71     new Hash[WSRequest] with Order[WSRequest] {
72       override def hash(x: WSRequest): Int = x.hashCode
73 
74       override def compare(x: WSRequest, y: WSRequest): Int =
75         reduceComparisons(
76           x.headers.compare(y.headers),
77           Eval.later(x.method.compare(y.method)),
78           Eval.later(x.uri.compare(y.uri)),
79         )
80     }
81 
82   implicit val catsShowForWSRequest: Show[WSRequest] =
83     Show.fromToString
84 
85   implicit def stdLibOrdering: Ordering[WSRequest] =
86     catsHashAndOrderForWSRequest.toOrdering
87 }

103 trait WSConnection[F[_]] { outer =>
104 
105   /** Send a single websocket frame. The sending side of this connection has to be open. */
106   def send(wsf: WSFrame): F[Unit]
107 
108   /** Send multiple websocket frames. Equivalent to multiple `send` calls, but at least as fast. */
109   def sendMany[G[_]: Foldable, A <: WSFrame](wsfs: G[A]): F[Unit]
110 
111   /** A `Pipe` which sends websocket frames and emits a `()` for each chunk sent. */
112   def sendPipe: Pipe[F, WSFrame, Unit] = _.chunks.evalMap(sendMany(_))
113 
114   /** Wait for a single websocket frame to be received. Returns `None` if the receiving side is
115     * closed.
116     */
117   def receive: F[Option[WSFrame]]
118 
119   /** A stream of the incoming websocket frames. */
120   def receiveStream: Stream[F, WSFrame] = Stream.repeatEval(receive).unNoneTerminate
121 
122   /** The negotiated subprotocol, if any. */
123   def subprotocol: Option[String]
124 
125   def mapK[G[_]](fk: F ~> G): WSConnection[G] = new WSConnection[G] {
126     def send(wsf: WSFrame): G[Unit] = fk(outer.send(wsf))
127     def sendMany[H[_]: Foldable, A <: WSFrame](wsfs: H[A]): G[Unit] = fk(outer.sendMany(wsfs))
128     def receive: G[Option[WSFrame]] = fk(outer.receive)
129     def subprotocol: Option[String] = outer.subprotocol
130   }
131 }

144   /** Send multiple websocket frames. Equivalent to multiple `send` calls, but at least as fast. */
145   def sendMany[G[_]: Foldable, A <: WSDataFrame](wsfs: G[A]): F[Unit]
146 
147   /** A `Pipe` which sends websocket frames and emits a `()` for each chunk sent. */
148   def sendPipe: Pipe[F, WSDataFrame, Unit] = _.chunks.evalMap(sendMany(_))
149 
150   /** Wait for a websocket frame to be received. Returns `None` if the receiving side is closed.
151     * Fragmentation is handled automatically, the `last` attribute can be ignored.
152     */
153   def receive: F[Option[WSDataFrame]]
154 
155   /** A stream of the incoming websocket frames. */
156   def receiveStream: Stream[F, WSDataFrame] = Stream.repeatEval(receive).unNoneTerminate
157 
158   /** The negotiated subprotocol, if any. */
159   def subprotocol: Option[String]
160 
161   /** The close frame, if available. */
162   def closeFrame: DeferredSource[F, WSFrame.Close]
163 
164   def mapK[G[_]](fk: F ~> G): WSConnectionHighLevel[G] =
165     new WSConnectionHighLevel[G] {
166       def send(wsf: WSDataFrame): G[Unit] = fk(outer.send(wsf))
167       def sendMany[H[_]: Foldable, A <: WSDataFrame](wsfs: H[A]): G[Unit] = fk(outer.sendMany(wsfs))
168       def receive: G[Option[WSDataFrame]] = fk(outer.receive)
169       def subprotocol: Option[String] = outer.subprotocol
170       def closeFrame: DeferredSource[G, WSFrame.Close] = new DeferredSource[G, WSFrame.Close] {
171         def get = fk(outer.closeFrame.get)
172         def tryGet = fk(outer.closeFrame.tryGet)
173       }
174     }

177 /** A websocket client capable of establishing [[WSClientHighLevel#connectHighLevel "high level" connections]].
178   * @see [[WSClient]] for a client also capable of "low-level" connections
179   */
180 trait WSClientHighLevel[F[_]] { outer =>
181 
182   /** Establish a "high level" websocket connection. You only get to handle Text and Binary frames.
183     * Pongs will be replied automatically. Received frames are grouped by the `last` attribute. The
184     * connection will be closed automatically.
185     */
186   def connectHighLevel(request: WSRequest): Resource[F, WSConnectionHighLevel[F]]
187 
188   def mapK[G[_]](
189       fk: F ~> G
190   )(implicit F: MonadCancel[F, _], G: MonadCancel[G, _]): WSClientHighLevel[G] =
191     new WSClientHighLevel[G] {
192       def connectHighLevel(request: WSRequest): Resource[G, WSConnectionHighLevel[G]] =
193         outer.connectHighLevel(request).map(_.mapK(fk)).mapK(fk)
194     }
195 }

214 object WSClient {
215   def apply[F[_]](
216       respondToPings: Boolean
217   )(f: WSRequest => Resource[F, WSConnection[F]])(implicit F: Concurrent[F]): WSClient[F] =
218     new WSClient[F] {
219       override def connect(request: WSRequest) = f(request)
220       override def connectHighLevel(request: WSRequest) =
221         for {
222           recvCloseFrame <- Resource.eval(Deferred[F, WSFrame.Close])
223           conn <- f(request)
224         } yield new WSConnectionHighLevel[F] {
225           override def send(wsf: WSDataFrame) = conn.send(wsf)
226           override def sendMany[G[_]: Foldable, A <: WSDataFrame](wsfs: G[A]): F[Unit] =
227             conn.sendMany(wsfs)
228           override def receive: F[Option[WSDataFrame]] = {
229             def receiveDataFrame: OptionT[F, WSDataFrame] =
230               OptionT(conn.receive).flatMap { wsf =>
231                 OptionT.liftF(wsf match {
232                   case WSFrame.Ping(data) if respondToPings => conn.send(WSFrame.Pong(data))
233                   case wsf: WSFrame.Close =>
234                     recvCloseFrame.complete(wsf) *> conn.send(wsf)
235                   case _ => F.unit
236                 }) >> (wsf match {
237                   case wsdf: WSDataFrame => OptionT.pure[F](wsdf)
238                   case _ => receiveDataFrame
239                 })
240               }
241             def defrag(text: Chain[String], binary: ByteVector): OptionT[F, WSDataFrame] =
242               receiveDataFrame.flatMap {
243                 case WSFrame.Text(t, finalFrame) =>
244                   val nextText = text :+ t
245                   if (finalFrame) {
246                     val sb = new StringBuilder(nextText.foldMap(_.length))
247                     nextText.iterator.foreach(sb ++= _)
248                     OptionT.pure[F](WSFrame.Text(sb.mkString))
249                   } else
250                     defrag(nextText, binary)
251                 case WSFrame.Binary(b, finalFrame) =>
252                   val nextBinary = binary ++ b
253                   if (finalFrame)
254                     OptionT.pure[F](WSFrame.Binary(nextBinary))
255                   else
256                     defrag(text, nextBinary)
257               }
258             defrag(Chain.empty, ByteVector.empty).value
259           }
260           override def subprotocol: Option[String] = conn.subprotocol
261           override def closeFrame: DeferredSource[F, WSFrame.Close] = recvCloseFrame
262         }
263     }
