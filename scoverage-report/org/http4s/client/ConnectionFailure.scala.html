19 import java.io.IOException
20 import java.net.InetSocketAddress

22 /** Indicates a failure to establish a client connection, preserving the request key
23   * that we tried to connect to.
24   */
25 class ConnectionFailure(
26     val requestKey: RequestKey,
27     val upstream: InetSocketAddress,
28     val cause: Throwable,
29 ) extends IOException(cause) {
30   override def getMessage(): String =
31     s"Error connecting to $requestKey using address ${upstream.getHostString}:${upstream.getPort} (unresolved: ${upstream.isUnresolved})"
32 }
33 
34 object ConnectionFailure {
35   def unapply(failure: ConnectionFailure): Option[(RequestKey, InetSocketAddress, Throwable)] =
36     Some((failure.requestKey, failure.upstream, failure.cause))
37 }
