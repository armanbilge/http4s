45   }

47   private[this] val AllMethods = List(HmacSha1, HmacSha256, HmacSha512)
48 
49   /** Map a [[SignatureMethod]] protocol parameter to a [[SignatureAlgorithm]] implementation
50     *
51     * @param method The signature method protocol parameter
52     * @return The implementation, or an [[IllegalArgumentException]] if none was found
53     */
54   private[oauth1] def unsafeFromMethod(method: SignatureMethod): SignatureAlgorithm =
55     AllMethods
56       .find(_.name == method.headerValue)
57       .getOrElse(
58         throw new IllegalArgumentException(
59           s"Unrecognized headerValue '${method.headerValue}', Valid options are: ${AllMethods.map(_.name)}"
60         )
61       )
62 
63 }

65 /** Implementations for Oauth1 signatures.
66   */
67 trait SignatureAlgorithm {
68 
69   /** @return The signature method name per the oauth1.0 spec
70     */
71   def name: String
72 
73   /** Apply the implementation's algorithm to the input
74     *
75     * @param input The input value
76     * @param secretKey The secret key
77     * @return The base64-encoded output
78     */
79   def generate[F[_]: MonadThrow](input: String, secretKey: String): F[String] =
80     MonadThrow[F].catchNonFatal(generate(input, secretKey): @nowarn("cat=deprecation"))
81 
82   @deprecated("Use generate[F[_]: MonadThrow] instead", "0.22.5")
83   def generate(input: String, secretKey: String): String =
84     generate[SyncIO](input, secretKey).unsafeRunSync()
85 
86   private[oauth1] def generateHMAC[F[_]: MonadThrow: Hmac](
87       input: String,
88       algorithm: HmacAlgorithm,
89       secretKey: String,
90   ): F[String] =
91     for {
92       keyData <- ByteVector.encodeUtf8(secretKey).liftTo[F]
93       sk = SecretKeySpec(keyData, algorithm)
94       data <- ByteVector.encodeUtf8(input).liftTo[F]
95       digest <- Hmac[F].digest(sk, data)
96     } yield digest.toBase64
97 
98 }

100 /** An implementation of the `HMAC-SHA1` oauth signature method.
101   *
102   * This uses the `HmacSHA1` implementation which every java platform is required to have.
103   */
104 object HmacSha1 extends SignatureAlgorithm {
105   override val name: String = `HMAC-SHA1`
106   override def generate[F[_]: MonadThrow](input: String, secretKey: String): F[String] =
107     generateHMAC(input, HmacAlgorithm.SHA1, secretKey)
108 }

120 /** An implementation of the `HMAC-SHA512` oauth signature method.
