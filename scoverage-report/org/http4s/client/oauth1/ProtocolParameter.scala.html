28 sealed trait ProtocolParameter {
29   val headerName: String
30   val headerValue: String
31   def toTuple: (String, String) =
32     headerName -> encode(headerValue)
33 }

60   object Timestamp {
61     def now[F[_]](implicit F: Clock[F]): F[Timestamp] = {
62       implicit val FA: Applicative[F] = F.applicative
63       F.realTime.map(time => Timestamp(time.toUnit(TimeUnit.SECONDS).toString()))
64     }
65   }

75     }
76   }
