49   /** Sign the request with an OAuth Authorization header
50     *
51     * __WARNING:__ POST requests with application/x-www-form-urlencoded bodies
52     *            will be entirely buffered due to signing requirements.
53     */
54   @deprecated(
55     "Preserved for binary compatibility - use the other `signRequest` function which passes a signature method",
56     "0.22.3",
57   )
58   def signRequest[F[_]](
59       req: Request[F],
60       consumer: Consumer,
61       callback: Option[Uri],
62       verifier: Option[String],
63       token: Option[Token],
64   )(implicit F: MonadThrow[F], W: EntityDecoder[F, UrlForm]): F[Request[F]] =
65     getUserParams(req).flatMap { case (req, params) =>
66       genAuthHeader(req.method, req.uri, params, consumer, callback, verifier, token, HmacSha1)
67         .map(auth => req.putHeaders(auth))
68     }

70   def signRequest[F[_]](
71       req: Request[F],
72       consumer: ProtocolParameter.Consumer,
73       token: Option[ProtocolParameter.Token],
74       realm: Option[Realm],
75       signatureMethod: SignatureMethod = ProtocolParameter.SignatureMethod(),
76       timestampGenerator: F[Timestamp],
77       version: ProtocolParameter.Version = Version(),
78       nonceGenerator: F[Nonce],
79       callback: Option[Callback] = None,
80       verifier: Option[Verifier] = None,
81   )(implicit F: MonadThrow[F], W: EntityDecoder[F, UrlForm]): F[Request[F]] =
82     for {
83       reqParams <- getUserParams(req)
84       // Working around lack of withFilter
85       (req, params) = reqParams
86       auth <- genAuthHeader(
87         req.method,
88         req.uri,
89         consumer,
90         token,
91         realm,
92         signatureMethod,
93         timestampGenerator,
94         version,
95         nonceGenerator,
96         callback,
97         verifier,
98         params.map { case (k, v) =>
99           Custom(k, v)
100         },
101       )
102     } yield req.putHeaders(auth)

104   def takeSigHeaders[F[_]: Monad](
105       consumer: ProtocolParameter.Consumer,
106       token: Option[ProtocolParameter.Token],
107       signatureMethod: SignatureMethod,
108       timestampGenerator: F[Timestamp],
109       version: ProtocolParameter.Version,
110       nonceGenerator: F[Nonce],
111       callback: Option[Callback],
112       verifier: Option[Verifier],
113   ): F[immutable.Seq[ProtocolParameter]] =
114     for {
115       timestamp <- timestampGenerator
116       nonce <- nonceGenerator
117     } yield {
118       val headers =
119         List(consumer, signatureMethod, timestamp, nonce, version)
120       headers ++ List(token, callback, verifier).flatten
121     }

123   private[oauth1] def genAuthHeader[F[_]: MonadThrow](
124       method: Method,
125       uri: Uri,
126       consumer: ProtocolParameter.Consumer,
127       token: Option[ProtocolParameter.Token],
128       realm: Option[Realm],
129       signatureMethod: SignatureMethod,
130       timestampGenerator: F[Timestamp],
131       version: ProtocolParameter.Version,
132       nonceGenerator: F[Nonce],
133       callback: Option[Callback],
134       verifier: Option[Verifier],
135       queryParams: Seq[ProtocolParameter],
136   ): F[Authorization] =
137     takeSigHeaders(
138       consumer,
139       token,
140       signatureMethod,
141       timestampGenerator,
142       version,
143       nonceGenerator,
144       callback,
145       verifier,
146     ).flatMap { headers =>
147       val baseStr = mkBaseString(
148         method,
149         uri,
150         (headers ++ queryParams).sorted.map(Show[ProtocolParameter].show).mkString("&"),
151       )
152       val alg = SignatureAlgorithm.unsafeFromMethod(signatureMethod)
153       makeSHASig(baseStr, consumer.secret, token.map(_.secret), alg).map { sig =>
154         val creds = Credentials.AuthParams(
155           ci"OAuth",
156           NonEmptyList(
157             "oauth_signature" -> encode(sig),
158             realm.fold(headers.map(_.toTuple))(_.toTuple +: headers.map(_.toTuple)).toList,
159           ),
160         )
161 
162         Authorization(creds)
163       }
164     }

166   // Generate an authorization header with the provided user params and OAuth requirements.
167   // Warning: Fixed to HMAC-SHA1
168   @deprecated("Preserved for binary compatibility", "0.22.3")
169   private[oauth1] def genAuthHeader(
170       method: Method,
171       uri: Uri,
172       userParams: immutable.Seq[(String, String)],
173       consumer: Consumer,
174       callback: Option[Uri],
175       verifier: Option[String],
176       token: Option[Token],
177   ): Authorization =
178     genAuthHeader[SyncIO](method, uri, userParams, consumer, callback, verifier, token, HmacSha1)
179       .unsafeRunSync()
180 
181   private[oauth1] def genAuthHeader[F[_]: MonadThrow](
182       method: Method,
183       uri: Uri,
184       userParams: immutable.Seq[(String, String)],
185       consumer: Consumer,
186       callback: Option[Uri],
187       verifier: Option[String],
188       token: Option[Token],
189       algorithm: SignatureAlgorithm,
190   ): F[Authorization] = {
191     val params = {
192       val params = new ListBuffer[(String, String)]
193       params += "oauth_consumer_key" -> encode(consumer.key)
194       params += "oauth_signature_method" -> algorithm.name
195       params += "oauth_timestamp" -> (System.currentTimeMillis / 1000).toString
196       params += "oauth_nonce" -> System.nanoTime.toString
197       params += "oauth_version" -> "1.0"
198       params += "oauth_callback" -> callback.map(c => encode(c.renderString)).getOrElse(OutOfBand)
199       token.foreach { t =>
200         params += "oauth_token" -> encode(t.value)
201       }
202       verifier.foreach { v =>
203         params += "oauth_verifier" -> encode(v)
204       }
205       params.result()
206     }
207 
208     val baseString = genBaseString(
209       method,
210       uri,
211       params ++ userParams.map { case (k, v) =>
212         (encode(k), encode(v))
213       },
214     )
215     makeSHASig(baseString, consumer.secret, token.map(_.secret), algorithm).map { sig =>
216       val creds =
217         Credentials.AuthParams(ci"OAuth", NonEmptyList("oauth_signature" -> encode(sig), params))
218 
219       Authorization(creds)
220     }
221   }

235   private[oauth1] def makeSHASig(
236       baseString: String,
237       consumerSecret: String,
238       tokenSecret: Option[String],
239   ): String =
240     makeSHASig[SyncIO](baseString, consumerSecret, tokenSecret, HmacSha1).unsafeRunSync()

242   private[oauth1] def makeSHASig[F[_]: MonadThrow](
243       baseString: String,
244       consumerSecret: String,
245       tokenSecret: Option[String],
246       algorithm: SignatureAlgorithm,
247   ): F[String] = {
248 
249     val key = encode(consumerSecret) + "&" + tokenSecret.map(t => encode(t)).getOrElse("")
250     algorithm.generate(baseString, key)
251   }
252 
253   // Needs to have all params already encoded
254   private[oauth1] def genBaseString(
255       method: Method,
256       uri: Uri,
257       params: immutable.Seq[(String, String)],
258   ): String = {
259     val paramsStr = params.map { case (k, v) => k + "=" + v }.sorted.mkString("&")
260     mkBaseString(method, uri, paramsStr)
261   }
262 
263   def mkBaseString(method: Method, uri: Uri, paramsStr: String): String =
264     immutable
265       .Seq(
266         method.name,
267         encode(uri.copy(query = Query.empty, fragment = None).renderString),
268         encode(paramsStr),
269       )
270       .mkString("&")
271 
272   private[oauth1] def encode(str: String): String =
273     Uri.encode(str, spaceIsPlus = false, toSkip = Uri.Unreserved)

275   private[oauth1] def getUserParams[F[_]](req: Request[F])(implicit
276       F: MonadThrow[F],
277       W: EntityDecoder[F, UrlForm],
278   ): F[(Request[F], immutable.Seq[(String, String)])] = {
279     val qparams = req.uri.query.pairs.map { case (k, ov) => (k, ov.getOrElse("")) }
280 
281     req.contentType match {
282       case Some(t)
283           if (req.method == Method.POST || req.method == Method.PUT) &&
284             t.mediaType == MediaType.application.`x-www-form-urlencoded` =>
285         req.as[UrlForm].map { urlform =>
286           val bodyparams = urlform.values.toSeq
287             .flatMap { case (k, vs) => if (vs.isEmpty) Seq(k -> "") else vs.toList.map((k, _)) }
288 
289           implicit val charset: Charset = req.charset.getOrElse(Charset.`UTF-8`)
290           req.withEntity(urlform) -> (qparams ++ bodyparams)
291         }
292 
293       case _ => F.pure(req -> qparams)
294     }
295   }
