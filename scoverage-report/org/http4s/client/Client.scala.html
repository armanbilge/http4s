33 /** A [[Client]] submits [[Request]]s to a server and processes the [[Response]].
34   *
35   * When a connection is "released" and the HTTP semantics of the
36   * request and response permit, the connection may be kept alive by
37   * the backend and used for a subsequent request.  When HTTP
38   * semantics require it, or at the backend's discretion, a released
39   * connection may also be closed.
40   */

42   def run(req: Request[F]): Resource[F, Response[F]]
43 
44   /** Submits a request, and provides a callback to process the response.
45     *
46     * @param req The request to submit
47     * @param f   A callback for the response to req.  The underlying HTTP connection
48     *            is released when the returned task completes.  Attempts to read the
49     *            response body afterward will result in an error.
50     * @return The result of applying f to the response to req
51     */

210   private[client] def translateImpl[G[_]](
211       fk: F ~> G
212   )(gK: G ~> F)(implicit F: MonadCancelThrow[F]): Client[G] = {
213     implicit val G: MonadCancelThrow[G] = liftMonadCancel(F)(fk)(gK)
214     Client[G]((req: Request[G]) =>
215       run(
216         req.mapK(gK)
217       ).mapK(fk)
218         .map(_.mapK(fk))
219     )

222   private[this] def liftMonadCancel[G[_]](
223       F: MonadCancelThrow[F]
224   )(fk: F ~> G)(gk: G ~> F): MonadCancelThrow[G] =
225     new MonadCancelThrow[G] {
226       def pure[A](x: A): G[A] = fk(F.pure(x))
227 
228       // Members declared in cats.ApplicativeError
229       def handleErrorWith[A](ga: G[A])(f: Throwable => G[A]): G[A] =
230         fk(F.handleErrorWith(gk(ga))(ex => gk(f(ex))))
231 
232       def raiseError[A](e: Throwable): G[A] = fk(F.raiseError[A](e))
233 
234       // Members declared in cats.FlatMap
235       def flatMap[A, B](ga: G[A])(f: A => G[B]): G[B] =
236         fk(F.flatMap(gk(ga))(a => gk(f(a))))
237 
238       def tailRecM[A, B](a: A)(f: A => G[Either[A, B]]): G[B] =
239         fk(F.tailRecM(a)(a => gk(f(a))))

241       // Members declared in cats.effect.kernel.MonadCancel
242       def canceled: G[Unit] = fk(F.canceled)
243 
244       def forceR[A, B](ga: G[A])(gb: G[B]): G[B] =
245         fk(F.forceR(gk(ga))(gk(gb)))
246 
247       def onCancel[A](ga: G[A], fin: G[Unit]): G[A] =
248         fk(F.onCancel(gk(ga), gk(fin)))

250       def rootCancelScope: CancelScope = F.rootCancelScope

252       def uncancelable[A](body: Poll[G] => G[A]): G[A] =

269   /** Creates a client from the specified [[HttpApp]].  Useful for
270     * generating pre-determined responses for requests in testing.
271     *
272     * @param app the [[HttpApp]] to respond to requests to this client
273     */
274   def fromHttpApp[F[_]](app: HttpApp[F])(implicit F: Async[F]): Client[F] = {
275     def until[A](disposed: Ref[F, Boolean])(source: Stream[F, A]): Stream[F, A] = {
276       def go(stream: Stream[F, A]): Pull[F, A, Unit] =
277         stream.pull.uncons.flatMap {
278           case Some((chunk, stream)) =>
279             Pull.eval(disposed.get).flatMap {
280               case true => Pull.raiseError[F](new IOException("response was disposed"))
281               case false => Pull.output(chunk) >> go(stream)
282             }
283           case None => Pull.done
284         }
285       go(source).stream
286     }
287 
288     def runOrDispose(req: Request[F]): F[Resource[F, Response[F]]] =
289       Ref[F].of(false).map { disposed =>
290         val req0 = addHostHeaderIfUriIsAbsolute(req.pipeBodyThrough(until(disposed)))
291         Resource
292           .eval(app(req0))
293           .onFinalize(disposed.set(true))
294           .map(_.pipeBodyThrough(until(disposed)))
295       }
296 
297     Client((req: Request[F]) => Resource.suspend(runOrDispose(req)))
298   }
299 
300   /** This method introduces an important way for the effectful backends to allow tracing. As Kleisli types
301     * form the backend of tracing and these transformations are non-trivial.
302     */
303   def liftKleisli[F[_]: MonadCancelThrow, A](client: Client[F]): Client[Kleisli[F, A, *]] =
304     Client { (req: Request[Kleisli[F, A, *]]) =>
305       Resource.eval(Kleisli.ask[F, A]).flatMap { a =>
306         client
307           .run(req.mapK(Kleisli.applyK(a)))
308           .mapK(Kleisli.liftK[F, A])
309           .map(_.mapK(Kleisli.liftK))
310       }
311     }

313   private def addHostHeaderIfUriIsAbsolute[F[_]](req: Request[F]): Request[F] =
314     req.uri.host match {
315       case Some(host) if !req.headers.contains[Host] =>
316         req.withHeaders(req.headers.put(Host(host.value, req.uri.port)))
317       case _ => req
318     }
