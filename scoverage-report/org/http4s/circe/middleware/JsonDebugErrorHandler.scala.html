28 object JsonDebugErrorHandler {
29   private[this] val messageFailureLogger =
30     org.log4s.getLogger("org.http4s.circe.middleware.jsondebugerrorhandler.message-failures")
31   private[this] val serviceErrorLogger =
32     org.log4s.getLogger("org.http4s.circe.middleware.jsondebugerrorhandler.service-errors")
33 
34   // Can be parametric on my other PR is merged.
35   def apply[F[_]: Concurrent, G[_]](
36       service: Kleisli[F, Request[G], Response[G]],
37       redactWhen: CIString => Boolean = Headers.SensitiveHeaders.contains,
38   ): Kleisli[F, Request[G], Response[G]] =
39     Kleisli { req =>
40       import cats.syntax.applicative._
41       import cats.syntax.applicativeError._
42       implicit def entEnc[M[_], N[_]]: EntityEncoder[M, JsonErrorHandlerResponse[N]] =
43         JsonErrorHandlerResponse.entEnc[M, N](redactWhen)
44 
45       service
46         .run(req)
47         .handleErrorWith {
48           case mf: MessageFailure =>
49             messageFailureLogger.debug(mf)(
50               s"""Message failure handling request: ${req.method} ${req.pathInfo} from ${req.remoteAddr
51                   .getOrElse("<unknown>")}"""
52             )
53             val firstResp = mf.toHttpResponse[G](req.httpVersion)
54             Response[G](
55               status = firstResp.status,
56               httpVersion = firstResp.httpVersion,
57               headers = firstResp.headers.redactSensitive(redactWhen),
58             ).withEntity(JsonErrorHandlerResponse[G](req, mf)).pure[F]
59           case t =>
60             serviceErrorLogger.error(t)(
61               s"""Error servicing request: ${req.method} ${req.pathInfo} from ${req.remoteAddr
62                   .getOrElse("<unknown>")}"""
63             )
64             Response[G](
65               Status.InternalServerError,
66               req.httpVersion,
67               Headers(Connection.close),
68             )
69               .withEntity(JsonErrorHandlerResponse[G](req, t))
70               .pure[F]
71         }
72     }
73 
74   private final case class JsonErrorHandlerResponse[F[_]](
75       req: Request[F],
76       caught: Throwable,
77   )
78   private object JsonErrorHandlerResponse {
79     def entEnc[F[_], G[_]](
80         redactWhen: CIString => Boolean
81     ): EntityEncoder[F, JsonErrorHandlerResponse[G]] =
82       jsonEncoderOf(
83         encoder(redactWhen)
84       )
85     def encoder[F[_]](
86         redactWhen: CIString => Boolean
87     ): Encoder[JsonErrorHandlerResponse[F]] =
88       (a: JsonErrorHandlerResponse[F]) =>
89         Json.obj(
90           "request" -> encodeRequest(a.req, redactWhen),
91           "throwable" -> encodeThrowable(a.caught),
92         )
93   }
94 
95   private def encodeRequest[F[_]](req: Request[F], redactWhen: CIString => Boolean): Json =
96     Json
97       .obj(
98         "method" -> req.method.name.asJson,
99         "uri" -> Json
100           .obj(
101             "scheme" -> req.uri.scheme.map(_.value).asJson,
102             "authority" -> req.uri.authority
103               .map(auth =>
104                 Json
105                   .obj(
106                     "host" -> auth.host.toString().asJson,
107                     "port" -> auth.port.asJson,
108                     "user_info" -> auth.userInfo
109                       .map(_.toString())
110                       .asJson,
111                   )
112                   .dropNullValues
113               )
114               .asJson,
115             "path" -> req.uri.path.renderString.asJson,
116             "query" -> req.uri.query.multiParams.asJson,
117           )
118           .dropNullValues,
119         "headers" -> req.headers
120           .redactSensitive(redactWhen)
121           .headers
122           .map { h =>
123             Json.obj(
124               "name" -> h.name.toString.asJson,
125               "value" -> h.value.asJson,
126             )
127           }
128           .asJson,
129         "path_info" -> req.pathInfo.renderString.asJson,
130         "remote_address" -> req.remoteAddr.toString.asJson,
131         "http_version" -> req.httpVersion.toString.asJson,
132       )
133       .dropNullValues
134 
135   private def encodeThrowable(a: Throwable): Json =
136     Json
137       .obj(
138         "message" -> Option(a.getMessage).asJson,
139         "stack_trace" -> Option(a.getStackTrace())
140           .map(_.toList)
141           .map(_.map(stackTraceElem => stackTraceElem.toString))
142           .asJson,
143         "localized_message" ->
144           Option(a.getLocalizedMessage()).asJson,
145         "cause" -> Option(a.getCause())
146           .map(encodeThrowable(_))
147           .asJson,
148         "suppressed" -> Option(a.getSuppressed())
149           .map(_.toList.map(encodeThrowable(_)))
150           .asJson,
151         "class_name" -> Option(a.getClass())
152           .flatMap(c =>
153             if (Platform.isJvm)
154               Option(c.getCanonicalName())
155             else
156               Option(c.getName())
157           )
158           .asJson,
159       )
160       .dropNullValues
161 }
