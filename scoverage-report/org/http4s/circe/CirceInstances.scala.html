55   private def jsonDecoderByteBufferImpl[F[_]: Concurrent](m: Media[F]): DecodeResult[F, Json] =
56     EntityDecoder.collectBinary(m).subflatMap { chunk =>
57       val bb = ByteBuffer.wrap(chunk.toArray)
58       if (bb.hasRemaining)
59         circeSupportParser
60           .parseFromByteBuffer(bb)
61           .toEither
62           .leftMap(e => circeParseExceptionMessage(ParsingFailure(e.getMessage(), e)))
63       else
64         Left(jawnEmptyBodyMessage)
65     }

274   private def jsonDecodeErrorHelper(
275       json: Json,
276       jsonToString: Json => String,
277       failures: NonEmptyList[DecodingFailure],
278   ): DecodeFailure = {
279 
280     val str: String = jsonToString(json)
281 
282     InvalidMessageBodyFailure(
283       s"Could not decode JSON: $str",
284       if (failures.tail.isEmpty) Some(failures.head) else Some(DecodingFailures(failures)),
285     )
286   }

290   private def fromJsonToChunk(printer: Printer)(json: Json): Chunk[Byte] =
291     Chunk.ByteBuffer.view(printer.printToByteBuffer(json))
292 
293   private def streamedJsonArray[F[_]](printer: Printer)(s: Stream[F, Json]): Stream[F, Byte] =
294     s.pull.uncons1.flatMap {
295       case None => Pull.output(emptyArray)
296       case Some((hd, tl)) =>
297         Pull.output(
298           Chunk.concat(Vector(CirceInstances.openBrace, fromJsonToChunk(printer)(hd)))
299         ) >> // Output First Json As Chunk with leading `[`
300           tl.repeatPull {
301             _.uncons.flatMap {
302               case None => Pull.pure(None)
303               case Some((hd, tl)) =>
304                 val interspersed = {
305                   val bldr = Vector.newBuilder[Chunk[Byte]]
306                   bldr.sizeHint(hd.size * 2)
307                   hd.foreach { o =>
308                     bldr += CirceInstances.comma
309                     bldr += fromJsonToChunk(printer)(o)
310                   }
311                   Chunk.concat(bldr.result())
312                 }
313                 Pull.output(interspersed) >> Pull.pure(Some(tl))
314             }
315           }.pull
316             .echo >>
317           Pull.output(closeBrace)
318     }.stream
319 
320   private final val openBrace: Chunk[Byte] =
321     Chunk.singleton('['.toByte)
322 
323   private final val closeBrace: Chunk[Byte] =
324     Chunk.singleton(']'.toByte)
325 
326   private final val emptyArray: Chunk[Byte] =
327     Chunk.array(Array('['.toByte, ']'.toByte))
328 
329   private final val comma: Chunk[Byte] =
330     Chunk.singleton(','.toByte)
