23 /** Derive [[EntityDecoder]] if implicit [[io.circe.Decoder]] is in the scope without need to explicitly call `jsonOfSensitive`
24   *
25   * Note that it varies from [[CirceEntityDecoder]] in that, when failing to decode [[io.circe.Json]] to an `A`,
26   * the JSON will not be included in the raised [[java.lang.Exception]]. In the event the JSON includes sensitive data, this trait is,
27   * arguably, a better choice since it eliminates the risk of logging sensitive data, e.g. due to logging a raised [[java.lang.Throwable]]
28   * that includes the sensitive JSON.
29   */
30 trait CirceSensitiveDataEntityDecoder {
31   implicit def circeEntityDecoder[F[_]: Concurrent, A: Decoder]: EntityDecoder[F, A] =
32     jsonOfSensitive[F, A](_ => "<REDACTED>")
33 }
