23 /** F-algebra for separating the Sync required for extracting
24   * the Json from the body. As such if F is Sync at some layer,
25   * then this can be used to extract without the lower layer
26   * needing to be aware of the strong constraint.
27   */
28 trait JsonDecoder[F[_]] {
29   def asJson(m: Message[F]): F[Json]
30   def asJsonDecode[A: Decoder](m: Message[F]): F[A]
31 }
32 
33 object JsonDecoder {
34   def apply[F[_]](implicit ev: JsonDecoder[F]): JsonDecoder[F] = ev
35 
36   implicit def impl[F[_]: Concurrent]: JsonDecoder[F] =
37     new JsonDecoder[F] {
38       def asJson(m: Message[F]): F[Json] = m.as[Json]
39       def asJsonDecode[A: Decoder](m: Message[F]): F[A] = m.decodeJson
40     }
41 }
