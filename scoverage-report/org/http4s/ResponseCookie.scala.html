41 /** @param extension The extension attributes of the cookie.  If there is more
42   * than one, they are joined by semi-colon, which must not appear in an
43   * attribute value.
44   */
45 final case class ResponseCookie(
46     name: String,
47     content: String,
48     expires: Option[HttpDate] = None,
49     maxAge: Option[Long] = None,
50     domain: Option[String] = None,
51     path: Option[String] = None,
52     sameSite: Option[SameSite] = None,
53     secure: Boolean = false,
54     httpOnly: Boolean = false,
55     extension: Option[String] = None,
56 ) extends Renderable { self =>
57   override lazy val renderString: String = super.renderString
58 
59   override def render(writer: Writer): writer.type = {
60     writer.append(name).append('=').append(content)
61     expires.foreach { e =>
62       writer.append("; Expires=").append(e)
63     }
64     maxAge.foreach(writer.append("; Max-Age=").append(_))
65     domain.foreach(writer.append("; Domain=").append(_))
66     path.foreach(writer.append("; Path=").append(_))
67     sameSite.foreach(writer.append("; SameSite=").append(_))
68     if (secure || sameSite.contains(SameSite.None)) writer.append("; Secure")
69     if (httpOnly) writer.append("; HttpOnly")
70     extension.foreach(writer.append("; ").append(_))
71     writer
72   }
73 
74   def clearCookie: ResponseCookie =
75     copy(content = "", expires = Some(HttpDate.Epoch))
76 
77   private def withExpires(expires: HttpDate): ResponseCookie =
78     copy(expires = Some(expires))
79 
80   private def withMaxAge(maxAge: Long): ResponseCookie =
81     copy(maxAge = Some(maxAge))
82 
83   private def withDomain(domain: String): ResponseCookie =
84     copy(domain = Some(domain))
85 
86   private def withPath(path: String): ResponseCookie =
87     copy(path = Some(path))
88 
89   private def withSecure(secure: Boolean): ResponseCookie =
90     copy(secure = secure)
91 
92   private def withHttpOnly(httpOnly: Boolean): ResponseCookie =
93     copy(httpOnly = httpOnly)
94 
95   private def withSameSite(sameSite: SameSite): ResponseCookie =
96     copy(sameSite = Some(sameSite))
97 
98   private def addExtension(extension: String): ResponseCookie =
99     copy(extension = self.extension.fold(Option(extension))(old => Some(old + "; " + extension)))
100 }

102 object ResponseCookie {
103   private[http4s] val parser: Parser[ResponseCookie] = {
104     import Parser.{char, charIn, failWith, ignoreCase, pure}
105     import Rfc2616.Rfc1123Date
106     import Rfc5234.digit
107     import RelaxedCookies.{cookieName, cookieValue}
108 
109     /* cookie-pair       = cookie-name "=" cookie-value */
110     val cookiePair = (cookieName <* char('=')) ~ cookieValue
111 
112     /* sane-cookie-date  = <rfc1123-date, defined in [RFC2616], Section 3.3.1> */
113     val saneCookieDate = Rfc2616.rfc1123Date.flatMap { (date: Rfc1123Date) =>
114       import date._
115       try {
116         val dt = ZonedDateTime.of(year, month, day, hour, min, sec, 0, ZoneOffset.UTC)
117         pure(org.http4s.HttpDate.unsafeFromZonedDateTime(dt))
118       } catch {
119         case _: DateTimeException =>
120           failWith(s"Invalid sane-cookie-date: $year-$month-$day $hour:$min:$sec")
121       }
122     }
123 
124     /* expires-av        = "Expires=" sane-cookie-date */
125     val expiresAv = ignoreCase("Expires=").with1 *> saneCookieDate.map {
126       (expires: HttpDate) => (cookie: ResponseCookie) => cookie.withExpires(expires)
127     }
128 
129     /* non-zero-digit    = %x31-39
130      *                   ; digits 1 through 9
131      */
132     val nonZeroDigit = charIn(0x31.toChar to 0x39.toChar)
133 
134     /* max-age-av        = "Max-Age=" non-zero-digit *DIGIT
135      *                   ; In practice, both expires-av and max-age-av
136      *                   ; are limited to dates representable by the
137      *                   ; user agent.
138      */
139     val maxAgeAv = ignoreCase("Max-Age=") *> (nonZeroDigit ~ digit.rep0).string.map {
140       (maxAge: String) => (cookie: ResponseCookie) => cookie.withMaxAge(maxAge.toLong)
141     }
142 
143     /* domain-value      = <subdomain>
144      *                   ; defined in [RFC1034], Section 3.5, as
145      *                   ; *enhanced by [RFC1123], Section 2.1
146      */
147     def domainValue = Rfc1034.subdomain
148 
149     /* domain-av         = "Domain=" domain-value
150      *
151      * But https://datatracker.ietf.org/doc/html/rfc6265#section-5.2.3 mandates
152      * a leading dot, which is invalid per domain-value.
153      */
154     val domainAv = ignoreCase("Domain=") *> (char('.').? ~ domainValue).string.map {
155       (domain: String) => (cookie: ResponseCookie) => cookie.withDomain(domain)
156     }
157 
158     /* av-octet = %x20-3A / %x3C-7E
159      *          ; any CHAR except CTLs or ";"
160      *
161      * as proposed by https://www.rfc-editor.org/errata/eid3444
162      */
163     val avOctet = charIn(0x20.toChar to 0x3a.toChar)
164       .orElse(charIn(0x3c.toChar to 0x7e.toChar))
165 
166     /* path-value = *av-octet
167      *
168      * as amended by https://www.rfc-editor.org/errata/eid3444
169      */
170     val pathValue = avOctet.rep0.string
171 
172     /* path-av           = "Path=" path-value */
173     val pathAv = ignoreCase("Path=") *> pathValue.map { case path: String =>
174       (cookie: ResponseCookie) => cookie.withPath(path)
175     }
176 
177     /* secure-av         = "Secure" */
178     val secureAv = ignoreCase("Secure").as { (cookie: ResponseCookie) =>
179       cookie.withSecure(true)
180     }
181 
182     /* httponly-av       = "HttpOnly" */
183     val httponlyAv = ignoreCase("HttpOnly").as { (cookie: ResponseCookie) =>
184       cookie.withHttpOnly(true)
185     }
186 
187     /* samesite-value    = "Strict" / "Lax" / "None"
188      * from https://datatracker.ietf.org/doc/html/draft-ietf-httpbis-rfc6265bis-06
189      *
190      * Anything but "strict" or "lax" is "none"
191      */
192     val samesiteValue =
193       ignoreCase("strict")
194         .as(SameSite.Strict)
195         .orElse(ignoreCase("lax").as(SameSite.Lax))
196         .orElse(avOctet.rep0.as(SameSite.None))
197 
198     /* samesite-av       = "SameSite" BWS "=" BWS samesite-value
199      * from https://datatracker.ietf.org/doc/html/draft-ietf-httpbis-rfc6265bis-06
200      */
201     val samesiteAv = ignoreCase("SameSite=") *> samesiteValue.map {
202       (sameSite: SameSite) => (cookie: ResponseCookie) => cookie.withSameSite(sameSite)
203     }
204 
205     /* extension-av = *av-octet
206      *
207      * as amended by https://www.rfc-editor.org/errata/eid3444
208      */
209     val extensionAv = avOctet.rep.string.map { (ext: String) => (cookie: ResponseCookie) =>
210       cookie.addExtension(ext)
211     }
212 
213     /* cookie-av         = expires-av / max-age-av / domain-av /
214      *                     path-av / secure-av / httponly-av /
215      *                     extension-av
216      */
217     val cookieAv = expiresAv
218       .orElse(maxAgeAv)
219       .orElse(domainAv)
220       .orElse(pathAv)
221       .orElse(secureAv)
222       .orElse(httponlyAv)
223       .orElse(samesiteAv)
224       .orElse(extensionAv)
225 
226     /* set-cookie-string = cookie-pair *( ";" SP cookie-av ) */
227     (cookiePair ~ (char(';') *> char(' ').rep0 *> cookieAv).rep0).map { case ((name, value), avs) =>
228       avs.foldLeft(ResponseCookie(name, value))((p, f) => f(p))
229     }
230   }
